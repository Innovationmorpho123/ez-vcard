066             * Gets the ENCODING sub type. This is used when the type value is encoded
067             * in a form other than plain text.
068             * <p>
069             * <b>Supported versions:</b> <code>2.1, 3.0</code>
070             * </p>
071             * @return the encoding or null if not found
072             */
073            public EncodingParameter getEncoding() {
074                    String value = first(EncodingParameter.NAME);
075                    if (value == null) {
076                            return null;
077                    }
078                    EncodingParameter encoding = EncodingParameter.valueOf(value);
079                    if (encoding == null) {
080                            encoding = new EncodingParameter(value);
081                    }
082                    return encoding;
083            }

231            /**
232             * <p>
233             * Gets the preference value. The lower the number, the more preferred this
234             * type is compared to other types in the vCard with the same name. If a
235             * type doesn't have a preference value, then it is considered the
236             * <b>least</b> preferred.
237             * </p>
238             * 
239             * <p>
240             * In the vCard below, the address on the second row is the most preferred
241             * because it has the lowest PREF value.
242             * </p>
243             * 
244             * <pre>
245             * ADR;TYPE=work;PREF=2:
246             * ADR;TYPE=work;PREF=1:
247             * ADR;TYPE=home:
248             * </pre>
249             * 
250             * <p>
251             * Preference values must be numeric and must be between 1 and 100.
252             * </p>
253             * 
254             * <p>
255             * <b>Supported versions:</b> <code>4.0</code>
256             * </p>
257             * @return the preference value or null if it doesn't exist or null if it
258             * couldn't be parsed into a number
259             */
260            public Integer getPref() {
261                    String pref = first("PREF");
262                    if (pref == null) {
263                            return null;
264                    }
265    
266                    try {
267                            return Integer.valueOf(pref);
268                    } catch (NumberFormatException e) {
269                            return null;
270                    }
271            }

274             * <p>
275             * Sets the preference value. The lower the number, the more preferred this
276             * type is compared to other types in the vCard with the same name. If a
277             * type doesn't have a preference value, then it is considered the
278             * <b>least</b> preferred.
279             * </p>
280             * 
281             * <p>
282             * In the vCard below, the address on the second row is the most preferred
283             * because it has the lowest PREF value.
284             * </p>
285             * 
286             * <pre>
287             * ADR;TYPE=work;PREF=2:
288             * ADR;TYPE=work;PREF=1:
289             * ADR;TYPE=home:
290             * </pre>
291             * 
292             * <p>
293             * Preference values must be numeric and must be between 1 and 100.
294             * </p>
295             * 
296             * <p>
297             * <b>Supported versions:</b> <code>4.0</code>
298             * </p>
299             * @param pref the preference value or null to remove
300             * @throws IllegalArgumentException if the value is not between 1 and 100
301             */
302            public void setPref(Integer pref) {
303                    if (pref != null && (pref < 1 || pref > 100)) {
304                            throw new IllegalArgumentException("Preference value must be between 1 and 100 inclusive.");
305                    }
306                    String value = (pref == null) ? null : pref.toString();
307                    replace("PREF", value);
308            }

311             * Gets the ALTID parameter value. This is used to specify alternative
312             * representations of the same type.
313             * 
314             * <p>
315             * For example, a vCard may contain multiple NOTE types that each have the
316             * same ALTID. This means that each NOTE contains a different representation
317             * of the same information. In the example below, the first three NOTEs have
318             * the same ALTID. They each contain the same message, but each is written
319             * in a different language. The other NOTEs have different (or absent)
320             * ALTIDs, which means they are not associated with the top three.
321             * </p>
322             * 
323             * <pre>
324             * NOTE;ALTID=1;LANGUAGE=en: Hello world!
325             * NOTE;ALTID=1;LANGUAGE=fr: Bonjour tout le monde!
326             * NOTE;ALTID=1;LANGUAGE=es: �Hola, mundo!
327             * NOTE;ALTID=2;LANGUAGE=de: Meine Lieblingsfarbe ist blau.
328             * NOTE;ALTID=2;LANGUAGE=en: My favorite color is blue.
329             * NOTE: This vCard will self-destruct in 5 seconds.
330             * </pre>
331             * 
332             * <p>
333             * <b>Supported versions:</b> <code>4.0</code>
334             * </p>
335             * @return the ALTID or null if it doesn't exist
336             */

372            /**
373             * Gets the GEO parameter value. This is used to associate global
374             * positioning information with a vCard type. It can be used with the ADR
375             * type.
376             * <p>
377             * <b>Supported versions:</b> <code>4.0</code>
378             * </p>
379             * @return the latitude (index 0) and longitude (index 1) or null if not
380             * present or null if the parameter value was in an incorrect format
381             */
382            public double[] getGeo() {
383                    String value = first("GEO");
384                    if (value == null) {
385                            return null;
386                    }
387    
388                    try {
389                            GeoUri geoUri = new GeoUri(value);
390                            return new double[] { geoUri.getCoordA(), geoUri.getCoordB() };
391                    } catch (IllegalArgumentException e) {
392                            return null;
393                    }
394            }

480            /**
481             * <p>
482             * Gets all PID parameter values. PIDs can exist on any property where
483             * multiple instances are allowed (such as EMAIL or ADR, but not N because
484             * only 1 instance of N is allowed).
485             * </p>
486             * <p>
487             * When used in conjunction with the CLIENTPIDMAP property, it allows an
488             * individual property instance to be uniquely identifiable. This feature is
489             * made use of when two different versions of the same vCard have to be
490             * merged together (called "synchronizing").
491             * </p>
492             * <p>
493             * <b>Supported versions:</b> <code>4.0</code>
494             * </p>
495             * @return the PID values or empty set if there are none. Index 0 is the
496             * local ID and index 1 is the ID used to reference the CLIENTPIDMAP
497             * property. Index 0 will never be null, but index 1 may be null.
498             */
499            public List<Integer[]> getPids() {
500                    List<String> values = get("PID");
501                    List<Integer[]> pids = new ArrayList<Integer[]>(values.size());
502                    for (String value : values) {
503                            try {
504                                    String split[] = value.split("\\.");
505                                    Integer pid[] = new Integer[2];
506                                    pid[0] = Integer.valueOf(split[0]);
507                                    if (split.length > 1) {
508                                            pid[1] = Integer.valueOf(split[1]);
509                                    }
510                                    pids.add(pid);
511                            } catch (NumberFormatException e) {
512                                    //skip
513                            }
514                    }
515                    return pids;
516            }

636             */

638                    String index = first("INDEX");
639                    if (index == null) {
640                            return null;
641                    }

644                            return Integer.valueOf(index);
645                    } catch (NumberFormatException e) {
646                            return null;
647                    }
