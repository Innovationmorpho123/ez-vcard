035    /**
036     * Represents the TYPE parameter of the {@link AddressType} and
037     * {@link LabelType} types.
038     * <p>
039     * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
040     * </p>
041     * @author Michael Angstadt
042     */
043    public class AddressTypeParameter extends TypeParameter {
044            /**
045             * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
046             */
047            public static final AddressTypeParameter HOME = new AddressTypeParameter("home");
048    
049            /**
050             * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
051             */
052            public static final AddressTypeParameter WORK = new AddressTypeParameter("work");
053    
054            /**
055             * <b>Supported versions:</b> <code>2.1, 3.0</code>
056             */
057            public static final AddressTypeParameter DOM = new AddressTypeParameter("dom");
058    
059            /**
060             * <b>Supported versions:</b> <code>2.1, 3.0</code>
061             */
062            public static final AddressTypeParameter INTL = new AddressTypeParameter("intl");
063    
064            /**
065             * <b>Supported versions:</b> <code>2.1, 3.0</code>
066             */
067            public static final AddressTypeParameter POSTAL = new AddressTypeParameter("postal");
068    
069            /**
070             * <b>Supported versions:</b> <code>2.1, 3.0</code>
071             */
072            public static final AddressTypeParameter PARCEL = new AddressTypeParameter("parcel");
073    
074            /**
075             * <b>Supported versions:</b> <code>2.1, 3.0</code>
076             */
077            public static final AddressTypeParameter PREF = new AddressTypeParameter("pref");
078    
079            /**
080             * Use of this constructor is discouraged and should only be used for
081             * defining non-standard TYPEs. Please use one of the predefined static
082             * objects.
083             * @param value the type value (e.g. "home")
084             */
085            public AddressTypeParameter(String value) {
086                    super(value);
087            }
088    
089            /**
090             * Searches the static objects in this class for one that has a certain type
091             * value.
092             * @param value the type value to search for (e.g. "home")
093             * @return the object or null if not found
094             */
095            public static AddressTypeParameter valueOf(String value) {
096                    return findByValue(value, AddressTypeParameter.class);
097            }
098    }
