055    /**
056     * Writes {@link VCard} objects to a templated HTML page (hCard format).
057     * @author Michael Angstadt
058     * @see <a
059     * href="http://microformats.org/wiki/hcard">http://microformats.org/wiki/hcard</a>
060     */
061    public class HCardPage {
062            private static final Template template;
063            static {
064                    Configuration cfg = new Configuration();
065                    cfg.setClassForTemplateLoading(HCardPage.class, "");
066                    cfg.setObjectWrapper(new DefaultObjectWrapper());
067                    cfg.setWhitespaceStripping(true);
068                    try {
069                            template = cfg.getTemplate("hcard-template.html");
070                    } catch (IOException e) {
071                            //should never be thrown because it's always on the classpath
072                            throw new RuntimeException(e);
073                    }
074            }
075    
076            private final List<VCard> vcards = new ArrayList<VCard>();
077    
078            /**
079             * Adds a vCard to the HTML page.
080             * @param vcard the vCard to add
081             */
082            public void addVCard(VCard vcard) {
083                    vcards.add(vcard);
084            }
085    
086            /**
087             * Writes the HTML document to a string.
088             * @return the HTML document
089             */
090            public String write() {
091                    StringWriter sw = new StringWriter();
092                    try {
093                            write(sw);
094                    } catch (IOException e) {
095                            //never thrown because we're writing to a string
096                    }
097                    return sw.toString();
098            }
099    
100            /**
101             * Writes the HTML document to an output stream.
102             * @param out the output stream
103             * @throws IOException if there's a problem writing to the output stream
104             */
105            public void write(OutputStream out) throws IOException {
106                    write(new OutputStreamWriter(out));
107            }
108    
109            /**
110             * Writes the HTML document to a file.
111             * @param file the file
112             * @throws IOException if there's a problem writing to the file
113             */
114            public void write(File file) throws IOException {
115                    FileWriter writer = null;
116                    try {
117                            writer = new FileWriter(file);
118                            write(writer);
119                    } finally {
120                            IOUtils.closeQuietly(writer);
121                    }
122            }
123    
124            /**
125             * Writes the HTML document to a writer.
126             * @param writer the writer
127             * @throws IOException if there's a problem writing to the writer
128             */
129            public void write(Writer writer) throws IOException {
130                    Map<String, Object> map = new HashMap<String, Object>();
131                    map.put("vcards", vcards);
132                    map.put("dataUri", new DataUriGenerator());
133                    map.put("translucentBg", readImage("translucent-bg.png", ImageTypeParameter.PNG));
134                    map.put("noProfile", readImage("no-profile.png", ImageTypeParameter.PNG));
135                    map.put("ezVCardVersion", Ezvcard.VERSION);
136                    map.put("ezVCardUrl", Ezvcard.URL);
137                    try {
138                            template.process(map, writer);
139                    } catch (TemplateException e) {
140                            //this should never be thrown because we're always using the same template (it is hard-coded and cannot be changed by the user)
141                            throw new RuntimeException(e);
142                    }
143                    writer.flush();
144            }
145    
146            /**
147             * Reads an image from the classpath.
148             * @param name the file name, relative to this class
149             * @param mediaType the media type of the image
150             * @return the image
151             * @throws IOException
152             */
153            private PhotoType readImage(String name, ImageTypeParameter mediaType) throws IOException {
154                    return new PhotoType(getClass().getResourceAsStream(name), mediaType);
155            }
156    
157            /**
158             * Generates data URIs for the freemarker template.
159             */
160            public static class DataUriGenerator {
161                    public String generate(String contentType, byte[] data) {
162                            return new DataUri(contentType, data).toString();
163                    }
164            }
165    }
