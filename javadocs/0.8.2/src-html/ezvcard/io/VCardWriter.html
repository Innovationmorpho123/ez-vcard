073            /**
074             * The characters that are not valid in parameter values and that should be
075             * removed.
076             */
077            private static final Map<VCardVersion, BitSet> invalidParamValueChars = new HashMap<VCardVersion, BitSet>();
078            static {
079                    BitSet controlChars = new BitSet(128);
080                    controlChars.set(0, 31);
081                    controlChars.set(127);
082                    controlChars.set('\t', false); //allow
083                    controlChars.set('\n', false); //allow
084                    controlChars.set('\r', false); //allow
085    
086                    //2.1
087                    {
088                            BitSet bitSet = new BitSet(128);
089                            bitSet.or(controlChars);
090    
091                            bitSet.set(',');
092                            bitSet.set('.');
093                            bitSet.set(':');
094                            bitSet.set('=');
095                            bitSet.set('[');
096                            bitSet.set(']');
097    
098                            invalidParamValueChars.put(VCardVersion.V2_1, bitSet);
099                    }
100    
101                    //3.0, 4.0
102                    {
103                            BitSet bitSet = new BitSet(128);
104                            bitSet.or(controlChars);
105    
106                            invalidParamValueChars.put(VCardVersion.V3_0, bitSet);
107                            invalidParamValueChars.put(VCardVersion.V4_0, bitSet);
108                    }
109            }

203             * Creates a vCard writer.
204             * @param writer the writer to write the vCard to
205             * @param targetVersion the version that the vCards should conform to
206             * @param foldingScheme the folding scheme to use or null not to fold at all
207             * @param newline the newline sequence to use

209            public VCardWriter(Writer writer, VCardVersion targetVersion, FoldingScheme foldingScheme, String newline) {
210                    if (writer instanceof FoldedLineWriter || foldingScheme == null) {
211                            //the check for FoldedLineWriter is for writing nested 2.1 vCards (i.e. the AGENT type)
212                            this.writer = writer;
213                    } else {
214                            this.writer = new FoldedLineWriter(writer, foldingScheme.getLineLength(), foldingScheme.getIndent(), newline);
215                    }
216                    this.targetVersion = targetVersion;
217                    this.foldingScheme = foldingScheme;
218                    this.newline = newline;
219            }

284             * </p>
285             * 
286             * <table border="1">

288             * <th>Character</th>
289             * <th>Replacement</th>

402            /**
403             * Writes a vCard to the stream.
404             * @param vcard the vCard to write
405             * @throws IOException if there's a problem writing to the output stream
406             */
407            public void write(final VCard vcard) throws IOException {
408                    warnings.clear();
409    
410                    if (targetVersion == VCardVersion.V2_1 || targetVersion == VCardVersion.V3_0) {
411                            if (vcard.getStructuredName() == null) {
412                                    warnings.add("vCard version " + targetVersion + " requires that a structured name property be defined.");
413                            }
414                    }
415    
416                    if (targetVersion == VCardVersion.V3_0 || targetVersion == VCardVersion.V4_0) {
417                            if (vcard.getFormattedName() == null) {
418                                    warnings.add("vCard version " + targetVersion + " requires that a formatted name property be defined.");
419                            }
420                    }
421    
422                    List<VCardType> typesToAdd = new ArrayList<VCardType>();
423                    typesToAdd.add(new TextType("BEGIN", "VCARD"));
424                    typesToAdd.add(new TextType("VERSION", targetVersion.getVersion()));
425    
426                    for (VCardType type : vcard) {
427                            if (addProdId && type instanceof ProdIdType) {
428                                    //do not add the PRODID in the vCard if "addProdId" is true
429                                    continue;
430                            }
431    
432                            //determine if this type is supported by the target version
433                            if (!supportsTargetVersion(type)) {
434                                    addWarning("This property is not supported by vCard version " + targetVersion + " and will not be added to the vCard.  Supported versions are: " + Arrays.toString(type.getSupportedVersions()), type.getTypeName());
435                                    continue;
436                            }
437    
438                            //check for correct KIND value if there are MEMBER types
439                            if (type instanceof MemberType && (vcard.getKind() == null || !vcard.getKind().isGroup())) {
440                                    addWarning("Value must be set to \"group\" if the vCard contains " + MemberType.NAME + " properties.", KindType.NAME);
441                                    continue;
442                            }
443    
444                            typesToAdd.add(type);
445    
446                            //add LABEL types for each ADR type if the target version is 2.1 or 3.0
447                            if (type instanceof AddressType && targetVersion != VCardVersion.V4_0) {
448                                    AddressType adr = (AddressType) type;
449                                    String labelStr = adr.getLabel();
450                                    if (labelStr != null) {
451                                            LabelType label = new LabelType(labelStr);
452                                            for (AddressTypeParameter t : adr.getTypes()) {
453                                                    label.addType(t);
454                                            }
455                                            typesToAdd.add(label);
456                                    }
457                            }
458                    }
459    
460                    //add an extended type saying it was generated by this library
461                    if (addProdId) {
462                            EzvcardProdIdType prodId = new EzvcardProdIdType(targetVersion);
463                            typesToAdd.add(prodId);
464                    }
465    
466                    typesToAdd.add(new TextType("END", "VCARD"));
467    
468                    List<String> warningsBuf = new ArrayList<String>();
469                    for (VCardType type : typesToAdd) {
470                            //marshal the value
471                            warningsBuf.clear();
472                            String value = null;
473                            VCard nested = null;
474                            try {
475                                    value = type.marshalText(targetVersion, warningsBuf, compatibilityMode);
476                            } catch (SkipMeException e) {
477                                    warningsBuf.add("Property has requested that it be skipped: " + e.getMessage());
478                                    continue;
479                            } catch (EmbeddedVCardException e) {
480                                    nested = e.getVCard();
481                            } finally {
482                                    for (String warning : warningsBuf) {
483                                            addWarning(warning, type.getTypeName());
484                                    }
485                            }
486    
487                            //marshal the sub types
488                            warningsBuf.clear();
489                            VCardSubTypes subTypes;
490                            try {
491                                    subTypes = type.marshalSubTypes(targetVersion, warningsBuf, compatibilityMode, vcard);
492                            } finally {
493                                    for (String warning : warningsBuf) {
494                                            addWarning(warning, type.getTypeName());
495                                    }
496                            }
497    
498                            //sanitize value for safe inclusion in the vCard
499                            if (value != null) {
500                                    if (targetVersion == VCardVersion.V2_1) {
501                                            if (VCardStringUtils.containsNewlines(value)) {
502                                                    //2.1 does not support the "\n" escape sequence (see "Delimiters" sub-section in section 2 of the specs)
503                                                    QuotedPrintableCodec codec = new QuotedPrintableCodec();
504                                                    try {
505                                                            value = codec.encode(value);
506                                                            subTypes.setEncoding(EncodingParameter.QUOTED_PRINTABLE);
507                                                    } catch (EncoderException e) {
508                                                            addWarning("A unexpected error occurred while encoding the property's value into \"quoted-printable\" encoding.  Value will not be encoded: " + e.getMessage(), type.getTypeName());
509                                                            value = VCardStringUtils.escapeNewlines(value);
510                                                    }
511                                            }
512                                    } else {
513                                            value = VCardStringUtils.escapeNewlines(value);
514                                    }
515                            }
516    
517                            StringBuilder sb = new StringBuilder();
518    
519                            //write the group
520                            if (type.getGroup() != null) {
521                                    sb.append(type.getGroup());
522                                    sb.append('.');
523                            }
524    
525                            //write the type name
526                            sb.append(type.getTypeName());
527    
528                            //write the Sub Types
529                            for (Map.Entry<String, List<String>> subType : subTypes) {
530                                    String subTypeName = subType.getKey();
531                                    List<String> subTypeValues = subType.getValue();
532                                    if (!subTypeValues.isEmpty()) {
533                                            if (targetVersion == VCardVersion.V2_1) {
534                                                    boolean typeSubType = TypeParameter.NAME.equalsIgnoreCase(subTypeName);
535                                                    for (String subTypeValue : subTypeValues) {
536                                                            subTypeValue = sanitizeSubTypeValue(subTypeValue, subTypeName, type.getTypeName());
537    
538                                                            if (typeSubType) {
539                                                                    //e.g. ADR;HOME;WORK:
540                                                                    sb.append(';').append(subTypeValue.toUpperCase());
541                                                            } else {
542                                                                    //e.g. ADR;FOO=bar;FOO=car:
543                                                                    sb.append(';').append(subTypeName).append('=').append(subTypeValue);
544                                                            }
545                                                    }
546                                            } else {
547                                                    //e.g. ADR;TYPE=home,work,"another,value":
548    
549                                                    sb.append(';').append(subTypeName).append('=');
550                                                    for (String subTypeValue : subTypeValues) {
551                                                            subTypeValue = sanitizeSubTypeValue(subTypeValue, subTypeName, type.getTypeName());
552    
553                                                            //surround with double quotes if contains special chars
554                                                            if (quoteMeRegex.matcher(subTypeValue).matches()) {
555                                                                    sb.append('"');
556                                                                    sb.append(subTypeValue);
557                                                                    sb.append('"');
558                                                            } else {
559                                                                    sb.append(subTypeValue);
560                                                            }
561    
562                                                            sb.append(',');
563                                                    }
564                                                    sb.deleteCharAt(sb.length() - 1); //chomp last comma
565                                            }
566                                    }
567                            }
568    
569                            sb.append(':');
570    
571                            writer.write(sb.toString());
572    
573                            //write the value
574                            if (nested == null) {
575                                    writer.write(value);
576                                    writer.write(newline);
577                            } else {
578                                    if (targetVersion == VCardVersion.V2_1) {
579                                            writer.write(newline);
580    
581                                            //write a nested vCard (2.1 style)
582                                            VCardWriter agentWriter = new VCardWriter(writer, targetVersion);
583                                            agentWriter.setAddProdId(false);
584                                            agentWriter.setCompatibilityMode(compatibilityMode);
585                                            try {
586                                                    agentWriter.write(nested);
587                                            } finally {
588                                                    for (String w : agentWriter.getWarnings()) {
589                                                            addWarning("Problem marshalling nested vCard value: " + w, type.getTypeName());
590                                                    }
591                                            }
592                                    } else {
593                                            //write an embedded vCard (3.0 style)
594                                            StringWriter sw = new StringWriter();
595                                            VCardWriter agentWriter = new VCardWriter(sw, targetVersion, null, "\n");
596                                            agentWriter.setAddProdId(false);
597                                            agentWriter.setCompatibilityMode(compatibilityMode);
598                                            try {
599                                                    agentWriter.write(nested);
600                                            } finally {
601                                                    for (String w : agentWriter.getWarnings()) {
602                                                            addWarning("Problem marshalling embedded vCard value: " + w, type.getTypeName());
603                                                    }
604                                            }
605    
606                                            String vCardStr = sw.toString();
607                                            vCardStr = VCardStringUtils.escape(vCardStr);
608                                            vCardStr = VCardStringUtils.escapeNewlines(vCardStr);
609                                            writer.write(vCardStr);
610                                            writer.write(newline);
611                                    }
612                            }
613                    }
614            }

616            private String sanitizeSubTypeValue(String value, String subTypeName, String typeName) {
617                    String modifiedValue = null;
618                    boolean subTypeValueChanged = false;
619    
620                    switch (targetVersion) {
621                    case V2_1:
622                            //remove invalid characters
623                            modifiedValue = removeInvalidSubTypeValueChars(value);
624    
625                            //replace newlines with spaces
626                            modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll(" ");
627    
628                            //check to see if value was changed
629                            subTypeValueChanged = (value != modifiedValue);
630    
631                            //escape backslashes
632                            modifiedValue = modifiedValue.replace("\\", "\\\\");
633    
634                            //escape semi-colons (see section 2)
635                            modifiedValue = modifiedValue.replace(";", "\\;");
636    
637                            break;
638    
639                    case V3_0:
640                    case V4_0:
641                            //remove invalid characters
642                            modifiedValue = removeInvalidSubTypeValueChars(value);
643    
644                            if (caretEncodingEnabled) {
645                                    subTypeValueChanged = (modifiedValue != value);
646                                    modifiedValue = applyCaretEncoding(modifiedValue);
647                            } else {
648                                    //replace double quotes with single quotes
649                                    modifiedValue = modifiedValue.replace('"', '\'');
650                                    subTypeValueChanged = (modifiedValue != value);
651    
652                                    //escape backslashes
653                                    modifiedValue = modifiedValue.replace("\\", "\\\\");
654    
655                                    //escape newlines
656                                    if (targetVersion == VCardVersion.V3_0) {
657                                            modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll(" ");
658                                    } else {
659                                            //for the "LABEL" parameter
660                                            modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll("\\\\\\n");
661                                    }
662                            }
663    
664                            break;
665                    }
666    
667                    if (subTypeValueChanged) {
668                            addWarning("\"" + subTypeName + "\" parameter contained one or more characters which are not allowed in vCard " + targetVersion.getVersion() + " parameter values.  These characters were removed.", typeName);
669                    }
670    
671                    return modifiedValue;
672            }
673    
674            private String removeInvalidSubTypeValueChars(String value) {
675                    BitSet invalidChars = invalidParamValueChars.get(targetVersion);
676                    StringBuilder sb = new StringBuilder(value.length());
677    
678                    for (int i = 0; i < value.length(); i++) {
679                            char ch = value.charAt(i);
680                            if (!invalidChars.get(ch)) {
681                                    sb.append(ch);
682                            }
683                    }
684    
685                    return (sb.length() == value.length()) ? value : sb.toString();
686            }
687    
688            private String applyCaretEncoding(String value) {
689                    value = value.replace("^", "^^");
690                    value = newlineRegex.matcher(value).replaceAll("^n");
691                    value = value.replace("\"", "^'");
692                    return value;
693            }

695            private boolean supportsTargetVersion(VCardType type) {
696                    for (VCardVersion version : type.getSupportedVersions()) {
697                            if (version == targetVersion) {
698                                    return true;
699                            }
700                    }
701                    return false;
702            }
