036    /**
037     * Thrown during the parsing of a jCard, when a jCard is not formatted in the
038     * correct way (the JSON syntax is valid, but it's not in the correct jCard
039     * format).
040     * @author Michael Angstadt
041     */
042    @SuppressWarnings("serial")
043    public class JCardParseException extends VCardException {
044            /**
045             * Creates a jCard parse exception.
046             * @param expected the JSON token that the parser was expecting
047             * @param actual the actual JSON token
048             */
049            public JCardParseException(JsonToken expected, JsonToken actual) {
050                    super("Expected " + expected + " but was " + actual + ".");
051            }
052    }
