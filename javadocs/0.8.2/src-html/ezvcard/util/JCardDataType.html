038    /**
039     * Contains all the jCard data types.
040     * @author Michael Angstadt
041     */
042    public class JCardDataType {
043            public static final JCardDataType TEXT = new JCardDataType("text");
044            public static final JCardDataType URI = new JCardDataType("uri");
045            public static final JCardDataType INTEGER = new JCardDataType("integer");
046            public static final JCardDataType BOOLEAN = new JCardDataType("boolean");
047            public static final JCardDataType FLOAT = new JCardDataType("float");
048            public static final JCardDataType DATE = new JCardDataType("date");
049            public static final JCardDataType TIME = new JCardDataType("time");
050            public static final JCardDataType DATE_TIME = new JCardDataType("date-time");
051            public static final JCardDataType DATE_AND_OR_TIME = new JCardDataType("date-and-or-time");
052            public static final JCardDataType TIMESTAMP = new JCardDataType("timestamp");
053            public static final JCardDataType UTC_OFFSET = new JCardDataType("utc-offset");
054            public static final JCardDataType LANGUAGE_TAG = new JCardDataType("language-tag");
055            public static final JCardDataType UNKNOWN = new JCardDataType("unknown");
056            static final Collection<JCardDataType> custom = Collections.synchronizedList(new ArrayList<JCardDataType>());
057    
058            private final String name;
059    
060            /**
061             * Creates a hCard data type.
062             * @param name the data type name (e.g. "text")
063             */
064            private JCardDataType(String name) {
065                    this.name = name;
066            }
067    
068            /**
069             * Gets the name of the data type
070             * @return the data type name (e.g. "text")
071             */
072            public String getName() {
073                    return name;
074            }
075    
076            @Override
077            public String toString() {
078                    return name;
079            }
080    
081            /**
082             * Searches for a data type object by name.
083             * @param name the data type (e.g. "text")
084             * @return the data type object
085             */
086            public static JCardDataType find(String name) {
087                    return find(name, false);
088            }
089    
090            /**
091             * Searches for a data type object by name, and creates a new object if one
092             * cannot be found.
093             * @param name the data type (e.g. "text")
094             * @return the data type object
095             */
096            public static JCardDataType get(String name) {
097                    return find(name, true);
098            }
099    
100            /**
101             * Searches for a data type object by name.
102             * @param name the data type (e.g. "text")
103             * @param create true to create a new object if one cannot be found, or
104             * false to return null if an object cannot be found
105             * @return the data type object
106             */
107            private static JCardDataType find(String name, boolean create) {
108                    SearchHandler find = new SearchHandler(name);
109                    forEachDataType(find);
110    
111                    JCardDataType match = find.match;
112                    if (create && match == null) {
113                            match = new JCardDataType(name);
114                            custom.add(match);
115                    }
116                    return match;
117            }
118    
119            /**
120             * Gets all available data types.
121             * @return all available data types
122             */
123            public static Collection<JCardDataType> all() {
124                    final Collection<JCardDataType> all = new ArrayList<JCardDataType>();
125    
126                    forEachDataType(new Handler() {
127                            public boolean handle(JCardDataType dataType) {
128                                    all.add(dataType);
129                                    return true;
130                            }
131                    });
132    
133                    return all;
134            }
135    
136            private static void forEachDataType(Handler handler) {
137                    Class<JCardDataType> clazz = JCardDataType.class;
138                    for (Field field : clazz.getFields()) {
139                            int modifiers = field.getModifiers();
140                            //@formatter:off
141                            if (Modifier.isStatic(modifiers) &&
142                                    Modifier.isPublic(modifiers) &&
143                                    field.getDeclaringClass() == clazz &&
144                                    field.getType() == clazz) {
145                                    //@formatter:on
146                                    try {
147                                            Object obj = field.get(null);
148                                            if (obj != null && !handler.handle((JCardDataType) obj)) {
149                                                    return;
150                                            }
151                                    } catch (Exception ex) {
152                                            //reflection error
153                                    }
154                            }
155                    }
156    
157                    for (JCardDataType dataType : custom) {
158                            if (!handler.handle(dataType)) {
159                                    return;
160                            }
161                    }
162            }
163    
164            private static interface Handler {
165                    /**
166                     * @param dataType
167                     * @return true to continue, false to break out of the loop
168                     */
169                    boolean handle(JCardDataType dataType);
170            }
171    
172            private static class SearchHandler implements Handler {
173                    public JCardDataType match = null;
174                    private final String name;
175    
176                    public SearchHandler(String name) {
177                            this.name = name;
178                    }
179    
180                    public boolean handle(JCardDataType dataType) {
181                            if (name.equalsIgnoreCase(dataType.name)) {
182                                    match = dataType;
183                                    return false;
184                            }
185                            return true;
186                    }
187            }
188    }
