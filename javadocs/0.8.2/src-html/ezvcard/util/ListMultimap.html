041    /**
042     * A multimap that uses {@link List} objects to store its values. The internal
043     * {@link Map} implementation is a {@link LinkedHashMap} that uses
044     * {@link ArrayList} for its values.
045     * @author Michael Angstadt
046     * @param <K> the key
047     * @param <V> the value
048     */
049    public class ListMultimap<K, V> implements Iterable<Map.Entry<K, List<V>>> {
050            private final Map<K, List<V>> map = new LinkedHashMap<K, List<V>>();
051    
052            /**
053             * Creates an empty multimap.
054             */
055            public ListMultimap() {
056                    //empty
057            }
058    
059            /**
060             * Creates a copy of an existing multimap.
061             * @param orig the multimap to copy from
062             */
063            public ListMultimap(ListMultimap<K, V> orig) {
064                    for (Map.Entry<K, List<V>> entry : orig) {
065                            List<V> values = new ArrayList<V>(entry.getValue());
066                            map.put(entry.getKey(), values);
067                    }
068            }
069    
070            /**
071             * Adds a value to the multimap.
072             * @param key the key
073             * @param value the value to add
074             */
075            public void put(K key, V value) {
076                    List<V> values = get(key, true);
077                    values.add(value);
078            }
079    
080            /**
081             * Adds multiple values to the multimap.
082             * @param key the key
083             * @param values the values to add
084             */
085            public void putAll(K key, Collection<V> values) {
086                    List<V> existingValues = get(key, true);
087                    existingValues.addAll(values);
088            }
089    
090            /**
091             * Gets the values associated with the key.
092             * @param key the key
093             * @return the list of values or empty list if the key doesn't exist
094             */
095            public List<V> get(K key) {
096                    return get(key, false);
097            }
098    
099            /**
100             * Gets the values associated with the key.
101             * @param key the key
102             * @param add true to add an empty element to the map if the key doesn't
103             * exist, false not to
104             * @return the list of values or empty list if the key doesn't exist
105             */
106            private List<V> get(K key, boolean add) {
107                    key = sanitizeKey(key);
108                    List<V> values = map.get(key);
109                    if (values == null) {
110                            values = new ArrayList<V>();
111                            if (add) {
112                                    map.put(key, values);
113                            }
114                    }
115                    return values;
116            }
117    
118            /**
119             * Gets the first value that's associated with a key.
120             * @param key the key
121             * @return the first value or null if the key doesn't exist
122             */
123            public V first(K key) {
124                    List<V> values = get(key);
125                    return (values == null || values.isEmpty()) ? null : values.get(0);
126            }
127    
128            /**
129             * Determines whether the given key exists.
130             * @param key the key
131             * @return true if the key exists, false if not
132             */
133            public boolean containsKey(K key) {
134                    return map.containsKey(key);
135            }
136    
137            /**
138             * Removes a particular value.
139             * @param key the key
140             * @param value the value to remove
141             * @return true if the multimap contained the value, false if not
142             */
143            public boolean remove(K key, V value) {
144                    List<V> values = map.get(sanitizeKey(key));
145                    if (values != null) {
146                            return values.remove(value);
147                    }
148                    return false;
149            }
150    
151            /**
152             * Removes all the values associated with a key
153             * @param key the key to remove
154             * @return the removed values or empty list if the key doesn't exist
155             */
156            public List<V> removeAll(K key) {
157                    List<V> removed = map.remove(sanitizeKey(key));
158                    return (removed == null) ? Collections.<V> emptyList() : removed;
159            }
160    
161            /**
162             * Replaces all values with the given value.
163             * @param key the key
164             * @param value the value with which to replace all existing values (null
165             * will remove all values)
166             * @return the values that were replaced
167             */
168            public List<V> replace(K key, V value) {
169                    List<V> replaced = removeAll(key);
170                    if (value != null) {
171                            put(key, value);
172                    }
173                    return replaced;
174            }
175    
176            /**
177             * Clears all entries from the multimap.
178             */
179            public void clear() {
180                    map.clear();
181            }
182    
183            /**
184             * Returns all the keys.
185             * @return all the keys
186             */
187            public Set<K> keySet() {
188                    return map.keySet();
189            }
190    
191            /**
192             * Returns all the values.
193             * @return all the values
194             */
195            public Collection<V> values() {
196                    Collection<V> list = new ArrayList<V>();
197                    for (List<V> value : map.values()) {
198                            list.addAll(value);
199                    }
200                    return list;
201            }
202    
203            /**
204             * Determines if the multimap is empty or not.
205             * @return true if it's empty, false if not
206             */
207            public boolean isEmpty() {
208                    return size() == 0;
209            }
210    
211            /**
212             * Returns the number of values in the map.
213             * @return the number of values
214             */
215            public int size() {
216                    int size = 0;
217                    for (List<V> value : map.values()) {
218                            size += value.size();
219                    }
220                    return size;
221            }
222    
223            /**
224             * Gets the underlying {@link Map} object.
225             * @return the underlying {@link Map} object
226             */
227            public Map<K, List<V>> getMap() {
228                    return map;
229            }
230    
231            /**
232             * Modifies a given key before it is used to interact with the internal map.
233             * This method is meant to be overridden by child classes if necessary.
234             * @param key the key
235             * @return the modified key (by default, the key is returned as-is)
236             */
237            protected K sanitizeKey(K key) {
238                    return key;
239            }
240    
241            //@Override
242            public Iterator<Map.Entry<K, List<V>>> iterator() {
243                    return map.entrySet().iterator();
244            }
245    }
