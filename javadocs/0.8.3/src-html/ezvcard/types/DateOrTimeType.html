051    /**
052     * Represents a property whose value contains a date and/or a time (for example,
053     * {@link BirthdayType}).
054     * @author Michael Angstadt
055     */
056    public class DateOrTimeType extends VCardType implements HasAltId {
057            private String text;
058            private Date date;
059            private PartialDate partialDate;
060    
061            /**
062             * True if the "date" or "reduceAccuracyDate" fields have a time component,
063             * false if they just contain a date.
064             */
065            private boolean dateHasTime;
066    
067            /**
068             * Creates a date-and-or-time property.
069             * @param typeName the name of the type (e.g. "BDAY")
070             */
071            public DateOrTimeType(String typeName) {
072                    super(typeName);
073            }
074    
075            /**
076             * Creates a date-and-or-time property.
077             * @param typeName the name of the type (e.g. "BDAY")
078             * @param date the date value
079             */
080            public DateOrTimeType(String typeName, Date date) {
081                    super(typeName);
082                    setDate(date, false);
083            }
084    
085            /**
086             * Creates a date-and-or-time property.
087             * @param typeName the name of the type (e.g. "BDAY")
088             * @param partialDate the date value
089             */
090            public DateOrTimeType(String typeName, PartialDate partialDate) {
091                    super(typeName);
092                    setPartialDate(partialDate);
093            }
094    
095            /**
096             * Gets the date value.
097             * @return the date value or null if not set
098             */
099            public Date getDate() {
100                    return date;
101            }
102    
103            /**
104             * Sets the value of this type to a complete date.
105             * @param date the date
106             * @param dateHasTime true if the date contains a time component, false if
107             * it's just a date
108             */
109            public void setDate(Date date, boolean dateHasTime) {
110                    this.date = date;
111                    this.dateHasTime = dateHasTime;
112                    text = null;
113                    partialDate = null;
114            }
115    
116            /**
117             * Gets the reduced accuracy or truncated date. This is only supported by
118             * vCard 4.0.
119             * @return the reduced accuracy or truncated date or null if not set
120             * @see "<a href="
121             * http://tools.ietf.org/html/rfc6350">RFC 6350</a> p.12-14 for examples"
122             */
123            public PartialDate getPartialDate() {
124                    return partialDate;
125            }
126    
127            /**
128             * <p>
129             * Sets the value of this type to a reduced accuracy or truncated date. This
130             * is only supported by vCard 4.0.
131             * </p>
132             * 
133             * <pre>
134             * BirthdayType bday = new BirthdayType();
135             * bday.setPartialDate(PartialDate.date(null, 4, 20)); //April 20
136             * </pre>
137             * @param partialDate the reduced accuracy or truncated date
138             * @see "<a href="
139             * http://tools.ietf.org/html/rfc6350">RFC 6350</a> p.12-14 for examples"
140             */
141            public void setPartialDate(PartialDate partialDate) {
142                    this.partialDate = partialDate;
143                    dateHasTime = partialDate.hasTimeComponent();
144                    text = null;
145                    date = null;
146            }
147    
148            /**
149             * Gets the text value of this type. This is only supported by vCard 4.0.
150             * @return the text value or null if not set
151             */
152            public String getText() {
153                    return text;
154            }
155    
156            /**
157             * Sets the value of this type to a text string. This is only supported by
158             * vCard 4.0.
159             * @param text the text value
160             */
161            public void setText(String text) {
162                    this.text = text;
163                    date = null;
164                    partialDate = null;
165            }
166    
167            /**
168             * <p>
169             * Gets the type of calendar that is used for a date or date-time property
170             * value.
171             * </p>
172             * <p>
173             * <b>Supported versions:</b> <code>4.0</code>
174             * </p>
175             * @return the type of calendar or null if not found
176             * @see VCardSubTypes#getCalscale
177             */
178            public CalscaleParameter getCalscale() {
179                    return subTypes.getCalscale();
180            }
181    
182            /**
183             * <p>
184             * Sets the type of calendar that is used for a date or date-time property
185             * value.
186             * </p>
187             * <p>
188             * <b>Supported versions:</b> <code>4.0</code>
189             * </p>
190             * @param calscale the type of calendar or null to remove
191             * @see VCardSubTypes#setCalscale
192             */
193            public void setCalscale(CalscaleParameter calscale) {
194                    subTypes.setCalscale(calscale);
195            }
196    
197            //@Override
198            public String getAltId() {
199                    return subTypes.getAltId();
200            }
201    
202            //@Override
203            public void setAltId(String altId) {
204                    subTypes.setAltId(altId);
205            }

207            @Override
208            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
209                    ValueParameter value = null;
210                    if (text != null && version == VCardVersion.V4_0) {
211                            value = ValueParameter.TEXT;
212                    }
213                    copy.setValue(value);
214            }
215    
216            @Override
217            protected void doMarshalText(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
218                    if (version == VCardVersion.V2_1 || version == VCardVersion.V3_0) {
219                            if (text != null) {
220                                    throw new SkipMeException("Text values are not supported in vCard version " + version + ".");
221                            } else if (partialDate != null) {
222                                    throw new SkipMeException("Reduced accuracy or truncated dates are not supported in vCard version " + version + ".");
223                            } else if (date != null) {
224                                    ISOFormat format = dateHasTime ? ISOFormat.TIME_BASIC : ISOFormat.DATE_BASIC;
225                                    sb.append(VCardDateFormatter.format(date, format));
226                            } else {
227                                    throw new SkipMeException("Property has no date value associated with it.");
228                            }
229                    } else {
230                            if (text != null) {
231                                    sb.append(VCardStringUtils.escape(text));
232                            } else if (partialDate != null) {
233                                    sb.append(partialDate.toDateAndOrTime(false));
234                            } else if (date != null) {
235                                    ISOFormat format = dateHasTime ? ISOFormat.TIME_BASIC : ISOFormat.DATE_BASIC;
236                                    sb.append(VCardDateFormatter.format(date, format));
237                            } else {
238                                    throw new SkipMeException("Property has no date, reduced accuracy date, or text value associated with it.");
239                            }
240                    }
241            }
242    
243            @Override
244            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
245                    value = VCardStringUtils.unescape(value);
246                    if (version == VCardVersion.V4_0 && subTypes.getValue() == ValueParameter.TEXT) {
247                            setText(value);
248                    } else {
249                            parseDate(value, version, warnings);
250                    }
251            }
252    
253            @Override
254            protected void doMarshalXml(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
255                    if (text != null) {
256                            parent.text(text);
257                    } else if (partialDate != null) {
258                            String name;
259                            if (partialDate.hasTimeComponent() && partialDate.hasDateComponent()) {
260                                    name = "date-time";
261                            } else if (partialDate.hasTimeComponent()) {
262                                    name = "time";
263                            } else if (partialDate.hasDateComponent()) {
264                                    name = "date";
265                            } else {
266                                    name = "date-and-or-time";
267                            }
268    
269                            parent.append(name, partialDate.toDateAndOrTime(false));
270                    } else if (date != null) {
271                            ISOFormat format = dateHasTime ? ISOFormat.TIME_BASIC : ISOFormat.DATE_BASIC;
272                            String name = dateHasTime ? "date-time" : "date";
273                            String value = VCardDateFormatter.format(date, format);
274                            parent.append(name, value);
275                    } else {
276                            throw new SkipMeException("Property has no date, reduced accuracy date, or text value associated with it.");
277                    }
278            }
279    
280            @Override
281            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
282                    String value = element.get("date", "date-time", "date-and-or-time");
283                    if (value != null) {
284                            parseDate(value, element.version(), warnings);
285                    } else {
286                            setText(element.text());
287                    }
288            }
289    
290            @Override
291            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
292                    String value = null;
293                    if ("time".equals(element.tagName())) {
294                            String datetime = element.attr("datetime");
295                            if (datetime.length() > 0) {
296                                    value = datetime;
297                            }
298                    }
299                    if (value == null) {
300                            value = element.value();
301                    }
302                    parseDate(value, VCardVersion.V3_0, warnings);
303            }
304    
305            @Override
306            protected JCardValue doMarshalJson(VCardVersion version, List<String> warnings) {
307                    JCardDataType dataType;
308                    String value;
309    
310                    if (text != null) {
311                            dataType = JCardDataType.TEXT;
312                            value = text;
313                    } else {
314                            if (date != null) {
315                                    dataType = dateHasTime ? JCardDataType.DATE_TIME : JCardDataType.DATE;
316                                    ISOFormat format = dateHasTime ? ISOFormat.TIME_EXTENDED : ISOFormat.DATE_EXTENDED;
317                                    value = VCardDateFormatter.format(date, format);
318                            } else if (partialDate != null) {
319                                    if (partialDate.hasTimeComponent() && partialDate.hasDateComponent()) {
320                                            dataType = JCardDataType.DATE_TIME;
321                                    } else if (partialDate.hasTimeComponent()) {
322                                            dataType = JCardDataType.TIME;
323                                    } else if (partialDate.hasDateComponent()) {
324                                            dataType = JCardDataType.DATE;
325                                    } else {
326                                            dataType = JCardDataType.DATE_AND_OR_TIME;
327                                    }
328    
329                                    value = partialDate.toDateAndOrTime(true);
330                            } else {
331                                    throw new SkipMeException("Property has no date, reduced accuracy date, or text value associated with it.");
332                            }
333                    }
334    
335                    return JCardValue.single(dataType, value);
336            }
337    
338            @Override
339            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
340                    String valueStr = value.getSingleValued();
341                    if (value.getDataType() == JCardDataType.TEXT) {
342                            setText(valueStr);
343                    } else {
344                            parseDate(valueStr, version, warnings);
345                    }
346            }
347    
348            private void parseDate(String value, VCardVersion version, List<String> warnings) {
349                    try {
350                            boolean hasTime = value.contains("T");
351                            setDate(VCardDateFormatter.parse(value), hasTime);
352                    } catch (IllegalArgumentException e) {
353                            if (version == VCardVersion.V4_0) {
354                                    try {
355                                            setPartialDate(new PartialDate(value));
356                                    } catch (IllegalArgumentException e2) {
357                                            warnings.add("Date string could not be parsed.  Assuming it's a text value: " + value);
358                                            setText(value);
359                                    }
360                            } else {
361                                    throw new SkipMeException("Date string could not be parsed: " + value);
362                            }
363                    }
364            }
