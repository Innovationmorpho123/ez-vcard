241             * Writes the XML document to a writer and pretty-prints it.
242             * @param writer the writer
243             * @param indent the number of indent spaces to use for pretty-printing
244             * @throws TransformerException if there's a problem writing to the writer
245             */
246            public void write(Writer writer, int indent) throws TransformerException {
247                    Map<String, String> properties = new HashMap<String, String>();
248                    if (indent >= 0) {
249                            properties.put(OutputKeys.INDENT, "yes");
250                            properties.put("{http://xml.apache.org/xslt}indent-amount", indent + "");
251                    }
252                    XmlUtils.toWriter(document, writer, properties);
253            }

255            /**
256             * Adds a vCard to the XML document.
257             * @param vcard the vCard to add
258             */
259            public void addVCard(VCard vcard) {
260                    warnings.clear();
261    
262                    if (vcard.getFormattedName() == null) {
263                            warnings.add("vCard version " + targetVersion + " requires that a formatted name property be defined.");
264                    }
265    
266                    ListMultimap<String, VCardType> typesToAdd = new ListMultimap<String, VCardType>(); //group the types by group name (null = no group name)
267    
268                    for (VCardType type : vcard) {
269                            if (addProdId && type instanceof ProdIdType) {
270                                    //do not add the PRODID in the vCard if "addProdId" is true
271                                    continue;
272                            }
273    
274                            //determine if this type is supported by the target version
275                            if (!supportsTargetVersion(type)) {
276                                    addWarning("This property is not supported by xCard (vCard version " + targetVersion + ") and will not be added to the xCard.  Supported versions are " + Arrays.toString(type.getSupportedVersions()), type.getTypeName());
277                                    continue;
278                            }
279    
280                            //check for correct KIND value if there are MEMBER types
281                            if (type instanceof MemberType && (vcard.getKind() == null || !vcard.getKind().isGroup())) {
282                                    addWarning("Value must be set to \"group\" if the vCard contains " + MemberType.NAME + " properties.", KindType.NAME);
283                                    continue;
284                            }
285    
286                            typesToAdd.put(type.getGroup(), type);
287                    }
288    
289                    //add an extended type saying it was generated by this library
290                    if (addProdId) {
291                            EzvcardProdIdType prodId = new EzvcardProdIdType(targetVersion);
292                            typesToAdd.put(prodId.getGroup(), prodId);
293                    }
294    
295                    //marshal each type object
296                    Element vcardElement = createElement("vcard");
297                    for (String groupName : typesToAdd.keySet()) {
298                            Element parent;
299                            if (groupName != null) {
300                                    Element groupElement = createElement("group");
301                                    groupElement.setAttribute("name", groupName);
302                                    vcardElement.appendChild(groupElement);
303                                    parent = groupElement;
304                            } else {
305                                    parent = vcardElement;
306                            }
307    
308                            List<String> warningsBuf = new ArrayList<String>();
309                            for (VCardType type : typesToAdd.get(groupName)) {
310                                    warningsBuf.clear();
311                                    try {
312                                            Element typeElement = marshalType(type, vcard, warningsBuf);
313                                            parent.appendChild(typeElement);
314                                    } catch (SkipMeException e) {
315                                            warningsBuf.add("Property has requested that it be skipped: " + e.getMessage());
316                                    } catch (EmbeddedVCardException e) {
317                                            warningsBuf.add("Property will not be marshalled because xCard does not supported embedded vCards.");
318                                    } finally {
319                                            for (String warning : warningsBuf) {
320                                                    addWarning(warning, type.getTypeName());
321                                            }
322                                    }
323                            }
324                    }
325                    root.appendChild(vcardElement);
326            }

328            /**
329             * Determines if a type supports the target version.
330             * @param type the type
331             * @return true if it supports the target version, false if not
332             */
333            private boolean supportsTargetVersion(VCardType type) {
334                    for (VCardVersion version : type.getSupportedVersions()) {
335                            if (version == targetVersion) {
336                                    return true;
337                            }
338                    }
339                    return false;
340            }
341    
342            /**
343             * Marshals a type object to an XML element.
344             * @param type the type object to marshal
345             * @param vcard the vcard the type belongs to
346             * @param warningsBuf the list to add the warnings to
347             * @return the XML element or null not to add anything to the final XML
348             * document
349             */
350            private Element marshalType(VCardType type, VCard vcard, List<String> warningsBuf) {
351                    QName qname = type.getQName();
352                    String ns, localPart;
353                    if (qname == null) {
354                            localPart = type.getTypeName().toLowerCase();
355                            ns = targetVersion.getXmlNamespace();
356                    } else {
357                            localPart = qname.getLocalPart();
358                            ns = qname.getNamespaceURI();
359                    }
360                    Element typeElement = createElement(localPart, ns);
361    
362                    //marshal the sub types
363                    VCardSubTypes subTypes = type.marshalSubTypes(targetVersion, warningsBuf, compatibilityMode, vcard);
364                    subTypes.setValue(null); //don't include the VALUE parameter (modification of the "VCardSubTypes" object is safe because it's a copy)
365                    if (!subTypes.isEmpty()) {
366                            Element parametersElement = createElement("parameters");
367                            for (Map.Entry<String, List<String>> param : subTypes) {
368                                    String paramName = param.getKey();
369                                    Element parameterElement = createElement(paramName.toLowerCase());
370                                    for (String paramValue : param.getValue()) {
371                                            String valueElementName = parameterChildElementNames.get(paramName.toLowerCase());
372                                            if (valueElementName == null) {
373                                                    valueElementName = "unknown";
374                                            }
375                                            Element parameterValueElement = createElement(valueElementName);
376                                            parameterValueElement.setTextContent(paramValue);
377                                            parameterElement.appendChild(parameterValueElement);
378                                    }
379                                    parametersElement.appendChild(parameterElement);
380                            }
381                            typeElement.appendChild(parametersElement);
382                    }
383    
384                    //marshal the value
385                    type.marshalXml(typeElement, targetVersion, warningsBuf, compatibilityMode);
386    
387                    return typeElement;
388            }

390            /**
391             * Creates a new XML element.
392             * @param name the name of the XML element
393             * @return the new XML element
394             */
395            private Element createElement(String name) {
396                    return createElement(name, targetVersion.getXmlNamespace());
397            }
398    
399            /**
400             * Creates a new XML element.
401             * @param name the name of the XML element
402             * @param ns the namespace of the XML element
403             * @return the new XML element
404             */
405            private Element createElement(String name, String ns) {
406                    return document.createElementNS(ns, name);
407            }
408    
409            private void addWarning(String message, String propertyName) {
410                    warnings.add(propertyName + " property: " + message);
411            }
