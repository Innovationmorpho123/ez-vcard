043    /**
044     * Parses a vCard data stream.
045     * @author Michael Angstadt
046     */
047    public class VCardRawReader implements Closeable {
048            private final FoldedLineReader reader;
049            private boolean caretDecodingEnabled = true;
050            private boolean eof = false;
051            private VCardVersion version = VCardVersion.V2_1;
052    
053            /**
054             * Creates a new reader.
055             * @param reader the reader to the data stream
056             */
057            public VCardRawReader(Reader reader) {
058                    this.reader = new FoldedLineReader(reader);
059            }
060    
061            /**
062             * Gets the line number of the last line that was read.
063             * @return the line number
064             */
065            public int getLineNum() {
066                    return reader.getLineNum();
067            }
068    
069            /**
070             * Starts or continues reading from the vCard data stream.
071             * @param listener handles the vCard data as it is read off the wire
072             * @throws IOException if there is an I/O problem
073             */
074            public void start(VCardDataStreamListener listener) throws IOException {
075                    String line;
076                    while ((line = reader.readLine()) != null) {
077                            try {
078                                    parseLine(line, listener);
079                            } catch (StopReadingException e) {
080                                    return;
081                            }
082                    }
083                    eof = true;
084            }
085    
086            private void parseLine(String line, VCardDataStreamListener listener) {
087                    String group = null;
088                    String propertyName = null;
089                    VCardSubTypes parameters = new VCardSubTypes();
090                    String value = null;
091    
092                    char escapeChar = 0; //is the next char escaped?
093                    boolean inQuotes = false; //are we inside of double quotes?
094                    StringBuilder buffer = new StringBuilder();
095                    String curParamName = null;
096                    for (int i = 0; i < line.length(); i++) {
097                            char ch = line.charAt(i);
098                            if (escapeChar != 0) {
099                                    if (escapeChar == '\\') {
100                                            if (ch == '\\') {
101                                                    buffer.append(ch);
102                                            } else if (ch == 'n' || ch == 'N') {
103                                                    //newlines appear as "\n" or "\N" (see RFC 2426 p.7)
104                                                    buffer.append(NEWLINE);
105                                            } else if (ch == '"' && version != VCardVersion.V2_1) {
106                                                    //double quotes don't need to be escaped in 2.1 parameter values because they have no special meaning
107                                                    buffer.append(ch);
108                                            } else if (ch == ';' && version == VCardVersion.V2_1) {
109                                                    //semi-colons can only be escaped in 2.1 parameter values (see section 2 of specs)
110                                                    //if a 3.0/4.0 param value has semi-colons, the value should be surrounded in double quotes
111                                                    buffer.append(ch);
112                                            } else {
113                                                    //treat the escape character as a normal character because it's not a valid escape sequence
114                                                    buffer.append(escapeChar).append(ch);
115                                            }
116                                    } else if (escapeChar == '^') {
117                                            if (ch == '^') {
118                                                    buffer.append(ch);
119                                            } else if (ch == 'n') {
120                                                    buffer.append(NEWLINE);
121                                            } else if (ch == '\'') {
122                                                    buffer.append('"');
123                                            } else {
124                                                    //treat the escape character as a normal character because it's not a valid escape sequence
125                                                    buffer.append(escapeChar).append(ch);
126                                            }
127                                    }
128                                    escapeChar = 0;
129                            } else if (ch == '\\' || (ch == '^' && version != VCardVersion.V2_1 && caretDecodingEnabled)) {
130                                    escapeChar = ch;
131                            } else if (ch == '.' && group == null && propertyName == null) {
132                                    group = buffer.toString();
133                                    buffer.setLength(0);
134                            } else if ((ch == ';' || ch == ':') && !inQuotes) {
135                                    if (propertyName == null) {
136                                            propertyName = buffer.toString();
137                                    } else {
138                                            //parameter value
139                                            String paramValue = buffer.toString();
140                                            if (version == VCardVersion.V2_1) {
141                                                    //2.1 allows whitespace to surround the "=", so remove it
142                                                    paramValue = VCardStringUtils.ltrim(paramValue);
143                                            }
144                                            parameters.put(curParamName, paramValue);
145                                            curParamName = null;
146                                    }
147                                    buffer.setLength(0);
148    
149                                    if (ch == ':') {
150                                            if (i < line.length() - 1) {
151                                                    value = line.substring(i + 1);
152                                            } else {
153                                                    value = "";
154                                            }
155                                            break;
156                                    }
157                            } else if (ch == ',' && !inQuotes && version != VCardVersion.V2_1) {
158                                    //multi-valued parameter
159                                    parameters.put(curParamName, buffer.toString());
160                                    buffer.setLength(0);
161                            } else if (ch == '=' && curParamName == null) {
162                                    //parameter name
163                                    String paramName = buffer.toString();
164                                    if (version == VCardVersion.V2_1) {
165                                            //2.1 allows whitespace to surround the "=", so remove it
166                                            paramName = VCardStringUtils.rtrim(paramName);
167                                    }
168                                    curParamName = paramName;
169                                    buffer.setLength(0);
170                            } else if (ch == '"' && version != VCardVersion.V2_1) {
171                                    //2.1 doesn't use the quoting mechanism
172                                    inQuotes = !inQuotes;
173                            } else {
174                                    buffer.append(ch);
175                            }
176                    }
177    
178                    if (propertyName == null || value == null) {
179                            listener.invalidLine(line);
180                            return;
181                    }
182                    if ("VERSION".equalsIgnoreCase(propertyName)) {
183                            VCardVersion version = VCardVersion.valueOfByStr(value.trim());
184                            if (version == null) {
185                                    listener.invalidVersion(value);
186                            } else {
187                                    this.version = version;
188                                    listener.readVersion(version);
189                            }
190                            return;
191                    }
192                    if ("BEGIN".equalsIgnoreCase(propertyName)) {
193                            listener.beginComponent(value.trim());
194                            return;
195                    }
196                    if ("END".equalsIgnoreCase(propertyName)) {
197                            listener.endComponent(value.trim());
198                            return;
199                    }
200                    listener.readProperty(group, propertyName, parameters, value);
201            }
202    
203            /**
204             * <p>
205             * Gets whether the reader will decode parameter values that use circumflex
206             * accent encoding (enabled by default). This escaping mechanism allows
207             * newlines and double quotes to be included in parameter values.
208             * </p>
209             * 
210             * <table border="1">
211             * <tr>
212             * <th>Raw Character</th>
213             * <th>Encoded Character</th>
214             * </tr>
215             * <tr>
216             * <td><code>"</code></td>
217             * <td><code>^'</code></td>
218             * </tr>
219             * <tr>
220             * <td><i>newline</i></td>
221             * <td><code>^n</code></td>
222             * </tr>
223             * <tr>
224             * <td><code>^</code></td>
225             * <td><code>^^</code></td>
226             * </tr>
227             * </table>
228             * 
229             * <p>
230             * Example:
231             * </p>
232             * 
233             * <pre>
234             * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
235             *  sburgh, PA 15212":40.446816;80.00566
236             * </pre>
237             * 
238             * @return true if circumflex accent decoding is enabled, false if not
239             * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
240             */
241            public boolean isCaretDecodingEnabled() {
242                    return caretDecodingEnabled;
243            }
244    
245            /**
246             * <p>
247             * Sets whether the reader will decode parameter values that use circumflex
248             * accent encoding (enabled by default). This escaping mechanism allows
249             * newlines and double quotes to be included in parameter values.
250             * </p>
251             * 
252             * <table border="1">
253             * <tr>
254             * <th>Raw Character</th>
255             * <th>Encoded Character</th>
256             * </tr>
257             * <tr>
258             * <td><code>"</code></td>
259             * <td><code>^'</code></td>
260             * </tr>
261             * <tr>
262             * <td><i>newline</i></td>
263             * <td><code>^n</code></td>
264             * </tr>
265             * <tr>
266             * <td><code>^</code></td>
267             * <td><code>^^</code></td>
268             * </tr>
269             * </table>
270             * 
271             * <p>
272             * Example:
273             * </p>
274             * 
275             * <pre>
276             * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
277             *  sburgh, PA 15212":geo:40.446816,-80.00566
278             * </pre>
279             * 
280             * @param enable true to use circumflex accent decoding, false not to
281             * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
282             */
283            public void setCaretDecodingEnabled(boolean enable) {
284                    caretDecodingEnabled = enable;
285            }
286    
287            /**
288             * Determines whether the end of the data stream has been reached.
289             * @return true if the end has been reached, false if not
290             */
291            public boolean eof() {
292                    return eof;
293            }
294    
295            /**
296             * Handles the vCard data as it is read off the data stream. Each one of
297             * this interface's methods may throw a {@link StopReadingException} at any
298             * time to force the parser to stop reading from the data stream. This will
299             * cause the reader to return from the {@link VCardRawReader#start} method.
300             * To continue reading from the data stream, simply call the
301             * {@link VCardRawReader#start} method again.
302             * @author Michael Angstadt
303             */
304            public static interface VCardDataStreamListener {
305                    /**
306                     * Called when a component begins (when a "BEGIN:NAME" property is
307                     * reached).
308                     * @param name the component name (e.g. "VCARD")
309                     * @throws StopReadingException to force the reader to stop reading from
310                     * the data stream
311                     */
312                    void beginComponent(String name);
313    
314                    /**
315                     * Called when a property is read.
316                     * @param name the property name (e.g. "VERSION")
317                     * @param parameters the parameters
318                     * @param value the property value
319                     * @throws StopReadingException to force the reader to stop reading from
320                     * the data stream
321                     */
322                    void readProperty(String group, String name, VCardSubTypes parameters, String value);
323    
324                    /**
325                     * Called when the vCard's VERSION property is read.
326                     * @param version the version that was read
327                     */
328                    void readVersion(VCardVersion version);
329    
330                    /**
331                     * Called when a component ends (when a "END:NAME" property is reached).
332                     * @param name the component name (e.g. "VCARD")
333                     * @throws StopReadingException to force the reader to stop reading from
334                     * the data stream
335                     */
336                    void endComponent(String name);
337    
338                    /**
339                     * Called when a line cannot be parsed.
340                     * @param line the unparseable line
341                     * @throws StopReadingException to force the reader to stop reading from
342                     * the data stream
343                     */
344                    void invalidLine(String line);
345    
346                    /**
347                     * Called when an invalid VERSION property is encountered.
348                     * @param version the invalid version
349                     */
350                    void invalidVersion(String version);
351            }
352    
353            /**
354             * Instructs an {@link VCardRawReader} to stop reading from the data stream
355             * when thrown from an {@link VCardDataStreamListener} implementation.
356             * @author Michael Angstadt
357             */
358            @SuppressWarnings("serial")
359            public static class StopReadingException extends VCardException {
360                    //empty
361            }
362    
363            /**
364             * Closes the underlying {@link Reader} object.
365             */
366            public void close() throws IOException {
367                    reader.close();
368            }
369    }
