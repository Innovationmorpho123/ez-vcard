034    /**
035     * Represents a LEVEL parameter for the {@link InterestType} property.
036     * 
037     * <p>
038     * <b>Supported versions:</b> <code>4.0</code>
039     * </p>
040     * @author Michael Angstadt
041     * @see <a href="http://tools.ietf.org/html/rfc6715">RFC 6715</a>
042     */
043    public class InterestLevelParameter extends LevelParameter {
044            public static final InterestLevelParameter LOW = new InterestLevelParameter("low");
045            public static final InterestLevelParameter MEDIUM = new InterestLevelParameter("medium");
046            public static final InterestLevelParameter HIGH = new InterestLevelParameter("high");
047    
048            /**
049             * Use of this constructor is discouraged and should only be used for
050             * defining non-standard LEVELs. Please use one of the predefined static
051             * objects.
052             * @param value the type value (e.g. "high")
053             */
054            public InterestLevelParameter(String value) {
055                    super(value);
056            }
057    
058            /**
059             * Searches the static objects in this class for one that has a certain type
060             * value.
061             * @param value the type value to search for (e.g. "high")
062             * @return the object or null if not found
063             */
064            public static InterestLevelParameter valueOf(String value) {
065                    return findByValue(value, InterestLevelParameter.class);
066            }
067    }
