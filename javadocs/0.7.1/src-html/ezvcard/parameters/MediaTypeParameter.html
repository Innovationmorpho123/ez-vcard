035    /**
036     * Represents a TYPE parameter that also has a media type associated with it.
037     * The TYPE parameter value is used in 2.1 and 3.0 vCards, while the media type
038     * value is used in 4.0 vCards.
039     * @author Michael Angstadt
040     */
041    public class MediaTypeParameter extends TypeParameter {
042            protected final String mediaType;
043            protected final String extension;
044    
045            /**
046             * @param value the TYPE parameter value (e.g. "JPEG")
047             * @param mediaType the media type (e.g. "image/jpeg")
048             * @param extension the file extension (e.g. "jpg")
049             */
050            public MediaTypeParameter(String value, String mediaType, String extension) {
051                    super(value);
052                    this.mediaType = mediaType;
053                    this.extension = extension;
054            }
055    
056            /**
057             * Gets the media type.
058             * @return the media type (e.g. "image/jpeg")
059             */
060            public String getMediaType() {
061                    return mediaType;
062            }
063    
064            /**
065             * Gets the file extension.
066             * @return the file extension (e.g. "jpg")
067             */
068            public String getExtension() {
069                    return extension;
070            }
071    
072            @Override
073            public int hashCode() {
074                    final int prime = 31;
075                    int result = super.hashCode();
076                    result = prime * result + ((extension == null) ? 0 : extension.hashCode());
077                    result = prime * result + ((mediaType == null) ? 0 : mediaType.hashCode());
078                    return result;
079            }
080    
081            @Override
082            public boolean equals(Object obj) {
083                    if (this == obj)
084                            return true;
085                    if (!super.equals(obj))
086                            return false;
087                    if (getClass() != obj.getClass())
088                            return false;
089                    MediaTypeParameter other = (MediaTypeParameter) obj;
090                    if (extension == null) {
091                            if (other.extension != null)
092                                    return false;
093                    } else if (!extension.equals(other.extension))
094                            return false;
095                    if (mediaType == null) {
096                            if (other.mediaType != null)
097                                    return false;
098                    } else if (!mediaType.equals(other.mediaType))
099                            return false;
100                    return true;
101            }
102    
103            /**
104             * Searches the static objects of a child class for one that has a certain
105             * media type value.
106             * @param mediaType the media type value to look for
107             * @param clazz the child class
108             * @return the object or null if not found
109             */
110            protected static <T extends MediaTypeParameter> T findByMediaType(String mediaType, Class<T> clazz) {
111                    for (Field f : clazz.getFields()) {
112                            int mods = f.getModifiers();
113                            if (Modifier.isStatic(mods) && Modifier.isPublic(mods) && f.getDeclaringClass() == clazz) {
114                                    try {
115                                            Object obj = f.get(null);
116                                            if (obj != null) {
117                                                    @SuppressWarnings("unchecked")
118                                                    T param = (T) obj;
119                                                    if (param.getMediaType().equalsIgnoreCase(mediaType)) {
120                                                            return param;
121                                                    }
122                                            }
123                                    } catch (Exception ex) {
124                                            //reflection error
125                                    }
126                            }
127                    }
128                    return null;
129            }
130    }
