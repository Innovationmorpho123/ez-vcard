060    /**
061     * Represents the TYPE parameter of the SOUND type.
062     * <p>
063     * vCard versions: 2.1, 3.0, 4.0
064     * </p>
065     * @author George El-Haddad Mar 10, 2010
066     * @author Michael Angstadt
067     */
068    public class SoundTypeParameter extends MediaTypeParameter {
069            public static final SoundTypeParameter AAC = new SoundTypeParameter("AAC", "audio/aac", "aac");
070            public static final SoundTypeParameter MIDI = new SoundTypeParameter("MIDI", "audio/midi", "mid");
071            public static final SoundTypeParameter MP3 = new SoundTypeParameter("MP3", "audio/mp3", "mp3");
072            public static final SoundTypeParameter MPEG = new SoundTypeParameter("MPEG", "audio/mpeg", "mpeg");
073            public static final SoundTypeParameter OGG = new SoundTypeParameter("OGG", "audio/ogg", "ogg");
074            public static final SoundTypeParameter WAV = new SoundTypeParameter("WAV", "audio/wav", "wav");
075    
076            /**
077             * Use of this constructor is discouraged and should only be used for
078             * defining non-standard TYPEs. Please use one of the predefined static
079             * objects.
080             * @param value the type value (e.g. "ogg")
081             * @param mediaType the media type (e.g. "audio/ogg")
082             * @param extension the file extension used for this type (e.g. "ogg")
083             */
084            public SoundTypeParameter(String value, String mediaType, String extension) {
085                    super(value, mediaType, extension);
086            }
087    
088            /**
089             * Searches the static objects in this class for one that has a certain type
090             * value.
091             * @param value the type value to search for (e.g. "ogg")
092             * @return the object or null if not found
093             */
094            public static SoundTypeParameter valueOf(String value) {
095                    return findByValue(value, SoundTypeParameter.class);
096            }
097    
098            /**
099             * Searches the static objects in this class for one that has a certain
100             * media type.
101             * @param mediaType the media type to search for (e.g. "audio/ogg")
102             * @return the object or null if not found
103             */
104            public static SoundTypeParameter findByMediaType(String mediaType) {
105                    return findByMediaType(mediaType, SoundTypeParameter.class);
106            }
107    }
