043    /**
044     * Holds the parameters (aka "sub types") of a vCard Type.
045     * @author Michael Angstadt
046     */
047    public class VCardSubTypes {
048            private final TreeMultimap<String, String> subTypes;
049    
050            public VCardSubTypes() {
051                    subTypes = new TreeMultimap<String, String>();
052            }
053    
054            /**
055             * Copy constructor.
056             * @param orig the object to copy
057             */
058            public VCardSubTypes(VCardSubTypes orig) {
059                    subTypes = new TreeMultimap<String, String>(orig.subTypes);
060            }
061    
062            /**
063             * Adds a value to a Sub Type.
064             * @param name the Sub Type name
065             * @param value the value to add
066             */
067            public void put(String name, String value) {
068                    subTypes.put(name.toUpperCase(), value);
069            }
070    
071            /**
072             * Adds a value to a Sub Type, replacing all existing values that the Sub
073             * Type has.
074             * @param name the Sub Type name
075             * @param value the values to replace all existing values with
076             * @return the values of the Sub Type that were replaced
077             */
078            public Set<String> replace(String name, String value) {
079                    Set<String> set = removeAll(name);
080                    if (value != null) {
081                            put(name, value);
082                    }
083                    return set;
084            }
085    
086            /**
087             * Removes a Sub Type.
088             * @param name the Sub Type name
089             * @return the values of the Sub Type that were removed
090             */
091            public Set<String> removeAll(String name) {
092                    return subTypes.remove(name.toUpperCase());
093            }
094    
095            /**
096             * Removes a value from a Sub Type.
097             * @param name the Sub Type name
098             * @param value the value to remove
099             */
100            public void remove(String name, String value) {
101                    subTypes.remove(name.toUpperCase(), value);
102            }

104            /**
105             * Gets the values of a Sub Type
106             * @param name the Sub Type name
107             * @return the values or an empty set if the Sub Type doesn't exist
108             */
109            public Set<String> get(String name) {
110                    return subTypes.get(name.toUpperCase());
111            }

113            /**
114             * Gets the first value of a Sub Type.
115             * @param name the Sub Type name

119                    Set<String> set = get(name);
120                    return set.isEmpty() ? null : set.iterator().next();
121            }

310            /**
311             * <p>
312             * Gets the preference value. The lower the number, the more preferred this
313             * type is compared to other types in the vCard with the same name. If a
314             * type doesn't have a preference value, then it is considered the
315             * <b>least</b> preferred.
316             * </p>
317             * 
318             * <p>
319             * In the vCard below, the address on the second row is the most preferred
320             * because it has the lowest PREF value.
321             * </p>
322             * 
323             * <pre>
324             * ADR;TYPE=work;PREF=2:
325             * ADR;TYPE=work;PREF=1:
326             * ADR;TYPE=home:
327             * </pre>
328             * 
329             * <p>
330             * Preference values must be numeric and must be between 1 and 100.
331             * </p>
332             * 
333             * <p>
334             * vCard versions: 4.0
335             * </p>
336             * @return the preference value or null if it doesn't exist or null if it
337             * couldn't be parsed into a number
338             */
339            public Integer getPref() {
340                    String pref = getFirst("PREF");
341                    if (pref == null) {
342                            return null;
343                    }
344    
345                    try {
346                            return Integer.valueOf(pref);
347                    } catch (NumberFormatException e) {
348                            return null;
349                    }
350            }

352            /**
353             * <p>
354             * Sets the preference value. The lower the number, the more preferred this
355             * type is compared to other types in the vCard with the same name. If a
356             * type doesn't have a preference value, then it is considered the
357             * <b>least</b> preferred.
358             * </p>
359             * 
360             * <p>
361             * In the vCard below, the address on the second row is the most preferred
362             * because it has the lowest PREF value.
363             * </p>
364             * 
365             * <pre>
366             * ADR;TYPE=work;PREF=2:
367             * ADR;TYPE=work;PREF=1:
368             * ADR;TYPE=home:
369             * </pre>
370             * 
371             * <p>
372             * Preference values must be numeric and must be between 1 and 100.
373             * </p>
374             * 
375             * <p>
376             * vCard versions: 4.0
377             * </p>
378             * @param pref the preference value or null to remove
379             * @throws IllegalArgumentException if the value is not between 1 and 100
380             */
381            public void setPref(Integer pref) {
382                    if (pref != null && (pref < 1 || pref > 100)) {
383                            throw new IllegalArgumentException("Preference value must be between 1 and 100 inclusive.");
384                    }
385                    String value = (pref == null) ? null : pref.toString();
386                    replace("PREF", value);
387            }

391             * representations of the same type.
392             * 
393             * <p>
394             * For example, a vCard may contain multiple NOTE types that each have the
395             * same ALTID. This means that each NOTE contains a different representation
396             * of the same information. In the example below, the first three NOTEs have
397             * the same ALTID. They each contain the same message, but each is written
398             * in a different language. The other NOTEs have different (or absent)
399             * ALTIDs, which means they are not associated with the top three.
400             * </p>
401             * 
402             * <pre>
403             * NOTE;ALTID=1;LANGUAGE=en: Hello world!
404             * NOTE;ALTID=1;LANGUAGE=fr: Bonjour tout le monde!
405             * NOTE;ALTID=1;LANGUAGE=es: �Hola, mundo!
406             * NOTE;ALTID=2;LANGUAGE=de: Meine Lieblingsfarbe ist blau.
407             * NOTE;ALTID=2;LANGUAGE=en: My favorite color is blue.
408             * NOTE: This vCard will self-destruct in 5 seconds.
409             * </pre>
410             * 
411             * <p>
412             * vCard versions: 4.0
413             * </p>
414             * @return the ALTID or null if it doesn't exist
415             */

452             * Gets the GEO parameter value. This is used to associate global
453             * positioning information with a vCard type. It can be used with the ADR
454             * type.
455             * <p>
456             * vCard versions: 4.0
457             * </p>
458             * @return the latitude (index 0) and longitude (index 1) or null if not
459             * present or null if the parameter value was in an incorrect format
460             */
461            public double[] getGeo() {
462                    String value = getFirst("GEO");
463                    if (value == null) {
464                            return null;
465                    }
466    
467                    try {
468                            GeoUri geoUri = new GeoUri(value);
469                            return new double[] { geoUri.getCoordA(), geoUri.getCoordB() };
470                    } catch (IllegalArgumentException e) {
471                            return null;
472                    }
473            }

552            /**
553             * <p>
554             * Gets all PID parameter values. PIDs can exist on any property where
555             * multiple instances are allowed (such as EMAIL or ADR, but not N because
556             * only 1 instance of N is allowed).
557             * </p>
558             * <p>
559             * When used in conjunction with the CLIENTPIDMAP property, it allows an
560             * individual property instance to be uniquely identifiable. This feature is
561             * made use of when two different versions of the same vCard have to be
562             * merged together (called "synchronizing").
563             * </p>
564             * <p>
565             * vCard versions: 4.0
566             * </p>
567             * @return the PID values or empty set if there are none. Index 0 is the
568             * local ID and index 1 is the ID used to reference the CLIENTPIDMAP
569             * property. Index 0 will never be null, but index 1 may be null.
570             */
571            public Set<Integer[]> getPids() {
572                    Set<String> values = get("PID");
573                    Set<Integer[]> pids = new HashSet<Integer[]>();
574                    for (String value : values) {
575                            try {
576                                    String split[] = value.split("\\.");
577                                    Integer pid[] = new Integer[2];
578                                    pid[0] = Integer.valueOf(split[0]);
579                                    if (split.length > 1) {
580                                            pid[1] = Integer.valueOf(split[1]);
581                                    }
582                                    pids.add(pid);
583                            } catch (NumberFormatException e) {
584                                    //skip
585                            }
586                    }
587                    return pids;
588            }

696            /**
697             * Gets the INDEX parameter. When present, it specifies the order in which a
698             * collection of multi-valued property instances should be sorted.
699             * Properties with low INDEX values are put at the beginning of the sorted
700             * list.
701             * 
702             * <p>
703             * vCard versions: 4.0
704             * </p>
705             * @return the INDEX value or null if it doesn't exist or null if it
706             * couldn't be parsed into a number
707             * @see <a href="http://tools.ietf.org/html/rfc6715">RFC 6715</a>
708             */
709            public Integer getIndex() {
710                    String index = getFirst("INDEX");
711                    if (index == null) {
712                            return null;
713                    }
714    
715                    try {
716                            return Integer.valueOf(index);
717                    } catch (NumberFormatException e) {
