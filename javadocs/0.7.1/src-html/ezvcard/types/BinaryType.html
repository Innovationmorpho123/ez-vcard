054    /**
055     * Represents a Type that contains binary data (for example, the "PHOTO" type).
056     * @author Michael Angstadt
057     * 
058     * @param <T> the class used for representing the content type of the resource
059     */
060    public abstract class BinaryType<T extends MediaTypeParameter> extends VCardType {
061            /**
062             * The decoded data.
063             */
064            protected byte[] data;
065    
066            /**
067             * The URL to the resource.
068             */
069            protected String url;
070    
071            /**
072             * The content type of the resource (e.g. "JPEG image").
073             */
074            protected T contentType;
075    
076            /**
077             * @param name the type name (e.g. "PHOTO")
078             */
079            public BinaryType(String name) {
080                    super(name);
081            }
082    
083            /**
084             * @param name the type name (e.g. "PHOTO")
085             * @param url the URL to the resource
086             * @param type the content type
087             */
088            public BinaryType(String name, String url, T type) {
089                    this(name);
090                    setUrl(url, type);
091            }
092    
093            /**
094             * @param name the type name (e.g. "PHOTO")
095             * @param data the binary data
096             * @param type the content type
097             */
098            public BinaryType(String name, byte[] data, T type) {
099                    this(name);
100                    setData(data, type);
101            }
102    
103            /**
104             * @param name the type name (e.g. "PHOTO")
105             * @param in an input stream to the binary data (will be closed)
106             * @param type the content type
107             * @throws IOException
108             */
109            public BinaryType(String name, InputStream in, T type) throws IOException {
110                    this(name, IOUtils.toByteArray(in, true), type);
111            }
112    
113            /**
114             * @param name the type name (e.g. "PHOTO")
115             * @param file the file containing the binary data
116             * @param type the content type
117             * @throws IOException
118             */
119            public BinaryType(String name, File file, T type) throws IOException {
120                    this(name, new FileInputStream(file), type);
121            }

124             * Gets the binary data of the resource.
125             * @return the binary data or null if there is none
126             */

151             * Sets the URL to the resource.
152             * @param url the URL
153             * @param type the content type (e.g. "JPEG image")
154             */
155            public void setUrl(String url, T type) {
156                    this.url = url;
157                    this.data = null;
158                    setContentType(type);
159            }

162             * Gets the content type of the resource.
163             * @return the content type (e.g. "JPEG image")

167            }

169            /**
170             * Sets the content type of the resource.
171             * @param contentType the content type (e.g. "JPEG image")

173            public void setContentType(T contentType) {

288            @Override
289            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
290                    MediaTypeParameter contentType = this.contentType;
291                    if (contentType == null) {
292                            contentType = new MediaTypeParameter(null, null, null);
293                    }
294    
295                    if (url != null) {
296                            ValueParameter vp = null;
297                            switch (version) {
298                            case V2_1:
299                                    vp = ValueParameter.URL;
300                                    break;
301                            case V3_0:
302                            case V4_0:
303                                    vp = ValueParameter.URI;
304                                    break;
305                            }
306                            copy.setValue(vp);
307    
308                            copy.setEncoding(null);
309    
310                            if (version == VCardVersion.V4_0) {
311                                    //don't null out TYPE, it could be set to "home" or "work"
312                                    copy.setMediaType(contentType.getMediaType());
313                            } else {
314                                    copy.setType(contentType.getValue());
315                                    copy.setMediaType(null);
316                            }
317                    }
318                    if (data != null) {
319                            copy.setMediaType(null);
320                            if (version == VCardVersion.V2_1) {
321                                    copy.setEncoding(EncodingParameter.BASE64);
322                                    copy.setValue(null);
323                                    copy.setType(contentType.getValue());
324                            } else if (version == VCardVersion.V3_0) {
325                                    copy.setEncoding(EncodingParameter.B);
326                                    copy.setValue(null);
327                                    copy.setType(contentType.getValue());
328                            } else {
329                                    copy.setEncoding(null);
330                                    copy.setValue(ValueParameter.URI);
331                                    //don't null out TYPE, it could be set to "home" or "work"
332                            }
333                    }
334            }

336            @Override
337            protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
338                    if (url != null) {
339                            sb.append(url);
340                    } else if (data != null) {
341                            String base64 = new String(Base64.encodeBase64(data));
342                            if (version == VCardVersion.V4_0) {
343                                    String mediaType = (contentType == null || contentType.getMediaType() == null) ? "application/octet-stream" : contentType.getMediaType();
344                                    DataUri uri = new DataUri(mediaType, data);
345                                    sb.append(uri.toString());
346                            } else {
347                                    sb.append(base64);
348                            }
349                    } else {
350                            throw new SkipMeException("Property has neither a URL nor binary data attached to it.");
351                    }
352            }
353    
354            @Override
355            protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
356                    //check for a 4.0 data URI
357                    try {
358                            DataUri uri = new DataUri(value);
359                            T contentType = buildMediaTypeObj(uri.getContentType());
360                            setData(uri.getData(), contentType);
361                            return;
362                    } catch (IllegalArgumentException e) {
363                            //not a data URI
364                    }
365    
366                    //get content type from 4.0 MEDIATYPE parameter
367                    T contentType = null;
368                    String mediaType = subTypes.getMediaType();
369                    if (mediaType != null) {
370                            contentType = buildMediaTypeObj(mediaType);
371                    } else {
372                            //get content type from 2.1/3.0 TYPE parameter
373                            String type = subTypes.getType();
374                            if (type != null && (version == VCardVersion.V2_1 || version == VCardVersion.V3_0)) {
375                                    contentType = buildTypeObj(getType());
376                            }
377                    }
378    
379                    //check for a URL
380                    ValueParameter valueSubType = subTypes.getValue();
381                    if (valueSubType == ValueParameter.URL || valueSubType == ValueParameter.URI) {
382                            setUrl(value, contentType);
383                            return;
384                    }
385    
386                    //check for 2.1/3.0 base64 data
387                    EncodingParameter encodingSubType = subTypes.getEncoding();
388                    if (encodingSubType != null) {
389                            if (encodingSubType != EncodingParameter.B && encodingSubType != EncodingParameter.BASE64) {
390                                    warnings.add("Unrecognized " + EncodingParameter.NAME + " parameter value \"" + encodingSubType + "\" in " + getTypeName() + " property.  Attempting to decode as base64.");
391                            }
392                            setData(Base64.decodeBase64(value), contentType);
393                            return;
394                    }
395    
396                    //see if the value is a URL incase they didn't set the VALUE parameter
397                    if (value.matches("(?i)http.*")) {
398                            setUrl(value, contentType);
399                            return;
400                    }
401    
402                    cannotUnmarshalValue(value, version, warnings, compatibilityMode, contentType);
403            }
404    
405            @Override
406            protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
407                    StringBuilder sb = new StringBuilder();
408                    doMarshalValue(sb, parent.getVersion(), warnings, compatibilityMode);
409                    parent.appendUri(sb.toString());
410            }
411    
412            @Override
413            protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
414                    String value = element.getUri();
415                    if (value != null) {
416                            doUnmarshalValue(value, element.getVersion(), warnings, compatibilityMode);
417                    }
418            }

420            @Override
421            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
422                    String elementName = element.tagName();
423                    String error = null;
424                    if ("object".equals(elementName)) {
425                            T mediaType = null;
426                            String type = element.attr("type");
427                            if (type.length() > 0) {
428                                    mediaType = buildMediaTypeObj(type);
429                            }
430    
431                            String data = element.absUrl("data");
432                            if (data.length() > 0) {
433                                    try {
434                                            DataUri uri = new DataUri(data);
435                                            mediaType = buildMediaTypeObj(uri.getContentType());
436                                            setData(uri.getData(), mediaType);
437                                    } catch (IllegalArgumentException e) {
438                                            //TODO create buildTypeObjFromExtension() method
439                                            setUrl(data, null);
440                                    }
441                            } else {
442                                    error = "<object> tag does not have a \"data\" attribute.";
443                            }
444                    } else {
445                            error = "Cannot parse HTML tag \"" + elementName + "\".";
446                    }
447    
448                    if (error != null) {
449                            throw new SkipMeException(error);
450                    }
451            }

453            /**
454             * Called if the unmarshalling code cannot determine how to unmarshal the
455             * value.
456             * @param value the value
457             * @param version the version of the vCard
458             * @param warnings the warnings
459             * @param compatibilityMode the compatibility mode
460             * @param contentType the content type of the resource of null if unknown
461             */
462            protected void cannotUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, T contentType) {
463                    warnings.add("No " + ValueParameter.NAME + " or " + EncodingParameter.NAME + " parameter given.  Attempting to decode as base64.");
464                    setData(Base64.decodeBase64(value), contentType);
465            }
