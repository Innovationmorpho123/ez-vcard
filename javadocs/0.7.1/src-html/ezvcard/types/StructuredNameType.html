043    /**
044     * <p>
045     * Contains the separated components of the person's name.
046     * </p>
047     * 
048     * <p>
049     * Multiple instances of this type can be added ONLY if each instance has an
050     * ALTID parameter and the value of the ALTID parameter is the same across all
051     * instances. However, this is a border-case; under most circumstances, you will
052     * only need to add one instance.
053     * </p>
054     * 
055     * <pre>
056     * VCard vcard = new VCard();
057     * StructuredNameType n = new StructuredNameType();
058     * n.setFamily(&quot;House&quot;);
059     * n.setGiven(&quot;Gregory&quot;);
060     * n.addPrefix(&quot;Dr&quot;);
061     * n.addSuffix(&quot;MD&quot;);
062     * vcard.setStructuredName(n);
063     * </pre>
064     * 
065     * <p>
066     * vCard property name: N
067     * </p>
068     * <p>
069     * vCard versions: 2.1, 3.0, 4.0
070     * </p>
071     * @author Michael Angstadt
072     */
073    public class StructuredNameType extends VCardType {
074            public static final String NAME = "N";
075    
076            private String family;
077            private String given;
078            private List<String> additional = new ArrayList<String>();
079            private List<String> prefixes = new ArrayList<String>();
080            private List<String> suffixes = new ArrayList<String>();
081    
082            public StructuredNameType() {
083                    super(NAME);
084            }
085    
086            /**
087             * Gets the family name (aka "last name").
088             * @return the family name or null if not set
089             */
090            public String getFamily() {
091                    return family;
092            }
093    
094            /**
095             * Sets the family name (aka "last name").
096             * @param family the family name or null to remove
097             */
098            public void setFamily(String family) {
099                    this.family = family;
100            }
101    
102            /**
103             * Gets the given name (aka "first name").
104             * @return the given name or null if not set
105             */
106            public String getGiven() {
107                    return given;
108            }
109    
110            /**
111             * Sets the given name (aka "first name").
112             * @param given the given name or null to remove
113             */
114            public void setGiven(String given) {
115                    this.given = given;
116            }
117    
118            /**
119             * Gets any additional names the person goes by.
120             * @return the additional names or empty list if there are none
121             */
122            public List<String> getAdditional() {
123                    return additional;
124            }
125    
126            /**
127             * Adds an additional name the person goes by.
128             * @param additional the additional name to add
129             */
130            public void addAdditional(String additional) {
131                    this.additional.add(additional);
132            }
133    
134            /**
135             * Gets the prefixes.
136             * @return the prefixes (e.g. "Mr.") or empty list if there are none
137             */
138            public List<String> getPrefixes() {
139                    return prefixes;
140            }
141    
142            /**
143             * Adds a prefix.
144             * @param prefix the prefix to add (e.g. "Mr.")
145             */
146            public void addPrefix(String prefix) {
147                    this.prefixes.add(prefix);
148            }
149    
150            /**
151             * Gets the suffixes.
152             * @return the suffixes (e.g. "Jr.") or empty list if there are none
153             */
154            public List<String> getSuffixes() {
155                    return suffixes;
156            }
157    
158            /**
159             * Adds a suffix.
160             * @param suffix the suffix to add (e.g. "Jr.")
161             */
162            public void addSuffix(String suffix) {
163                    this.suffixes.add(suffix);
164            }
165    
166            /**
167             * Gets the string that determines how this name should be sorted.
168             * <p>
169             * vCard versions: 4.0
170             * </p>
171             * @return the sort string (e.g. "Doe,John") or null if not set
172             * @see VCardSubTypes#getSortAs
173             */
174            public String getSortAs() {
175                    return subTypes.getSortAs();
176            }
177    
178            /**
179             * Sets the string that determines how this name should be sorted.
180             * <p>
181             * vCard versions: 4.0
182             * </p>
183             * @param sortAs the sort string (e.g. "Doe,John") or null to remove
184             * @see VCardSubTypes#setSortAs
185             */
186            public void setSortAs(String sortAs) {
187                    subTypes.setSortAs(sortAs);
188            }
189    
190            /**
191             * Gets the language the name is written in.
192             * @return the language or null if not set
193             * @see VCardSubTypes#getLanguage
194             */
195            public String getLanguage() {
196                    return subTypes.getLanguage();
197            }
198    
199            /**
200             * Sets the language the name is written in.
201             * @param language the language or null to remove
202             * @see VCardSubTypes#setLanguage
203             */
204            public void setLanguage(String language) {
205                    subTypes.setLanguage(language);
206            }
207    
208            /**
209             * Gets the ALTID.
210             * <p>
211             * vCard versions: 4.0
212             * </p>
213             * @return the ALTID or null if it doesn't exist
214             * @see VCardSubTypes#getAltId
215             */
216            public String getAltId() {
217                    return subTypes.getAltId();
218            }
219    
220            /**
221             * Sets the ALTID.
222             * <p>
223             * vCard versions: 4.0
224             * </p>
225             * @param altId the ALTID or null to remove
226             * @see VCardSubTypes#setAltId
227             */
228            public void setAltId(String altId) {
229                    subTypes.setAltId(altId);
230            }
231    
232            @Override
233            protected void doMarshalValue(StringBuilder value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
234                    if (family != null) {
235                            value.append(VCardStringUtils.escape(family));
236                    }
237    
238                    value.append(';');
239                    if (given != null) {
240                            value.append(VCardStringUtils.escape(given));
241                    }
242    
243                    value.append(';');
244                    if (!additional.isEmpty()) {
245                            for (String s : additional) {
246                                    value.append(VCardStringUtils.escape(s)).append(',');
247                            }
248                            value.deleteCharAt(value.length() - 1);
249                    }
250    
251                    value.append(';');
252                    if (!prefixes.isEmpty()) {
253                            for (String s : prefixes) {
254                                    value.append(VCardStringUtils.escape(s)).append(',');
255                            }
256                            value.deleteCharAt(value.length() - 1);
257                    }
258    
259                    value.append(';');
260                    if (!suffixes.isEmpty()) {
261                            for (String s : suffixes) {
262                                    value.append(VCardStringUtils.escape(s)).append(',');
263                            }
264                            value.deleteCharAt(value.length() - 1);
265                    }
266            }
267    
268            @Override
269            protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
270                    //preserve empty items and don't unescape escaped characters(e.g. "additional" might have escaped commas)
271                    String split[] = VCardStringUtils.splitBy(value, ';', false, false);
272    
273                    int i = 0;
274    
275                    family = (split.length > i && split[i].length() > 0) ? VCardStringUtils.unescape(split[i]) : null;
276                    i++;
277    
278                    given = (split.length > i && split[i].length() > 0) ? VCardStringUtils.unescape(split[i]) : null;
279                    i++;
280    
281                    if (split.length > i && split[i].length() > 0) {
282                            additional = new ArrayList<String>(Arrays.asList(VCardStringUtils.splitBy(split[i], ',', true, true)));
283                    } else {
284                            additional = new ArrayList<String>();
285                    }
286                    i++;
287    
288                    if (split.length > i && split[i].length() > 0) {
289                            prefixes = new ArrayList<String>(Arrays.asList(VCardStringUtils.splitBy(split[i], ',', true, true)));
290                    } else {
291                            prefixes = new ArrayList<String>();
292                    }
293                    i++;
294    
295                    if (split.length > i && split[i].length() > 0) {
296                            suffixes = new ArrayList<String>(Arrays.asList(VCardStringUtils.splitBy(split[i], ',', true, true)));
297                    } else {
298                            suffixes = new ArrayList<String>();
299                    }
300                    i++;
301            }
302    
303            @Override
304            protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
305                    if (family != null) {
306                            parent.append("surname", family);
307                    }
308                    if (given != null) {
309                            parent.append("given", given);
310                    }
311                    parent.append("additional", additional);
312                    parent.append("prefix", prefixes);
313                    parent.append("suffix", suffixes);
314            }
315    
316            @Override
317            protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
318                    family = element.get("surname");
319                    given = element.get("given");
320    
321                    additional.clear();
322                    additional.addAll(element.getAll("additional"));
323    
324                    prefixes.clear();
325                    prefixes.addAll(element.getAll("prefix"));
326    
327                    suffixes.clear();
328                    suffixes.addAll(element.getAll("suffix"));
329            }
330    
331            @Override
332            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
333                    family = element.firstValue("family-name");
334                    given = element.firstValue("given-name");
335    
336                    additional.clear();
337                    additional.addAll(element.allValues("additional-name"));
338    
339                    prefixes.clear();
340                    prefixes.addAll(element.allValues("honorific-prefix"));
341    
342                    suffixes.clear();
343                    suffixes.addAll(element.allValues("honorific-suffix"));
344            }
