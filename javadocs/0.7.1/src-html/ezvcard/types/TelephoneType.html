047    /**
048     * A telephone number.
049     * 
050     * <pre>
051     * VCard vcard = new VCard();
052     * TelephoneType tel = new TelephoneType(&quot;+1 123-555-6789&quot;);
053     * tel.addType(TelephoneTypeParameter.HOME);
054     * tel.setPref(2); //the second-most preferred
055     * vcard.addTelephoneNumber(tel);
056     * tel = new TelephoneType(&quot;+1 800-555-9876;ext=111&quot;);
057     * tel.addType(TelephoneTypeParameter.WORK);
058     * tel.setPref(1); //the most preferred
059     * vcard.addTelephoneNumber(tel);
060     * </pre>
061     * 
062     * <p>
063     * vCard property name: TEL
064     * </p>
065     * <p>
066     * vCard versions: 2.1, 3.0, 4.0
067     * </p>
068     * @author Michael Angstadt
069     */
070    public class TelephoneType extends MultiValuedTypeParameterType<TelephoneTypeParameter> {
071            public static final String NAME = "TEL";
072    
073            private String value;
074    
075            public TelephoneType() {
076                    this(null);
077            }
078    
079            /**
080             * @param telNumber the telephone number
081             */
082            public TelephoneType(String telNumber) {
083                    super(NAME);
084                    this.value = telNumber;
085            }
086    
087            /**
088             * Gets the telephone number.
089             * @return the telephone number
090             */
091            public String getValue() {
092                    return value;
093            }
094    
095            /**
096             * Sets the telephone number.
097             * @param value the telephone number
098             */
099            public void setValue(String value) {
100                    this.value = value;
101            }
102    
103            /**
104             * Gets all PID parameter values.
105             * <p>
106             * vCard versions: 4.0
107             * </p>
108             * @return the PID values or empty set if there are none
109             * @see VCardSubTypes#getPids
110             */
111            public Set<Integer[]> getPids() {
112                    return subTypes.getPids();
113            }
114    
115            /**
116             * Adds a PID value.
117             * <p>
118             * vCard versions: 4.0
119             * </p>
120             * @param localId the local ID
121             * @param clientPidMapRef the ID used to reference the property's globally
122             * unique identifier in the CLIENTPIDMAP property.
123             * @see VCardSubTypes#addPid(int, int)
124             */
125            public void addPid(int localId, int clientPidMapRef) {
126                    subTypes.addPid(localId, clientPidMapRef);
127            }
128    
129            /**
130             * Removes all PID values.
131             * <p>
132             * vCard versions: 4.0
133             * </p>
134             * @see VCardSubTypes#removePids
135             */
136            public void removePids() {
137                    subTypes.removePids();
138            }
139    
140            /**
141             * Gets the preference value.
142             * <p>
143             * vCard versions: 4.0
144             * </p>
145             * @return the preference value or null if it doesn't exist
146             * @see VCardSubTypes#getPref
147             */
148            public Integer getPref() {
149                    return subTypes.getPref();
150            }
151    
152            /**
153             * Sets the preference value.
154             * <p>
155             * vCard versions: 4.0
156             * </p>
157             * @param pref the preference value or null to remove
158             * @see VCardSubTypes#setPref
159             */
160            public void setPref(Integer pref) {
161                    subTypes.setPref(pref);
162            }
163    
164            /**
165             * Gets the ALTID.
166             * <p>
167             * vCard versions: 4.0
168             * </p>
169             * @return the ALTID or null if it doesn't exist
170             * @see VCardSubTypes#getAltId
171             */
172            public String getAltId() {
173                    return subTypes.getAltId();
174            }
175    
176            /**
177             * Sets the ALTID.
178             * <p>
179             * vCard versions: 4.0
180             * </p>
181             * @param altId the ALTID or null to remove
182             * @see VCardSubTypes#setAltId
183             */
184            public void setAltId(String altId) {
185                    subTypes.setAltId(altId);
186            }
187    
188            @Override
189            protected TelephoneTypeParameter buildTypeObj(String type) {
190                    TelephoneTypeParameter param = TelephoneTypeParameter.valueOf(type);
191                    if (param == null) {
192                            param = new TelephoneTypeParameter(type);
193                    }
194                    return param;
195            }
196    
197            @Override
198            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
199                    if (version == VCardVersion.V4_0) {
200                            copy.setValue(ValueParameter.URI);
201                    } else {
202                            copy.setValue(null);
203                    }
204    
205                    //replace "TYPE=pref" with "PREF=1"
206                    if (version == VCardVersion.V4_0) {
207                            if (getTypes().contains(TelephoneTypeParameter.PREF)) {
208                                    copy.removeType(TelephoneTypeParameter.PREF.getValue());
209                                    copy.setPref(1);
210                            }
211                    } else {
212                            copy.setPref(null);
213    
214                            //find the TEL with the lowest PREF value in the vCard
215                            TelephoneType mostPreferred = null;
216                            for (TelephoneType tel : vcard.getTelephoneNumbers()) {
217                                    Integer pref = tel.getPref();
218                                    if (pref != null) {
219                                            if (mostPreferred == null || pref < mostPreferred.getPref()) {
220                                                    mostPreferred = tel;
221                                            }
222                                    }
223                            }
224                            if (this == mostPreferred) {
225                                    copy.addType(TelephoneTypeParameter.PREF.getValue());
226                            }
227                    }
228            }
229    
230            @Override
231            protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
232                    String value = writeValue(version);
233                    sb.append(VCardStringUtils.escape(value));
234            }
235    
236            @Override
237            protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
238                    value = VCardStringUtils.unescape(value.trim());
239                    parseValue(value);
240            }
241    
242            @Override
243            protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
244                    String value = writeValue(parent.getVersion());
245                    parent.appendUri(value);
246            }
247    
248            @Override
249            protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
250                    String value = element.get("text", "uri");
251                    if (value != null) {
252                            parseValue(value);
253                    }
254            }
255    
256            @Override
257            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
258                    List<String> types = element.types();
259                    for (String type : types) {
260                            subTypes.addType(type);
261                    }
262    
263                    String tel = null;
264                    String href = element.attr("href");
265                    if (href.length() > 0) {
266                            Pattern p = Pattern.compile("^tel:(.*?)$", Pattern.CASE_INSENSITIVE);
267                            Matcher m = p.matcher(href);
268                            if (m.find()) {
269                                    tel = m.group(1);
270                            }
271                    }
272                    if (tel == null) {
273                            tel = element.value();
274                    }
275                    setValue(tel);
276            }
277    
278            private void parseValue(String value) {
279                    if (value.matches("(?i)tel:.*")) {
280                            //remove "tel:"
281                            value = (value.length() > 4) ? value.substring(4) : "";
282                    }
283                    setValue(value);
284            }
285    
286            private String writeValue(VCardVersion version) {
287                    String value = this.value;
288                    if (version == VCardVersion.V4_0) {
289                            value = "tel:" + value;
290                    }
291                    return value;
292            }
