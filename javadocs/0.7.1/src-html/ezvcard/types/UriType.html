038    /**
039     * Represents a type whose value is a URI.
040     * @author Michael Angstadt
041     */
042    public class UriType extends TextType {
043            /**
044             * @param name the type name (e.g. "URL")
045             */
046            public UriType(String name) {
047                    super(name);
048            }
049    
050            /**
051             * @param name the type name (e.g. "URL")
052             * @param uri the type value
053             */
054            public UriType(String name, String uri) {
055                    super(name, uri);
056            }
057    
058            @Override
059            protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
060                    setValue(element.getUri());
061            }
062    
063            @Override
064            protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
065                    parent.appendUri(getValue());
066            }
067    
068            @Override
069            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
070                    String href = element.absUrl("href");
071                    if (href.length() > 0) {
072                            setValue(href);
073                    } else {
074                            super.doUnmarshalHtml(element, warnings);
075                    }
076            }
077    }
