063    /**
064     * Unmarshals vCards into {@link VCard} objects.
065     * @author Michael Angstadt
066     */
067    public class VCardReader implements Closeable, IParser {
068            private CompatibilityMode compatibilityMode = CompatibilityMode.RFC;
069            private List<String> warnings = new ArrayList<String>();
070            private Map<String, Class<? extends VCardType>> extendedTypeClasses = new HashMap<String, Class<? extends VCardType>>();
071            private FoldedLineReader reader;
072

074             * @param str the string to read the vCards from
075             */

077                    this(new StringReader(str));
078            }

132            public void registerExtendedType(Class<? extends VCardType> clazz) {

142            public List<String> getWarnings() {
143                    return new ArrayList<String>(warnings);
144            }

146            //@Override
147            public VCard readNext() throws IOException {
148                    warnings.clear();
149    
150                    VCard vcard = new VCard();
151    
152                    List<LabelType> labels = new ArrayList<LabelType>();
153                    VCardVersion version = null;
154                    boolean endFound = false;
155                    int typesRead = 0;
156                    String line;
157                    List<String> warningsBuf = new ArrayList<String>();
158    
159                    while (!endFound && (line = reader.readLine()) != null) {
160                            //parse the components out of the line
161                            VCardLine parsedLine = VCardLine.parse(line);
162                            if (parsedLine == null) {
163                                    warnings.add("Skipping malformed vCard line: \"" + line + "\"");
164                                    continue;
165                            }
166    
167                            //build the sub types 
168                            VCardSubTypes subTypes = new VCardSubTypes();
169                            for (String[] subTypeStr : parsedLine.getSubTypes()) {
170                                    //if the parameter is name-less, make a guess at what the name is
171                                    //v3.0 and v4.0 requires all sub types to have names, but v2.1 does not
172                                    String subTypeName = subTypeStr[0];
173                                    String subTypeValue = subTypeStr[1];
174                                    if (subTypeName == null) {
175                                            if (ValueParameter.valueOf(subTypeValue) != null) {
176                                                    subTypeName = ValueParameter.NAME;
177                                            } else if (EncodingParameter.valueOf(subTypeValue) != null) {
178                                                    subTypeName = EncodingParameter.NAME;
179                                            } else {
180                                                    //otherwise, assume it's a TYPE
181                                                    subTypeName = TypeParameter.NAME;
182                                            }
183    
184                                            subTypes.put(subTypeName, subTypeValue);
185                                    } else {
186                                            subTypeName = subTypeName.toUpperCase();
187    
188                                            //split the value up if it's a comma-delimited list (i.e. the "TYPE" sub type)
189                                            String commaSplit[] = VCardStringUtils.splitBy(subTypeValue, ',', true, true);
190                                            for (String s : commaSplit) {
191                                                    subTypes.put(subTypeName, s);
192                                            }
193                                    }
194                            }
195    
196                            String typeName = parsedLine.getTypeName().toUpperCase();
197                            String value = VCardStringUtils.ltrim(parsedLine.getValue());
198                            String groupName = parsedLine.getGroup();
199    
200                            //if the value is encoded in "quoted-printable", decode it
201                            //"quoted-printable" encoding is only supported in v2.1
202                            if (subTypes.getEncoding() == EncodingParameter.QUOTED_PRINTABLE) {
203                                    QuotedPrintableCodec codec = new QuotedPrintableCodec();
204                                    String charset = subTypes.getCharset();
205                                    try {
206                                            value = (charset == null) ? codec.decode(value) : codec.decode(value, charset);
207                                    } catch (DecoderException e) {
208                                            warnings.add("The value of the " + typeName + " type was marked as \"quoted-printable\", but it could not be decoded.  Assuming that the value is plain text.");
209                                    }
210                                    subTypes.setEncoding(null); //remove encoding sub type
211                            }
212    
213                            //vCard should start with "BEGIN"
214                            if (typesRead == 0 && !"BEGIN".equals(typeName)) {
215                                    warnings.add("vCard does not start with \"BEGIN\".");
216                            }
217    
218                            typesRead++;
219    
220                            if ("BEGIN".equals(typeName)) {
221                                    if (!"vcard".equalsIgnoreCase(value)) {
222                                            warnings.add("The value of the BEGIN property should be \"vcard\", but it is \"" + value + "\".");
223                                    }
224                            } else if ("END".equals(typeName)) {
225                                    endFound = true;
226                                    if (!"vcard".equalsIgnoreCase(value)) {
227                                            warnings.add("The value of the END property should be \"vcard\", but it is \"" + value + "\".");
228                                    }
229                            } else if ("VERSION".equals(typeName)) {
230                                    if (version == null) {
231                                            version = VCardVersion.valueOfByStr(value);
232                                            if (version == null) {
233                                                    warnings.add("Invalid value of VERSION property: " + value);
234                                            }
235                                    } else {
236                                            warnings.add("Additional VERSION property encountered: \"" + value + "\".  It will be ignored.");
237                                    }
238                                    vcard.setVersion(version);
239                            } else {
240                                    //create the type object
241                                    VCardType type = createTypeObject(typeName);
242                                    type.setGroup(groupName);
243    
244                                    //unmarshal the text string into the object
245                                    warningsBuf.clear();
246                                    try {
247                                            type.unmarshalValue(subTypes, value, version, warningsBuf, compatibilityMode);
248    
249                                            //add to vcard
250                                            if (type instanceof LabelType) {
251                                                    //LABELs must be treated specially so they can be matched up with their ADRs
252                                                    labels.add((LabelType) type);
253                                            } else {
254                                                    addToVCard(type, vcard);
255                                            }
256                                    } catch (SkipMeException e) {
257                                            warningsBuf.add(type.getTypeName() + " property will not be unmarshalled: " + e.getMessage());
258                                    } catch (EmbeddedVCardException e) {
259                                            //parse an embedded vCard (i.e. the AGENT type)
260    
261                                            VCardReader agentReader;
262                                            if (value.length() == 0 || version == null || version == VCardVersion.V2_1) {
263                                                    //vCard will be added as a nested vCard (2.1 style)
264                                                    agentReader = new VCardReader(reader);
265                                            } else {
266                                                    //vCard will be contained within the type value (3.0 style)
267                                                    value = VCardStringUtils.unescape(value);
268                                                    agentReader = new VCardReader(new StringReader(value));
269                                            }
270    
271                                            agentReader.setCompatibilityMode(compatibilityMode);
272                                            try {
273                                                    VCard agentVcard = agentReader.readNext();
274                                                    e.injectVCard(agentVcard);
275                                            } finally {
276                                                    for (String w : agentReader.getWarnings()) {
277                                                            warnings.add("Problem unmarshalling nested vCard value from " + type.getTypeName() + ": " + w);
278                                                    }
279                                            }
280    
281                                            addToVCard(type, vcard);
282                                    } finally {
283                                            warnings.addAll(warningsBuf);
284                                    }
285                            }
286                    }
287    
288                    if (typesRead == 0) {
289                            //end of stream reached
290                            return null;
291                    }
292    
293                    //assign labels to their addresses
294                    for (LabelType label : labels) {
295                            boolean orphaned = true;
296                            for (AddressType adr : vcard.getAddresses()) {
297                                    if (adr.getLabel() == null && adr.getTypes().equals(label.getTypes())) {
298                                            adr.setLabel(label.getValue());
299                                            orphaned = false;
300                                            break;
301                                    }
302                            }
303                            if (orphaned) {
304                                    vcard.addOrphanedLabel(label);
305                            }
306                    }
307    
308                    if (!endFound) {
309                            warnings.add("vCard does not terminate with the END property.");
310                    }
311    
312                    return vcard;
313            }
314    
315            /**
316             * Creates the appropriate {@link VCardType} instance, given the type name.
317             * This method does not unmarshal the type, it just creates the type object.
318             * @param name the type name (e.g. "FN")
319             * @return the Type that was created
320             */
321            private VCardType createTypeObject(String name) {
322                    Class<? extends VCardType> clazz = TypeList.getTypeClass(name);
323                    VCardType t;
324                    if (clazz != null) {
325                            try {
326                                    //create a new instance of the class
327                                    t = clazz.newInstance();
328                            } catch (Exception e) {
329                                    //it is the responsibility of the EZ-vCard developer to ensure that this exception is never thrown
330                                    //all type classes defined in the EZ-vCard library MUST have public, no-arg constructors
331                                    throw new RuntimeException(e);
332                            }
333                    } else {
334                            Class<? extends VCardType> extendedTypeClass = extendedTypeClasses.get(name);
335                            if (extendedTypeClass != null) {
336                                    try {
337                                            t = extendedTypeClass.newInstance();
338                                    } catch (Exception e) {
339                                            //this should never happen because the type class is checked to see if it has a public, no-arg constructor in the "registerExtendedType" method
340                                            throw new RuntimeException("Extended type class \"" + extendedTypeClass.getName() + "\" must have a public, no-arg constructor.");
341                                    }
342                            } else {
343                                    t = new RawType(name); //use RawType instead of TextType because we don't want to unescape any characters that might be meaningful to this type
344                                    if (!name.startsWith("X-")) {
345                                            warnings.add("Non-standard type \"" + name + "\" found.  Treating it as an extended type.");
346                                    }
347                            }
348                    }
349                    return t;
350            }
351    
352            /**
353             * Adds a type object to the vCard.
354             * @param t the type object
355             * @param vcard the vCard
356             */
357            private void addToVCard(VCardType t, VCard vcard) {
358                    Method method = TypeList.getAddMethod(t.getClass());
359                    if (method != null) {
360                            try {
361                                    method.invoke(vcard, t);
362                            } catch (Exception e) {
363                                    //this should NEVER be thrown because the method MUST be public
364                                    throw new RuntimeException(e);
365                            }
366                    } else {
367                            vcard.addExtendedType(t);
368                    }
369            }
370    
371            /**
372             * Gets the type name from a type class.
373             * @param clazz the type class
374             * @return the type name
375             */
376            private String getTypeNameFromTypeClass(Class<? extends VCardType> clazz) {
377                    try {
378                            VCardType t = clazz.newInstance();
379                            return t.getTypeName().toUpperCase();
380                    } catch (Exception e) {
381                            //there is no public, no-arg constructor
382                            throw new RuntimeException(e);
383                    }
384            }

387             * Closes the underlying {@link Reader} object.
388             */
389            public void close() throws IOException {
390                    reader.close();
391            }
