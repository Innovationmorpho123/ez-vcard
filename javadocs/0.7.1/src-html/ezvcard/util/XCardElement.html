041    /**
042     * Wraps xCard functionality around an XML {@link Element}.
043     * @author Michael Angstadt
044     */
045    public class XCardElement {
046            private final Element element;
047            private List<Element> children;
048            private final VCardVersion version;
049            private final String namespace;
050    
051            /**
052             * @param element the XML element
053             */
054            public XCardElement(Element element) {
055                    this(element, VCardVersion.V4_0);
056            }
057    
058            /**
059             * @param element the XML element
060             * @param version the vCard version
061             */
062            public XCardElement(Element element, VCardVersion version) {
063                    this.element = element;
064                    this.version = version;
065                    namespace = version.getXmlNamespace();
066            }
067    
068            /**
069             * Gets the value of the first <code>&lt;text&gt;</code> child element.
070             * @return the text value or null if not found
071             */
072            public String getText() {
073                    return get("text");
074            }
075    
076            /**
077             * Gets the value of the first <code>&lt;uri&gt;</code> child element.
078             * @return the URI or null if not found
079             */
080            public String getUri() {
081                    return get("uri");
082            }
083    
084            /**
085             * Gets the value of the first <code>&lt;date-and-or-time&gt;</code> child
086             * element.
087             * @return the value or null if not found
088             */
089            public String getDateAndOrTime() {
090                    return get("date-and-or-time");
091            }
092    
093            /**
094             * Gets the value of the first <code>&lt;timestamp&gt;</code> child element.
095             * @return the timestamp or null if not found
096             */
097            public String getTimestamp() {
098                    return get("timestamp");
099            }
100    
101            /**
102             * Gets the value of the first child element with one of the given names.
103             * @param names the possible names of the element
104             * @return the element's text or null if not found
105             */
106            public String get(String... names) {
107                    List<String> localNamesList = Arrays.asList(names);
108                    for (Element child : getChildren()) {
109                            if (localNamesList.contains(child.getLocalName()) && namespace.equals(child.getNamespaceURI())) {
110                                    return child.getTextContent();
111                            }
112                    }
113                    return null;
114            }
115    
116            /**
117             * Gets the value of all non-empty child elements that have the given name.
118             * @param localName the element name
119             * @return the values of the child elements
120             */
121            public List<String> getAll(String localName) {
122                    List<String> childrenText = new ArrayList<String>();
123                    for (Element child : getChildren()) {
124                            if (localName.equals(child.getLocalName()) && namespace.equals(child.getNamespaceURI())) {
125                                    String text = child.getTextContent();
126                                    if (text.length() > 0) {
127                                            childrenText.add(child.getTextContent());
128                                    }
129                            }
130                    }
131                    return childrenText;
132            }
133    
134            /**
135             * Adds a <code>&lt;text&gt;</code> child element.
136             * @param text the text
137             * @return the created element
138             */
139            public Element appendText(String text) {
140                    return append("text", text);
141            }
142    
143            /**
144             * Adds a <code>&lt;uri&gt;</code> child element.
145             * @param uri the URI
146             * @return the created element
147             */
148            public Element appendUri(String uri) {
149                    return append("uri", uri);
150            }
151    
152            /**
153             * Adds a <code>&lt;date-and-or-time&gt;</code> child element.
154             * @param dateAndOrTime the date/time
155             * @return the created element
156             */
157            public Element appendDateAndOrTime(String dateAndOrTime) {
158                    return append("date-and-or-time", dateAndOrTime);
159            }
160    
161            /**
162             * Adds a <code>&lt;timestamp&gt;</code> child element.
163             * @param timestamp the timestamp
164             * @return the created element
165             */
166            public Element appendTimestamp(String timestamp) {
167                    return append("timestamp", timestamp);
168            }
169    
170            /**
171             * Adds a child element.
172             * @param name the name of the child element
173             * @param value the value of the child element.
174             * @return the created element
175             */
176            public Element append(String name, String value) {
177                    Element child = element.getOwnerDocument().createElementNS(namespace, name);
178                    child.setTextContent(value);
179                    element.appendChild(child);
180                    return child;
181            }
182    
183            /**
184             * Adds multiple child elements, each with the same name.
185             * @param name the name for all the child elements
186             * @param values the values of each child element
187             * @return the created elements
188             */
189            public List<Element> append(String name, Collection<String> values) {
190                    List<Element> elements = new ArrayList<Element>(values.size());
191                    for (String value : values) {
192                            elements.add(append(name, value));
193                    }
194                    return elements;
195            }
196    
197            /**
198             * Gets the wrapped XML element.
199             * @return the wrapped XML element
200             */
201            public Element getElement() {
202                    return element;
203            }
204    
205            /**
206             * Gets the vCard version.
207             * @return the vCard version
208             */
209            public VCardVersion getVersion() {
210                    return version;
211            }
212    
213            /**
214             * Gets the child elements of the XML element.
215             * @return the child elements
216             */
217            private List<Element> getChildren() {
218                    if (children == null) {
219                            children = XmlUtils.toElementList(element.getChildNodes());
220                    }
221                    return children;
222            }
223    }
