056    /**
057     * Generic XML utility methods.
058     * @author Michael Angstadt
059     */
060    public class XmlUtils {
061            /**
062             * Parses an XML string into a DOM.
063             * @param xml the XML string
064             * @return the parsed DOM
065             * @throws SAXException if the string is not valid XML
066             */
067            public static Document toDocument(String xml) throws SAXException {
068                    try {
069                            DocumentBuilderFactory fact = DocumentBuilderFactory.newInstance();
070                            fact.setNamespaceAware(true);
071                            DocumentBuilder db = fact.newDocumentBuilder();
072                            InputSource source = new InputSource(new StringReader(xml));
073                            return db.parse(source);
074                    } catch (ParserConfigurationException e) {
075                            //no complex configurations
076                    } catch (IOException e) {
077                            //reading from string
078                    }
079                    return null;
080            }
081    
082            /**
083             * Converts a DOM to a string.
084             * @param document the DOM
085             * @return the string
086             */
087            public static String toString(Document document) {
088                    try {
089                            Transformer transformer = TransformerFactory.newInstance().newTransformer();
090                            DOMSource source = new DOMSource(document);
091                            StringWriter writer = new StringWriter();
092                            StreamResult result = new StreamResult(writer);
093                            transformer.transform(source, result);
094                            return writer.toString();
095                    } catch (TransformerConfigurationException e) {
096                            //no complex configurations
097                    } catch (TransformerFactoryConfigurationError e) {
098                            //no complex configurations
099                    } catch (TransformerException e) {
100                            //writing to string
101                    }
102                    return null;
103            }
104    
105            /**
106             * Gets all the elements out of a {@link NodeList}.
107             * @param nodeList the node list
108             * @return the elements
109             */
110            public static List<Element> toElementList(NodeList nodeList) {
111                    List<Element> elements = new ArrayList<Element>();
112                    for (int i = 0; i < nodeList.getLength(); i++) {
113                            Node node = nodeList.item(i);
114                            if (node instanceof Element) {
115                                    elements.add((Element) node);
116                            }
117                    }
118                    return elements;
119            }
120    
121            /**
122             * Gets the root element of a document.
123             * @param parent the document
124             * @return the root element
125             */
126            public static Element getRootElement(Document parent) {
127                    return getFirstChildElement((Node) parent);
128            }
129    
130            /**
131             * Gets the first child element of an element.
132             * @param parent the parent element
133             * @return the first child element or null if there are no child elements
134             */
135            public static Element getFirstChildElement(Element parent) {
136                    return getFirstChildElement((Node) parent);
137            }
138    
139            /**
140             * Gets the first child element of a node.
141             * @param parent the node
142             * @return the first child element or null if there are no child elements
143             */
144            private static Element getFirstChildElement(Node parent) {
145                    NodeList nodeList = parent.getChildNodes();
146                    for (int i = 0; i < nodeList.getLength(); i++) {
147                            Node node = nodeList.item(i);
148                            if (node instanceof Element) {
149                                    return (Element) node;
150                            }
151                    }
152                    return null;
153            }
154    
155            private XmlUtils() {
156                    //hide
157            }
158    }
