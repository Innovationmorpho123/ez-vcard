052    /**
053     * A public key for encryption.
054     * 
055     * <p>
056     * <b>Adding a key</b>
057     * </p>
058     * 
059     * <pre>
060     * VCard vcard = new VCard();
061     * 
062     * //URL (vCard 4.0 only; KEYs cannot have URLs in vCard 2.1 and 3.0)
063     * KeyType key = new KeyType("http://www.mywebsite.com/pubkey.pgp", KeyTypeParameter.PGP);
064     * vcard.addKey(key);
065     * 
066     * //binary data
067     * byte data[] = ...
068     * key = new KeyType(data, KeyTypeParameter.PGP);
069     * vcard.addKey(key);
070     * 
071     * //plain text value
072     * key = new KeyType();
073     * key.setText("...", KeyTypeParameter.PGP);
074     * vcard.addKey(key);
075     * 
076     * //if "KeyTypeParameter" does not have the pre-defined constant that you need, then create a new instance
077     * //arg 1: the value of the 2.1/3.0 TYPE parameter
078     * //arg 2: the value to use for the 4.0 MEDIATYPE parameter and for 4.0 data URIs
079     * //arg 3: the file extension of the data type (optional)
080     * KeyTypeParameter param = new KeyTypeParameter("mykey", "application/my-key", "mkey");
081     * key = new KeyType("http://www.mywebsite.com/pubkey.enc", param);
082     * vcard.addKey(key);
083     * </pre>
084     * 
085     * <p>
086     * <b>Getting the keys</b>
087     * </p>
088     * 
089     * <pre>
090     * VCard vcard = ...
091     * 
092     * int fileCount = 0;
093     * for (KeyType key : vcard.getKeys()){
094     *   //the key will have either a URL or a binary data
095     *   //only 4.0 vCards are allowed to use URLs for keys
096     *   if (key.getData() == null){
097     *     System.out.println("Key URL: " + key.getUrl());
098     *   } else {
099     *     KeyTypeParameter type = key.getContentType();
100     *     
101     *     if (type == null) {
102     *       //the vCard may not have any content type data associated with the key
103     *       System.out.println("Saving a key file...");
104     *     } else {
105     *       System.out.println("Saving a \"" + type.getMediaType() + "\" file...");
106     *     }
107     *     
108     *     String folder;
109     *     if (type == KeyTypeParameter.PGP){ //it is safe to use "==" instead of "equals()"
110     *       folder = "pgp-keys";
111     *     } else {
112     *       folder = "other-keys";
113     *     }
114     *     
115     *     byte data[] = key.getData();
116     *     String filename = "key" + fileCount;
117     *     if (type != null && type.getExtension() != null){
118     *      filename += "." + type.getExtension();
119     *     }
120     *     OutputStream out = new FileOutputStream(new File(folder, filename));
121     *     out.write(data);
122     *     out.close();
123     *     fileCount++;
124     *   }
125     * }
126     * </pre>
127     * 
128     * <p>
129     * <b>Property name:</b> <code>KEY</code>
130     * </p>
131     * <p>
132     * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
133     * </p>
134     * @author Michael Angstadt
135     */
136    public class KeyType extends BinaryType<KeyTypeParameter> {
137            public static final String NAME = "KEY";
138    
139            private String text;
140    
141            public KeyType() {
142                    super(NAME);
143            }
144    
145            /**
146             * @param data the binary data
147             * @param type the type of key (e.g. PGP)
148             */
149            public KeyType(byte data[], KeyTypeParameter type) {
150                    super(NAME, data, type);
151            }
152    
153            /**
154             * @param url the URL to the key (vCard 4.0 only)
155             * @param type the type of key (e.g. PGP)
156             */
157            public KeyType(String url, KeyTypeParameter type) {
158                    super(NAME, url, type);
159            }
160    
161            /**
162             * @param in an input stream to the binary data (will be closed)
163             * @param type the content type (e.g. PGP)
164             * @throws IOException if there's a problem reading from the input stream
165             */
166            public KeyType(InputStream in, KeyTypeParameter type) throws IOException {
167                    super(NAME, in, type);
168            }
169    
170            /**
171             * @param file the key file
172             * @param type the content type (e.g. PGP)
173             * @throws IOException if there's a problem reading from the file
174             */
175            public KeyType(File file, KeyTypeParameter type) throws IOException {
176                    super(NAME, file, type);
177            }
178    
179            /**
180             * Sets a plain text representation of the key.
181             * @param text the key in plain text
182             * @param type the key type
183             */
184            public void setText(String text, KeyTypeParameter type) {
185                    this.text = text;
186                    data = null;
187                    url = null;
188                    setContentType(type);
189            }
190    
191            /**
192             * Gets the plain text representation of the key.
193             * @return the key in plain text
194             */
195            public String getText() {
196                    return text;
197            }
198    
199            @Override
200            protected KeyTypeParameter buildTypeObj(String type) {
201                    KeyTypeParameter param = KeyTypeParameter.valueOf(type);
202                    if (param == null) {
203                            param = new KeyTypeParameter(type, "application/" + type, null);
204                    }
205                    return param;
206            }
207    
208            @Override
209            protected KeyTypeParameter buildMediaTypeObj(String mediaType) {
210                    KeyTypeParameter p = KeyTypeParameter.findByMediaType(mediaType);
211                    if (p == null) {
212                            int slashPos = mediaType.indexOf('/');
213                            String type;
214                            if (slashPos == -1 || slashPos < mediaType.length() - 1) {
215                                    type = "";
216                            } else {
217                                    type = mediaType.substring(slashPos + 1);
218                            }
219                            p = new KeyTypeParameter(type, mediaType, null);
220                    }
221                    return p;
222            }
223    
224            @Override
225            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
226                    if (text != null) {
227                            MediaTypeParameter contentType = getContentType();
228                            if (contentType == null) {
229                                    contentType = new MediaTypeParameter(null, null, null);
230                            }
231    
232                            copy.setValue(ValueParameter.TEXT);
233                            copy.setEncoding(null);
234                            if (version == VCardVersion.V4_0) {
235                                    //don't null out TYPE, it could be set to "home", "work", etc
236                                    copy.setMediaType(contentType.getMediaType());
237                            } else {
238                                    copy.setType(contentType.getValue());
239                                    copy.setMediaType(null);
240                            }
241                    }
242            }
243    
244            @Override
245            protected void doMarshalText(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
246                    if (text != null) {
247                            sb.append(VCardStringUtils.escape(text));
248                    } else {
249                            if ((version == VCardVersion.V2_1 || version == VCardVersion.V3_0) && getUrl() != null) {
250                                    warnings.add("vCard version " + version + " does not allow URLs to be used in the " + NAME + " type.");
251                            }
252                            super.doMarshalText(sb, version, warnings, compatibilityMode);
253                    }
254            }
255    
256            @Override
257            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
258                    if (subTypes.getValue() == ValueParameter.TEXT) {
259                            parseText(VCardStringUtils.unescape(value), version);
260                    } else {
261                            super.doUnmarshalText(value, version, warnings, compatibilityMode);
262                    }
263            }
264    
265            @Override
266            protected void doMarshalXml(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
267                    if (text != null) {
268                            parent.text(text);
269                    } else {
270                            super.doMarshalXml(parent, warnings, compatibilityMode);
271                    }
272            }
273    
274            @Override
275            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
276                    String value = element.text();
277                    if (value != null) {
278                            parseText(value, element.version());
279                    } else {
280                            super.doUnmarshalXml(element, warnings, compatibilityMode);
281                    }
282            }
283    
284            @Override
285            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
286                    String elementName = element.tagName();
287                    if ("a".equals(elementName)) {
288                            String href = element.absUrl("href");
289                            if (href.length() > 0) {
290                                    try {
291                                            DataUri uri = new DataUri(href);
292                                            KeyTypeParameter mediaType = buildMediaTypeObj(uri.getContentType());
293                                            setData(uri.getData(), mediaType);
294                                    } catch (IllegalArgumentException e) {
295                                            //TODO create buildTypeObjFromExtension() method
296                                            setUrl(href, null);
297                                    }
298                            } else {
299                                    throw new SkipMeException("<a> tag does not have a \"href\" attribute.");
300                            }
301                    } else {
302                            super.doUnmarshalHtml(element, warnings);
303                    }
304            }
305    
306            @Override
307            protected JCardValue doMarshalJson(VCardVersion version, List<String> warnings) {
308                    if (text != null) {
309                            return JCardValue.text(text);
310                    } else {
311                            return super.doMarshalJson(version, warnings);
312                    }
313            }
314    
315            @Override
316            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
317                    if (value.getDataType() == JCardDataType.TEXT) {
318                            parseText(value.getFirstValueAsString(), version);
319                    } else {
320                            super.doUnmarshalJson(value, version, warnings);
321                    }
322            }
323    
324            @Override
325            protected void cannotUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, KeyTypeParameter contentType) {
326                    //unmarshal it as a plain text key
327                    setText(value, contentType);
328            }
329    
330            private void parseText(String value, VCardVersion version) {
331                    setText(value, parseContentType(version));
332            }
333    }
