035    /**
036     * Defines a URI representing the person's work place, which can be used to
037     * lookup information on the person's co-workers.
038     * 
039     * <p>
040     * <b>Code sample</b>
041     * </p>
042     * 
043     * <pre>
044     * VCard vcard = new VCard();
045     * OrgDirectoryType orgDirectory = new OrgDirectoryType(&quot;http://www.company.com/staff&quot;);
046     * vcard.addOrgDirectory(orgDirectory);
047     * </pre>
048     * 
049     * <p>
050     * <b>Property name:</b> <code>ORG-DIRECTORY</code>
051     * </p>
052     * <p>
053     * <b>Supported versions:</b> <code>4.0</code>
054     * </p>
055     * @author Michael Angstadt
056     * @see <a href="http://tools.ietf.org/html/rfc6715">RFC 6715</a>
057     */
058    public class OrgDirectoryType extends UriType implements HasAltId {
059            public static final String NAME = "ORG-DIRECTORY";
060    
061            public OrgDirectoryType() {
062                    this(null);
063            }
064    
065            /**
066             * @param uri the URI
067             */
068            public OrgDirectoryType(String uri) {
069                    super(NAME, uri);
070            }
071    
072            @Override
073            public VCardVersion[] getSupportedVersions() {
074                    return new VCardVersion[] { VCardVersion.V4_0 };
075            }
076    
077            /**
078             * Gets the INDEX parameter.
079             * @return the INDEX or null if not set
080             * @see VCardSubTypes#getIndex
081             */
082            public Integer getIndex() {
083                    return subTypes.getIndex();
084            }
085    
086            /**
087             * Sets the INDEX parameter.
088             * @param index the INDEX or null to remove
089             * @see VCardSubTypes#setIndex
090             */
091            public void setIndex(Integer index) {
092                    subTypes.setIndex(index);
093            }
094    
095            /**
096             * Gets the TYPE parameter.
097             * @return the TYPE value (typically, this will be either "work" or "home")
098             * or null if it doesn't exist
099             */
100            public String getType() {
101                    return subTypes.getType();
102            }
103    
104            /**
105             * Sets the TYPE parameter.
106             * @param type the TYPE value (this should be either "work" or "home") or
107             * null to remove
108             */
109            public void setType(String type) {
110                    subTypes.setType(type);
111            }
112    
113            /**
114             * Gets the LANGUAGE parameter.
115             * @return the language or null if not set
116             * @see VCardSubTypes#getLanguage
117             */
118            public String getLanguage() {
119                    return subTypes.getLanguage();
120            }
121    
122            /**
123             * Sets the LANGUAGE parameter.
124             * @param language the language or null to remove
125             * @see VCardSubTypes#setLanguage
126             */
127            public void setLanguage(String language) {
128                    subTypes.setLanguage(language);
129            }
130    
131            /**
132             * Gets the preference value.
133             * <p>
134             * <b>Supported versions:</b> <code>4.0</code>
135             * </p>
136             * @return the preference value or null if it doesn't exist
137             * @see VCardSubTypes#getPref
138             */
139            public Integer getPref() {
140                    return subTypes.getPref();
141            }
142    
143            /**
144             * Sets the preference value.
145             * <p>
146             * <b>Supported versions:</b> <code>4.0</code>
147             * </p>
148             * @param pref the preference value or null to remove
149             * @see VCardSubTypes#setPref
150             */
151            public void setPref(Integer pref) {
152                    subTypes.setPref(pref);
153            }
154    
155            //@Override
156            public String getAltId() {
157                    return subTypes.getAltId();
158            }
159    
160            //@Override
161            public void setAltId(String altId) {
162                    subTypes.setAltId(altId);
163            }
164    }
