040    /**
041     * A list of keywords (aka "tags") that can be used to describe the person.
042     * 
043     * <p>
044     * <b>Code sample</b>
045     * </p>
046     * 
047     * <pre>
048     * VCard vcard = new VCard();
049     * CategoriesType categories = new CategoriesType();
050     * categories.addValue(&quot;programmer&quot;);
051     * categories.addValue(&quot;developer&quot;);
052     * categories.addValue(&quot;java-coder&quot;);
053     * categories.addValue(&quot;ladies-man&quot;);
054     * vcard.setCategories(categories);
055     * </pre>
056     * 
057     * <p>
058     * <b>Property name:</b> <code>CATEGORIES</code>
059     * </p>
060     * <p>
061     * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
062     * </p>
063     * @author Michael Angstadt
064     */
065    public class CategoriesType extends TextListType implements HasAltId {
066            public static final String NAME = "CATEGORIES";
067    
068            public CategoriesType() {
069                    super(NAME, ',');
070            }
071    
072            /**
073             * Gets all PID parameter values.
074             * <p>
075             * <b>Supported versions:</b> <code>4.0</code>
076             * </p>
077             * @return the PID values or empty set if there are none
078             * @see VCardSubTypes#getPids
079             */
080            public List<Integer[]> getPids() {
081                    return subTypes.getPids();
082            }
083    
084            /**
085             * Adds a PID value.
086             * <p>
087             * <b>Supported versions:</b> <code>4.0</code>
088             * </p>
089             * @param localId the local ID
090             * @param clientPidMapRef the ID used to reference the property's globally
091             * unique identifier in the CLIENTPIDMAP property.
092             * @see VCardSubTypes#addPid(int, int)
093             */
094            public void addPid(int localId, int clientPidMapRef) {
095                    subTypes.addPid(localId, clientPidMapRef);
096            }
097    
098            /**
099             * Removes all PID values.
100             * <p>
101             * <b>Supported versions:</b> <code>4.0</code>
102             * </p>
103             * @see VCardSubTypes#removePids
104             */
105            public void removePids() {
106                    subTypes.removePids();
107            }
108    
109            /**
110             * Gets the preference value.
111             * <p>
112             * <b>Supported versions:</b> <code>4.0</code>
113             * </p>
114             * @return the preference value or null if it doesn't exist
115             * @see VCardSubTypes#getPref
116             */
117            public Integer getPref() {
118                    return subTypes.getPref();
119            }
120    
121            /**
122             * Sets the preference value.
123             * <p>
124             * <b>Supported versions:</b> <code>4.0</code>
125             * </p>
126             * @param pref the preference value or null to remove
127             * @see VCardSubTypes#setPref
128             */
129            public void setPref(Integer pref) {
130                    subTypes.setPref(pref);
131            }
132    
133            //@Override
134            public String getAltId() {
135                    return subTypes.getAltId();
136            }
137    
138            //@Override
139            public void setAltId(String altId) {
140                    subTypes.setAltId(altId);
141            }
142    
143            /**
144             * Gets the TYPE parameter.
145             * <p>
146             * <b>Supported versions:</b> <code>4.0</code>
147             * </p>
148             * @return the TYPE value (typically, this will be either "work" or "home")
149             * or null if it doesn't exist
150             */
151            public String getType() {
152                    return subTypes.getType();
153            }
154    
155            /**
156             * Sets the TYPE parameter.
157             * <p>
158             * <b>Supported versions:</b> <code>4.0</code>
159             * </p>
160             * @param type the TYPE value (this should be either "work" or "home") or
161             * null to remove
162             */
163            public void setType(String type) {
164                    subTypes.setType(type);
165            }
166    
167            @Override
168            protected void doMarshalText(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
169                    if (!values.isEmpty()) {
170                            for (String value : values) {
171                                    sb.append(VCardStringUtils.escape(value));
172                                    if (compatibilityMode == CompatibilityMode.KDE_ADDRESS_BOOK) {
173                                            //KDE escapes all the comma delimiters
174                                            sb.append('\\');
175                                    }
176                                    sb.append(',');
177                            }
178                            if (compatibilityMode == CompatibilityMode.KDE_ADDRESS_BOOK) {
179                                    sb.deleteCharAt(sb.length() - 1);
180                            }
181                            sb.deleteCharAt(sb.length() - 1);
182                    }
183            }
184    
185            @Override
186            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
187                    if (compatibilityMode == CompatibilityMode.KDE_ADDRESS_BOOK) {
188                            //KDE escapes all the comma delimiters
189                            value = VCardStringUtils.unescape(value);
190                    }
191                    super.doUnmarshalText(value, version, warnings, compatibilityMode);
192            }
193    
194            @Override
195            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
196                    String rel = element.attr("rel");
197                    if (rel.length() > 0) {
198                            addValue(rel);
199                    } else {
200                            addValue(element.value());
201                    }
202            }
203    }
