051    /**
052     * Represents a vCard key/value pair entry (called a "type" or "property").
053     * @author Michael Angstadt
054     */
055    public abstract class VCardType implements Comparable<VCardType> {
056            /**
057             * The name of the type.
058             */
059            protected final String typeName;
060    
061            /**
062             * The group that this type belongs to or null if it doesn't belong to a
063             * group.
064             */
065            protected String group;
066    
067            /**
068             * The list of attributes that are associated with this type (called
069             * "sub types" or "parameters").
070             */
071            protected VCardSubTypes subTypes = new VCardSubTypes();

097             * Converts this type object to a string for sending over the wire. It is
098             * NOT responsible for folding.
099             * @param version the version vCard that is being generated
100             * @param warnings allows the programmer to alert the user to any
101             * note-worthy (but non-critical) issues that occurred during the
102             * marshalling process
103             * @param compatibilityMode allows the programmer to customize the
104             * marshalling process depending on the expected consumer of the vCard
105             * @return the string for sending over the wire
106             * @throws SkipMeException if this type should NOT be marshalled into the
107             * vCard
108             * @throws EmbeddedVCardException if the value of this type is an embedded
109             * vCard (i.e. the AGENT type)
110             */
111            public final String marshalText(VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
112                    StringBuilder sb = new StringBuilder();
113                    doMarshalText(sb, version, warnings, compatibilityMode);
114                    return sb.toString();
115            }

135             * Marshals this type for inclusion in an xCard (XML document).

138             * type.
139             * @param version the version vCard that is being generated
140             * @param warnings allows the programmer to alert the user to any
141             * note-worthy (but non-critical) issues that occurred during the
142             * marshalling process

153            /**
154             * Marshals this type for inclusion in an xCard (XML document). All child
155             * classes SHOULD override this, but are not required to.
156             * @param parent the XML element that the type's value will be inserted
157             * into. For example, this would be the "&lt;fn&gt;" element for the "FN"
158             * type.
159             * @param warnings allows the programmer to alert the user to any
160             * note-worthy (but non-critical) issues that occurred during the
161             * marshalling process
162             * @param compatibilityMode allows the programmer to customize the
163             * marshalling process depending on the expected consumer of the vCard
164             * @throws SkipMeException if this type should NOT be marshalled into the
165             * vCard

167            protected void doMarshalXml(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
168                    String value = marshalText(parent.version(), warnings, compatibilityMode);
169                    parent.append("unknown", value);
170            }

172            /**
173             * Marshals this type for inclusion in a jCard (JSON document).
174             * @param version the version vCard that is being generated
175             * @param warnings allows the programmer to alert the user to any
176             * note-worthy (but non-critical) issues that occurred during the

183                    return doMarshalJson(version, warnings);
184            }

186            /**
187             * Marshals this type for inclusion in a jCard (JSON document).
188             * @param version the version vCard that is being generated
189             * @param warnings allows the programmer to alert the user to any
190             * note-worthy (but non-critical) issues that occurred during the
191             * marshalling process
192             * @return the marshalled jCard value
193             * @throws SkipMeException if this type should NOT be marshalled into the
194             * vCard
195             */
196            protected JCardValue doMarshalJson(VCardVersion version, List<String> warnings) {
197                    JCardValue value = new JCardValue();
198                    String valueStr = marshalText(version, warnings, CompatibilityMode.RFC);
199                    value.addValues(valueStr);
200    
201                    //determine the data type based on the VALUE parameter
202                    JCardDataType dataType;
203                    ValueParameter valueParam = subTypes.getValue();
204                    if (valueParam == null) {
205                            dataType = JCardDataType.UNKNOWN;
206                    } else {
207                            dataType = JCardDataType.find(valueParam.getValue());
208                            if (dataType == null) {
209                                    dataType = JCardDataType.UNKNOWN;
210                            }
211                    }
212                    value.setDataType(dataType);
213    
214                    return value;
215            }

382             * @param subTypes the sub types that were parsed
383             * @param value includes the data type and property value(s)
384             * @param version the version of the jCard that is being read
385             * @param warnings allows the programmer to alert the user to any

396            /**
397             * Unmarshals the type from a jCard (JSON).
398             * @param value includes the data type and property value(s)
399             * @param version the version of the jCard that is being read
400             * @param warnings allows the programmer to alert the user to any
401             * note-worthy (but non-critical) issues that occurred during the
402             * unmarshalling process
403             * @throws SkipMeException if this type should NOT be added to the
404             * {@link VCard} object
405             */
406            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
407                    StringBuilder sb = new StringBuilder();
408    
409                    char delimiter = value.isStructured() ? ';' : ',';
410                    boolean firstOutter = true;
411                    for (List<String> valueStrs : value.getValuesAsStrings()) {
412                            if (!firstOutter) {
413                                    sb.append(delimiter);
414                            }
415    
416                            boolean firstInner = true;
417                            for (String valueStr : valueStrs) {
418                                    valueStr = VCardStringUtils.escape(valueStr);
419    
420                                    if (!firstInner) {
421                                            sb.append(',');
422                                    }
423                                    sb.append(valueStr);
424    
425                                    firstInner = false;
426                            }
427    
428                            firstOutter = false;
429                    }
430    
431                    doUnmarshalText(sb.toString(), version, warnings, CompatibilityMode.RFC);
432            }

492            /**
493             * Sorts by PREF parameter ascending. Types that do not have a PREF
494             * parameter are pushed to the end of the list.
495             */
496            public int compareTo(VCardType that) {
497                    Integer pref0 = this.getSubTypes().getPref();
498                    Integer pref1 = that.getSubTypes().getPref();
499                    if (pref0 == null && pref1 == null) {
500                            return 0;
501                    }
502                    if (pref0 == null) {
503                            return 1;
504                    }
505                    if (pref1 == null) {
506                            return -1;
507                    }
508                    return pref1.compareTo(pref0);
509            }
