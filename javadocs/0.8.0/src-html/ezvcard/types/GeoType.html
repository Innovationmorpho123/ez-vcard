045    /**
046     * A set of latitude/longitude coordinates. There is no rule for what these
047     * coordinates must represent, but the meaning could vary depending on the value
048     * of the vCard KIND type.
049     * 
050     * <ul>
051     * <li>"individual": the location of the person's home or workplace.</li>
052     * <li>"group": the location of the group's meeting place.</li>
053     * <li>"org": the coordinates of the organization's headquarters.</li>
054     * <li>"location": the coordinates of the location itself.</li>
055     * </ul>
056     * 
057     * <p>
058     * <b>Code sample</b>
059     * </p>
060     * 
061     * <pre>
062     * VCard vcard = new VCard();
063     * GeoType geo = new GeoType(-123.456, 12.54);
064     * vcard.setGeo(geo);
065     * </pre>
066     * 
067     * <p>
068     * <b>Property name:</b> <code>GEO</code>
069     * </p>
070     * <p>
071     * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
072     * </p>
073     * @author Michael Angstadt
074     */
075    public class GeoType extends VCardType implements HasAltId {
076            public static final String NAME = "GEO";
077            private GeoUri uri = new GeoUri();
078    
079            public GeoType() {
080                    this(null, null);
081            }
082    
083            /**
084             * @param latitude the latitude
085             * @param longitude the longitude
086             */
087            public GeoType(Double latitude, Double longitude) {
088                    super(NAME);
089                    setLatitude(latitude);
090                    setLongitude(longitude);
091            }
092    
093            /**
094             * Gets the latitude.
095             * @return the latitude
096             */
097            public Double getLatitude() {
098                    return uri.getCoordA();
099            }
100    
101            /**
102             * Sets the latitude.
103             * @param latitude the latitude
104             */
105            public void setLatitude(Double latitude) {
106                    uri.setCoordA(latitude);
107            }
108    
109            /**
110             * Gets the longitude.
111             * @return the longitude
112             */
113            public Double getLongitude() {
114                    return uri.getCoordB();
115            }
116    
117            /**
118             * Sets the longitude.
119             * @param longitude the longitude
120             */
121            public void setLongitude(Double longitude) {
122                    uri.setCoordB(longitude);
123            }
124    
125            /**
126             * Gets the raw object used for storing the GEO information. This can be
127             * used to supplement the GEO value with additional information (such as the
128             * altitude). Geo URIs are only supported by vCard version 4.0. Everything
129             * but latitude and longitude will be lost when marshalling to an earlier
130             * vCard version.
131             * @return the geo URI object
132             * @see <a href="http://tools.ietf.org/html/rfc5870">RFC 5870</a>
133             */
134            public GeoUri getGeoUri() {
135                    return uri;
136            }
137

140             * <p>
141             * <b>Supported versions:</b> <code>4.0</code>
142             * </p>

144             * or null if it doesn't exist
145             */
146            public String getType() {
147                    return subTypes.getType();
148            }
149

156             * null to remove

255            @Override
256            protected void doMarshalText(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
257                    sb.append(write(version));
258            }
259    
260            @Override
261            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
262                    value = VCardStringUtils.unescape(value);
263                    parse(value, version, warnings);
264            }
265    
266            @Override
267            protected void doMarshalXml(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
268                    parent.uri(write(parent.version()));
269            }
270    
271            @Override
272            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
273                    String value = element.uri();
274                    if (value != null) {
275                            parse(value, element.version(), warnings);
276                    } else {
277                            throw new SkipMeException("No URI found.");
278                    }
279            }
280    
281            @Override
282            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
283                    uri = new GeoUri();
284    
285                    String latitude = element.firstValue("latitude");
286                    if (latitude == null) {
287                            throw new SkipMeException("Latitude missing.");
288                    }
289                    try {
290                            setLatitude(Double.parseDouble(latitude));
291                    } catch (NumberFormatException e) {
292                            throw new SkipMeException("Could not parse latitude: " + latitude);
293                    }
294    
295                    String longitude = element.firstValue("longitude");
296                    if (longitude == null) {
297                            throw new SkipMeException("Longitude missing.");
298                    }
299                    try {
300                            setLongitude(Double.parseDouble(longitude));
301                    } catch (NumberFormatException e) {
302                            throw new SkipMeException("Could not parse longitude: " + longitude);
303                    }
304            }
305    
306            @Override
307            protected JCardValue doMarshalJson(VCardVersion version, List<String> warnings) {
308                    return JCardValue.uri(write(version));
309            }
310    
311            @Override
312            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
313                    parse(value.getFirstValueAsString(), version, warnings);
314            }
315    
316            private void parse(String value, VCardVersion version, List<String> warnings) {
317                    if (version == VCardVersion.V4_0) {
318                            try {
319                                    uri = new GeoUri(value);
320                            } catch (IllegalArgumentException e) {
321                                    throw new SkipMeException("Invalid geo URI: " + value);
322                            }
323                    } else {
324                            String split[] = value.split(";");
325    
326                            if (split.length != 2) {
327                                    throw new SkipMeException("Invalid value: " + value);
328                            }
329    
330                            uri = new GeoUri();
331                            String latitude = split[0];
332                            String longitude = split[1];
333    
334                            try {
335                                    setLatitude(Double.valueOf(latitude));
336                            } catch (NumberFormatException e) {
337                                    throw new SkipMeException("Could not parse latitude: " + latitude);
338                            }
339    
340                            try {
341                                    setLongitude(Double.valueOf(longitude));
342                            } catch (NumberFormatException e) {
343                                    throw new SkipMeException("Could not parse longtude: " + longitude);
344                            }
345                    }
346            }
347    
348            private String write(VCardVersion version) {
349                    if (getLatitude() == null || getLongitude() == null) {
350                            throw new SkipMeException("Latitude and/or longitude is missing.");
351                    }
352    
353                    if (version == VCardVersion.V4_0) {
354                            return uri.toString(6);
355                    } else {
356                            NumberFormat nf = GeoUri.buildNumberFormat(6);
357                            return nf.format(getLatitude()) + ';' + nf.format(getLongitude());
358                    }
359            }
