048/**
049 * Writes data to an vCard data stream.
050 * @author Michael Angstadt
051 */
052public class VCardRawWriter implements Closeable, Flushable {
053        /**
054         * Regular expression used to determine if a parameter value needs to be
055         * quoted.
056         */
057        private static final Pattern quoteMeRegex = Pattern.compile(".*?[,:;].*");
058
059        /**
060         * Regular expression used to detect newline character sequences.
061         */
062        private static final Pattern newlineRegex = Pattern.compile("\\r\\n|\\r|\\n");
063
064        /**
065         * Regular expression used to determine if a property name contains any
066         * invalid characters.
067         */
068        private static final Pattern propertyNameRegex = Pattern.compile("(?i)[-a-z0-9]+");
069
070        /**
071         * The characters that are not valid in parameter values and that should be
072         * removed.
073         */
074        private static final Map<VCardVersion, BitSet> invalidParamValueChars;
075        static {
076                BitSet controlChars = new BitSet(128);
077                controlChars.set(0, 31);
078                controlChars.set(127);
079                controlChars.set('\t', false); //allow
080                controlChars.set('\n', false); //allow
081                controlChars.set('\r', false); //allow
082
083                Map<VCardVersion, BitSet> map = new HashMap<VCardVersion, BitSet>();
084
085                //2.1
086                {
087                        BitSet bitSet = new BitSet(128);
088                        bitSet.or(controlChars);
089
090                        bitSet.set(',');
091                        bitSet.set('.');
092                        bitSet.set(':');
093                        bitSet.set('=');
094                        bitSet.set('[');
095                        bitSet.set(']');
096
097                        map.put(VCardVersion.V2_1, bitSet);
098                }
099
100                //3.0, 4.0
101                {
102                        BitSet bitSet = new BitSet(128);
103                        bitSet.or(controlChars);
104
105                        map.put(VCardVersion.V3_0, bitSet);
106                        map.put(VCardVersion.V4_0, bitSet);
107                }
108
109                invalidParamValueChars = Collections.unmodifiableMap(map);
110        }

316        /**
317         * Writes a property to the vCard data stream.
318         * @param group the group or null if there is no group
319         * @param propertyName the property name (e.g. "FN")
320         * @param parameters the property parameters
321         * @param value the property value (will be converted to "quoted-printable"
322         * encoding if the {@link Encoding#QUOTED_PRINTABLE} parameter is set)
323         * @throws IllegalArgumentException if the group or property name contains
324         * invalid characters
325         * @throws IOException if there's an I/O problem
326         */
327        public void writeProperty(String group, String propertyName, VCardParameters parameters, String value) throws IOException {
328                //validate the group name
329                if (group != null && !propertyNameRegex.matcher(group).matches()) {
330                        throw new IllegalArgumentException("Group contains invalid characters.  Valid characters are letters, numbers, and hyphens: " + group);
331                }
332
333                //validate the property name
334                if (!propertyNameRegex.matcher(propertyName).matches()) {
335                        throw new IllegalArgumentException("Property name contains invalid characters.  Valid characters are letters, numbers, and hyphens: " + propertyName);
336                }
337
338                value = sanitizeValue(parameters, value);
339
340                //determine if the property value must be encoded in quoted printable
341                //and determine the charset to use when encoding to quoted-printable
342                boolean quotedPrintable = (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE);
343                Charset charset = null;
344                if (quotedPrintable) {
345                        String charsetParam = parameters.getCharset();
346                        if (charsetParam == null) {
347                                charset = Charset.forName("UTF-8");
348                        } else {
349                                try {
350                                        charset = Charset.forName(charsetParam);
351                                } catch (Throwable t) {
352                                        charset = Charset.forName("UTF-8");
353                                }
354                        }
355                        parameters.setCharset(charset.name());
356                }
357
358                //write the group
359                if (group != null) {
360                        writer.append(group);
361                        writer.append('.');
362                }
363
364                //write the property name
365                writer.append(propertyName);
366
367                //write the parameters
368                for (Map.Entry<String, List<String>> parameter : parameters) {
369                        String parameterName = parameter.getKey();
370                        List<String> parameterValues = parameter.getValue();
371                        if (parameterValues.isEmpty()) {
372                                continue;
373                        }
374
375                        if (version == VCardVersion.V2_1) {
376                                boolean isTypeParameter = VCardParameters.TYPE.equalsIgnoreCase(parameterName);
377                                for (String parameterValue : parameterValues) {
378                                        parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
379
380                                        if (isTypeParameter) {
381                                                //e.g. ADR;HOME;WORK:
382                                                writer.append(';').append(parameterValue.toUpperCase());
383                                        } else {
384                                                //e.g. ADR;FOO=bar;FOO=car:
385                                                writer.append(';').append(parameterName).append('=').append(parameterValue);
386                                        }
387                                }
388                                continue;
389                        }
390
391                        //e.g. ADR;TYPE=home,work,"another,value":
392                        boolean first = true;
393                        writer.append(';').append(parameterName).append('=');
394                        for (String parameterValue : parameterValues) {
395                                if (!first) {
396                                        writer.append(',');
397                                }
398
399                                parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
400
401                                //surround with double quotes if contains special chars
402                                if (quoteMeRegex.matcher(parameterValue).matches()) {
403                                        writer.append('"');
404                                        writer.append(parameterValue);
405                                        writer.append('"');
406                                } else {
407                                        writer.append(parameterValue);
408                                }
409
410                                first = false;
411                        }
412                }
413
414                writer.append(':');
415                writer.append(value, quotedPrintable, charset);
416                writer.append(writer.getNewline());
417        }

419        /**
420         * Sanitizes a property value for safe inclusion in a vCard.
421         * @param parameters the parameters
422         * @param value the value to sanitize
423         * @return the sanitized value
424         */
425        private String sanitizeValue(VCardParameters parameters, String value) {
426                if (value == null) {
427                        return "";
428                }
429
430                if (version == VCardVersion.V2_1 && containsNewlines(value)) {
431                        //2.1 does not support the "\n" escape sequence (see "Delimiters" sub-section in section 2 of the specs)
432                        parameters.setEncoding(Encoding.QUOTED_PRINTABLE);
433                        return value;
434                }
435
436                return escapeNewlines(value);
437        }

439        /**
440         * Removes or escapes all invalid characters in a parameter value.
441         * @param parameterValue the parameter value
442         * @param parameterName the parameter name
443         * @param propertyName the name of the property to which the parameter
444         * belongs
445         * @return the sanitized parameter value
446         */
447        private String sanitizeParameterValue(String parameterValue, String parameterName, String propertyName) {
448                String modifiedValue = null;
449                boolean valueChanged = false;
450
451                //Note: String reference comparisons ("==") are used because the Pattern class returns the same instance if the String wasn't changed
452
453                switch (version) {
454                case V2_1:
455                        //remove invalid characters
456                        modifiedValue = removeInvalidParameterValueChars(parameterValue);
457
458                        //replace newlines with spaces
459                        modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll(" ");
460
461                        //check to see if value was changed
462                        valueChanged = (parameterValue != modifiedValue);
463
464                        //escape backslashes
465                        modifiedValue = modifiedValue.replace("\\", "\\\\");
466
467                        //escape semi-colons (see section 2)
468                        modifiedValue = modifiedValue.replace(";", "\\;");
469
470                        break;
471
472                case V3_0:
473                        //remove invalid characters
474                        modifiedValue = removeInvalidParameterValueChars(parameterValue);
475
476                        if (caretEncodingEnabled) {
477                                valueChanged = (modifiedValue != parameterValue);
478
479                                //apply caret encoding
480                                modifiedValue = applyCaretEncoding(modifiedValue);
481                        } else {
482                                //replace double quotes with single quotes
483                                modifiedValue = modifiedValue.replace('"', '\'');
484
485                                //replace newlines with spaces
486                                modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll(" ");
487
488                                valueChanged = (modifiedValue != parameterValue);
489                        }
490
491                        break;
492
493                case V4_0:
494                        //remove invalid characters
495                        modifiedValue = removeInvalidParameterValueChars(parameterValue);
496
497                        if (caretEncodingEnabled) {
498                                valueChanged = (modifiedValue != parameterValue);
499
500                                //apply caret encoding
501                                modifiedValue = applyCaretEncoding(modifiedValue);
502                        } else {
503                                //replace double quotes with single quotes
504                                modifiedValue = modifiedValue.replace('"', '\'');
505
506                                valueChanged = (modifiedValue != parameterValue);
507
508                                //backslash-escape newlines (for the "LABEL" parameter)
509                                modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll("\\\\\\n");
510                        }
511
512                        break;
513                }
514
515                if (valueChanged && problemsListener != null) {
516                        problemsListener.onParameterValueChanged(propertyName, parameterName, parameterValue, modifiedValue);
517                }
518
519                return modifiedValue;
520        }

522        /**
523         * Removes invalid characters from a parameter value.
524         * @param value the parameter value
525         * @return the sanitized parameter value
526         */
527        private String removeInvalidParameterValueChars(String value) {
528                BitSet invalidChars = invalidParamValueChars.get(version);
529                StringBuilder sb = null;
530
531                for (int i = 0; i < value.length(); i++) {
532                        char ch = value.charAt(i);
533                        if (invalidChars.get(ch)) {
534                                if (sb == null) {
535                                        sb = new StringBuilder(value.length());
536                                        sb.append(value.substring(0, i));
537                                }
538                                continue;
539                        }
540
541                        if (sb != null) {
542                                sb.append(ch);
543                        }
544                }
545
546                return (sb == null) ? value : sb.toString();
547        }

549        /**
550         * Applies circumflex accent encoding to a string.
551         * @param value the string
552         * @return the encoded string
553         */
554        private String applyCaretEncoding(String value) {
555                value = value.replace("^", "^^");
556                value = newlineRegex.matcher(value).replaceAll("^n");
557                value = value.replace("\"", "^'");
558                return value;
559        }

561        /**
562         * <p>
563         * Escapes all newline character sequences. The newline character sequences
564         * are:
565         * </p>
566         * <ul>
567         * <li>{@code \r\n}</li>
568         * <li>{@code \r}</li>
569         * <li>{@code \n}</li>
570         * </ul>
571         * @param text the text to escape
572         * @return the escaped text
573         */
574        private String escapeNewlines(String text) {
575                return newlineRegex.matcher(text).replaceAll("\\\\n");
576        }
577
578        /**
579         * <p>
580         * Determines if a string has at least one newline character sequence. The
581         * newline character sequences are:
582         * </p>
583         * <ul>
584         * <li>{@code \r\n}</li>
585         * <li>{@code \r}</li>
586         * <li>{@code \n}</li>
587         * </ul>
588         * @param text the text to escape
589         * @return the escaped text
590         */
591        private boolean containsNewlines(String text) {
592                return newlineRegex.matcher(text).find();
593        }

611        /**
612         * A listener whose methods are invoked when non-critical issues occur
613         * during the writing process.
614         * @author Michael Angstadt
615         */
616        public static interface ProblemsListener {
617                /**
618                 * Called when a parameter value is changed in a lossy way, due to it
619                 * containing invalid characters. If a character can be escaped (such as
620                 * the "^" character when caret encoding is enabled), then this does not
621                 * count as the parameter being modified because it can be decoded
622                 * without losing any information.
623                 * @param propertyName the name of the property to which the parameter
624                 * belongs
625                 * @param parameterName the parameter name
626                 * @param originalValue the original parameter value
627                 * @param modifiedValue the modified parameter value
628                 */
629                void onParameterValueChanged(String propertyName, String parameterName, String originalValue, String modifiedValue);
630        }
