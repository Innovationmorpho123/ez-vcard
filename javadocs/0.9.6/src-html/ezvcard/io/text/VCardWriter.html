056/**
057 * <p>
058 * Writes {@link VCard} objects to a plain-text vCard data stream.
059 * </p>
060 * <p>
061 * <b>Example:</b>
062 * 
063 * <pre class="brush:java">
064 * VCard vcard1 = ...
065 * VCard vcard2 = ...
066 * 
067 * File file = new File("vcard.vcf");
068 * VCardWriter vcardWriter = new VCardWriter(file, VCardVersion.V3_0);
069 * vcardWriter.write(vcard1);
070 * vcardWriter.write(vcard2);
071 * vcardWriter.close();
072 * </pre>
073 * 
074 * </p>
075 * 
076 * <p>
077 * <b>Changing the line folding settings:</b>
078 * 
079 * <pre class="brush:java">
080 * VCardWriter writer = new VCardWriter(...);
081 * 
082 * //disable line folding
083 * writer.getRawWriter().getFoldedLineWriter().setLineLength(null);
084 * 
085 * //change line length
086 * writer.getRawWriter().getFoldedLineWriter().setLineLength(50);
087 * 
088 * //change folded line indent string
089 * writer.getRawWriter().getFoldedLineWriter().setIndent("\t");
090 * 
091 * //change newline character
092 * writer.getRawWriter().getFoldedLineWriter().setNewline("**");
093 * </pre>
094 * 
095 * </p>
096 * @author Michael Angstadt
097 */
098public class VCardWriter extends StreamWriter implements Flushable {
099        private final VCardRawWriter writer;
100        private final LinkedList<Boolean> prodIdStack = new LinkedList<Boolean>();
101
102        /**
103         * Creates a vCard writer.
104         * @param out the output stream to write the vCard to
105         * @param targetVersion the version that the vCards should conform to (if
106         * set to "4.0", vCards will be written in UTF-8 encoding)
107         */
108        public VCardWriter(OutputStream out, VCardVersion targetVersion) {
109                this((targetVersion == VCardVersion.V4_0) ? utf8Writer(out) : new OutputStreamWriter(out), targetVersion);
110        }
111
112        /**
113         * Creates a vCard writer.
114         * @param file the file to write the vCard to
115         * @param targetVersion the version that the vCards should conform to (if
116         * set to "4.0", vCards will be written in UTF-8 encoding)
117         * @throws IOException if there's a problem opening the file
118         */
119        public VCardWriter(File file, VCardVersion targetVersion) throws IOException {
120                this(file, false, targetVersion);
121        }
122
123        /**
124         * Creates a vCard writer.
125         * @param file the file to write the vCard to
126         * @param append true to append to the end of the file, false to overwrite
127         * it
128         * @param targetVersion the version that the vCards should conform to (if
129         * set to "4.0", vCards will be written in UTF-8 encoding)
130         * @throws IOException if there's a problem opening the file
131         */
132        public VCardWriter(File file, boolean append, VCardVersion targetVersion) throws IOException {
133                this((targetVersion == VCardVersion.V4_0) ? utf8Writer(file, append) : new FileWriter(file, append), targetVersion);
134        }
135
136        /**
137         * Creates a vCard writer.
138         * @param writer the writer to write the vCard to
139         * @param targetVersion the version that the vCards should conform to
140         */
141        public VCardWriter(Writer writer, VCardVersion targetVersion) {
142                this.writer = new VCardRawWriter(writer, targetVersion);
143        }
144
145        /**
146         * Gets the writer that this object wraps.
147         * @return the raw writer
148         */
149        public VCardRawWriter getRawWriter() {
150                return writer;
151        }
152
153        /**
154         * Gets the version that the vCards should adhere to.
155         * @return the vCard version
156         */
157        @Override
158        public VCardVersion getTargetVersion() {
159                return writer.getVersion();
160        }
161
162        /**
163         * Sets the version that the vCards should adhere to.
164         * @param targetVersion the vCard version
165         */
166        public void setTargetVersion(VCardVersion targetVersion) {
167                writer.setVersion(targetVersion);
168        }
169
170        /**
171         * <p>
172         * Gets whether the writer will apply circumflex accent encoding on
173         * parameter values (disabled by default, only applies to 3.0 and 4.0
174         * vCards). This escaping mechanism allows for newlines and double quotes to
175         * be included in parameter values.
176         * </p>
177         * 
178         * <p>
179         * When disabled, the writer will replace newlines with spaces and double
180         * quotes with single quotes.
181         * </p>
182         * @return true if circumflex accent encoding is enabled, false if not
183         * @see VCardRawWriter#isCaretEncodingEnabled()
184         */
185        public boolean isCaretEncodingEnabled() {
186                return writer.isCaretEncodingEnabled();
187        }
188
189        /**
190         * <p>
191         * Sets whether the writer will apply circumflex accent encoding on
192         * parameter values (disabled by default, only applies to 3.0 and 4.0
193         * vCards). This escaping mechanism allows for newlines and double quotes to
194         * be included in parameter values.
195         * </p>
196         * 
197         * <p>
198         * When disabled, the writer will replace newlines with spaces and double
199         * quotes with single quotes.
200         * </p>
201         * @param enable true to use circumflex accent encoding, false not to
202         * @see VCardRawWriter#setCaretEncodingEnabled(boolean)
203         */
204        public void setCaretEncodingEnabled(boolean enable) {
205                writer.setCaretEncodingEnabled(enable);
206        }
207
208        @Override
209        @SuppressWarnings({ "rawtypes", "unchecked" })
210        protected void _write(VCard vcard, List<VCardProperty> propertiesToAdd) throws IOException {
211                VCardVersion targetVersion = getTargetVersion();
212                writer.writeBeginComponent("VCARD");
213                writer.writeVersion();
214
215                for (VCardProperty property : propertiesToAdd) {
216                        VCardPropertyScribe scribe = index.getPropertyScribe(property);
217
218                        //marshal the value
219                        String value = null;
220                        VCard nestedVCard = null;
221                        try {
222                                value = scribe.writeText(property, targetVersion);
223                        } catch (SkipMeException e) {
224                                continue;
225                        } catch (EmbeddedVCardException e) {
226                                nestedVCard = e.getVCard();
227                        }
228
229                        //marshal the parameters
230                        VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
231
232                        //is the value a nested vCard?
233                        if (nestedVCard != null) {
234                                if (targetVersion == VCardVersion.V2_1) {
235                                        //write a nested vCard (2.1 style)
236                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
237                                        prodIdStack.add(addProdId);
238                                        addProdId = false;
239                                        write(nestedVCard);
240                                        addProdId = prodIdStack.removeLast();
241                                } else {
242                                        //write an embedded vCard (3.0 style)
243                                        StringWriter sw = new StringWriter();
244                                        VCardWriter agentWriter = new VCardWriter(sw, targetVersion);
245                                        agentWriter.getRawWriter().getFoldedLineWriter().setLineLength(null);
246                                        agentWriter.getRawWriter().getFoldedLineWriter().setNewline("\n");
247                                        agentWriter.setAddProdId(false);
248                                        agentWriter.setVersionStrict(versionStrict);
249                                        try {
250                                                agentWriter.write(nestedVCard);
251                                        } catch (IOException e) {
252                                                //writing to a string
253                                        } finally {
254                                                IOUtils.closeQuietly(agentWriter);
255                                        }
256
257                                        String vCardStr = sw.toString();
258                                        vCardStr = VCardPropertyScribe.escape(vCardStr);
259                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, vCardStr);
260                                }
261                                continue;
262                        }
263
264                        if (value != null) {
265                                //set the data type
266                                //only add a VALUE parameter if the data type is (1) not "unknown" and (2) different from the property's default data type
267                                VCardDataType dataType = scribe.dataType(property, targetVersion);
268                                if (dataType != null) {
269                                        VCardDataType defaultDataType = scribe.defaultDataType(targetVersion);
270                                        if (dataType != defaultDataType) {
271                                                if (defaultDataType == VCardDataType.DATE_AND_OR_TIME && (dataType == VCardDataType.DATE || dataType == VCardDataType.DATE_TIME || dataType == VCardDataType.TIME)) {
272                                                        //do not write VALUE if the default data type is "date-and-or-time" and the property's data type is time-based
273                                                } else {
274                                                        parameters.setValue(dataType);
275                                                }
276                                        }
277                                }
278
279                                writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
280                                continue;
281                        }
282                }
283
284                writer.writeEndComponent("VCARD");
285        }
286
287        /**
288         * Flushes the underlying {@link Writer} object.
289         * @throws IOException if there's a problem flushing the writer
290         */
291        public void flush() throws IOException {
292                writer.flush();
293        }
294
295        /**
296         * Closes the underlying {@link Writer} object.
297         * @throws IOException if there's a problem closing the writer
298         */
299        public void close() throws IOException {
300                writer.close();
301        }
