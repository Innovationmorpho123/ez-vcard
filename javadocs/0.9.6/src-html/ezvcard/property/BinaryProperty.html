044/**
045 * Represents a property whose value contains binary data.
046 * @author Michael Angstadt
047 * @param <T> the class used for representing the content type of the resource
048 */
049public abstract class BinaryProperty<T extends MediaTypeParameter> extends VCardProperty implements HasAltId {
050        /**
051         * The decoded data.
052         */
053        protected byte[] data;
054
055        /**
056         * The URL to the resource.
057         */
058        protected String url;
059
060        /**
061         * The content type of the resource (for example, a JPEG image).
062         */
063        protected T contentType;
064
065        public BinaryProperty() {
066                //empty
067        }
068
069        /**
070         * Creates a binary property.
071         * @param url the URL to the resource
072         * @param type the content type
073         */
074        public BinaryProperty(String url, T type) {
075                setUrl(url, type);
076        }
077
078        /**
079         * Creates a binary property.
080         * @param data the binary data
081         * @param type the content type
082         */
083        public BinaryProperty(byte[] data, T type) {
084                setData(data, type);
085        }
086
087        /**
088         * Creates a binary property.
089         * @param in an input stream to the binary data (will be closed)
090         * @param type the content type
091         * @throws IOException if there is a problem reading from the input stream
092         */
093        public BinaryProperty(InputStream in, T type) throws IOException {
094                this(IOUtils.toByteArray(in, true), type);
095        }
096
097        /**
098         * Creates a binary property.
099         * @param file the file containing the binary data
100         * @param type the content type
101         * @throws IOException if there is a problem reading from the file
102         */
103        public BinaryProperty(File file, T type) throws IOException {
104                this(new FileInputStream(file), type);
105        }
106
107        /**
108         * Gets the binary data of the resource.
109         * @return the binary data or null if there is none
110         */
111        public byte[] getData() {
112                return data;
113        }
114
115        /**
116         * Sets the binary data of the resource.
117         * @param data the binary data
118         * @param type the content type (e.g. "JPEG image")
119         */
120        public void setData(byte[] data, T type) {
121                this.url = null;
122                this.data = data;
123                setContentType(type);
124        }
125
126        /**
127         * Gets the URL to the resource
128         * @return the URL or null if there is none
129         */
130        public String getUrl() {
131                return url;
132        }
133
134        /**
135         * Sets the URL to the resource.
136         * @param url the URL
137         * @param type the content type (e.g. "JPEG image")
138         */
139        public void setUrl(String url, T type) {
140                this.url = url;
141                this.data = null;
142                setContentType(type);
143        }
144
145        /**
146         * Gets the content type of the resource.
147         * @return the content type (e.g. "JPEG image")
148         */
149        public T getContentType() {
150                return contentType;
151        }
152
153        /**
154         * Sets the content type of the resource.
155         * @param contentType the content type (e.g. "JPEG image")
156         */
157        public void setContentType(T contentType) {
158                this.contentType = contentType;
159        }
160
161        /**
162         * Gets the vCard 4.0 TYPE parameter. This should NOT be used to get the
163         * TYPE parameter for 2.1/3.0 vCards. Use {@link #getContentType} instead.
164         * <p>
165         * <b>Supported versions:</b> {@code 4.0}
166         * </p>
167         * @return the TYPE value (typically, this will be either "work" or "home")
168         * or null if it doesn't exist
169         */
170        public String getType() {
171                return parameters.getType();
172        }
173
174        /**
175         * Sets the vCard 4.0 TYPE parameter. This should NOT be used to set the
176         * TYPE parameter for 2.1/3.0 vCards. Use {@link #setContentType} instead.
177         * <p>
178         * <b>Supported versions:</b> {@code 4.0}
179         * </p>
180         * @param type the TYPE value (should be either "work" or "home") or null to
181         * remove
182         */
183        public void setType(String type) {
184                parameters.setType(type);
185        }
186
187        @Override
188        public List<Integer[]> getPids() {
189                return super.getPids();
190        }
191
192        @Override
193        public void addPid(int localId, int clientPidMapRef) {
194                super.addPid(localId, clientPidMapRef);
195        }
196
197        @Override
198        public void removePids() {
199                super.removePids();
200        }
201
202        @Override
203        public Integer getPref() {
204                return super.getPref();
205        }
206
207        @Override
208        public void setPref(Integer pref) {
209                super.setPref(pref);
210        }
211
212        //@Override
213        public String getAltId() {
214                return parameters.getAltId();
215        }
216
217        //@Override
218        public void setAltId(String altId) {
219                parameters.setAltId(altId);
220        }
221
222        @Override
223        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
224                if (url == null && data == null) {
225                        warnings.add(new Warning(8));
226                }
227        }
228}
