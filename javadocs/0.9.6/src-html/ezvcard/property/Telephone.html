042/**
043 * <p>
044 * Defines a telephone number.
045 * </p>
046 * 
047 * <p>
048 * <b>Code sample</b>
049 * </p>
050 * 
051 * <pre class="brush:java">
052 * VCard vcard = new VCard();
053 * 
054 * //text
055 * Telephone tel = new Telephone(&quot;(123) 555-6789&quot;);
056 * tel.addType(TelephoneType.HOME);
057 * tel.setPref(2); //the second-most preferred
058 * vcard.addTelephoneNumber(tel);
059 * 
060 * //URI (vCard version 4.0 only)
061 * TelUri uri = new TelUri.Builder(&quot;+1-800-555-9876&quot;).extension(&quot;111&quot;).build();
062 * tel = new Telephone(uri);
063 * tel.addType(TelephoneType.WORK);
064 * tel.setPref(1); //the most preferred
065 * vcard.addTelephoneNumber(tel);
066 * </pre>
067 * 
068 * <p>
069 * <b>Property name:</b> {@code TEL}
070 * </p>
071 * <p>
072 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
073 * </p>
074 * @author Michael Angstadt
075 */
076public class Telephone extends VCardProperty implements HasAltId {
077        private String text;
078        private TelUri uri;
079
080        /**
081         * Creates a telephone property.
082         * @param text the telephone number (e.g. "(123) 555-6789")
083         */
084        public Telephone(String text) {
085                setText(text);
086        }
087
088        /**
089         * Creates a telephone property.
090         * @param uri a "tel" URI representing the telephone number (vCard 4.0 only)
091         */
092        public Telephone(TelUri uri) {
093                setUri(uri);
094        }
095
096        /**
097         * Gets the telephone number as a text value.
098         * @return the telephone number or null if the text value is not set
099         */
100        public String getText() {
101                return text;
102        }
103
104        /**
105         * Sets the telephone number as a text value.
106         * @param text the telephone number
107         */
108        public void setText(String text) {
109                this.text = text;
110                uri = null;
111        }
112
113        /**
114         * Gets a "tel" URI representing the phone number.
115         * <p>
116         * <b>Supported versions:</b> {@code 4.0}
117         * </p>
118         * @return the "tel" URI or null if it is not set
119         */
120        public TelUri getUri() {
121                return uri;
122        }
123
124        /**
125         * Sets a "tel" URI representing the phone number.
126         * <p>
127         * <b>Supported versions:</b> {@code 4.0}
128         * </p>
129         * @param uri the "tel" URI
130         */
131        public void setUri(TelUri uri) {
132                text = null;
133                this.uri = uri;
134        }
135
136        /**
137         * Gets all the TYPE parameters.
138         * @return the TYPE parameters or empty set if there are none
139         */
140        public Set<TelephoneType> getTypes() {
141                Set<String> values = parameters.getTypes();
142                Set<TelephoneType> types = new HashSet<TelephoneType>(values.size());
143                for (String value : values) {
144                        types.add(TelephoneType.get(value));
145                }
146                return types;
147        }
148
149        /**
150         * Adds a TYPE parameter.
151         * @param type the TYPE parameter to add
152         */
153        public void addType(TelephoneType type) {
154                parameters.addType(type.getValue());
155        }
156
157        /**
158         * Removes a TYPE parameter.
159         * @param type the TYPE parameter to remove
160         */
161        public void removeType(TelephoneType type) {
162                parameters.removeType(type.getValue());
163        }
164
165        @Override
166        public List<Integer[]> getPids() {
167                return super.getPids();
168        }
169
170        @Override
171        public void addPid(int localId, int clientPidMapRef) {
172                super.addPid(localId, clientPidMapRef);
173        }
174
175        @Override
176        public void removePids() {
177                super.removePids();
178        }
179
180        @Override
181        public Integer getPref() {
182                return super.getPref();
183        }
184
185        @Override
186        public void setPref(Integer pref) {
187                super.setPref(pref);
188        }
189
190        //@Override
191        public String getAltId() {
192                return parameters.getAltId();
193        }
194
195        //@Override
196        public void setAltId(String altId) {
197                parameters.setAltId(altId);
198        }
199
200        @Override
201        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
202                if (uri == null && text == null) {
203                        warnings.add(new Warning(8));
204                }
205
206                if (uri != null && (version == VCardVersion.V2_1 || version == VCardVersion.V3_0)) {
207                        warnings.add(new Warning(19));
208                }
209
210                for (TelephoneType type : getTypes()) {
211                        if (type == TelephoneType.PREF) {
212                                //ignore because it is converted to a PREF parameter for 4.0 vCards
213                                continue;
214                        }
215
216                        if (!type.isSupported(version)) {
217                                warnings.add(new Warning(9, type.getValue()));
218                        }
219                }
220        }
221}
