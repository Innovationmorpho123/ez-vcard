042/**
043 * Represents a property with a date and/or time value.
044 * @author Michael Angstadt
045 */
046public class DateOrTimeProperty extends VCardProperty implements HasAltId {
047        private String text;
048        private Date date;
049        private PartialDate partialDate;
050        private boolean dateHasTime;
051
052        /**
053         * Creates a date-and-or-time property.
054         * @param date the date value
055         */
056        public DateOrTimeProperty(Date date) {
057                this(date, false);
058        }
059
060        /**
061         * Creates a date-and-or-time property.
062         * @param date the date value
063         * @param hasTime true to include the date's time component, false if it's
064         * strictly a date
065         */
066        public DateOrTimeProperty(Date date, boolean hasTime) {
067                setDate(date, hasTime);
068        }
069
070        /**
071         * Creates a date-and-or-time property.
072         * @param partialDate the partial date value (vCard 4.0 only)
073         */
074        public DateOrTimeProperty(PartialDate partialDate) {
075                setPartialDate(partialDate);
076        }
077
078        /**
079         * Creates a date-and-or-time property.
080         * @param text the text value (vCard 4.0 only)
081         */
082        public DateOrTimeProperty(String text) {
083                setText(text);
084        }
085
086        /**
087         * Gets the date value.
088         * @return the date value or null if not set
089         */
090        public Date getDate() {
091                return date;
092        }
093
094        /**
095         * Sets the value of this property to a complete date.
096         * @param date the date
097         * @param hasTime true to include the date's time component, false if it's
098         * strictly a date
099         */
100        public void setDate(Date date, boolean hasTime) {
101                this.date = date;
102                this.dateHasTime = (date == null) ? false : hasTime;
103                text = null;
104                partialDate = null;
105        }
106
107        /**
108         * Gets the reduced accuracy or truncated date. This is only supported by
109         * vCard 4.0.
110         * @return the reduced accuracy or truncated date or null if not set
111         * @see "<a href="
112         * http://tools.ietf.org/html/rfc6350">RFC 6350</a> p.12-14 for examples"
113         */
114        public PartialDate getPartialDate() {
115                return partialDate;
116        }
117
118        /**
119         * <p>
120         * Sets the value of this property to a reduced accuracy or truncated date.
121         * This is only supported by vCard 4.0.
122         * </p>
123         * 
124         * <pre class="brush:java">
125         * Birthday bday = new Birthday();
126         * bday.setPartialDate(PartialDate.date(null, 4, 20)); //April 20
127         * </pre>
128         * @param partialDate the reduced accuracy or truncated date
129         * @see "<a href="
130         * http://tools.ietf.org/html/rfc6350">RFC 6350</a> p.12-14 for examples"
131         */
132        public void setPartialDate(PartialDate partialDate) {
133                this.partialDate = partialDate;
134                dateHasTime = (partialDate == null) ? false : partialDate.hasTimeComponent();
135                text = null;
136                date = null;
137        }
138
139        /**
140         * Gets the text value of this type. This is only supported by vCard 4.0.
141         * @return the text value or null if not set
142         */
143        public String getText() {
144                return text;
145        }
146
147        /**
148         * Sets the value of this property to a text string. This is only supported
149         * by vCard 4.0.
150         * @param text the text value
151         */
152        public void setText(String text) {
153                this.text = text;
154                date = null;
155                partialDate = null;
156                dateHasTime = false;
157        }
158
159        /**
160         * Determines whether the "date" or "partialDate" fields have a time
161         * component.
162         * @return true if the date has a time component, false if it's strictly a
163         * date, and false if a text value is defined
164         */
165        public boolean hasTime() {
166                return dateHasTime;
167        }
168
169        /**
170         * <p>
171         * Gets the type of calendar that is used for a date or date-time property
172         * value.
173         * </p>
174         * <p>
175         * <b>Supported versions:</b> {@code 4.0}
176         * </p>
177         * @return the type of calendar or null if not found
178         * @see VCardParameters#getCalscale
179         */
180        public Calscale getCalscale() {
181                return parameters.getCalscale();
182        }
183
184        /**
185         * <p>
186         * Sets the type of calendar that is used for a date or date-time property
187         * value.
188         * </p>
189         * <p>
190         * <b>Supported versions:</b> {@code 4.0}
191         * </p>
192         * @param calscale the type of calendar or null to remove
193         * @see VCardParameters#setCalscale
194         */
195        public void setCalscale(Calscale calscale) {
196                parameters.setCalscale(calscale);
197        }
198
199        //@Override
200        public String getAltId() {
201                return parameters.getAltId();
202        }
203
204        //@Override
205        public void setAltId(String altId) {
206                parameters.setAltId(altId);
207        }
208
209        @Override
210        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
211                if (date == null && partialDate == null && text == null) {
212                        warnings.add(new Warning(8));
213                }
214
215                if (version == VCardVersion.V2_1 || version == VCardVersion.V3_0) {
216                        if (text != null) {
217                                warnings.add(new Warning(11));
218                        }
219                        if (partialDate != null) {
220                                warnings.add(new Warning(12));
221                        }
222                }
223        }
224}
