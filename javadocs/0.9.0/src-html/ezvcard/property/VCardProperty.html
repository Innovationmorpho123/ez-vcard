042    /**
043     * Represents a vCard key/value pair entry (called a "type" or "property").
044     * @author Michael Angstadt
045     */
046    public abstract class VCardProperty implements Comparable<VCardProperty> {
047            /**
048             * The group that this property belongs to or null if it doesn't belong to a
049             * group.
050             */
051            protected String group;
052    
053            /**
054             * The property's parameters.
055             */
056            protected VCardParameters parameters = new VCardParameters();
057    
058            /**
059             * Gets the vCard versions that support this property.
060             * @return the vCard versions that support this property.
061             */
062            public final Set<VCardVersion> getSupportedVersions() {
063                    return _supportedVersions();
064            }
065    
066            /**
067             * <p>
068             * Gets the vCard versions that support this property.
069             * </p>
070             * <p>
071             * This method should be overridden by child classes if the property does
072             * not support all vCard versions. The default implementation of this method
073             * returns all vCard versions.
074             * </p>
075             * @return the vCard versions that support this property.
076             */
077            protected Set<VCardVersion> _supportedVersions() {
078                    return EnumSet.copyOf(Arrays.asList(VCardVersion.values()));
079            }
080    
081            /**
082             * Checks the property for data consistency problems or deviations from the
083             * spec. These problems will not prevent the property from being written to
084             * a data stream, but may prevent it from being parsed correctly by the
085             * consuming application. These problems can largely be avoided by reading
086             * the Javadocs of the property class, or by being familiar with the vCard
087             * standard.
088             * @param version the version to check the property against (use 4.0 for
089             * xCard and jCard)
090             * @param vcard the vCard this property belongs to
091             * @see VCard#validate
092             * @return a list of warnings or an empty list if no problems were found
093             */
094            public final List<String> validate(VCardVersion version, VCard vcard) {
095                    List<String> warnings = new ArrayList<String>(0);
096    
097                    //check the supported versions
098                    Set<VCardVersion> supportedVersions = getSupportedVersions();
099                    if (!supportedVersions.contains(version)) {
100                            warnings.add("Property is not supported by version " + version.getVersion() + ".  Supported versions are: " + supportedVersions);
101                    }
102    
103                    //check parameters
104                    warnings.addAll(parameters.validate(version));
105    
106                    _validate(warnings, version, vcard);
107    
108                    return warnings;
109            }
110    
111            /**
112             * Checks the property for data consistency problems or deviations from the
113             * spec. Meant to be overridden by child classes that wish to provide
114             * validation logic.
115             * @param warnings the list to add the warnings to
116             * @param version the version to check the property against
117             * @param vcard the vCard this property belongs to
118             */
119            protected void _validate(List<String> warnings, VCardVersion version, VCard vcard) {
120                    //empty
121            }
122    
123            /**
124             * Gets all of the property's parameters.
125             * @return the property's parameters
126             */
127            public VCardParameters getParameters() {
128                    return parameters;
129            }
130    
131            /**
132             * Sets the property's parameters.
133             * @param parameters the parameters
134             */
135            public void setParameters(VCardParameters parameters) {
136                    this.parameters = parameters;
137            }
138    
139            /**
140             * Gets the first value of a parameter.
141             * @param name the parameter name (case insensitive, e.g. "LANGUAGE")
142             * @return the parameter value or null if not found
143             */
144            public String getParameter(String name) {
145                    return parameters.first(name);
146            }
147    
148            /**
149             * Gets all values of a parameter.
150             * @param name the parameter name (case insensitive, e.g. "LANGUAGE")
151             * @return the parameter values
152             */
153            public List<String> getParameters(String name) {
154                    return parameters.get(name);
155            }
156    
157            /**
158             * Replaces all existing values of a parameter with the given value.
159             * @param name the parameter name (case insensitive, e.g. "LANGUAGE")
160             * @param value the parameter value
161             */
162            public void setParameter(String name, String value) {
163                    parameters.replace(name, value);
164            }
165    
166            /**
167             * Adds a value to a parameter.
168             * @param name the parameter name (case insensitive, e.g. "LANGUAGE")
169             * @param value the parameter value
170             */
171            public void addParameter(String name, String value) {
172                    parameters.put(name, value);
173            }
174    
175            /**
176             * Removes a parameter from the property.
177             * @param name the parameter name (case insensitive, e.g. "LANGUAGE")
178             */
179            public void removeParameter(String name) {
180                    parameters.removeAll(name);
181            }
182    
183            /**
184             * Gets this property's group.
185             * @return the group or null if it does not belong to a group
186             */
187            public String getGroup() {
188                    return group;
189            }
190    
191            /**
192             * Sets this property's group.
193             * @param group the group or null to remove the property's group
194             */
195            public void setGroup(String group) {
196                    this.group = group;
197            }
198    
199            /**
200             * Sorts by PREF parameter ascending. Properties that do not have a PREF
201             * parameter are pushed to the end of the list.
202             */
203            public int compareTo(VCardProperty that) {
204                    Integer pref0 = this.getParameters().getPref();
205                    Integer pref1 = that.getParameters().getPref();
206                    if (pref0 == null && pref1 == null) {
207                            return 0;
208                    }
209                    if (pref0 == null) {
210                            return 1;
211                    }
212                    if (pref1 == null) {
213                            return -1;
214                    }
215                    return pref1.compareTo(pref0);
216            }
217    
218            //Note: The following parameter helper methods are package-scoped to prevent them from cluttering up the Javadocs
219    
220            /**
221             * <p>
222             * Gets all PID values.
223             * </p>
224             * <p>
225             * <b>Supported versions:</b> {@code 4.0}
226             * </p>
227             * @return the PID values or empty set if there are none
228             * @see VCardParameters#getPids
229             */
230            List<Integer[]> getPids() {
231                    return parameters.getPids();
232            }
233    
234            /**
235             * <p>
236             * Adds a PID value.
237             * </p>
238             * <p>
239             * <b>Supported versions:</b> {@code 4.0}
240             * </p>
241             * @param localId the local ID
242             * @param clientPidMapRef the ID used to reference the property's globally
243             * unique identifier in the CLIENTPIDMAP property.
244             * @see VCardParameters#addPid(int, int)
245             */
246            void addPid(int localId, int clientPidMapRef) {
247                    parameters.addPid(localId, clientPidMapRef);
248            }
249    
250            /**
251             * <p>
252             * Removes all PID values.
253             * </p>
254             * <p>
255             * <b>Supported versions:</b> {@code 4.0}
256             * </p>
257             * @see VCardParameters#removePids
258             */
259            void removePids() {
260                    parameters.removePids();
261            }
262    
263            /**
264             * <p>
265             * Gets the preference value. The lower the number, the more preferred this
266             * property instance is compared with other properties in the same vCard of
267             * the same type. If a property doesn't have a preference value, then it is
268             * considered the least preferred.
269             * </p>
270             * <p>
271             * <b>Supported versions:</b> {@code 4.0}
272             * </p>
273             * @return the preference value or null if it doesn't exist
274             * @see VCardParameters#getPref
275             */
276            Integer getPref() {
277                    return parameters.getPref();
278            }
279    
280            /**
281             * <p>
282             * Sets the preference value. The lower the number, the more preferred this
283             * property instance is compared with other properties in the same vCard of
284             * the same type. If a property doesn't have a preference value, then it is
285             * considered the least preferred.
286             * </p>
287             * <p>
288             * <b>Supported versions:</b> {@code 4.0}
289             * </p>
290             * @param pref the preference value or null to remove
291             * @see VCardParameters#setPref
292             */
293            void setPref(Integer pref) {
294                    parameters.setPref(pref);
295            }
296    
297            /**
298             * Gets the language that the property value is written in.
299             * @return the language or null if not set
300             * @see VCardParameters#getLanguage
301             */
302            String getLanguage() {
303                    return parameters.getLanguage();
304            }
305    
306            /**
307             * Sets the language that the property value is written in.
308             * @param language the language or null to remove
309             * @see VCardParameters#setLanguage
310             */
311            void setLanguage(String language) {
312                    parameters.setLanguage(language);
313            }
314    
315            /**
316             * Gets the sorted position of this property when it is grouped together
317             * with other properties of the same type. Properties with low index values
318             * are put at the beginning of the sorted list and properties with high
319             * index values are put at the end of the list.
320             * @return the index or null if not set
321             * @see VCardParameters#setIndex
322             */
323            Integer getIndex() {
324                    return parameters.getIndex();
325            }
326    
327            /**
328             * Sets the sorted position of this property when it is grouped together
329             * with other properties of the same type. Properties with low index values
330             * are put at the beginning of the sorted list and properties with high
331             * index values are put at the end of the list.
332             * @param index the index or null to remove
333             * @see VCardParameters#setIndex
334             */
335            void setIndex(Integer index) {
336                    parameters.setIndex(index);
337            }
338    }
