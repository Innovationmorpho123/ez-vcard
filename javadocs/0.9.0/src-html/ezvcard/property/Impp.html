043    /**
044     * An instant message handle. The handle is represented as a URI in the format "
045     * {@code <IM-PROTOCOL>:<IM-HANDLE>}". For example, someone with a Yahoo!
046     * Messenger handle of "johndoe@yahoo.com" would have an IMPP vCard property
047     * value of "ymsgr:johndoe@yahoo.com".
048     * 
049     * <p>
050     * <b>Code sample</b>
051     * </p>
052     * 
053     * <pre class="brush:java">
054     * VCard vcard = new VCard();
055     * 
056     * //URI
057     * Impp impp = new Impp(&quot;aim:johndoe@aol.com&quot;);
058     * vcard.addImpp(impp);
059     * 
060     * //static factory methods
061     * impp = Impp.msn(&quot;janedoe@msn.com&quot;);
062     * vcard.addImpp(impp);
063     * </pre>
064     * 
065     * <p>
066     * <b>Property name:</b> {@code IMPP}
067     * </p>
068     * <p>
069     * <b>Supported versions:</b> {@code 3.0, 4.0}
070     * </p>
071     * @author Michael Angstadt
072     */
073    public class Impp extends VCardProperty implements HasAltId {
074            private static final String AIM = "aim";
075            private static final String ICQ = "icq";
076            private static final String IRC = "irc";
077            private static final String MSN = "msnim";
078            private static final String SIP = "sip";
079            private static final String SKYPE = "skype";
080            private static final String XMPP = "xmpp";
081            private static final String YAHOO = "ymsgr";
082    
083            private URI uri;
084    
085            /**
086             * Creates an IMPP property. Note that this class has static factory methods
087             * for creating IMPP properties of common IM protocols.
088             * @param uri the IM URI (e.g. "aim:johndoe@aol.com")
089             * @throws IllegalArgumentException if the URI is not a valid URI
090             */
091            public Impp(String uri) {
092                    setUri(uri);
093            }
094    
095            /**
096             * Creates an IMPP property. Note that this class has static factory methods
097             * for creating IMPP properties of common IM protocols.
098             * @param uri the IM URI (e.g. "aim:johndoe@aol.com")
099             */
100            public Impp(URI uri) {
101                    setUri(uri);
102            }
103    
104            /**
105             * Creates an IMPP property. Note that this class has static factory methods
106             * for creating IMPP properties of common IM protocols.
107             * @param protocol the IM protocol (e.g. "aim")
108             * @param handle the IM handle (e.g. "johndoe@aol.com")
109             */
110            public Impp(String protocol, String handle) {
111                    setUri(protocol, handle);
112            }

260            private boolean isProtocol(String protocol) {
261                    return uri != null && protocol.equals(uri.getScheme());
262            }
263    
264            /**
265             * Gets the IM URI.
266             * @return the IM URI
267             */
268            public URI getUri() {
269                    return uri;
270            }
271    
272            /**
273             * Sets the IM URI.
274             * @param uri the IM URI (e.g. "aim:theuser@aol.com")
275             * @throws IllegalArgumentException if the URI is not a valid URI
276             */
277            public void setUri(String uri) {
278                    setUri((uri == null) ? null : URI.create(uri));
279            }
280    
281            /**
282             * Sets the IM URI.
283             * @param uri the IM URI (e.g. "aim:theuser@aol.com")
284             */
285            public void setUri(URI uri) {
286                    this.uri = uri;
287            }
288    
289            /**
290             * Sets the IM URI.
291             * @param protocol the IM protocol (e.g. "aim")
292             * @param handle the IM handle (e.g. "theuser@aol.com")
293             */
294            public void setUri(String protocol, String handle) {
295                    try {
296                            this.uri = new URI(protocol, handle, null);
297                    } catch (URISyntaxException e) {
298                            throw new IllegalArgumentException(e);
299                    }
300            }
301    
302            /**
303             * Gets the IM protocol. Use {@link #setUri(String, String)} to set the
304             * protocol.
305             * @return the IM protocol (e.g. "aim") or null if not set
306             */
307            public String getProtocol() {
308                    if (uri == null) {
309                            return null;
310                    }
311                    return uri.getScheme();
312            }
313    
314            /**
315             * Gets the IM handle. Use {@link #setUri(String, String)} to set the
316             * handle.
317             * @return the IM handle (e.g. "johndoe@aol.com") or null if not set
318             */
319            public String getHandle() {
320                    if (uri == null) {
321                            return null;
322                    }
323                    return uri.getSchemeSpecificPart();
324            }

326            /**
327             * Gets all the TYPE parameters.
328             * @return the TYPE parameters or empty set if there are none
329             */
330            public Set<ImppType> getTypes() {
331                    Set<String> values = parameters.getTypes();
332                    Set<ImppType> types = new HashSet<ImppType>(values.size());
333                    for (String value : values) {
334                            types.add(ImppType.get(value));
335                    }
336                    return types;
337            }
