366            /**
367             * <p>
368             * Gets the preference value. The lower the number, the more preferred this
369             * property instance is compared with other properties in the same vCard of
370             * the same type. If a property doesn't have a preference value, then it is
371             * considered the least preferred.
372             * </p>
373             * 
374             * <p>
375             * In the vCard below, the address on the second row is the most preferred
376             * because it has the lowest PREF value.
377             * </p>
378             * 
379             * <pre>
380             * ADR;TYPE=work;PREF=2:
381             * ADR;TYPE=work;PREF=1:
382             * ADR;TYPE=home:
383             * </pre>
384             * 
385             * <p>
386             * Preference values must be numeric and must be between 1 and 100.
387             * </p>
388             * 
389             * <p>
390             * <b>Supported versions:</b> {@code 4.0}
391             * </p>
392             * @throws IllegalStateException if the parameter value is malformed and
393             * cannot be parsed
394             * @return the preference value or null if it doesn't exist or null if it
395             * couldn't be parsed into a number
396             */
397            public Integer getPref() {
398                    String pref = first(PREF);
399                    if (pref == null) {
400                            return null;
401                    }
402    
403                    try {
404                            return Integer.valueOf(pref);
405                    } catch (NumberFormatException e) {
406                            throw new IllegalStateException(PREF + " parameter value is malformed and could not be parsed. Retrieve its raw text value instead.", e);
407                    }
408            }

411             * <p>
412             * Sets the preference value. The lower the number, the more preferred this
413             * property instance is compared with other properties in the same vCard of
414             * the same type. If a property doesn't have a preference value, then it is
415             * considered the least preferred.
416             * </p>
417             * 
418             * <p>
419             * In the vCard below, the address on the second row is the most preferred
420             * because it has the lowest PREF value.
421             * </p>
422             * 
423             * <pre>
424             * ADR;TYPE=work;PREF=2:
425             * ADR;TYPE=work;PREF=1:
426             * ADR;TYPE=home:
427             * </pre>
428             * 
429             * <p>
430             * Preference values must be numeric and must be between 1 and 100.
431             * </p>
432             * 
433             * <p>
434             * <b>Supported versions:</b> {@code 4.0}
435             * </p>
436             * @param pref the preference value or null to remove
437             * @throws IllegalArgumentException if the value is not between 1 and 100
438             */
439            public void setPref(Integer pref) {

441                            throw new IllegalArgumentException("Preference value must be between 1 and 100 inclusive.");

444                    replace(PREF, value);
445            }

516             * <p>
517             * Gets the GEO parameter value. This is used to associate global
518             * positioning information with a vCard property. It can be used with the
519             * {@link Address} property.
520             * </p>
521             * <p>
522             * <b>Supported versions:</b> {@code 4.0}
523             * </p>
524             * @throws IllegalStateException if the parameter value is malformed and
525             * cannot be parsed
526             * @return the latitude (index 0) and longitude (index 1) or null if not
527             * present or null if the parameter value was in an incorrect format
528             */
529            public double[] getGeo() {
530                    String value = first(GEO);
531                    if (value == null) {
532                            return null;
533                    }
534    
535                    try {
536                            GeoUri geoUri = GeoUri.parse(value);
537                            return new double[] { geoUri.getCoordA(), geoUri.getCoordB() };
538                    } catch (IllegalArgumentException e) {
539                            throw new IllegalStateException(GEO + " parameter value is malformed and could not be parsed. Retrieve its raw text value instead.", e);
540                    }
541            }

631            /**
632             * <p>
633             * Gets all PID parameter values. PIDs can exist on any property where
634             * multiple instances are allowed (such as EMAIL or ADR, but not N because
635             * only 1 instance of N is allowed).
636             * </p>
637             * <p>
638             * When used in conjunction with the CLIENTPIDMAP property, it allows an
639             * individual property instance to be uniquely identifiable. This feature is
640             * made use of when two different versions of the same vCard have to be
641             * merged together (called "synchronizing").
642             * </p>
643             * <p>
644             * <b>Supported versions:</b> {@code 4.0}
645             * </p>
646             * @throws IllegalStateException if the parameter value is malformed and
647             * cannot be parsed
648             * @return the PID values or empty set if there are none. Index 0 is the
649             * local ID and index 1 is the ID used to reference the CLIENTPIDMAP
650             * property. Index 0 will never be null, but index 1 may be null.
651             */
652            public List<Integer[]> getPids() {
653                    List<String> values = get(PID);
654                    List<Integer[]> pids = new ArrayList<Integer[]>(values.size());
655                    for (String value : values) {
656                            String split[] = value.split("\\.");
657                            try {
658                                    Integer localId = Integer.valueOf(split[0]);
659                                    Integer clientPidMapRef = (split.length > 1) ? Integer.valueOf(split[1]) : null;
660                                    pids.add(new Integer[] { localId, clientPidMapRef });
661                            } catch (NumberFormatException e) {
662                                    throw new IllegalStateException(PID + " parameter value is malformed and could not be parsed. Retrieve its raw text value instead.", e);
663                            }
664                    }
665                    return pids;
666            }

668            /**
669             * <p>
670             * Adds a PID parameter value. PIDs can exist on any property where multiple
671             * instances are allowed (such as EMAIL or ADR, but not N because only 1
672             * instance of N is allowed).
673             * </p>
674             * <p>
675             * When used in conjunction with the CLIENTPIDMAP property, it allows an
676             * individual property instance to be uniquely identifiable. This feature is
677             * made use of when two different versions of the same vCard have to be
678             * merged together (called "synchronizing").
679             * </p>
680             * <p>
681             * <b>Supported versions:</b> {@code 4.0}
682             * </p>
683             * @param localId the local ID

837            /**
838             * Checks this parameters list for data consistency problems or deviations
839             * from the spec. These problems will not prevent the vCard from being
840             * written to a data stream, but may prevent it from being parsed correctly
841             * by the consuming application.
842             * @param version the vCard version to validate against
843             * @return a list of warnings or an empty list if no problems were found
844             */
845            public List<String> validate(VCardVersion version) {
846                    List<String> warnings = new ArrayList<String>(0);
847    
848                    String nonStandard = "%s parameter has a non-standard value (\"%s\").  Standard values are: %s";
849                    String valueNotSupported = "%s parameter value (\"%s\") is not supported by version " + version.getVersion() + ".";
850    
851                    String value = first(CALSCALE);
852                    if (value != null && Calscale.find(value) == null) {
853                            warnings.add(String.format(nonStandard, CALSCALE, value, Calscale.all()));
854                    }
855    
856                    value = first(ENCODING);
857                    if (value != null) {
858                            Encoding encoding = Encoding.find(value);
859                            if (encoding == null) {
860                                    warnings.add(String.format(nonStandard, ENCODING, value, Encoding.all()));
861                            } else if (!encoding.isSupported(version)) {
862                                    warnings.add(String.format(valueNotSupported, ENCODING, value));
863                            }
864                    }
865    
866                    value = first(VALUE);
867                    if (value != null) {
868                            VCardDataType dataType = VCardDataType.find(value);
869                            if (dataType == null) {
870                                    warnings.add(String.format(nonStandard, VALUE, value, VCardDataType.all()));
871                            } else if (!dataType.isSupported(version)) {
872                                    warnings.add(String.format(valueNotSupported, VALUE, value));
873                            }
874                    }
875    
876                    String malformed = "%s parameter has a malformed value (\"%s\").";
877    
878                    try {
879                            getGeo();
880                    } catch (IllegalStateException e) {
881                            warnings.add(String.format(malformed, GEO, first(GEO)));
882                    }
883    
884                    try {
885                            getIndex();
886                    } catch (IllegalStateException e) {
887                            warnings.add(String.format(malformed, INDEX, first(INDEX)));
888                    }
889    
890                    try {
891                            getPids();
892                    } catch (IllegalStateException e) {
893                            warnings.add(String.format(malformed, PID, first(PID)));
894                    }
895    
896                    try {
897                            getPref();
898                    } catch (IllegalStateException e) {
899                            warnings.add(String.format(malformed, PREF, first(PREF)));
900                    }
901    
902                    String paramNotSupported = "%s parameter is not supported by version " + version.getVersion() + ".";
903                    for (Map.Entry<String, Set<VCardVersion>> entry : supportedVersions.entrySet()) {
904                            String name = entry.getKey();
905                            value = first(name);
906                            if (value == null) {
907                                    continue;
908                            }
909    
910                            Set<VCardVersion> versions = entry.getValue();
911                            if (!versions.contains(version)) {
912                                    warnings.add(String.format(paramNotSupported, name));
913                            }
914                    }
915    
916                    return warnings;
917            }
