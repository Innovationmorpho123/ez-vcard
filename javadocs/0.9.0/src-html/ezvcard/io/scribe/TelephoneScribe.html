040    /**
041     * Marshals {@link Telephone} properties.
042     * @author Michael Angstadt
043     */
044    public class TelephoneScribe extends VCardPropertyScribe<Telephone> {
045            public TelephoneScribe() {
046                    super(Telephone.class, "TEL");
047            }
048    
049            @Override
050            protected VCardDataType _defaultDataType(VCardVersion version) {
051                    return VCardDataType.TEXT;
052            }
053    
054            @Override
055            protected VCardDataType _dataType(Telephone property, VCardVersion version) {
056                    if (version == VCardVersion.V4_0) {
057                            if (property.getText() != null) {
058                                    return VCardDataType.TEXT;
059                            }
060                            if (property.getUri() != null) {
061                                    return VCardDataType.URI;
062                            }
063                    }
064    
065                    return VCardDataType.TEXT;
066            }
067    
068            @Override
069            protected void _prepareParameters(Telephone property, VCardParameters copy, VCardVersion version, VCard vcard) {
070                    handlePrefParam(property, copy, version, vcard);
071            }
072    
073            @Override
074            protected String _writeText(Telephone property, VCardVersion version) {
075                    String text = property.getText();
076                    if (text != null) {
077                            return escape(text);
078                    }
079    
080                    TelUri uri = property.getUri();
081                    if (uri != null) {
082                            if (version == VCardVersion.V4_0) {
083                                    return uri.toString();
084                            }
085    
086                            String ext = uri.getExtension();
087                            if (ext == null) {
088                                    return escape(uri.getNumber());
089                            }
090                            return escape(uri.getNumber() + " x" + ext);
091                    }
092    
093                    return "";
094            }
095    
096            @Override
097            protected Telephone _parseText(String value, VCardDataType dataType, VCardVersion version, VCardParameters parameters, List<String> warnings) {
098                    value = unescape(value);
099                    return parse(value, dataType, warnings);
100            }
101    
102            @Override
103            protected void _writeXml(Telephone property, XCardElement parent) {
104                    String text = property.getText();
105                    if (text != null) {
106                            parent.append(VCardDataType.TEXT, text);
107                            return;
108                    }
109    
110                    TelUri uri = property.getUri();
111                    if (uri != null) {
112                            parent.append(VCardDataType.URI, uri.toString());
113                            return;
114                    }
115    
116                    parent.append(VCardDataType.TEXT, "");
117            }
118    
119            @Override
120            protected Telephone _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
121                    String text = element.first(VCardDataType.TEXT);
122                    if (text != null) {
123                            return new Telephone(text);
124                    }
125    
126                    String uri = element.first(VCardDataType.URI);
127                    if (uri != null) {
128                            try {
129                                    return new Telephone(TelUri.parse(uri));
130                            } catch (IllegalArgumentException e) {
131                                    warnings.add("Could not parse property value as a URI.  Assuming it's text.");
132                                    return new Telephone(uri);
133                            }
134                    }
135    
136                    throw missingXmlElements(VCardDataType.TEXT, VCardDataType.URI);
137            }
138    
139            @Override
140            protected Telephone _parseHtml(HCardElement element, List<String> warnings) {
141                    Telephone property;
142                    String href = element.attr("href");
143                    try {
144                            property = new Telephone(TelUri.parse(href));
145                    } catch (IllegalArgumentException e) {
146                            //not a tel URI
147                            property = new Telephone(element.value());
148                    }
149    
150                    List<String> types = element.types();
151                    for (String type : types) {
152                            property.getParameters().addType(type);
153                    }
154    
155                    return property;
156            }
157    
158            @Override
159            protected JCardValue _writeJson(Telephone property) {
160                    String text = property.getText();
161                    if (text != null) {
162                            return JCardValue.single(text);
163                    }
164    
165                    TelUri uri = property.getUri();
166                    if (uri != null) {
167                            return JCardValue.single(uri.toString());
168                    }
169    
170                    return JCardValue.single("");
171            }
172    
173            @Override
174            protected Telephone _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
175                    String valueStr = value.asSingle();
176                    return parse(valueStr, dataType, warnings);
177            }
178    
179            private Telephone parse(String value, VCardDataType dataType, List<String> warnings) {
180                    try {
181                            return new Telephone(TelUri.parse(value));
182                    } catch (IllegalArgumentException e) {
183                            if (dataType == VCardDataType.URI) {
184                                    warnings.add("Could not parse property value as a URI.  Assuming it's text.");
185                            }
186                    }
187    
188                    return new Telephone(value);
189            }
190    }
