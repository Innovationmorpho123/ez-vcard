412            /**
413             * <p>
414             * Unmarshals a property from an XML document (xCard).
415             * </p>
416             * <p>
417             * This method should be overridden by child classes that wish to support
418             * xCard. The default implementation of this method will find the first
419             * child element with the xCard namespace. The element's name will be used
420             * as the property's data type and its text content will be passed into the
421             * {@link #_parseText} method. If no such child element is found, then the
422             * parent element's text content will be passed into {@link #_parseText} and
423             * the data type will be null.
424             * </p>
425             * @param element the property's XML element
426             * @param parameters the parsed parameters. These parameters will be
427             * assigned to the property object once this method returns. Therefore, do
428             * not assign any parameters to the property object itself whilst inside of
429             * this method, or else they will be overwritten.
430             * @param warnings allows the programmer to alert the user to any
431             * note-worthy (but non-critical) issues that occurred during the
432             * unmarshalling process
433             * @return the unmarshalled property object
434             * @throws CannotParseException if the marshaller could not parse the
435             * property's value
436             * @throws SkipMeException if the property should not be added to the final
437             * {@link VCard} object
438             */
439            protected T _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
440                    String value = null;
441                    VCardDataType dataType = null;
442                    Element rawElement = element.element();
443    
444                    //get the text content of the first child element with the xCard namespace
445                    List<Element> children = XmlUtils.toElementList(rawElement.getChildNodes());
446                    for (Element child : children) {
447                            if (!element.version().getXmlNamespace().equals(child.getNamespaceURI())) {
448                                    continue;
449                            }
450    
451                            dataType = VCardDataType.get(child.getLocalName());
452                            value = child.getTextContent();
453                            break;
454                    }
455    
456                    if (dataType == null) {
457                            //get the text content of the property element
458                            value = rawElement.getTextContent();
459                    }
460    
461                    value = escape(value);
462                    return _parseText(value, dataType, element.version(), parameters, warnings);
463            }

493            /**
494             * <p>
495             * Unmarshals a property from a JSON data stream (jCard).
496             * </p>
497             * <p>
498             * This method should be overridden by child classes that wish to support
499             * jCard. The default implementation of this method will convert the jCard
500             * property value to a string and pass it into the {@link #_parseText}
501             * method.
502             * </p>
503             * 
504             * <hr>
505             * 
506             * <p>
507             * The following paragraphs describe the way in which this method's default
508             * implementation converts a jCard value to a string:
509             * </p>
510             * <p>
511             * If the jCard value consists of a single, non-array, non-object value,
512             * then the value is converted to a string. Special characters (backslashes,
513             * commas, and semicolons) are escaped in order to simulate what the value
514             * might look like in a plain-text vCard.<br>
515             * <code>["x-foo", {}, "text", "the;value"] --&gt; "the\;value"</code><br>
516             * <code>["x-foo", {}, "text", 2] --&gt; "2"</code>
517             * </p>
518             * <p>
519             * If the jCard value consists of multiple, non-array, non-object values,
520             * then all the values are appended together in a single string, separated
521             * by commas. Special characters (backslashes, commas, and semicolons) are
522             * escaped for each value in order to prevent commas from being treated as
523             * delimiters, and to simulate what the value might look like in a
524             * plain-text vCard.<br>
525             * <code>["x-foo", {}, "text", "one", "two,three"] --&gt;
526             * "one,two\,three"</code>
527             * </p>
528             * <p>
529             * If the jCard value is a single array, then this array is treated as a
530             * "structured value", and converted its plain-text representation. Special
531             * characters (backslashes, commas, and semicolons) are escaped for each
532             * value in order to prevent commas and semicolons from being treated as
533             * delimiters.<br>
534             * <code>["x-foo", {}, "text", ["one", ["two", "three"], "four;five"]]
535             * --&gt; "one;two,three;four\;five"</code>
536             * </p>
537             * <p>
538             * If the jCard value starts with a JSON object, then it is converted to an
539             * empty string (JSON objects are not supported by this method).<br>
540             * <code>["x-foo", , "text", {"one": 1}] --&gt; ""}</code>
541             * </p>
542             * <p>
543             * For all other cases, behavior is undefined.
544             * </p>
545             * @param value the property's JSON value
546             * @param dataType the data type
547             * @param parameters the parsed parameters. These parameters will be
548             * assigned to the property object once this method returns. Therefore, do
549             * not assign any parameters to the property object itself whilst inside of
550             * this method, or else they will be overwritten.
551             * @param warnings allows the programmer to alert the user to any
552             * note-worthy (but non-critical) issues that occurred during the
553             * unmarshalling process
554             * @return the unmarshalled property object
555             * @throws CannotParseException if the marshaller could not parse the
556             * property's value
557             * @throws SkipMeException if the property should not be added to the final
558             * {@link VCard} object
559             */
560            protected T _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
561                    return _parseText(jcardValueToString(value), dataType, VCardVersion.V4_0, parameters, warnings);
562            }

564            private static String jcardValueToString(JCardValue value) {
565                    if (value.getValues().size() > 1) {
566                            List<String> multi = value.asMulti();
567                            if (!multi.isEmpty()) {
568                                    return list(multi);
569                            }
570                    }
571    
572                    if (!value.getValues().isEmpty() && value.getValues().get(0).getArray() != null) {
573                            List<List<String>> structured = value.asStructured();
574                            if (!structured.isEmpty()) {
575                                    return structured(structured.toArray());
576                            }
577                    }
578    
579                    return escape(value.asSingle());
580            }

582            /**
583             * Unescapes all special characters that are escaped with a backslash, as
584             * well as escaped newlines.
585             * @param text the text to unescape
586             * @return the unescaped text
587             */
588            public static String unescape(String text) {
589                    if (text == null) {
590                            return null;
591                    }
592    
593                    StringBuilder sb = null; //only instantiate the StringBuilder if the string needs to be modified
594                    boolean escaped = false;
595                    for (int i = 0; i < text.length(); i++) {
596                            char ch = text.charAt(i);
597    
598                            if (escaped) {
599                                    if (sb == null) {
600                                            sb = new StringBuilder(text.length());
601                                            sb.append(text.substring(0, i - 1));
602                                    }
603    
604                                    escaped = false;
605    
606                                    if (ch == 'n' || ch == 'N') {
607                                            //newlines appear as "\n" or "\N" (see RFC 5545 p.46)
608                                            sb.append(NEWLINE);
609                                            continue;
610                                    }
611    
612                                    sb.append(ch);
613                                    continue;
614                            }
615    
616                            if (ch == '\\') {
617                                    escaped = true;
618                                    continue;
619                            }
620    
621                            if (sb != null) {
622                                    sb.append(ch);
623                            }
624                    }
625                    return (sb == null) ? text : sb.toString();
626            }

628            /**
629             * <p>
630             * Escapes all special characters within a vCard value. These characters
631             * are:
632             * </p>
633             * <ul>
634             * <li>backslashes ({@code \})</li>
635             * <li>commas ({@code ,})</li>
636             * <li>semi-colons ({@code ;})</li>
637             * </ul>
638             * <p>
639             * Newlines are not escaped by this method. They are escaped when the vCard
640             * is serialized (in the {@link VCardRawWriter} class).
641             * </p>
642             * @param text the text to escape
643             * @return the escaped text
644             */
645            public static String escape(String text) {
646                    if (text == null) {
647                            return null;
648                    }
649    
650                    String chars = "\\,;";
651                    StringBuilder sb = null; //only instantiate the StringBuilder if the string needs to be modified
652                    for (int i = 0; i < text.length(); i++) {
653                            char ch = text.charAt(i);
654                            if (chars.indexOf(ch) >= 0) {
655                                    if (sb == null) {
656                                            sb = new StringBuilder(text.length());
657                                            sb.append(text.substring(0, i));
658                                    }
659                                    sb.append('\\');
660                            }
661    
662                            if (sb != null) {
663                                    sb.append(ch);
664                            }
665                    }
666                    return (sb == null) ? text : sb.toString();
667            }

680             * Factory class for splitting strings.
681             */
682            protected static class Splitter {
683                    private String string;
684                    private String delimiter;
685                    private boolean removeEmpties = false;
686                    private boolean unescape = false;
687                    private int limit = -1;
688    
689                    /**
690                     * Creates a new splitter object.
691                     * @param string the string to split (e.g. "one,two,three")
692                     * @param delimiter the delimiter (e.g. ",")
693                     */
694                    public Splitter(String string, String delimiter) {
695                            this.string = string;
696                            this.delimiter = delimiter;
697                    }

699                    /**
700                     * Sets whether to remove empty elements.
701                     * @param removeEmpties true to remove empty elements, false not to
702                     * (default is false)
703                     * @return this
704                     */
705                    public Splitter removeEmpties(boolean removeEmpties) {
706                            this.removeEmpties = removeEmpties;
707                            return this;

711                     * Sets whether to unescape each split string.
712                     * @param unescape true to unescape, false not to (default is false)

715                    public Splitter unescape(boolean unescape) {
716                            this.unescape = unescape;
717                            return this;
718                    }

720                    /**
721                     * Sets the max number of split strings it should parse.
722                     * @param limit the max number of split strings
723                     * @return this
724                     */
725                    public Splitter limit(int limit) {
726                            this.limit = limit;
727                            return this;
728                    }

730                    /**
731                     * Performs the split operation.
732                     * @return the split string
733                     */
734                    public List<String> split() {
735                            //from: http://stackoverflow.com/q/820172">http://stackoverflow.com/q/820172
736                            String split[] = string.split("\\s*(?<!\\\\)" + Pattern.quote(delimiter) + "\\s*", limit);
737    
738                            List<String> list = new ArrayList<String>(split.length);
739                            for (String s : split) {
740                                    if (s.length() == 0 && removeEmpties) {
741                                            continue;
742                                    }
743    
744                                    if (unescape) {
745                                            s = VCardPropertyScribe.unescape(s);
746                                    }
747    
748                                    list.add(s);
749                            }
750                            return list;
751                    }

839             * <p>
840             * Writes a structured value.
841             * </p>
842             * <p>
843             * This method accepts a list of {@link Object} instances.
844             * {@link Collection} objects will be treated as multi-valued components.
845             * Null objects will be treated as empty components. All other objects will
846             * have their {@code toString()} method invoked to generate the string
847             * value.
848             * </p>
849             * @param values the values to write
850             * @return the structured value string
851             */
852            protected static String structured(Object... values) {
853                    return join(Arrays.asList(values), ";", new JoinCallback<Object>() {
854                            public void handle(StringBuilder sb, Object value) {
855                                    if (value == null) {
856                                            return;
857                                    }
858    
859                                    if (value instanceof Collection) {
860                                            Collection<?> list = (Collection<?>) value;
861                                            sb.append(list(list));
862                                            return;
863                                    }
864    
865                                    sb.append(escape(value.toString()));
866                            }
867                    });
868            }

870            /**
871             * Iterates over the fields in a structured value.
872             */
873            protected static class StructuredIterator {
874                    private final Iterator<List<String>> it;
875    
876                    /**
877                     * Constructs a new structured iterator.
878                     * @param it the iterator to wrap
879                     */
880                    public StructuredIterator(Iterator<List<String>> it) {
881                            this.it = it;
882                    }
883    
884                    /**
885                     * Gets the first value of the next component.
886                     * @return the first value, null if the value is an empty string, or
887                     * null if there are no more components
888                     */
889                    public String nextString() {
890                            if (!hasNext()) {
891                                    return null;
892                            }
893    
894                            List<String> list = it.next();
895                            if (list.isEmpty()) {
896                                    return null;
897                            }
898    
899                            String value = list.get(0);
900                            return (value.length() == 0) ? null : value;
901                    }
902    
903                    /**
904                     * Gets the next component.
905                     * @return the next component, an empty list if the component is empty,
906                     * or an empty list of there are no more components
907                     */
908                    public List<String> nextComponent() {
909                            if (!hasNext()) {
910                                    return new ArrayList<String>(0); //the lists should be mutable so they can be directly assigned to the property object's fields
911                            }
912    
913                            List<String> list = it.next();
914                            if (list.size() == 1 && list.get(0).length() == 0) {
915                                    return new ArrayList<String>(0);
916                            }
917    
918                            return list;
919                    }
920    
921                    public boolean hasNext() {
922                            return it.hasNext();
923                    }
924            }

1072    
1073                    switch (elements.length) {

1076                            break;
1077                    case 1:

1079                            break;
1080                    case 2:

1082                            break;
1083                    default:
1084                            StringBuilder sb = new StringBuilder();
1085    
1086                            sb.append("Property value empty (no ");
1087                            StringUtils.join(Arrays.asList(elements).subList(0, elements.length - 1), ", ", sb, new JoinCallback<String>() {

1101            /**
1102             * Utility method for switching between the "PREF" and "TYPE=PREF"
1103             * parameters, depending on the target vCard version. Meant to be called
1104             * from a scribe's {@link #_prepareParameters} method.
1105             * @param property the property that's being marshalled
1106             * @param copy the parameters that are being marshalled
1107             * @param version the vCard version
1108             * @param vcard the vCard that's being marshalled
1109             */
1110            protected static void handlePrefParam(VCardProperty property, VCardParameters copy, VCardVersion version, VCard vcard) {
1111                    switch (version) {
1112                    case V2_1:
1113                    case V3_0:
1114                            copy.setPref(null);
1115    
1116                            //find the property with the lowest PREF value in the vCard
1117                            VCardProperty mostPreferred = null;
1118                            for (VCardProperty p : vcard.getProperties(property.getClass())) {
1119                                    Integer pref = p.getParameters().getPref();
1120                                    if (pref == null) {
1121                                            continue;
1122                                    }
1123    
1124                                    if (mostPreferred == null || pref < mostPreferred.getParameters().getPref()) {
1125                                            mostPreferred = p;
1126                                    }
1127                            }
1128    
1129                            if (property == mostPreferred) {
1130                                    copy.addType("pref");
1131                            }
1132    
1133                            break;
1134                    case V4_0:
1135                            for (String type : property.getParameters().getTypes()) {
1136                                    if ("pref".equalsIgnoreCase(type)) {
1137                                            copy.removeType(type);
1138                                            copy.setPref(1);
1139                                            break;
1140                                    }
1141                            }
1142                            break;
1143                    }
1144            }
