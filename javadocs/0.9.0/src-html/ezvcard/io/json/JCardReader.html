060    /**
061     * <p>
062     * Parses {@link VCard} objects from a JSON data stream (jCard format).
063     * </p>
064     * <p>
065     * <b>Example:</b>
066     * 
067     * <pre class="brush:java">
068     * File file = new File("vcards.json");
069     * JCardReader jcardReader = new JCardReader(file);
070     * VCard vcard;
071     * while ((vcard = jcardReader.readNext()) != null){
072     *   ...
073     * }
074     * jcardReader.close();
075     * </pre>
076     * 
077     * </p>
078     * @author Michael Angstadt
079     * @see <a
080     * href="http://tools.ietf.org/html/draft-kewisch-vcard-in-json-04">jCard
081     * draft</a>
082     */
083    public class JCardReader implements Closeable {
084            private ScribeIndex index = new ScribeIndex();
085            private final List<String> warnings = new ArrayList<String>();
086            private final JCardRawReader reader;
087    
088            /**
089             * Creates a reader that parses jCards from a JSON string.
090             * @param json the JSON string
091             */
092            public JCardReader(String json) {
093                    this(new StringReader(json));
094            }
095    
096            /**
097             * Creates a reader that parses jCards from an input stream.
098             * @param in the input stream to read the vCards from
099             */
100            public JCardReader(InputStream in) {
101                    this(utf8Reader(in));
102            }
103    
104            /**
105             * Creates a reader that parses jCards from a file.
106             * @param file the file to read the vCards from
107             * @throws FileNotFoundException if the file doesn't exist
108             */
109            public JCardReader(File file) throws FileNotFoundException {
110                    this(utf8Reader(file));
111            }
112    
113            /**
114             * Creates a reader that parses jCards from a reader.
115             * @param reader the reader to read the vCards from
116             */
117            public JCardReader(Reader reader) {
118                    this.reader = new JCardRawReader(reader);
119            }
120    
121            /**
122             * Reads the next vCard from the data stream.
123             * @return the next vCard or null if there are no more
124             * @throws IOException if there's a problem reading from the stream
125             */
126            public VCard readNext() throws IOException {
127                    if (reader.eof()) {
128                            return null;
129                    }
130    
131                    warnings.clear();
132    
133                    JCardDataStreamListenerImpl listener = new JCardDataStreamListenerImpl();
134                    reader.readNext(listener);
135                    VCard vcard = listener.vcard;
136                    if (vcard != null && !listener.versionFound) {
137                            addWarning("No \"version\" property found.");
138                    }
139                    return vcard;
140            }
141    
142            /**
143             * <p>
144             * Registers a property scribe. This is the same as calling:
145             * </p>
146             * <p>
147             * {@code getScribeIndex().register(scribe)}
148             * </p>
149             * @param scribe the scribe to register
150             */
151            public void registerScribe(VCardPropertyScribe<? extends VCardProperty> scribe) {
152                    index.register(scribe);
153            }
154    
155            /**
156             * Gets the scribe index.
157             * @return the scribe index
158             */
159            public ScribeIndex getScribeIndex() {
160                    return index;
161            }
162    
163            /**
164             * Sets the scribe index.
165             * @param index the scribe index
166             */
167            public void setScribeIndex(ScribeIndex index) {
168                    this.index = index;
169            }
170    
171            /**
172             * Gets the warnings from the last vCard that was unmarshalled. This list is
173             * reset every time a new vCard is read.
174             * @return the warnings or empty list if there were no warnings
175             */
176            public List<String> getWarnings() {
177                    return new ArrayList<String>(warnings);
178            }
179    
180            private void addWarning(String message) {
181                    addWarning(message, null);
182            }
183    
184            private void addWarning(String message, String propertyName) {
185                    StringBuilder sb = new StringBuilder();
186                    sb.append("Line ").append(reader.getLineNum());
187                    if (propertyName != null) {
188                            sb.append(" (").append(propertyName).append(" property)");
189                    }
190                    sb.append(": ").append(message);
191    
192                    warnings.add(sb.toString());
193            }
194    
195            public void close() throws IOException {
196                    reader.close();
197            }
198    
199            private class JCardDataStreamListenerImpl implements JCardDataStreamListener {
200                    private VCard vcard = null;
201                    private boolean versionFound = false;
202    
203                    public void beginVCard() {
204                            vcard = new VCard();
205                            vcard.setVersion(VCardVersion.V4_0);
206                    }
207    
208                    public void readProperty(String group, String propertyName, VCardParameters parameters, VCardDataType dataType, JCardValue value) {
209                            if ("version".equalsIgnoreCase(propertyName)) {
210                                    //don't unmarshal "version" because we don't treat it as a property
211                                    versionFound = true;
212    
213                                    VCardVersion version = VCardVersion.valueOfByStr(value.asSingle());
214                                    if (version != VCardVersion.V4_0) {
215                                            addWarning("Version must be \"" + VCardVersion.V4_0.getVersion() + "\"", propertyName);
216                                    }
217                                    return;
218                            }
219    
220                            VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(propertyName);
221                            if (scribe == null) {
222                                    scribe = new RawPropertyScribe(propertyName);
223                            }
224    
225                            VCardProperty property;
226                            try {
227                                    Result<? extends VCardProperty> result = scribe.parseJson(value, dataType, parameters);
228    
229                                    for (String warning : result.getWarnings()) {
230                                            addWarning(warning, propertyName);
231                                    }
232    
233                                    property = result.getProperty();
234                                    property.setGroup(group);
235                            } catch (SkipMeException e) {
236                                    addWarning("Property has requested that it be skipped: " + e.getMessage(), propertyName);
237                                    return;
238                            } catch (CannotParseException e) {
239                                    scribe = new RawPropertyScribe(propertyName);
240                                    Result<? extends VCardProperty> result = scribe.parseJson(value, dataType, parameters);
241    
242                                    property = result.getProperty();
243                                    property.setGroup(group);
244    
245                                    String valueStr = ((RawProperty) property).getValue();
246                                    addWarning("Property value could not be parsed.  Property will be saved as an extended property instead." + NEWLINE + "  Value: " + valueStr + NEWLINE + "  Reason: " + e.getMessage(), propertyName);
247                            } catch (EmbeddedVCardException e) {
248                                    addWarning("Property will not be unmarshalled because jCard does not supported embedded vCards.", propertyName);
249                                    return;
250                            }
251    
252                            vcard.addProperty(property);
253                    }
254            }
255    }
