080    /**
081    * <p>
082    * Represents an XML document that contains vCard objects ("xCard" standard).
083    * This class can be used to read and write xCard documents.
084    * </p>
085    * <p>
086    * <b>Examples:</b>
087    * 
088    * <pre class="brush:java">
089    * String xml =
090    * "&lt;vcards xmlns=\"urn:ietf:params:xml:ns:vcard-4.0\"&gt;" +
091    *   "&lt;vcard&gt;" +
092    *     "&lt;fn&gt;" +
093    *       "&lt;text&gt;John Doe&lt;/text&gt;" +
094    *     "&lt;/fn&gt;" +
095    *     "&lt;n&gt;" +
096    *       "&lt;surname&gt;Doe&lt;/surname&gt;" +
097    *        "&lt;given&gt;Johnathan&lt;/given&gt;" +
098    *        "&lt;additional&gt;Jonny&lt;/additional&gt;" +
099    *        "&lt;additional&gt;John&lt;/additional&gt;" +
100    *        "&lt;prefix&gt;Mr.&lt;/prefix&gt;" +
101    *        "&lt;suffix /&gt;" +
102    *      "&lt;/n&gt;" +
103    *    "&lt;/vcard&gt;" +
104    * "&lt;/vcards&gt;";
105    *     
106    * //parsing an existing xCard document
107    * XCardDocument xcard = new XCardDocument(xml);
108    * List&lt;VCard&gt; vcards = xcard.parseAll();
109    * 
110    * //creating an empty xCard document
111    * XCardDocument xcard = new XCardDocument();
112    * 
113    * //VCard objects can be added at any time
114    * VCard vcard = ...
115    * xcard.add(vcard);
116    * 
117    * //retrieving the raw XML DOM
118    * Document document = xcard.getDocument();
119    * 
120    * //call one of the "write()" methods to output the xCard document
121    * File file = new File("johndoe.xml");
122    * xcard.write(file);
123    * </pre>
124    * 
125    * </p>
126    * @author Michael Angstadt
127    * @see <a href="http://tools.ietf.org/html/rfc6351">RFC 6351</a>
128    */
129    //@formatter:on

232            private void init(Document document) {
233                    XPath xpath = XPathFactory.newInstance().newXPath();
234                    xpath.setNamespaceContext(nsContext);
235    
236                    try {
237                            //find the <vcards> element
238                            String prefix = nsContext.getPrefix();
239                            root = (Element) xpath.evaluate("//" + prefix + ":vcards", document, XPathConstants.NODE);
240                    } catch (XPathExpressionException e) {
241                            //never thrown, xpath expression is hard coded
242                    }
243            }

348            /**
349             * Parses all the {@link VCard} objects from the xCard document.
350             * @return the vCard objects
351             */
352            public List<VCard> parseAll() {
353                    parseWarnings.clear();
354    
355                    if (root == null) {
356                            return Collections.emptyList();
357                    }
358    
359                    List<VCard> vcards = new ArrayList<VCard>();
360                    for (Element vcardElement : getVCardElements()) {
361                            List<String> warnings = new ArrayList<String>();
362                            parseWarnings.add(warnings);
363                            VCard vcard = parseVCardElement(vcardElement, warnings);
364                            vcards.add(vcard);
365                    }
366    
367                    return vcards;
368            }
369    
370            /**
371             * Parses the first the {@link VCard} object from the xCard document.
372             * @return the vCard object
373             */
374            public VCard parseFirst() {
375                    parseWarnings.clear();
376    
377                    if (root == null) {
378                            return null;
379                    }
380    
381                    List<Element> vcardElements = getVCardElements();
382                    if (vcardElements.isEmpty()) {
383                            return null;
384                    }
385    
386                    List<String> warnings = new ArrayList<String>();
387                    parseWarnings.add(warnings);
388                    return parseVCardElement(vcardElements.get(0), warnings);
389            }
390    
391            private VCard parseVCardElement(Element vcardElement, List<String> warnings) {
392                    VCard vcard = new VCard();
393                    vcard.setVersion(version4);
394    
395                    String ns = version4.getXmlNamespace();
396                    List<Element> children = XmlUtils.toElementList(vcardElement.getChildNodes());
397                    for (Element child : children) {
398                            if ("group".equals(child.getLocalName()) && ns.equals(child.getNamespaceURI())) {
399                                    String group = child.getAttribute("name");
400                                    if (group.length() == 0) {
401                                            group = null;
402                                    }
403                                    List<Element> propElements = XmlUtils.toElementList(child.getChildNodes());
404                                    for (Element propElement : propElements) {
405                                            parseAndAddElement(propElement, group, vcard, warnings);
406                                    }
407                            } else {
408                                    parseAndAddElement(child, null, vcard, warnings);
409                            }
410                    }
411    
412                    return vcard;
413            }

415            /**
416             * Parses a property element from the XML document and adds the property to
417             * the vCard.
418             * @param element the element to parse
419             * @param group the group name or null if the property does not belong to a
420             * group
421             * @param vcard the vCard object
422             * @param warningsBuf the list to add the warnings to
423             */
424            private void parseAndAddElement(Element element, String group, VCard vcard, List<String> warnings) {
425                    VCardParameters parameters = parseSubTypes(element);
426    
427                    VCardProperty property;
428                    String propertyName = element.getLocalName();
429                    String ns = element.getNamespaceURI();
430                    VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(new QName(ns, propertyName));
431                    try {
432                            Result<? extends VCardProperty> result = scribe.parseXml(element, parameters);
433    
434                            property = result.getProperty();
435                            property.setGroup(group);
436    
437                            for (String warning : result.getWarnings()) {
438                                    addWarning(warning, propertyName, warnings);
439                            }
440                    } catch (SkipMeException e) {
441                            addWarning("Property has requested that it be skipped: " + e.getMessage(), propertyName, warnings);
442                            return;
443                    } catch (CannotParseException e) {
444                            String xml = XmlUtils.toString(element);
445                            addWarning("Property value could not be parsed.  It will be unmarshalled as a " + Xml.class.getSimpleName() + " property instead." + NEWLINE + "  XML: " + xml + NEWLINE + "  Reason: " + e.getMessage(), propertyName, warnings);
446    
447                            scribe = index.getPropertyScribe(Xml.class);
448                            Result<? extends VCardProperty> result = scribe.parseXml(element, parameters);
449                            property = result.getProperty();
450                            property.setGroup(group);
451                    } catch (UnsupportedOperationException e) {
452                            scribe = index.getPropertyScribe(Xml.class);
453                            Result<? extends VCardProperty> result = scribe.parseXml(element, parameters);
454                            property = result.getProperty();
455                            property.setGroup(group);
456    
457                            addWarning("Property class \"" + property.getClass().getName() + "\" does not support xCard unmarshalling.  It will be unmarshalled as a " + Xml.class.getSimpleName() + " property instead.", propertyName, warnings);
458                    } catch (EmbeddedVCardException e) {
459                            addWarning("Property will not be unmarshalled because xCard does not supported embedded vCards.", propertyName, warnings);
460                            return;
461                    }
462    
463                    vcard.addProperty(property);
464            }

466            /**
467             * Parses the property parameters (aka "sub types").
468             * @param element the property's XML element
469             * @return the parsed parameters
470             */
471            private VCardParameters parseSubTypes(Element element) {
472                    VCardParameters subTypes = new VCardParameters();
473    
474                    List<Element> parametersElements = XmlUtils.toElementList(element.getElementsByTagNameNS(version4.getXmlNamespace(), "parameters"));
475                    for (Element parametersElement : parametersElements) { // foreach "<parameters>" element (there should only be 1 though)
476                            List<Element> paramElements = XmlUtils.toElementList(parametersElement.getChildNodes());
477                            for (Element paramElement : paramElements) {
478                                    String name = paramElement.getLocalName().toUpperCase();
479                                    List<Element> valueElements = XmlUtils.toElementList(paramElement.getChildNodes());
480                                    if (valueElements.isEmpty()) {
481                                            String value = paramElement.getTextContent();
482                                            subTypes.put(name, value);
483                                    } else {
484                                            for (Element valueElement : valueElements) {
485                                                    String value = valueElement.getTextContent();
486                                                    subTypes.put(name, value);
487                                            }
488                                    }
489                            }
490                    }
491    
492                    return subTypes;
493            }

589            /**
590             * Adds a vCard to the XML document.
591             * @param vcard the vCard to add
592             * @throws IllegalArgumentException if a scribe hasn't been registered for a
593             * custom property class (see: {@link #registerScribe})
594             */
595            public void add(VCard vcard) {
596                    ListMultimap<String, VCardProperty> typesToAdd = new ListMultimap<String, VCardProperty>(); //group the types by group name (null = no group name)
597    
598                    for (VCardProperty type : vcard) {
599                            if (addProdId && type instanceof ProductId) {
600                                    //do not add the PRODID in the vCard if "addProdId" is true
601                                    continue;
602                            }
603    
604                            if (versionStrict && !type.getSupportedVersions().contains(version4)) {
605                                    //do not add the property to the vCard if it is not supported by the target version
606                                    continue;
607                            }
608    
609                            //check for scribes before writing anything to the stream
610                            if (index.getPropertyScribe(type) == null) {
611                                    throw new IllegalArgumentException("No scribe found for property class \"" + type.getClass().getName() + "\".");
612                            }
613    
614                            typesToAdd.put(type.getGroup(), type);
615                    }
616    
617                    //add an extended type saying it was generated by this library
618                    if (addProdId) {
619                            ProductId prodId = new ProductId("ez-vcard " + Ezvcard.VERSION);
620                            typesToAdd.put(prodId.getGroup(), prodId);
621                    }
622    
623                    //marshal each type object
624                    Element vcardElement = createElement("vcard");
625                    for (String groupName : typesToAdd.keySet()) {
626                            Element parent;
627                            if (groupName != null) {
628                                    Element groupElement = createElement("group");
629                                    groupElement.setAttribute("name", groupName);
630                                    vcardElement.appendChild(groupElement);
631                                    parent = groupElement;
632                            } else {
633                                    parent = vcardElement;
634                            }
635    
636                            for (VCardProperty type : typesToAdd.get(groupName)) {
637                                    try {
638                                            Element typeElement = marshalType(type, vcard);
639                                            parent.appendChild(typeElement);
640                                    } catch (SkipMeException e) {
641                                            //skip property
642                                    } catch (EmbeddedVCardException e) {
643                                            //skip property
644                                    }
645                            }
646                    }
647                    root.appendChild(vcardElement);
648            }

650            /**
651             * Marshals a type object to an XML element.
652             * @param type the type object to marshal
653             * @param vcard the vcard the type belongs to
654             * @return the XML element
655             */
656            @SuppressWarnings({ "rawtypes", "unchecked" })
657            private Element marshalType(VCardProperty type, VCard vcard) {
658                    VCardPropertyScribe scribe = index.getPropertyScribe(type);
659                    if (scribe == null) {
660                            throw new IllegalArgumentException("No marshaller found for property class \"" + type.getClass().getName() + "\".");
661                    }
662    
663                    VCardParameters parameters = scribe.prepareParameters(type, version4, vcard);
664    
665                    QName qname = scribe.getQName();
666                    Element typeElement = createElement(qname.getLocalPart(), qname.getNamespaceURI());
667    
668                    //marshal the sub types
669                    if (!parameters.isEmpty()) {
670                            Element parametersElement = marshalSubTypes(parameters);
671                            typeElement.appendChild(parametersElement);
672                    }
673    
674                    //marshal the value
675                    scribe.writeXml(type, typeElement);
676    
677                    return typeElement;
678            }
679    
680            private Element marshalSubTypes(VCardParameters parameters) {
681                    Element parametersElement = createElement("parameters");
682    
683                    for (Map.Entry<String, List<String>> param : parameters) {
684                            String parameterName = param.getKey().toLowerCase();
685                            Element parameterElement = createElement(parameterName);
686    
687                            for (String paramValue : param.getValue()) {
688                                    VCardDataType dataType = parameterDataTypes.get(parameterName);
689                                    String dataTypeElementName = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
690                                    Element dataTypeElement = createElement(dataTypeElementName);
691                                    dataTypeElement.setTextContent(paramValue);
692                                    parameterElement.appendChild(dataTypeElement);
693                            }
694    
695                            parametersElement.appendChild(parameterElement);
696                    }
697    
698                    return parametersElement;
699            }
700    
701            private List<Element> getVCardElements() {
702                    return getChildElements(root, "vcard");
703            }
704    
705            private List<Element> getChildElements(Element parent, String localName) {
706                    List<Element> elements = new ArrayList<Element>();
707                    for (Element child : XmlUtils.toElementList(parent.getChildNodes())) {
708                            if (localName.equals(child.getLocalName()) && version4.getXmlNamespace().equals(child.getNamespaceURI())) {
709                                    elements.add(child);
710                            }
711                    }
712                    return elements;
713            }
