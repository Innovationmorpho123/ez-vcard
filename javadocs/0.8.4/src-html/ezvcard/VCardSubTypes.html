363            /**
364             * <p>
365             * Gets the preference value. The lower the number, the more preferred this
366             * property instance is compared with other properties in the same vCard of
367             * the same type. If a property doesn't have a preference value, then it is
368             * considered the least preferred.
369             * </p>
370             * 
371             * <p>
372             * In the vCard below, the address on the second row is the most preferred
373             * because it has the lowest PREF value.
374             * </p>
375             * 
376             * <pre>
377             * ADR;TYPE=work;PREF=2:
378             * ADR;TYPE=work;PREF=1:
379             * ADR;TYPE=home:
380             * </pre>
381             * 
382             * <p>
383             * Preference values must be numeric and must be between 1 and 100.
384             * </p>
385             * 
386             * <p>
387             * <b>Supported versions:</b> {@code 4.0}
388             * </p>
389             * @throws IllegalStateException if the parameter value is malformed and
390             * cannot be parsed
391             * @return the preference value or null if it doesn't exist or null if it
392             * couldn't be parsed into a number
393             */
394            public Integer getPref() {
395                    String pref = first(PREF);
396                    if (pref == null) {
397                            return null;
398                    }
399    
400                    try {
401                            return Integer.valueOf(pref);
402                    } catch (NumberFormatException e) {
403                            throw new IllegalStateException(PREF + " parameter value is malformed and could not be parsed. Retrieve its raw text value instead.", e);
404                    }
405            }

407            /**
408             * <p>
409             * Sets the preference value. The lower the number, the more preferred this
410             * property instance is compared with other properties in the same vCard of
411             * the same type. If a property doesn't have a preference value, then it is
412             * considered the least preferred.
413             * </p>
414             * 
415             * <p>
416             * In the vCard below, the address on the second row is the most preferred
417             * because it has the lowest PREF value.
418             * </p>
419             * 
420             * <pre>
421             * ADR;TYPE=work;PREF=2:
422             * ADR;TYPE=work;PREF=1:
423             * ADR;TYPE=home:
424             * </pre>
425             * 
426             * <p>
427             * Preference values must be numeric and must be between 1 and 100.
428             * </p>
429             * 
430             * <p>
431             * <b>Supported versions:</b> {@code 4.0}
432             * </p>
433             * @param pref the preference value or null to remove
434             * @throws IllegalArgumentException if the value is not between 1 and 100
435             */
436            public void setPref(Integer pref) {
437                    if (pref != null && (pref < 1 || pref > 100)) {
438                            throw new IllegalArgumentException("Preference value must be between 1 and 100 inclusive.");
439                    }
440                    String value = (pref == null) ? null : pref.toString();
441                    replace(PREF, value);
442            }

462             * NOTE;ALTID=1;LANGUAGE=es: �Hola, mundo!
463             * NOTE;ALTID=2;LANGUAGE=de: Meine Lieblingsfarbe ist blau.

510            /**
511             * <p>
512             * Gets the GEO parameter value. This is used to associate global
513             * positioning information with a vCard type. It can be used with the ADR
514             * type.
515             * </p>
516             * <p>
517             * <b>Supported versions:</b> {@code 4.0}
518             * </p>
519             * @throws IllegalStateException if the parameter value is malformed and
520             * cannot be parsed
521             * @return the latitude (index 0) and longitude (index 1) or null if not
522             * present or null if the parameter value was in an incorrect format
523             */
524            public double[] getGeo() {
525                    String value = first(GEO);
526                    if (value == null) {
527                            return null;
528                    }
529    
530                    try {
531                            GeoUri geoUri = GeoUri.parse(value);
532                            return new double[] { geoUri.getCoordA(), geoUri.getCoordB() };
533                    } catch (IllegalArgumentException e) {
534                            throw new IllegalStateException(GEO + " parameter value is malformed and could not be parsed. Retrieve its raw text value instead.", e);
535                    }
536            }

625            /**
626             * <p>
627             * Gets all PID parameter values. PIDs can exist on any property where
628             * multiple instances are allowed (such as EMAIL or ADR, but not N because
629             * only 1 instance of N is allowed).
630             * </p>
631             * <p>
632             * When used in conjunction with the CLIENTPIDMAP property, it allows an
633             * individual property instance to be uniquely identifiable. This feature is
634             * made use of when two different versions of the same vCard have to be
635             * merged together (called "synchronizing").
636             * </p>
637             * <p>
638             * <b>Supported versions:</b> {@code 4.0}
639             * </p>
640             * @throws IllegalStateException if the parameter value is malformed and
641             * cannot be parsed
642             * @return the PID values or empty set if there are none. Index 0 is the
643             * local ID and index 1 is the ID used to reference the CLIENTPIDMAP
644             * property. Index 0 will never be null, but index 1 may be null.
645             */
646            public List<Integer[]> getPids() {
647                    List<String> values = get(PID);
648                    List<Integer[]> pids = new ArrayList<Integer[]>(values.size());
649                    for (String value : values) {
650                            String split[] = value.split("\\.");
651                            try {
652                                    Integer localId = Integer.valueOf(split[0]);
653                                    Integer clientPidMapRef = (split.length > 1) ? Integer.valueOf(split[1]) : null;
654                                    pids.add(new Integer[] { localId, clientPidMapRef });
655                            } catch (NumberFormatException e) {
656                                    throw new IllegalStateException(PID + " parameter value is malformed and could not be parsed. Retrieve its raw text value instead.", e);
657                            }
658                    }
659                    return pids;
660            }

797                    if (index == null) {

805                    }

831            /**
832             * Checks this parameters list for data consistency problems or deviations
833             * from the spec. These problems will not prevent the vCard from being
834             * written to a data stream, but may prevent it from being parsed correctly
835             * by the consuming application.
836             * @param version the vCard version to validate against
837             * @return a list of warnings or an empty list if no problems were found
838             */
839            public List<String> validate(VCardVersion version) {
840                    List<String> warnings = new ArrayList<String>(0);
841    
842                    String nonStandard = "%s parameter has a non-standard value (\"%s\").  Standard values are: %s";
843                    String valueNotSupported = "%s parameter value (\"%s\") is not supported by version " + version.getVersion() + ".";
844    
845                    String value = first(CALSCALE);
846                    if (value != null && CalscaleParameter.find(value) == null) {
847                            warnings.add(String.format(nonStandard, CALSCALE, value, CalscaleParameter.all()));
848                    }
849    
850                    value = first(ENCODING);
851                    if (value != null) {
852                            EncodingParameter encoding = EncodingParameter.find(value);
853                            if (encoding == null) {
854                                    warnings.add(String.format(nonStandard, ENCODING, value, EncodingParameter.all()));
855                            } else if (!encoding.isSupported(version)) {
856                                    warnings.add(String.format(valueNotSupported, ENCODING, value));
857                            }
858                    }
859    
860                    value = first(VALUE);
861                    if (value != null) {
862                            VCardDataType dataType = VCardDataType.find(value);
863                            if (dataType == null) {
864                                    warnings.add(String.format(nonStandard, VALUE, value, VCardDataType.all()));
865                            } else if (!dataType.isSupported(version)) {
866                                    warnings.add(String.format(valueNotSupported, VALUE, value));
867                            }
868                    }
869    
870                    String malformed = "%s parameter has a malformed value (\"%s\").";
871    
872                    try {
873                            getGeo();
874                    } catch (IllegalStateException e) {
875                            warnings.add(String.format(malformed, GEO, first(GEO)));
876                    }
877    
878                    try {
879                            getIndex();
880                    } catch (IllegalStateException e) {
881                            warnings.add(String.format(malformed, INDEX, first(INDEX)));
882                    }
883    
884                    try {
885                            getPids();
886                    } catch (IllegalStateException e) {
887                            warnings.add(String.format(malformed, PID, first(PID)));
888                    }
889    
890                    try {
891                            getPref();
892                    } catch (IllegalStateException e) {
893                            warnings.add(String.format(malformed, PREF, first(PREF)));
894                    }
895    
896                    String paramNotSupported = "%s parameter is not supported by version " + version.getVersion() + ".";
897                    for (Map.Entry<String, Set<VCardVersion>> entry : supportedVersions.entrySet()) {
898                            String name = entry.getKey();
899                            value = first(name);
900                            if (value == null) {
901                                    continue;
902                            }
903    
904                            Set<VCardVersion> versions = entry.getValue();
905                            if (!versions.contains(version)) {
906                                    warnings.add(String.format(paramNotSupported, name));
907                            }
908                    }
909    
910                    return warnings;
911            }
