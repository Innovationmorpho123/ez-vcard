050    /**
051     * Represents a property whose value contains a date and/or a time (for example,
052     * {@link BirthdayType}).
053     * @author Michael Angstadt
054     */
055    public class DateOrTimeType extends VCardType implements HasAltId {
056            private String text;
057            private Date date;
058            private PartialDate partialDate;
059            private boolean dateHasTime;
060    
061            /**
062             * Creates a date-and-or-time property.
063             * @param typeName the name of the type (e.g. "BDAY")
064             */
065            public DateOrTimeType(String typeName) {
066                    super(typeName);
067            }
068    
069            /**
070             * Creates a date-and-or-time property.
071             * @param typeName the name of the type (e.g. "BDAY")
072             * @param date the date value
073             */
074            public DateOrTimeType(String typeName, Date date) {
075                    super(typeName);
076                    setDate(date, false);
077            }
078    
079            /**
080             * Creates a date-and-or-time property.
081             * @param typeName the name of the type (e.g. "BDAY")
082             * @param partialDate the date value
083             */
084            public DateOrTimeType(String typeName, PartialDate partialDate) {
085                    super(typeName);
086                    setPartialDate(partialDate);
087            }
088    
089            /**
090             * Gets the date value.
091             * @return the date value or null if not set
092             */
093            public Date getDate() {
094                    return date;
095            }
096    
097            /**
098             * Sets the value of this type to a complete date.
099             * @param date the date
100             * @param dateHasTime true if the date contains a time component, false if
101             * it's just a date
102             */
103            public void setDate(Date date, boolean dateHasTime) {
104                    this.date = date;
105                    this.dateHasTime = dateHasTime;
106                    text = null;
107                    partialDate = null;
108            }
109    
110            /**
111             * Gets the reduced accuracy or truncated date. This is only supported by
112             * vCard 4.0.
113             * @return the reduced accuracy or truncated date or null if not set
114             * @see "<a href="
115             * http://tools.ietf.org/html/rfc6350">RFC 6350</a> p.12-14 for examples"
116             */
117            public PartialDate getPartialDate() {
118                    return partialDate;
119            }
120    
121            /**
122             * <p>
123             * Sets the value of this type to a reduced accuracy or truncated date. This
124             * is only supported by vCard 4.0.
125             * </p>
126             * 
127             * <pre class="brush:java">
128             * BirthdayType bday = new BirthdayType();
129             * bday.setPartialDate(PartialDate.date(null, 4, 20)); //April 20
130             * </pre>
131             * @param partialDate the reduced accuracy or truncated date
132             * @see "<a href="
133             * http://tools.ietf.org/html/rfc6350">RFC 6350</a> p.12-14 for examples"
134             */
135            public void setPartialDate(PartialDate partialDate) {
136                    this.partialDate = partialDate;
137                    dateHasTime = partialDate.hasTimeComponent();
138                    text = null;
139                    date = null;
140            }
141    
142            /**
143             * Gets the text value of this type. This is only supported by vCard 4.0.
144             * @return the text value or null if not set
145             */
146            public String getText() {
147                    return text;
148            }
149    
150            /**
151             * Sets the value of this type to a text string. This is only supported by
152             * vCard 4.0.
153             * @param text the text value
154             */
155            public void setText(String text) {
156                    this.text = text;
157                    date = null;
158                    partialDate = null;
159                    dateHasTime = false;
160            }
161    
162            /**
163             * Determines whether the "date" or "partialDate" fields have a time
164             * component.
165             * @return true if the date has a time component, false if it's strictly a
166             * date, and false if a text value is defined
167             */
168            public boolean hasTime() {
169                    return dateHasTime;
170            }
171    
172            /**
173             * <p>
174             * Gets the type of calendar that is used for a date or date-time property
175             * value.
176             * </p>
177             * <p>
178             * <b>Supported versions:</b> {@code 4.0}
179             * </p>
180             * @return the type of calendar or null if not found
181             * @see VCardSubTypes#getCalscale
182             */
183            public CalscaleParameter getCalscale() {
184                    return subTypes.getCalscale();
185            }
186    
187            /**
188             * <p>
189             * Sets the type of calendar that is used for a date or date-time property
190             * value.
191             * </p>
192             * <p>
193             * <b>Supported versions:</b> {@code 4.0}
194             * </p>
195             * @param calscale the type of calendar or null to remove
196             * @see VCardSubTypes#setCalscale
197             */
198            public void setCalscale(CalscaleParameter calscale) {
199                    subTypes.setCalscale(calscale);
200            }
201    
202            //@Override
203            public String getAltId() {
204                    return subTypes.getAltId();
205            }
206    
207            //@Override
208            public void setAltId(String altId) {
209                    subTypes.setAltId(altId);
210            }
211    
212            @Override
213            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, CompatibilityMode compatibilityMode, VCard vcard) {
214                    VCardDataType dataType = null;
215                    if (text != null && version == VCardVersion.V4_0) {
216                            dataType = VCardDataType.TEXT;
217                    }
218                    copy.setValue(dataType);
219            }
220    
221            @Override
222            protected void doMarshalText(StringBuilder sb, VCardVersion version, CompatibilityMode compatibilityMode) {
223                    if (date != null) {
224                            ISOFormat format = dateHasTime ? ISOFormat.TIME_BASIC : ISOFormat.DATE_BASIC;
225                            sb.append(VCardDateFormatter.format(date, format));
226                            return;
227                    }
228    
229                    if (version == VCardVersion.V4_0) {
230                            if (text != null) {
231                                    sb.append(VCardStringUtils.escape(text));
232                                    return;
233                            }
234                            if (partialDate != null) {
235                                    sb.append(partialDate.toDateAndOrTime(false));
236                                    return;
237                            }
238                    }
239            }
240    
241            @Override
242            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
243                    value = VCardStringUtils.unescape(value);
244                    if (version == VCardVersion.V4_0 && subTypes.getValue() == VCardDataType.TEXT) {
245                            setText(value);
246                            return;
247                    }
248    
249                    parseDate(value, version, warnings);
250            }
251    
252            @Override
253            protected void doMarshalXml(XCardElement parent, CompatibilityMode compatibilityMode) {
254                    if (date != null) {
255                            ISOFormat format = dateHasTime ? ISOFormat.TIME_BASIC : ISOFormat.DATE_BASIC;
256                            String value = VCardDateFormatter.format(date, format);
257                            VCardDataType dataType = dateHasTime ? VCardDataType.DATE_TIME : VCardDataType.DATE;
258    
259                            parent.append(dataType, value);
260                            return;
261                    }
262    
263                    if (partialDate != null) {
264                            VCardDataType dataType;
265                            if (partialDate.hasTimeComponent() && partialDate.hasDateComponent()) {
266                                    dataType = VCardDataType.DATE_TIME;
267                            } else if (partialDate.hasTimeComponent()) {
268                                    dataType = VCardDataType.TIME;
269                            } else if (partialDate.hasDateComponent()) {
270                                    dataType = VCardDataType.DATE;
271                            } else {
272                                    dataType = VCardDataType.DATE_AND_OR_TIME;
273                            }
274    
275                            parent.append(dataType, partialDate.toDateAndOrTime(false));
276                            return;
277                    }
278    
279                    if (text != null) {
280                            parent.append(VCardDataType.TEXT, text);
281                            return;
282                    }
283    
284                    parent.append(VCardDataType.DATE_AND_OR_TIME, "");
285            }
286    
287            @Override
288            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
289                    String value = element.first(VCardDataType.DATE, VCardDataType.DATE_TIME, VCardDataType.DATE_AND_OR_TIME);
290                    if (value != null) {
291                            parseDate(value, element.version(), warnings);
292                            return;
293                    }
294    
295                    value = element.first(VCardDataType.TEXT);
296                    if (value != null) {
297                            setText(value);
298                            return;
299                    }
300    
301                    throw missingXmlElements(VCardDataType.DATE, VCardDataType.DATE_TIME, VCardDataType.DATE_AND_OR_TIME, VCardDataType.TEXT);
302            }
303    
304            @Override
305            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
306                    String value = null;
307                    if ("time".equals(element.tagName())) {
308                            String datetime = element.attr("datetime");
309                            if (datetime.length() > 0) {
310                                    value = datetime;
311                            }
312                    }
313                    if (value == null) {
314                            value = element.value();
315                    }
316                    parseDate(value, VCardVersion.V3_0, warnings);
317            }
318    
319            @Override
320            protected JCardValue doMarshalJson(VCardVersion version) {
321                    if (date != null) {
322                            VCardDataType dataType = dateHasTime ? VCardDataType.DATE_TIME : VCardDataType.DATE;
323    
324                            ISOFormat format = dateHasTime ? ISOFormat.TIME_EXTENDED : ISOFormat.DATE_EXTENDED;
325                            String value = VCardDateFormatter.format(date, format);
326    
327                            return JCardValue.single(dataType, value);
328                    }
329    
330                    if (partialDate != null) {
331                            VCardDataType dataType;
332                            if (partialDate.hasTimeComponent() && partialDate.hasDateComponent()) {
333                                    dataType = VCardDataType.DATE_TIME;
334                            } else if (partialDate.hasTimeComponent()) {
335                                    dataType = VCardDataType.TIME;
336                            } else if (partialDate.hasDateComponent()) {
337                                    dataType = VCardDataType.DATE;
338                            } else {
339                                    dataType = VCardDataType.DATE_AND_OR_TIME;
340                            }
341    
342                            String value = partialDate.toDateAndOrTime(true);
343    
344                            return JCardValue.single(dataType, value);
345                    }
346    
347                    if (text != null) {
348                            return JCardValue.single(VCardDataType.TEXT, text);
349                    }
350    
351                    return JCardValue.single(VCardDataType.DATE_AND_OR_TIME, "");
352            }
353    
354            @Override
355            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
356                    String valueStr = value.asSingle();
357                    if (value.getDataType() == VCardDataType.TEXT) {
358                            setText(valueStr);
359                            return;
360                    }
361    
362                    parseDate(valueStr, version, warnings);
363            }
364    
365            @Override
366            protected void _validate(List<String> warnings, VCardVersion version, VCard vcard) {
367                    if (date == null && partialDate == null && text == null) {
368                            warnings.add("Property has no value associated with it.");
369                    }
370    
371                    if (version == VCardVersion.V2_1 || version == VCardVersion.V3_0) {
372                            if (text != null) {
373                                    warnings.add("Text values are not supported in version " + version + ".");
374                            }
375                            if (partialDate != null) {
376                                    warnings.add("Reduced accuracy or truncated dates are not supported in version " + version + ".");
377                            }
378                    }
379            }
380    
381            private void parseDate(String value, VCardVersion version, List<String> warnings) {
382                    try {
383                            boolean hasTime = value.contains("T");
384                            setDate(VCardDateFormatter.parse(value), hasTime);
385                    } catch (IllegalArgumentException e) {
386                            if (version == VCardVersion.V2_1 || version == VCardVersion.V3_0) {
387                                    throw new CannotParseException("Date string could not be parsed.");
388                            }
389    
390                            try {
391                                    setPartialDate(new PartialDate(value));
392                            } catch (IllegalArgumentException e2) {
393                                    warnings.add("Date string could not be parsed.  Treating it as a text value.");
394                                    setText(value);
395                            }
396                    }
397            }
