035    /**
036     * Thrown during the marshalling/unmarshalling of a vCard property to show that
037     * the property should NOT be included in the marshalled vCard or the
038     * unmarshalled {@link VCard} object.
039     * @author Michael Angstadt
040     */
041    @SuppressWarnings("serial")
042    public class SkipMeException extends VCardException {
043            public SkipMeException() {
044                    super();
045            }
046    
047            /**
048             * @param msg the reason why this property cannot be marshalled/unmarshalled
049             */
050            public SkipMeException(String msg) {
051                    super(msg);
052            }
053    }
