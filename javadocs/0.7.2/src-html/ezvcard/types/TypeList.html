039    /**
040     * Contains listings of all the vCard type classes in this library.
041     * @author Michael Angstadt
042     */
043    public class TypeList {
044            /**
045             * Maps each type name to its corresponding type class. For example, "NOTE"
046             * is mapped to "NoteType".
047             */
048            private final static Map<String, Class<? extends VCardType>> nameToTypeClass;
049    
050            /**
051             * Maps each type class to the "VCard" class method that is used to add an
052             * instance of the type class to a "VCard" object. For example the
053             * "NoteType" class is mapped to the "VCard.addNote" method.
054             */
055            private final static Map<Class<? extends VCardType>, Method> typeClassToAddMethod;
056    
057            static {
058                    /*
059                     * This Map lists ALL of the type classes for the standard vCard types.
060                     * 
061                     * Key = The type class.
062                     * 
063                     * Value = The name of the method in the "VCard" class that's used for
064                     * adding instances of these type classes to an instance of the "VCard"
065                     * class. If null, it will guess the method name using the name of the
066                     * type class
067                     * 
068                     * Requirements for each type class:
069                     * 
070                     * 1. MUST have a public, no-arg constructor
071                     * 
072                     * 2. MUST have a public, static "NAME" field
073                     * 
074                     * Requirements for each "VCard" class method:
075                     * 
076                     * 1. MUST be public
077                     */
078                    Map<Class<? extends VCardType>, String> typeClasses = new HashMap<Class<? extends VCardType>, String>();
079                    typeClasses.put(AddressType.class, null);
080                    typeClasses.put(AgentType.class, null);
081                    typeClasses.put(AnniversaryType.class, null);
082                    typeClasses.put(BirthdayType.class, null);
083                    typeClasses.put(BirthplaceType.class, null);
084                    typeClasses.put(CalendarRequestUriType.class, null);
085                    typeClasses.put(CalendarUriType.class, null);
086                    typeClasses.put(CategoriesType.class, null);
087                    typeClasses.put(ClassificationType.class, null);
088                    typeClasses.put(ClientPidMapType.class, null);
089                    typeClasses.put(DeathdateType.class, null);
090                    typeClasses.put(DeathplaceType.class, null);
091                    typeClasses.put(EmailType.class, null);
092                    typeClasses.put(ExpertiseType.class, null);
093                    typeClasses.put(FbUrlType.class, null);
094                    typeClasses.put(FormattedNameType.class, null);
095                    typeClasses.put(GenderType.class, null);
096                    typeClasses.put(GeoType.class, null);
097                    typeClasses.put(HobbyType.class, null);
098                    typeClasses.put(ImppType.class, null);
099                    typeClasses.put(InterestType.class, null);
100                    typeClasses.put(KeyType.class, null);
101                    typeClasses.put(KindType.class, null);
102                    typeClasses.put(LabelType.class, "addOrphanedLabel");
103                    typeClasses.put(LanguageType.class, null);
104                    typeClasses.put(LogoType.class, null);
105                    typeClasses.put(MailerType.class, null);
106                    typeClasses.put(MemberType.class, null);
107                    typeClasses.put(NicknameType.class, null);
108                    typeClasses.put(NoteType.class, null);
109                    typeClasses.put(OrganizationType.class, null);
110                    typeClasses.put(OrgDirectoryType.class, null);
111                    typeClasses.put(PhotoType.class, null);
112                    typeClasses.put(ProdIdType.class, null);
113                    typeClasses.put(ProfileType.class, null);
114                    typeClasses.put(RelatedType.class, null);
115                    typeClasses.put(RevisionType.class, null);
116                    typeClasses.put(RoleType.class, null);
117                    typeClasses.put(SortStringType.class, null);
118                    typeClasses.put(SoundType.class, null);
119                    typeClasses.put(SourceDisplayTextType.class, null);
120                    typeClasses.put(SourceType.class, null);
121                    typeClasses.put(StructuredNameType.class, null);
122                    typeClasses.put(TelephoneType.class, "addTelephoneNumber");
123                    typeClasses.put(TimezoneType.class, null);
124                    typeClasses.put(TitleType.class, null);
125                    typeClasses.put(UidType.class, null);
126                    typeClasses.put(UrlType.class, null);
127                    typeClasses.put(XmlType.class, null);
128    
129                    Map<String, Class<? extends VCardType>> _nameToType = new HashMap<String, Class<? extends VCardType>>();
130                    Map<Class<? extends VCardType>, Method> _typeToMethod = new HashMap<Class<? extends VCardType>, Method>();
131                    for (Map.Entry<Class<? extends VCardType>, String> entry : typeClasses.entrySet()) {
132                            Class<? extends VCardType> clazz = entry.getKey();
133                            String methodName = entry.getValue();
134                            try {
135                                    String typeName = (String) clazz.getField("NAME").get(null);
136                                    typeName = typeName.toUpperCase();
137    
138                                    Method method;
139                                    if (methodName == null) {
140                                            //get the suffix of the method name using the class name
141                                            //e.g. "NoteType" --> "Note"
142                                            String simpleName = clazz.getSimpleName();
143                                            int pos = simpleName.lastIndexOf("Type");
144                                            String methodNameSuffix = simpleName.substring(0, pos);
145    
146                                            //try "addNote" first
147                                            methodName = "add" + methodNameSuffix;
148                                            try {
149                                                    method = VCard.class.getMethod(methodName, clazz);
150                                            } catch (Exception e) {
151                                                    //try "setNote"
152                                                    methodName = "set" + methodNameSuffix;
153                                                    method = VCard.class.getMethod(methodName, clazz);
154                                            }
155                                    } else {
156                                            method = VCard.class.getMethod(methodName, clazz);
157                                    }
158    
159                                    _nameToType.put(typeName, clazz);
160                                    _typeToMethod.put(clazz, method);
161                            } catch (Exception e) {
162                                    //a reflection problem occurred
163                                    //this exception should NEVER be thrown if the above rules are followed
164                                    throw new RuntimeException(e);
165                            }
166                    }
167    
168                    nameToTypeClass = Collections.unmodifiableMap(_nameToType);
169                    typeClassToAddMethod = Collections.unmodifiableMap(_typeToMethod);
170            }
171    
172            /**
173             * Gets the class that represents a vCard type.
174             * @param typeName the name of the vCard type (e.g. "ADR")
175             * @return the type class or null if not found
176             */
177            public static Class<? extends VCardType> getTypeClass(String typeName) {
178                    typeName = typeName.toUpperCase();
179                    return nameToTypeClass.get(typeName);
180            }
181    
182            /**
183             * Gets the class that represents a vCard type using its hCard type name.
184             * @param typeName the hCard name of the vCard type (e.g. "adr")
185             * @return the type class or null if not found
186             */
187            public static Class<? extends VCardType> getTypeClassByHCardTypeName(String typeName) {
188                    //TODO refactor
189                    if ("category".equalsIgnoreCase(typeName)) {
190                            return CategoriesType.class;
191                    }
192                    return getTypeClass(typeName);
193            }
194    
195            /**
196             * Gets the method of the {@link VCard} class that's used to add a vCard
197             * type object to the {@link VCard}.
198             * @param typeClass the type class
199             * @return the method or null if not found
200             */
201            public static Method getAddMethod(Class<? extends VCardType> typeClass) {
202                    return typeClassToAddMethod.get(typeClass);
203            }
204    }
