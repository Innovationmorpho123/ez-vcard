056/**
057 * <p>
058 * Writes {@link VCard} objects to a plain-text vCard data stream.
059 * </p>
060 * <p>
061 * <b>Example:</b>
062 * 
063 * <pre class="brush:java">
064 * VCard vcard1 = ...
065 * VCard vcard2 = ...
066 * 
067 * File file = new File("vcard.vcf");
068 * VCardWriter vcardWriter = new VCardWriter(file);
069 * vcardWriter.write(vcard1);
070 * vcardWriter.write(vcard2);
071 * vcardWriter.close();
072 * </pre>
073 * 
074 * </p>
075 * @author Michael Angstadt
076 */
077public class VCardWriter extends AbstractVCardWriter implements Closeable, Flushable {
078        private final VCardRawWriter writer;
079
080        /**
081         * Creates a vCard writer (writes v3.0 vCards and uses the standard folding
082         * scheme and newline sequence).
083         * @param out the output stream to write the vCard to
084         */
085        public VCardWriter(OutputStream out) {
086                this(new OutputStreamWriter(out));
087        }
088
089        /**
090         * Creates a vCard writer (uses the standard folding scheme and newline
091         * sequence).
092         * @param out the output stream to write the vCard to
093         * @param targetVersion the version that the vCards should conform to (if
094         * set to "4.0", vCards will be written in UTF-8 encoding)
095         */
096        public VCardWriter(OutputStream out, VCardVersion targetVersion) {
097                this((targetVersion == VCardVersion.V4_0) ? utf8Writer(out) : new OutputStreamWriter(out), targetVersion);
098        }
099

101         * Creates a vCard writer.
102         * @param out the output stream to write the vCard to
103         * @param targetVersion the version that the vCards should conform to (if
104         * set to "4.0", vCards will be written in UTF-8 encoding)
105         * @param foldingScheme the folding scheme to use or null not to fold at all
106         * @param newline the newline sequence to use
107         */
108        public VCardWriter(OutputStream out, VCardVersion targetVersion, FoldingScheme foldingScheme, String newline) {

263        /**
264         * Writes a vCard to the stream.
265         * @param vcard the vCard to write
266         * @throws IOException if there's a problem writing to the output stream
267         * @throws IllegalArgumentException if a scribe hasn't been registered for a
268         * custom property class (see: {@link #registerScribe registerScribe})
269         */
270        public void write(VCard vcard) throws IOException {
271                write(vcard, addProdId);
272        }
273
274        @SuppressWarnings({ "rawtypes", "unchecked" })
275        private void write(VCard vcard, boolean addProdId) throws IOException {
276                VCardVersion targetVersion = writer.getVersion();
277                List<VCardProperty> propertiesToAdd = prepare(vcard, targetVersion, addProdId);
278
279                writer.writeBeginComponent("VCARD");
280                writer.writeVersion();
281
282                for (VCardProperty property : propertiesToAdd) {
283                        VCardPropertyScribe scribe = index.getPropertyScribe(property);
284
285                        //marshal the value
286                        String value = null;
287                        VCard nestedVCard = null;
288                        try {
289                                value = scribe.writeText(property, targetVersion);
290                        } catch (SkipMeException e) {
291                                continue;
292                        } catch (EmbeddedVCardException e) {
293                                nestedVCard = e.getVCard();
294                        }
295
296                        //marshal the parameters
297                        VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
298
299                        //is the value a nested vCard?
300                        if (nestedVCard != null) {
301                                if (targetVersion == VCardVersion.V2_1) {
302                                        //write a nested vCard (2.1 style)
303                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
304                                        write(nestedVCard, false);
305                                } else {
306                                        //write an embedded vCard (3.0 style)
307                                        StringWriter sw = new StringWriter();
308                                        VCardWriter agentWriter = new VCardWriter(sw, targetVersion, null, "\n");
309                                        agentWriter.setAddProdId(false);
310                                        agentWriter.setVersionStrict(versionStrict);
311                                        try {
312                                                agentWriter.write(nestedVCard);
313                                        } catch (IOException e) {
314                                                //writing to a string
315                                        } finally {
316                                                IOUtils.closeQuietly(agentWriter);
317                                        }
318
319                                        String vCardStr = sw.toString();
320                                        vCardStr = VCardPropertyScribe.escape(vCardStr);
321                                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, vCardStr);
322                                }
323                                continue;
324                        }
325
326                        if (value != null) {
327                                //set the data type
328                                //only add a VALUE parameter if the data type is (1) not "unknown" and (2) different from the property's default data type
329                                VCardDataType dataType = scribe.dataType(property, targetVersion);
330                                if (dataType != null) {
331                                        VCardDataType defaultDataType = scribe.defaultDataType(targetVersion);
332                                        if (dataType != defaultDataType) {
333                                                if (defaultDataType == VCardDataType.DATE_AND_OR_TIME && (dataType == VCardDataType.DATE || dataType == VCardDataType.DATE_TIME || dataType == VCardDataType.TIME)) {
334                                                        //do not write VALUE if the default data type is "date-and-or-time" and the property's data type is time-based
335                                                } else {
336                                                        parameters.setValue(dataType);
337                                                }
338                                        }
339                                }
340
341                                writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
342                                continue;
343                        }
344                }
345
346                writer.writeEndComponent("VCARD");
347        }
