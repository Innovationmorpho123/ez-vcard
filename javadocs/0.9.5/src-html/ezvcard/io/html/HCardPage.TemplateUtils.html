057/**
058 * <p>
059 * Writes {@link VCard} objects to a templated HTML page (hCard format).
060 * </p>
061 * <p>
062 * <b>Example:</b>
063 * 
064 * <pre class="brush:java">
065 * VCard vcard1 = ...
066 * VCard vcard2 = ...
067 * 
068 * HCardPage page = new HCardPage();
069 * page.add(vcard1);
070 * page.add(vcard2);
071 * 
072 * File file = new File("hcard.html");
073 * page.write(file);
074 * </pre>
075 * 
076 * </p>
077 * @author Michael Angstadt
078 * @see <a
079 * href="http://microformats.org/wiki/hcard">http://microformats.org/wiki/hcard</a>
080 */
081public class HCardPage {
082        private static final Template template;
083        static {
084                Configuration cfg = new Configuration();
085                cfg.setClassForTemplateLoading(HCardPage.class, "");
086                cfg.setObjectWrapper(new DefaultObjectWrapper());
087                cfg.setWhitespaceStripping(true);
088                try {
089                        template = cfg.getTemplate("hcard-template.html");
090                } catch (IOException e) {
091                        //should never be thrown because it's always on the classpath
092                        throw new RuntimeException(e);
093                }
094        }
095
096        private final List<VCard> vcards = new ArrayList<VCard>();
097
098        /**
099         * Adds a vCard to the HTML page.
100         * @param vcard the vCard to add
101         */
102        public void add(VCard vcard) {
103                vcards.add(vcard);
104        }
105
106        /**
107         * Writes the HTML document to a string.
108         * @return the HTML document
109         */
110        public String write() {
111                StringWriter sw = new StringWriter();
112                try {
113                        write(sw);
114                } catch (IOException e) {
115                        //never thrown because we're writing to a string
116                }
117                return sw.toString();
118        }
119
120        /**
121         * Writes the HTML document to an output stream.
122         * @param out the output stream
123         * @throws IOException if there's a problem writing to the output stream
124         */
125        public void write(OutputStream out) throws IOException {
126                write(new OutputStreamWriter(out));
127        }
128
129        /**
130         * Writes the HTML document to a file.
131         * @param file the file
132         * @throws IOException if there's a problem writing to the file
133         */
134        public void write(File file) throws IOException {
135                FileWriter writer = null;
136                try {
137                        writer = new FileWriter(file);
138                        write(writer);
139                } finally {
140                        IOUtils.closeQuietly(writer);
141                }
142        }
143
144        /**
145         * Writes the HTML document to a writer.
146         * @param writer the writer
147         * @throws IOException if there's a problem writing to the writer
148         */
149        public void write(Writer writer) throws IOException {
150                Map<String, Object> map = new HashMap<String, Object>();
151                map.put("vcards", vcards);
152                map.put("utils", new TemplateUtils());
153                map.put("translucentBg", readImage("translucent-bg.png", ImageType.PNG));
154                map.put("noProfile", readImage("no-profile.png", ImageType.PNG));
155                map.put("ezVCardVersion", Ezvcard.VERSION);
156                map.put("ezVCardUrl", Ezvcard.URL);
157                map.put("scribeIndex", new ScribeIndex());
158                try {
159                        template.process(map, writer);
160                } catch (TemplateException e) {
161                        //this should never be thrown because we're always using the same template (it is hard-coded and cannot be changed by the user)
162                        throw new RuntimeException(e);
163                }
164                writer.flush();
165        }
166
167        /**
168         * Reads an image from the classpath.
169         * @param name the file name, relative to this class
170         * @param mediaType the media type of the image
171         * @return the image
172         * @throws IOException
173         */
174        private Photo readImage(String name, ImageType mediaType) throws IOException {
175                return new Photo(getClass().getResourceAsStream(name), mediaType);
176        }
177
178        /**
179         * Utility functions for the freemarker template.
180         */
181        public static class TemplateUtils {
182                private final Pattern newlineRegex = Pattern.compile("\\r\\n|\\r|\\n");
183
184                public String base64(String contentType, byte[] data) {
185                        return new DataUri(contentType, data).toString();
186                }
187
188                public String lineBreaks(String value) {
189                        return newlineRegex.matcher(value).replaceAll("<br />");
190                }
191        }
192}
