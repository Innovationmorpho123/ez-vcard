082/**
083* <p>
084* Represents an XML document that contains vCard objects ("xCard" standard).
085* This class can be used to read and write xCard documents.
086* </p>
087* <p>
088* <b>Examples:</b>
089* 
090* <pre class="brush:java">
091* String xml =
092* "&lt;vcards xmlns=\"urn:ietf:params:xml:ns:vcard-4.0\"&gt;" +
093*   "&lt;vcard&gt;" +
094*     "&lt;fn&gt;" +
095*       "&lt;text&gt;John Doe&lt;/text&gt;" +
096*     "&lt;/fn&gt;" +
097*     "&lt;n&gt;" +
098*       "&lt;surname&gt;Doe&lt;/surname&gt;" +
099*        "&lt;given&gt;Johnathan&lt;/given&gt;" +
100*        "&lt;additional&gt;Jonny&lt;/additional&gt;" +
101*        "&lt;additional&gt;John&lt;/additional&gt;" +
102*        "&lt;prefix&gt;Mr.&lt;/prefix&gt;" +
103*        "&lt;suffix /&gt;" +
104*      "&lt;/n&gt;" +
105*    "&lt;/vcard&gt;" +
106* "&lt;/vcards&gt;";
107*     
108* //parsing an existing xCard document
109* XCardDocument xcard = new XCardDocument(xml);
110* List&lt;VCard&gt; vcards = xcard.parseAll();
111* 
112* //creating an empty xCard document
113* XCardDocument xcard = new XCardDocument();
114* 
115* //VCard objects can be added at any time
116* VCard vcard = ...
117* xcard.add(vcard);
118* 
119* //retrieving the raw XML DOM
120* Document document = xcard.getDocument();
121* 
122* //call one of the "write()" methods to output the xCard document
123* File file = new File("johndoe.xml");
124* xcard.write(file);
125* </pre>
126* 
127* </p>
128* @author Michael Angstadt
129* @see <a href="http://tools.ietf.org/html/rfc6351">RFC 6351</a>
130*/
131//@formatter:on

166                Document document = XmlUtils.createDocument();

231        /**
232         * Wraps an existing XML DOM object.
233         * @param document the XML DOM that contains the xCard document
234         */
235        public XCardDocument(Document document) {
236                this.document = document;
237
238                XCardNamespaceContext nsContext = new XCardNamespaceContext(version4, "v");
239                XPath xpath = XPathFactory.newInstance().newXPath();
240                xpath.setNamespaceContext(nsContext);
241
242                try {
243                        //find the <vcards> element
244                        root = (Element) xpath.evaluate("//" + nsContext.getPrefix() + ":" + VCARDS.getLocalPart(), document, XPathConstants.NODE);
245                } catch (XPathExpressionException e) {
246                        //never thrown because xpath expression is hard coded
247                        throw new RuntimeException(e);
248                }
249        }

251        /**
252         * Gets the warnings from the last parse operation.
253         * @return the warnings (it is a "list of lists"--each parsed {@link VCard}
254         * object has its own warnings list)
255         * @see #parseAll
256         * @see #parseFirst
257         */
258        public List<List<String>> getParseWarnings() {
259                List<List<String>> warnings = new ArrayList<List<String>>(parseWarnings.size());
260                for (ParseWarnings parseWarning : parseWarnings) {
261                        warnings.add(parseWarning.copy());
262                }
263                return warnings;
264        }

266        /**
267         * Registers the data type of an experimental parameter. Experimental
268         * parameters use the "unknown" data type by default.
269         * @param parameterName the parameter name (e.g. "x-foo")
270         * @param dataType the data type or null to remove
271         */
272        public void registerParameterDataType(String parameterName, VCardDataType dataType) {
273                parameterName = parameterName.toLowerCase();
274                if (dataType == null) {
275                        parameterDataTypes.remove(parameterName);
276                } else {
277                        parameterDataTypes.put(parameterName, dataType);
278                }
279        }

289        /**
290         * Parses all the {@link VCard} objects from the xCard document.
291         * @return the vCard objects
292         */
293        public List<VCard> parseAll() {
294                parseWarnings.clear();
295
296                if (root == null) {
297                        return Collections.emptyList();
298                }
299
300                List<VCard> vcards = new ArrayList<VCard>();
301                for (Element vcardElement : getChildElements(root, VCARD)) {
302                        ParseWarnings warnings = new ParseWarnings();
303                        parseWarnings.add(warnings);
304                        VCard vcard = parseVCardElement(vcardElement, warnings);
305                        vcards.add(vcard);
306                }
307
308                return vcards;
309        }
310
311        /**
312         * Parses the first the {@link VCard} object from the xCard document.
313         * @return the vCard object
314         */
315        public VCard parseFirst() {
316                parseWarnings.clear();
317
318                if (root == null) {
319                        return null;
320                }
321
322                List<Element> vcardElements = getChildElements(root, VCARD);
323                if (vcardElements.isEmpty()) {
324                        return null;
325                }
326
327                ParseWarnings warnings = new ParseWarnings();
328                parseWarnings.add(warnings);
329                return parseVCardElement(vcardElements.get(0), warnings);
330        }
331
332        private VCard parseVCardElement(Element vcardElement, ParseWarnings warnings) {
333                VCard vcard = new VCard();
334                vcard.setVersion(version4);
335
336                List<Element> children = XmlUtils.toElementList(vcardElement.getChildNodes());
337                for (Element child : children) {
338                        if (GROUP.getNamespaceURI().equals(child.getNamespaceURI()) && GROUP.getLocalPart().equals(child.getLocalName())) {
339                                String group = child.getAttribute("name");
340                                if (group.length() == 0) {
341                                        group = null;
342                                }
343                                List<Element> grandChildren = XmlUtils.toElementList(child.getChildNodes());
344                                for (Element grandChild : grandChildren) {
345                                        parseAndAddElement(grandChild, group, vcard, warnings);
346                                }
347                                continue;
348                        }
349
350                        parseAndAddElement(child, null, vcard, warnings);
351                }
352
353                return vcard;
354        }

356        /**
357         * Parses a property element from the XML document and adds the property to
358         * the vCard.
359         * @param element the element to parse
360         * @param group the group name or null if the property does not belong to a
361         * group
362         * @param vcard the vCard object
363         * @param warningsBuf the list to add the warnings to
364         */
365        private void parseAndAddElement(Element element, String group, VCard vcard, ParseWarnings warnings) {
366                VCardParameters parameters = parseParameters(element);
367
368                VCardProperty property;
369                String propertyName = element.getLocalName();
370                String ns = element.getNamespaceURI();
371                QName qname = new QName(ns, propertyName);
372                VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(qname);
373                try {
374                        Result<? extends VCardProperty> result = scribe.parseXml(element, parameters);
375
376                        property = result.getProperty();
377                        property.setGroup(group);
378
379                        for (String warning : result.getWarnings()) {
380                                warnings.add(null, propertyName, warning);
381                        }
382                } catch (SkipMeException e) {
383                        warnings.add(null, propertyName, 22, e.getMessage());
384                        return;
385                } catch (CannotParseException e) {
386                        String xml = XmlUtils.toString(element);
387                        warnings.add(null, propertyName, 33, xml, e.getMessage());
388
389                        scribe = index.getPropertyScribe(Xml.class);
390                        Result<? extends VCardProperty> result = scribe.parseXml(element, parameters);
391                        property = result.getProperty();
392                        property.setGroup(group);
393                } catch (EmbeddedVCardException e) {
394                        warnings.add(null, propertyName, 34);
395                        return;
396                }
397
398                vcard.addProperty(property);
399        }

401        /**
402         * Parses the property parameters (aka "sub types").
403         * @param element the property's XML element
404         * @return the parsed parameters
405         */
406        private VCardParameters parseParameters(Element element) {
407                VCardParameters parameters = new VCardParameters();
408
409                List<Element> roots = XmlUtils.toElementList(element.getElementsByTagNameNS(PARAMETERS.getNamespaceURI(), PARAMETERS.getLocalPart()));
410                for (Element root : roots) { // foreach "<parameters>" element (there should only be 1 though)
411                        List<Element> parameterElements = XmlUtils.toElementList(root.getChildNodes());
412                        for (Element parameterElement : parameterElements) {
413                                String name = parameterElement.getLocalName().toUpperCase();
414                                List<Element> valueElements = XmlUtils.toElementList(parameterElement.getChildNodes());
415                                if (valueElements.isEmpty()) {
416                                        String value = parameterElement.getTextContent();
417                                        parameters.put(name, value);
418                                        continue;
419                                }
420
421                                for (Element valueElement : valueElements) {
422                                        String value = valueElement.getTextContent();
423                                        parameters.put(name, value);
424                                }
425                        }
426                }
427
428                return parameters;
429        }

525        /**
526         * Adds a vCard to the XML document.
527         * @param vcard the vCard to add
528         * @throws IllegalArgumentException if a scribe hasn't been registered for a
529         * custom property class (see: {@link #registerScribe})
530         */
531        public void add(VCard vcard) {
532                List<VCardProperty> propertiesToAdd = prepare(vcard, version4);
533                ListMultimap<String, VCardProperty> propertiesByGroup = new ListMultimap<String, VCardProperty>(); //group the types by group name (null = no group name)
534                for (VCardProperty property : propertiesToAdd) {
535                        propertiesByGroup.put(property.getGroup(), property);
536                }
537
538                //marshal each type object
539                Element vcardElement = createElement(VCARD);
540                for (Map.Entry<String, List<VCardProperty>> entry : propertiesByGroup) {
541                        String groupName = entry.getKey();
542                        Element parent;
543                        if (groupName != null) {
544                                Element groupElement = createElement(GROUP);
545                                groupElement.setAttribute("name", groupName);
546                                vcardElement.appendChild(groupElement);
547                                parent = groupElement;
548                        } else {
549                                parent = vcardElement;
550                        }
551
552                        for (VCardProperty property : entry.getValue()) {
553                                try {
554                                        Element propertyElement = marshalProperty(property, vcard);
555                                        parent.appendChild(propertyElement);
556                                } catch (SkipMeException e) {
557                                        //skip property
558                                } catch (EmbeddedVCardException e) {
559                                        //skip property
560                                }
561                        }
562                }
563
564                if (root == null) {
565                        root = createElement(VCARDS);
566                        Element documentRoot = XmlUtils.getRootElement(document);
567                        if (documentRoot == null) {
568                                document.appendChild(root);
569                        } else {
570                                documentRoot.appendChild(root);
571                        }
572                }
573                root.appendChild(vcardElement);
574        }

576        /**
577         * Marshals a type object to an XML element.
578         * @param type the type object to marshal
579         * @param vcard the vcard the type belongs to
580         * @return the XML element
581         */
582        @SuppressWarnings({ "rawtypes", "unchecked" })
583        private Element marshalProperty(VCardProperty type, VCard vcard) {
584                VCardPropertyScribe scribe = index.getPropertyScribe(type);
585                VCardParameters parameters = scribe.prepareParameters(type, version4, vcard);
586
587                QName qname = scribe.getQName();
588                Element propertyElement = createElement(qname);
589
590                //marshal the parameters
591                if (!parameters.isEmpty()) {
592                        Element parametersElement = marshalParameters(parameters);
593                        propertyElement.appendChild(parametersElement);
594                }
595
596                //marshal the value
597                scribe.writeXml(type, propertyElement);
598
599                return propertyElement;
600        }
601
602        private Element marshalParameters(VCardParameters parameters) {
603                Element parametersElement = createElement(PARAMETERS);
604
605                for (Map.Entry<String, List<String>> parameter : parameters) {
606                        String parameterName = parameter.getKey().toLowerCase();
607                        Element parameterElement = createElement(parameterName);
608
609                        for (String parameterValue : parameter.getValue()) {
610                                VCardDataType dataType = parameterDataTypes.get(parameterName);
611                                String dataTypeElementName = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
612                                Element dataTypeElement = createElement(dataTypeElementName);
613                                dataTypeElement.setTextContent(parameterValue);
614                                parameterElement.appendChild(dataTypeElement);
615                        }
616
617                        parametersElement.appendChild(parameterElement);
618                }
619
620                return parametersElement;
621        }
622
623        private List<Element> getChildElements(Element parent, QName qname) {
624                List<Element> elements = new ArrayList<Element>();
625                for (Element child : XmlUtils.toElementList(parent.getChildNodes())) {
626                        if (qname.getLocalPart().equals(child.getLocalName()) && qname.getNamespaceURI().equals(child.getNamespaceURI())) {
627                                elements.add(child);
628                        }
629                }
630                return elements;
631        }
