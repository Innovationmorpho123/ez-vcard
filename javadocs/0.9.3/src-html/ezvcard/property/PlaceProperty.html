041    /**
042     * Represents the location of a physical place.
043     * @author Michael Angstadt
044     * @see <a href="http://tools.ietf.org/html/rfc6474">RFC 6474</a>
045     */
046    public class PlaceProperty extends VCardProperty implements HasAltId {
047            protected GeoUri geoUri;
048            protected String uri;
049            protected String text;
050    
051            /**
052             * Creates a new place property.
053             */
054            public PlaceProperty() {
055                    //empty
056            }
057    
058            /**
059             * Creates a new place property.
060             * @param latitude the latitude coordinate of the place
061             * @param longitude the longitude coordinate of the place
062             */
063            public PlaceProperty(double latitude, double longitude) {
064                    setCoordinates(latitude, longitude);
065            }
066    
067            /**
068             * Creates a new place property.
069             * @param text a text value representing the place
070             */
071            public PlaceProperty(String text) {
072                    setText(text);
073            }
074    
075            @Override
076            public Set<VCardVersion> _supportedVersions() {
077                    return EnumSet.of(VCardVersion.V4_0);
078            }
079    
080            /**
081             * Gets the latitude of the location.
082             * @return the latitude or null if there is no geo information
083             */
084            public Double getLatitude() {
085                    return (geoUri == null) ? null : geoUri.getCoordA();
086            }
087    
088            /**
089             * Gets the longitude of the location.
090             * @return the longitude or null if there is no geo information
091             */
092            public Double getLongitude() {
093                    return (geoUri == null) ? null : geoUri.getCoordB();
094            }
095    
096            /**
097             * Gets the location's geo position.
098             * @return the geo position or null if there is no geo information
099             */
100            public GeoUri getGeoUri() {
101                    return geoUri;
102            }
103    
104            /**
105             * Sets the property's value to a set of geo coordinates.
106             * @param latitude the latitude
107             * @param longitude the longitude
108             */
109            public void setCoordinates(double latitude, double longitude) {
110                    setGeoUri(new GeoUri.Builder(latitude, longitude).build());
111            }
112    
113            /**
114             * Sets the property's value to a set of geo coordinates.
115             * @param geoUri the geo URI
116             */
117            public void setGeoUri(GeoUri geoUri) {
118                    this.geoUri = geoUri;
119                    uri = null;
120                    text = null;
121            }
122    
123            /**
124             * Gets the URI representing the location.
125             * @return the URI or null if no URI is set
126             */
127            public String getUri() {
128                    return uri;
129            }
130    
131            /**
132             * Sets the property's value to a URI.
133             * @param uri the URI
134             */
135            public void setUri(String uri) {
136                    this.uri = uri;
137                    geoUri = null;
138                    text = null;
139            }
140    
141            /**
142             * Gets the text value representing the location.
143             * @return the text value or null if no text value is set
144             */
145            public String getText() {
146                    return text;
147            }
148    
149            /**
150             * Sets the property's value to a text value.
151             * @param text the text
152             */
153            public void setText(String text) {
154                    this.text = text;
155                    geoUri = null;
156                    uri = null;
157            }
158    
159            //@Override
160            public String getAltId() {
161                    return parameters.getAltId();
162            }
163    
164            //@Override
165            public void setAltId(String altId) {
166                    parameters.setAltId(altId);
167            }
168    
169            @Override
170            public String getLanguage() {
171                    return super.getLanguage();
172            }
173    
174            @Override
175            public void setLanguage(String language) {
176                    super.setLanguage(language);
177            }
178    
179            @Override
180            protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
181                    if (uri == null && text == null && geoUri == null) {
182                            warnings.add(new Warning(8));
183                    }
184            }
185    }
