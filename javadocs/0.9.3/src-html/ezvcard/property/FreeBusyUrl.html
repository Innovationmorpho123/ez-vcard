038    /**
039     * <p>
040     * Defines a URL that shows when the person is free/busy on their calendar.
041     * </p>
042     * 
043     * <p>
044     * <b>Code sample</b>
045     * </p>
046     * 
047     * <pre class="brush:java">
048     * VCard vcard = new VCard();
049     * 
050     * FreeBusyUrl fburl = new FreeBusyUrl(&quot;http://www.example.com/freebusy/janedoe&quot;);
051     * vcard.addFbUrl(fburl);
052     * </pre>
053     * 
054     * <p>
055     * <b>Property name:</b> {@code FBURL}
056     * </p>
057     * <p>
058     * <b>Supported versions:</b> {@code 4.0}
059     * </p>
060     * @author Michael Angstadt
061     */
062    public class FreeBusyUrl extends UriProperty implements HasAltId {
063            /**
064             * Creates a free/busy URL property.
065             * @param uri the URI
066             */
067            public FreeBusyUrl(String uri) {
068                    super(uri);
069            }
070    
071            @Override
072            public Set<VCardVersion> _supportedVersions() {
073                    return EnumSet.of(VCardVersion.V4_0);
074            }
075    
076            /**
077             * Gets the MEDIATYPE parameter.
078             * <p>
079             * <b>Supported versions:</b> {@code 4.0}
080             * </p>
081             * @return the media type or null if not set
082             */
083            public String getMediaType() {
084                    return parameters.getMediaType();
085            }
086    
087            /**
088             * Sets the MEDIATYPE parameter.
089             * <p>
090             * <b>Supported versions:</b> {@code 4.0}
091             * </p>
092             * @param mediaType the media type or null to remove
093             */
094            public void setMediaType(String mediaType) {
095                    parameters.setMediaType(mediaType);
096            }
097    
098            @Override
099            public List<Integer[]> getPids() {
100                    return super.getPids();
101            }
102    
103            @Override
104            public void addPid(int localId, int clientPidMapRef) {
105                    super.addPid(localId, clientPidMapRef);
106            }
107    
108            @Override
109            public void removePids() {
110                    super.removePids();
111            }
112    
113            @Override
114            public Integer getPref() {
115                    return super.getPref();
116            }
117    
118            @Override
119            public void setPref(Integer pref) {
120                    super.setPref(pref);
121            }
122    
123            //@Override
124            public String getAltId() {
125                    return parameters.getAltId();
126            }
127    
128            //@Override
129            public void setAltId(String altId) {
130                    parameters.setAltId(altId);
131            }
132    
133            /**
134             * Gets the TYPE parameter.
135             * <p>
136             * <b>Supported versions:</b> {@code 4.0}
137             * </p>
138             * @return the TYPE value (typically, this will be either "work" or "home")
139             * or null if it doesn't exist
140             */
141            public String getType() {
142                    return parameters.getType();
143            }
144    
145            /**
146             * Sets the TYPE parameter.
147             * <p>
148             * <b>Supported versions:</b> {@code 4.0}
149             * </p>
150             * @param type the TYPE value (this should be either "work" or "home") or
151             * null to remove
152             */
153            public void setType(String type) {
154                    parameters.setType(type);
155            }
156    }
