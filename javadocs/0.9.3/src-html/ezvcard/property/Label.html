040    /**
041     * <p>
042     * Defines the exact text to put on the mailing label when sending snail mail to
043     * the person. Note that instances of this class should NEVER be added to a
044     * vCard! Instead, use the {@link Address#setLabel} method to assign a mailing
045     * label to an {@link Address} property.
046     * </p>
047     * 
048     * <p>
049     * <b>Version interoperability</b>
050     * </p>
051     * 
052     * <p>
053     * The label property is not supported in vCard version 4.0. Instead, labels are
054     * included as <i>parameters</i> to their corresponding {@link Address}
055     * properties. When marshalling a vCard, ez-vcard will use either the label
056     * property or the LABEL parameter, depending on the requested vCard version.
057     * </p>
058     * 
059     * <p>
060     * <b>Orphaned labels</b>
061     * </p>
062     * 
063     * <p>
064     * ez-vcard defines an "orphaned label" as a label property that could not be
065     * assigned to an address (a label is assigned to an address if its list of TYPE
066     * parameters is identical to the address's list of TYPE parameters). The
067     * {@link VCard#addOrphanedLabel} method can be used to add such labels to a
068     * vCard, but its use is strongly discouraged. The
069     * {@link VCard#getOrphanedLabels} method can be useful when parsing version 2.1
070     * or 3.0 vCards in order to retrieve any label properties that the parser could
071     * not assign to an address.
072     * </p>
073     * 
074     * <p>
075     * <b>Property name:</b> {@code LABEL}
076     * </p>
077     * <p>
078     * <b>Supported versions:</b> {@code 2.1, 3.0}
079     * </p>
080     * @author Michael Angstadt
081     */
082    public class Label extends TextProperty {
083            /**
084             * Creates a label property.
085             * @param label the label value
086             */
087            public Label(String label) {
088                    super(label);
089            }
090    
091            @Override
092            public Set<VCardVersion> _supportedVersions() {
093                    return EnumSet.of(VCardVersion.V2_1, VCardVersion.V3_0);
094            }
095    
096            /**
097             * Gets all the TYPE parameters.
098             * @return the TYPE parameters or empty set if there are none
099             */
100            public Set<AddressType> getTypes() {
101                    Set<String> values = parameters.getTypes();
102                    Set<AddressType> types = new HashSet<AddressType>(values.size());
103                    for (String value : values) {
104                            types.add(AddressType.get(value));
105                    }
106                    return types;
107            }
108    
109            /**
110             * Adds a TYPE parameter.
111             * @param type the TYPE parameter to add
112             */
113            public void addType(AddressType type) {
114                    parameters.addType(type.getValue());
115            }
116    
117            /**
118             * Removes a TYPE parameter.
119             * @param type the TYPE parameter to remove
120             */
121            public void removeType(AddressType type) {
122                    parameters.removeType(type.getValue());
123            }
124    
125            @Override
126            public String getLanguage() {
127                    return super.getLanguage();
128            }
129    
130            @Override
131            public void setLanguage(String language) {
132                    super.setLanguage(language);
133            }
134    }
