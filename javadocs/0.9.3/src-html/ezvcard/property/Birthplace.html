032    /**
033     * <p>
034     * Defines the location of the person's birth.
035     * </p>
036     * 
037     * <p>
038     * <b>Code sample (creating)</b>
039     * </p>
040     * 
041     * <pre class="brush:java">
042     * VCard vcard = new VCard();
043     * 
044     * //text
045     * Birthplace birthplace = new Birthplace(&quot;Maida Vale, London, United Kingdom&quot;);
046     * vcard.setBirthplace(birthplace);
047     * 
048     * //geo coordinates
049     * birthplace = new Birthplace(51.5274, -0.1899);
050     * vcard.setBirthplace(birthplace);
051     * 
052     * //URI
053     * birthplace = new Birthplace();
054     * birthplace.setUri(&quot;http://en.wikipedia.org/wiki/Maida_Vale&quot;);
055     * vcard.setBirthplace(birthplace);
056     * </pre>
057     * 
058     * <p>
059     * <b>Code sample (retrieving)</b>
060     * </p>
061     * 
062     * <pre class="brush:java">
063     * VCard vcard = ...
064     * Birthplace birthplace = vcard.getBirthplace();
065     * 
066     * String text = birthplace.getText();
067     * if (text != null){
068     *   //property value is plain text
069     * }
070     * 
071     * Double latitude = birthplace.getLatitude();
072     * Double longitude = birthplace.getLongitude();
073     * if (latitude != null){
074     *   //property value is a set of geo coordinates
075     * }
076     * 
077     * String uri = birthplace.getUri();
078     * if (uri != null){
079     *   //property value is a URI
080     * }
081     * </pre>
082     * 
083     * <p>
084     * <b>Property name:</b> {@code BIRTHPLACE}
085     * </p>
086     * <p>
087     * <b>Supported versions:</b> {@code 4.0}
088     * </p>
089     * @author Michael Angstadt
090     * @see <a href="http://tools.ietf.org/html/rfc6474">RFC 6474</a>
091     */
092    public class Birthplace extends PlaceProperty {
093            /**
094             * Creates a new birthplace property.
095             */
096            public Birthplace() {
097                    super();
098            }
099    
100            /**
101             * Creates a new birthplace property.
102             * @param latitude the latitude coordinate of the place
103             * @param longitude the longitude coordinate of the place
104             */
105            public Birthplace(double latitude, double longitude) {
106                    super(latitude, longitude);
107            }
108    
109            /**
110             * Creates a new birthplace property.
111             * @param text a text value representing the place
112             */
113            public Birthplace(String text) {
114                    super(text);
115            }
116    }
