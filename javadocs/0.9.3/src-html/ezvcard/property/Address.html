042    /**
043     * <p>
044     * Defines a mailing address.
045     * </p>
046     * 
047     * <p>
048     * <b>Code sample (creating)</b>
049     * </p>
050     * 
051     * <pre class="brush:java">
052     * VCard vcard = new VCard();
053     * 
054     * Address adr = new Address();
055     * adr.setStreetAddress(&quot;123 Main St.&quot;);
056     * adr.setLocality(&quot;Austin&quot;);
057     * adr.setRegion(&quot;TX&quot;);
058     * adr.setPostalCode(&quot;12345&quot;);
059     * adr.setCountry(&quot;USA&quot;);
060     * adr.addType(AddressType.WORK);
061     * 
062     * //optionally, set the text to print on the mailing label
063     * adr.setLabel(&quot;123 Main St.\nAustin, TX 12345\nUSA&quot;);
064     * 
065     * vcard.addAddress(adr);
066     * </pre>
067     * 
068     * <p>
069     * <b>Code sample (retrieving)</b>
070     * </p>
071     * 
072     * <pre class="brush:java">
073     * VCard vcard = ...
074     * for (Address adr : vcard.getAddresses()){
075     *   String street = adr.getStreetAddress();
076     *   String city = adr.getLocality();
077     *   //etc.
078     * }
079     * </pre>
080     * 
081     * <p>
082     * <b>Property name:</b> {@code ADR}
083     * </p>
084     * <p>
085     * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
086     * </p>
087     * @author Michael Angstadt
088     */

212            /**
213             * Gets all the TYPE parameters.
214             * @return the TYPE parameters or empty set if there are none
215             */
216            public Set<AddressType> getTypes() {
217                    Set<String> values = parameters.getTypes();
218                    Set<AddressType> types = new HashSet<AddressType>(values.size());
219                    for (String value : values) {
220                            types.add(AddressType.get(value));
221                    }
222                    return types;
223            }
224    
225            /**
226             * Adds a TYPE parameter.
227             * @param type the TYPE parameter to add
228             */
229            public void addType(AddressType type) {
230                    parameters.addType(type.getValue());
231            }
232    
233            /**
234             * Removes a TYPE parameter.
235             * @param type the TYPE parameter to remove
236             */
237            public void removeType(AddressType type) {
238                    parameters.removeType(type.getValue());
239            }

349            public void setTimezone(String timezone) {
350                    parameters.setTimezone(timezone);
351            }
352    
353            @Override
354            protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
355                    for (AddressType type : getTypes()) {
356                            if (type == AddressType.PREF) {
357                                    //ignore because it is converted to a PREF parameter for 4.0 vCards
358                                    continue;
359                            }
360    
361                            if (!type.isSupported(version)) {
362                                    warnings.add(new Warning(9, type.getValue()));
363                            }
364                    }
365            }
