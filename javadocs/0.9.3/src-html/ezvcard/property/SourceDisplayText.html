037    /**
038     * <p>
039     * Defines a textual representation of the {@link Source} property.
040     * </p>
041     * 
042     * <p>
043     * <b>Code sample</b>
044     * </p>
045     * 
046     * <pre class="brush:java">
047     * VCard vcard = new VCard();
048     * 
049     * SourceDisplayText name = new SourceDisplayText(&quot;Download the latest version of my vCard&quot;);
050     * vcard.setSourceDisplayText(name);
051     * </pre>
052     * 
053     * <p>
054     * <b>Property name:</b> {@code NAME}
055     * </p>
056     * <p>
057     * <b>Supported versions:</b> {@code 3.0}
058     * </p>
059     * @author Michael Angstadt
060     */
061    public class SourceDisplayText extends TextProperty {
062            /**
063             * Creates a source display text property.
064             * @param displayText the display text
065             */
066            public SourceDisplayText(String displayText) {
067                    super(displayText);
068            }
069    
070            @Override
071            public Set<VCardVersion> _supportedVersions() {
072                    return EnumSet.of(VCardVersion.V3_0);
073            }
074    }
