039    /**
040     * <p>
041     * Defines a set of latitude/longitude coordinates. There is no rule as to what
042     * these coordinates represent, but their meaning could vary depending on the
043     * value of the vCard's {@link Kind} property:
044     * </p>
045     * 
046     * <table class="simpleTable">
047     * <tr>
048     * <th>KIND value</th>
049     * <th>GEO meaning</th>
050     * </tr>
051     * <tr>
052     * <td>"individual"</td>
053     * <td>The location of the person's home or workplace.</td>
054     * </tr>
055     * <tr>
056     * <td>"group"</td>
057     * <td>The location of the group's meeting place.</td>
058     * </tr>
059     * <tr>
060     * <td>"org"</td>
061     * <td>The coordinates of the organization's headquarters.</td>
062     * </tr>
063     * <tr>
064     * <td>"location"</td>
065     * <td>The coordinates of the location itself.</td>
066     * </tr>
067     * </table>
068     * 
069     * <p>
070     * <b>Code sample</b>
071     * </p>
072     * 
073     * <pre class="brush:java">
074     * VCard vcard = new VCard();
075     * 
076     * Geo geo = new Geo(40.7127, -74.0059);
077     * vcard.setGeo(geo);
078     * </pre>
079     * 
080     * <p>
081     * <b>Property name:</b> {@code GEO}
082     * </p>
083     * <p>
084     * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
085     * </p>
086     * @author Michael Angstadt
087     */
088    public class Geo extends VCardProperty implements HasAltId {
089            private GeoUri uri;
090    
091            /**
092             * Creates a geo property.
093             * @param latitude the latitude
094             * @param longitude the longitude
095             */
096            public Geo(Double latitude, Double longitude) {
097                    this(new GeoUri.Builder(latitude, longitude).build());
098            }
099    
100            /**
101             * Creates a geo property.
102             * @param uri the geo URI
103             */
104            public Geo(GeoUri uri) {
105                    this.uri = uri;
106            }
107    
108            /**
109             * Gets the latitude.
110             * @return the latitude
111             */
112            public Double getLatitude() {
113                    return (uri == null) ? null : uri.getCoordA();
114            }
115    
116            /**
117             * Sets the latitude.
118             * @param latitude the latitude
119             */
120            public void setLatitude(Double latitude) {
121                    if (uri == null) {
122                            uri = new GeoUri.Builder(latitude, null).build();
123                    } else {
124                            uri = new GeoUri.Builder(uri).coordA(latitude).build();
125                    }
126            }
127    
128            /**
129             * Gets the longitude.
130             * @return the longitude
131             */
132            public Double getLongitude() {
133                    return (uri == null) ? null : uri.getCoordB();
134            }
135    
136            /**
137             * Sets the longitude.
138             * @param longitude the longitude
139             */
140            public void setLongitude(Double longitude) {
141                    if (uri == null) {
142                            uri = new GeoUri.Builder(null, longitude).build();
143                    } else {
144                            uri = new GeoUri.Builder(uri).coordB(longitude).build();
145                    }
146            }
147    
148            /**
149             * Gets the raw object used for storing the GEO information. This can be
150             * used to supplement the GEO value with additional information (such as
151             * altitude or level of accuracy). Geo URIs are only supported by vCard
152             * version 4.0. Only latitude and longitude values are used when marshalling
153             * to earlier vCard versions.
154             * @return the geo URI object or null if not set
155             * @see <a href="http://tools.ietf.org/html/rfc5870">RFC 5870</a>
156             */
157            public GeoUri getGeoUri() {
158                    return uri;
159            }
160    
161            /**
162             * Sets the raw object used for storing the GEO information. This can be
163             * used to supplement the GEO value with additional information (such as
164             * altitude or level of accuracy). Geo URIs are only supported by vCard
165             * version 4.0. Only latitude and longitude values are used when marshalling
166             * to earlier vCard versions.
167             * @param uri the geo URI object
168             * @see <a href="http://tools.ietf.org/html/rfc5870">RFC 5870</a>
169             */
170            public void setGeoUri(GeoUri uri) {
171                    this.uri = uri;
172            }
173    
174            /**
175             * Gets the TYPE parameter.
176             * <p>
177             * <b>Supported versions:</b> {@code 4.0}
178             * </p>
179             * @return the TYPE value (typically, this will be either "work" or "home")
180             * or null if it doesn't exist
181             */
182            public String getType() {
183                    return parameters.getType();
184            }
185    
186            /**
187             * Sets the TYPE parameter.
188             * <p>
189             * <b>Supported versions:</b> {@code 4.0}
190             * </p>
191             * @param type the TYPE value (this should be either "work" or "home") or
192             * null to remove
193             */
194            public void setType(String type) {
195                    parameters.setType(type);
196            }
197    
198            /**
199             * Gets the MEDIATYPE parameter.
200             * <p>
201             * <b>Supported versions:</b> {@code 4.0}
202             * </p>
203             * @return the media type or null if not set
204             */
205            public String getMediaType() {
206                    return parameters.getMediaType();
207            }
208    
209            /**
210             * Sets the MEDIATYPE parameter.
211             * <p>
212             * <b>Supported versions:</b> {@code 4.0}
213             * </p>
214             * @param mediaType the media type or null to remove
215             */
216            public void setMediaType(String mediaType) {
217                    parameters.setMediaType(mediaType);
218            }
219    
220            @Override
221            public List<Integer[]> getPids() {
222                    return super.getPids();
223            }
224    
225            @Override
226            public void addPid(int localId, int clientPidMapRef) {
227                    super.addPid(localId, clientPidMapRef);
228            }
229    
230            @Override
231            public void removePids() {
232                    super.removePids();
233            }
234    
235            @Override
236            public Integer getPref() {
237                    return super.getPref();
238            }
239    
240            @Override
241            public void setPref(Integer pref) {
242                    super.setPref(pref);
243            }
244    
245            //@Override
246            public String getAltId() {
247                    return parameters.getAltId();
248            }
249    
250            //@Override
251            public void setAltId(String altId) {
252                    parameters.setAltId(altId);
253            }
254    
255            @Override
256            protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
257                    if (getLatitude() == null) {
258                            warnings.add(new Warning(13));
259                    }
260                    if (getLongitude() == null) {
261                            warnings.add(new Warning(14));
262                    }
263            }
264    }
