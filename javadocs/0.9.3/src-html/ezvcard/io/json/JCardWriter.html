055    /**
056     * <p>
057     * Writes {@link VCard} objects to a JSON data stream (jCard format).
058     * </p>
059     * <p>
060     * <b>Example:</b>
061     * 
062     * <pre class="brush:java">
063     * VCard vcard1 = ...
064     * VCard vcard2 = ...
065     * 
066     * File file = new File("vcard.json");
067     * JCardWriter jcardWriter = new JCardWriter(file);
068     * jcardWriter.write(vcard1);
069     * jcardWriter.write(vcard2);
070     * jcardWriter.close(); //"close()" must be called in order to terminate the JSON object
071     * </pre>
072     * 
073     * </p>
074     * @author Michael Angstadt
075     * @see <a href="http://tools.ietf.org/html/rfc7095">RFC 7095</a>
076     */
077    public class JCardWriter implements Closeable, Flushable {
078            private ScribeIndex index = new ScribeIndex();
079            private final JCardRawWriter writer;
080            private final VCardVersion targetVersion = VCardVersion.V4_0;
081            private boolean addProdId = true;
082            private boolean versionStrict = true;
083    
084            /**
085             * Creates a writer that writes jCards to an output stream (UTF-8 encoding
086             * will be used).
087             * @param out the output stream to write the vCard to
088             */
089            public JCardWriter(OutputStream out) {
090                    this(utf8Writer(out));
091            }
092    
093            /**
094             * Creates a writer that writes jCards to an output stream (UTF-8 encoding
095             * will be used).
096             * @param out the output stream to write the vCard to
097             * @param wrapInArray true to enclose all written vCards in a JSON array,
098             * false not to
099             */
100            public JCardWriter(OutputStream out, boolean wrapInArray) {
101                    this(utf8Writer(out), wrapInArray);
102            }
103    
104            /**
105             * Creates a writer that writes jCards to a file (UTF-8 encoding will be
106             * used).
107             * @param file the file to write the vCard to
108             * @throws IOException if there's a problem opening the file
109             */
110            public JCardWriter(File file) throws IOException {
111                    this(utf8Writer(file));
112            }
113    
114            /**
115             * Creates a writer that writes jCards to a file (UTF-8 encoding will be
116             * used).
117             * @param file the file to write the vCard to
118             * @param wrapInArray true to enclose all written vCards in a JSON array,
119             * false not to
120             * @throws IOException if there's a problem opening the file
121             */
122            public JCardWriter(File file, boolean wrapInArray) throws IOException {
123                    this(utf8Writer(file), wrapInArray);
124            }
125    
126            /**
127             * Creates a writer that writes jCards to a writer.
128             * @param writer the writer to write the vCard to
129             */
130            public JCardWriter(Writer writer) {
131                    this(writer, false);
132            }
133    
134            /**
135             * Creates a writer that writes jCards to a writer.
136             * @param writer the writer to write the vCard to
137             * @param wrapInArray true to enclose all written vCards in a JSON array,
138             * false not to
139             */
140            public JCardWriter(Writer writer, boolean wrapInArray) {
141                    this.writer = new JCardRawWriter(writer, wrapInArray);
142            }
143    
144            /**
145             * Writes a vCard to the stream.
146             * @param vcard the vCard to write
147             * @throws IOException if there's a problem writing to the output stream
148             * @throws IllegalArgumentException if a scribe hasn't been registered for a
149             * custom property class (see: {@link #registerScribe})
150             */
151            @SuppressWarnings({ "rawtypes", "unchecked" })
152            public void write(VCard vcard) throws IOException {
153                    List<VCardProperty> propertiesToAdd = new ArrayList<VCardProperty>();
154                    for (VCardProperty property : vcard) {
155                            if (addProdId && property instanceof ProductId) {
156                                    //do not add the PRODID in the vCard if "addProdId" is true
157                                    continue;
158                            }
159    
160                            if (versionStrict && !property.getSupportedVersions().contains(targetVersion)) {
161                                    //do not add the property to the vCard if it is not supported by the target version
162                                    continue;
163                            }
164    
165                            //check for scribes before writing anything to the stream
166                            if (!index.hasPropertyScribe(property)) {
167                                    throw new IllegalArgumentException("No scribe found for property class \"" + property.getClass().getName() + "\".");
168                            }
169    
170                            propertiesToAdd.add(property);
171                    }
172    
173                    if (addProdId) {
174                            VCardProperty prodId = new ProductId("ezvcard " + Ezvcard.VERSION);
175                            propertiesToAdd.add(prodId);
176                    }
177    
178                    writer.writeStartVCard();
179                    writer.writeProperty("version", VCardDataType.TEXT, JCardValue.single(targetVersion.getVersion()));
180    
181                    for (VCardProperty property : propertiesToAdd) {
182                            VCardPropertyScribe scribe = index.getPropertyScribe(property);
183    
184                            //marshal the value
185                            JCardValue value;
186                            try {
187                                    value = scribe.writeJson(property);
188                            } catch (SkipMeException e) {
189                                    //property has requested not to be written
190                                    continue;
191                            } catch (EmbeddedVCardException e) {
192                                    //don't write because jCard does not support embedded vCards
193                                    continue;
194                            }
195    
196                            String group = property.getGroup();
197                            String name = scribe.getPropertyName().toLowerCase();
198                            VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
199                            VCardDataType dataType = scribe.dataType(property, targetVersion);
200    
201                            writer.writeProperty(group, name, parameters, dataType, value);
202                    }
203    
204                    writer.writeEndVCard();
205            }
206    
207            /**
208             * Gets whether or not a "PRODID" property will be added to each vCard,
209             * saying that the vCard was generated by this library.
210             * @return true if this property will be added, false if not (defaults to
211             * true)
212             */
213            public boolean isAddProdId() {
214                    return addProdId;
215            }
216    
217            /**
218             * Sets whether or not to add a "PRODID" property to each vCard, saying that
219             * the vCard was generated by this library.
220             * @param addProdId true to add this property, false not to (defaults to
221             * true)
222             */
223            public void setAddProdId(boolean addProdId) {
224                    this.addProdId = addProdId;
225            }
226    
227            /**
228             * Gets whether properties that do not support jCard (vCard version 4.0)
229             * will be excluded from the written vCard.
230             * @return true to exclude properties that do not support jCard, false to
231             * include them anyway (defaults to true)
232             */
233            public boolean isVersionStrict() {
234                    return versionStrict;
235            }
236    
237            /**
238             * Sets whether properties that do not support jCard (vCard version 4.0)
239             * will be excluded from the written vCard.
240             * @param versionStrict true to exclude properties that do not support
241             * jCard, false to include them anyway (defaults to true)
242             */
243            public void setVersionStrict(boolean versionStrict) {
244                    this.versionStrict = versionStrict;
245            }
246    
247            /**
248             * Gets whether or not the JSON will be pretty-printed.
249             * @return true if it will be pretty-printed, false if not (defaults to
250             * false)
251             */
252            public boolean isIndent() {
253                    return writer.isIndent();
254            }
255    
256            /**
257             * Sets whether or not to pretty-print the JSON.
258             * @param indent true to pretty-print it, false not to (defaults to false)
259             */
260            public void setIndent(boolean indent) {
261                    writer.setIndent(indent);
262            }
263    
264            /**
265             * <p>
266             * Registers a property scribe. This is the same as calling:
267             * </p>
268             * <p>
269             * {@code getScribeIndex().register(scribe)}
270             * </p>
271             * @param scribe the scribe to register
272             */
273            public void registerScribe(VCardPropertyScribe<? extends VCardProperty> scribe) {
274                    index.register(scribe);
275            }
276    
277            /**
278             * Gets the scribe index.
279             * @return the scribe index
280             */
281            public ScribeIndex getScribeIndex() {
282                    return index;
283            }
284    
285            /**
286             * Sets the scribe index.
287             * @param index the scribe index
288             */
289            public void setScribeIndex(ScribeIndex index) {
290                    this.index = index;
291            }
292    
293            /**
294             * Flushes the jCard data stream.
295             * @throws IOException if there's a problem flushing the stream
296             */
297            public void flush() throws IOException {
298                    writer.flush();
299            }
300    
301            /**
302             * Ends the jCard data stream, but does not close the underlying writer.
303             * @throws IOException if there's a problem closing the stream
304             */
305            public void closeJsonStream() throws IOException {
306                    writer.closeJsonStream();
307            }
308    
309            /**
310             * Ends the jCard data stream and closes the underlying writer.
311             * @throws IOException if there's a problem closing the stream
312             */
313            public void close() throws IOException {
314                    writer.close();
315            }
316    }
