038    /**
039     * <p>
040     * Manages a collection of property scribes (marshallers) to use when reading or
041     * writing a vCard. The same instance of this object can be re-used across
042     * multiple vCard reader/writer objects. This is useful if you have custom
043     * scribe classes defined, as it allows you to only define them once instead of
044     * each time a vCard reader/writer object is created.
045     * </p>
046     * <p>
047     * <b>Example:</b>
048     * 
049     * <pre class="brush:java">
050     * //init the index
051     * ScribeIndex index = new ScribeIndex();
052     * index.register(new CustomPropertyScribe());
053     * index.register(new AnotherCustomPropertyScribe());
054     * 
055     * //inject the ScribeIndex into a plain-text vCard reader class and read the vCard data stream
056     * VCardReader vcardReader = new VCardReader(...);
057     * vcardReader.setScribeIndex(index);
058     * List&lt;VCard&gt; vcards = new ArrayList&lt;VCard&gt;();
059     * VCard vcard;
060     * while ((vcards = vcardReader.readNext()) != null){
061     *   vcards.add(vcard);
062     * }
063     * vcardReader.close();
064     * 
065     * //inject the same ScribeIndex instance into a jCard writer and write the vCards
066     * JCardWriter jcardWriter = new JCardWriter(...);
067     * jcardWriter.setScribeIndex(index);
068     * for (VCard vcard : vcards){
069     *   jcardWriter.write(vcard);
070     * }
071     * jcardWriter.close();
072     * </pre>
073     * 
074     * </p>
075     * @author Michael Angstadt
076     */
077    public class ScribeIndex {
078            //define standard property marshallers
079            private static final Map<String, VCardPropertyScribe<? extends VCardProperty>> standardByName = new HashMap<String, VCardPropertyScribe<? extends VCardProperty>>();
080            private static final Map<Class<? extends VCardProperty>, VCardPropertyScribe<? extends VCardProperty>> standardByClass = new HashMap<Class<? extends VCardProperty>, VCardPropertyScribe<? extends VCardProperty>>();
081            private static final Map<QName, VCardPropertyScribe<? extends VCardProperty>> standardByQName = new HashMap<QName, VCardPropertyScribe<? extends VCardProperty>>();

141            private final Map<String, VCardPropertyScribe<? extends VCardProperty>> extendedByName = new HashMap<String, VCardPropertyScribe<? extends VCardProperty>>(0);
142            private final Map<Class<? extends VCardProperty>, VCardPropertyScribe<? extends VCardProperty>> extendedByClass = new HashMap<Class<? extends VCardProperty>, VCardPropertyScribe<? extends VCardProperty>>(0);
143            private final Map<QName, VCardPropertyScribe<? extends VCardProperty>> extendedByQName = new HashMap<QName, VCardPropertyScribe<? extends VCardProperty>>(0);
144    
145            /**
146             * Gets a property scribe by name.
147             * @param propertyName the property name (case-insensitive, e.g. "FN")
148             * @return the property scribe or null if not found
149             */
150            public VCardPropertyScribe<? extends VCardProperty> getPropertyScribe(String propertyName) {
151                    propertyName = propertyName.toUpperCase();
152    
153                    VCardPropertyScribe<? extends VCardProperty> marshaller = extendedByName.get(propertyName);
154                    if (marshaller != null) {
155                            return marshaller;
156                    }
157    
158                    return standardByName.get(propertyName);
159            }
160    
161            /**
162             * Determines if a scribe exists for a given property instance.
163             * @param property the property
164             * @return true if a scribe exists, false if not
165             */
166            public boolean hasPropertyScribe(VCardProperty property) {
167                    if (property instanceof RawProperty) {
168                            return true;
169                    }
170    
171                    return getPropertyScribe(property.getClass()) != null;
172            }
173    
174            /**
175             * Gets a property scribe by class.
176             * @param clazz the property class
177             * @return the property scribe or null if not found
178             */
179            public VCardPropertyScribe<? extends VCardProperty> getPropertyScribe(Class<? extends VCardProperty> clazz) {
180                    VCardPropertyScribe<? extends VCardProperty> marshaller = extendedByClass.get(clazz);
181                    if (marshaller != null) {
182                            return marshaller;
183                    }
184    
185                    return standardByClass.get(clazz);
186            }
187    
188            /**
189             * Gets the appropriate property scribe for a given property instance.
190             * @param property the property instance
191             * @return the property scribe or null if not found
192             */
193            public VCardPropertyScribe<? extends VCardProperty> getPropertyScribe(VCardProperty property) {
194                    if (property instanceof RawProperty) {
195                            RawProperty raw = (RawProperty) property;
196                            return new RawPropertyScribe(raw.getPropertyName());
197                    }
198    
199                    return getPropertyScribe(property.getClass());
200            }
201    
202            /**
203             * Gets a property scribe by XML local name and namespace.
204             * @param qname the XML local name and namespace
205             * @return the property scribe or a {@link XmlScribe} if not found
206             */
207            public VCardPropertyScribe<? extends VCardProperty> getPropertyScribe(QName qname) {
208                    VCardPropertyScribe<? extends VCardProperty> marshaller = extendedByQName.get(qname);
209                    if (marshaller != null) {
210                            return marshaller;
211                    }
212    
213                    marshaller = standardByQName.get(qname);
214                    if (marshaller != null) {
215                            return marshaller;
216                    }
217    
218                    if (VCardVersion.V4_0.getXmlNamespace().equals(qname.getNamespaceURI())) {
219                            return new RawPropertyScribe(qname.getLocalPart().toUpperCase());
220                    }
221    
222                    return getPropertyScribe(Xml.class);
223            }
224    
225            /**
226             * Registers a property scribe.
227             * @param scribe the scribe to register
228             */
229            public void register(VCardPropertyScribe<? extends VCardProperty> scribe) {
230                    extendedByName.put(scribe.getPropertyName().toUpperCase(), scribe);
231                    extendedByClass.put(scribe.getPropertyClass(), scribe);
232                    extendedByQName.put(scribe.getQName(), scribe);
233            }
234    
235            /**
236             * Unregisters a property scribe.
237             * @param scribe the scribe to unregister
238             */
239            public void unregister(VCardPropertyScribe<? extends VCardProperty> scribe) {
240                    extendedByName.remove(scribe.getPropertyName().toUpperCase());
241                    extendedByClass.remove(scribe.getPropertyClass());
242                    extendedByQName.remove(scribe.getQName());
243            }
244    
245            private static void registerStandard(VCardPropertyScribe<? extends VCardProperty> scribe) {
246                    standardByName.put(scribe.getPropertyName().toUpperCase(), scribe);
247                    standardByClass.put(scribe.getPropertyClass(), scribe);
248                    standardByQName.put(scribe.getQName(), scribe);
249            }
