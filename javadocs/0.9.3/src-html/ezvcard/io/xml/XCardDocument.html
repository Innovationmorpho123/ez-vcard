080    /**
081    * <p>
082    * Represents an XML document that contains vCard objects ("xCard" standard).
083    * This class can be used to read and write xCard documents.
084    * </p>
085    * <p>
086    * <b>Examples:</b>
087    * 
088    * <pre class="brush:java">
089    * String xml =
090    * "&lt;vcards xmlns=\"urn:ietf:params:xml:ns:vcard-4.0\"&gt;" +
091    *   "&lt;vcard&gt;" +
092    *     "&lt;fn&gt;" +
093    *       "&lt;text&gt;John Doe&lt;/text&gt;" +
094    *     "&lt;/fn&gt;" +
095    *     "&lt;n&gt;" +
096    *       "&lt;surname&gt;Doe&lt;/surname&gt;" +
097    *        "&lt;given&gt;Johnathan&lt;/given&gt;" +
098    *        "&lt;additional&gt;Jonny&lt;/additional&gt;" +
099    *        "&lt;additional&gt;John&lt;/additional&gt;" +
100    *        "&lt;prefix&gt;Mr.&lt;/prefix&gt;" +
101    *        "&lt;suffix /&gt;" +
102    *      "&lt;/n&gt;" +
103    *    "&lt;/vcard&gt;" +
104    * "&lt;/vcards&gt;";
105    *     
106    * //parsing an existing xCard document
107    * XCardDocument xcard = new XCardDocument(xml);
108    * List&lt;VCard&gt; vcards = xcard.parseAll();
109    * 
110    * //creating an empty xCard document
111    * XCardDocument xcard = new XCardDocument();
112    * 
113    * //VCard objects can be added at any time
114    * VCard vcard = ...
115    * xcard.add(vcard);
116    * 
117    * //retrieving the raw XML DOM
118    * Document document = xcard.getDocument();
119    * 
120    * //call one of the "write()" methods to output the xCard document
121    * File file = new File("johndoe.xml");
122    * xcard.write(file);
123    * </pre>
124    * 
125    * </p>
126    * @author Michael Angstadt
127    * @see <a href="http://tools.ietf.org/html/rfc6351">RFC 6351</a>
128    */
129    //@formatter:on

232            private void init(Document document) {
233                    XPath xpath = XPathFactory.newInstance().newXPath();
234                    xpath.setNamespaceContext(nsContext);
235    
236                    try {
237                            //find the <vcards> element
238                            String prefix = nsContext.getPrefix();
239                            root = (Element) xpath.evaluate("//" + prefix + ":vcards", document, XPathConstants.NODE);
240                    } catch (XPathExpressionException e) {
241                            //never thrown, xpath expression is hard coded
242                    }
243            }

348            /**
349             * Parses all the {@link VCard} objects from the xCard document.
350             * @return the vCard objects
351             */
352            public List<VCard> parseAll() {
353                    parseWarnings.clear();
354    
355                    if (root == null) {
356                            return Collections.emptyList();
357                    }
358    
359                    List<VCard> vcards = new ArrayList<VCard>();
360                    for (Element vcardElement : getVCardElements()) {
361                            List<String> warnings = new ArrayList<String>();
362                            parseWarnings.add(warnings);
363                            VCard vcard = parseVCardElement(vcardElement, warnings);
364                            vcards.add(vcard);
365                    }
366    
367                    return vcards;
368            }
369    
370            /**
371             * Parses the first the {@link VCard} object from the xCard document.
372             * @return the vCard object
373             */
374            public VCard parseFirst() {
375                    parseWarnings.clear();
376    
377                    if (root == null) {
378                            return null;
379                    }
380    
381                    List<Element> vcardElements = getVCardElements();
382                    if (vcardElements.isEmpty()) {
383                            return null;
384                    }
385    
386                    List<String> warnings = new ArrayList<String>();
387                    parseWarnings.add(warnings);
388                    return parseVCardElement(vcardElements.get(0), warnings);
389            }
390    
391            private VCard parseVCardElement(Element vcardElement, List<String> warnings) {
392                    VCard vcard = new VCard();
393                    vcard.setVersion(version4);
394    
395                    String ns = version4.getXmlNamespace();
396                    List<Element> children = XmlUtils.toElementList(vcardElement.getChildNodes());
397                    for (Element child : children) {
398                            if ("group".equals(child.getLocalName()) && ns.equals(child.getNamespaceURI())) {
399                                    String group = child.getAttribute("name");
400                                    if (group.length() == 0) {
401                                            group = null;
402                                    }
403                                    List<Element> grandChildren = XmlUtils.toElementList(child.getChildNodes());
404                                    for (Element grandChild : grandChildren) {
405                                            parseAndAddElement(grandChild, group, vcard, warnings);
406                                    }
407                            } else {
408                                    parseAndAddElement(child, null, vcard, warnings);
409                            }
410                    }
411    
412                    return vcard;
413            }

415            /**
416             * Parses a property element from the XML document and adds the property to
417             * the vCard.
418             * @param element the element to parse
419             * @param group the group name or null if the property does not belong to a
420             * group
421             * @param vcard the vCard object
422             * @param warningsBuf the list to add the warnings to
423             */
424            private void parseAndAddElement(Element element, String group, VCard vcard, List<String> warnings) {
425                    VCardParameters parameters = parseParameters(element);
426    
427                    VCardProperty property;
428                    String propertyName = element.getLocalName();
429                    String ns = element.getNamespaceURI();
430                    QName qname = new QName(ns, propertyName);
431                    VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(qname);
432                    try {
433                            Result<? extends VCardProperty> result = scribe.parseXml(element, parameters);
434    
435                            property = result.getProperty();
436                            property.setGroup(group);
437    
438                            for (String warning : result.getWarnings()) {
439                                    addWarning(propertyName, warning, warnings);
440                            }
441                    } catch (SkipMeException e) {
442                            addWarning(propertyName, warnings, 22, e.getMessage());
443                            return;
444                    } catch (CannotParseException e) {
445                            String xml = XmlUtils.toString(element);
446                            addWarning(propertyName, warnings, 33, xml, e.getMessage());
447    
448                            scribe = index.getPropertyScribe(Xml.class);
449                            Result<? extends VCardProperty> result = scribe.parseXml(element, parameters);
450                            property = result.getProperty();
451                            property.setGroup(group);
452                    } catch (EmbeddedVCardException e) {
453                            addWarning(propertyName, warnings, 34);
454                            return;
455                    }
456    
457                    vcard.addProperty(property);
458            }

460            /**
461             * Parses the property parameters (aka "sub types").
462             * @param element the property's XML element
463             * @return the parsed parameters
464             */
465            private VCardParameters parseParameters(Element element) {
466                    VCardParameters parameters = new VCardParameters();
467    
468                    List<Element> roots = XmlUtils.toElementList(element.getElementsByTagNameNS(version4.getXmlNamespace(), "parameters"));
469                    for (Element root : roots) { // foreach "<parameters>" element (there should only be 1 though)
470                            List<Element> parameterElements = XmlUtils.toElementList(root.getChildNodes());
471                            for (Element parameterElement : parameterElements) {
472                                    String name = parameterElement.getLocalName().toUpperCase();
473                                    List<Element> valueElements = XmlUtils.toElementList(parameterElement.getChildNodes());
474                                    if (valueElements.isEmpty()) {
475                                            String value = parameterElement.getTextContent();
476                                            parameters.put(name, value);
477                                            continue;
478                                    }
479    
480                                    for (Element valueElement : valueElements) {
481                                            String value = valueElement.getTextContent();
482                                            parameters.put(name, value);
483                                    }
484                            }
485                    }
486    
487                    return parameters;
488            }

584            /**
585             * Adds a vCard to the XML document.
586             * @param vcard the vCard to add
587             * @throws IllegalArgumentException if a scribe hasn't been registered for a
588             * custom property class (see: {@link #registerScribe})
589             */
590            public void add(VCard vcard) {
591                    ListMultimap<String, VCardProperty> propertiesToAdd = new ListMultimap<String, VCardProperty>(); //group the types by group name (null = no group name)
592                    for (VCardProperty property : vcard) {
593                            if (addProdId && property instanceof ProductId) {
594                                    //do not add the PRODID in the vCard if "addProdId" is true
595                                    continue;
596                            }
597    
598                            if (versionStrict && !property.getSupportedVersions().contains(version4)) {
599                                    //do not add the property to the vCard if it is not supported by the target version
600                                    continue;
601                            }
602    
603                            //check for scribes before writing anything to the stream
604                            if (!index.hasPropertyScribe(property)) {
605                                    throw new IllegalArgumentException("No scribe found for property class \"" + property.getClass().getName() + "\".");
606                            }
607    
608                            propertiesToAdd.put(property.getGroup(), property);
609                    }
610    
611                    //add a PRODID property saying it was generated by this library
612                    if (addProdId) {
613                            ProductId prodId = new ProductId("ez-vcard " + Ezvcard.VERSION);
614                            propertiesToAdd.put(prodId.getGroup(), prodId);
615                    }
616    
617                    //marshal each type object
618                    Element vcardElement = createElement("vcard");
619                    for (String groupName : propertiesToAdd.keySet()) {
620                            Element parent;
621                            if (groupName != null) {
622                                    Element groupElement = createElement("group");
623                                    groupElement.setAttribute("name", groupName);
624                                    vcardElement.appendChild(groupElement);
625                                    parent = groupElement;
626                            } else {
627                                    parent = vcardElement;
628                            }
629    
630                            for (VCardProperty property : propertiesToAdd.get(groupName)) {
631                                    try {
632                                            Element propertyElement = marshalProperty(property, vcard);
633                                            parent.appendChild(propertyElement);
634                                    } catch (SkipMeException e) {
635                                            //skip property
636                                    } catch (EmbeddedVCardException e) {
637                                            //skip property
638                                    }
639                            }
640                    }
641                    root.appendChild(vcardElement);
642            }

644            /**
645             * Marshals a type object to an XML element.
646             * @param type the type object to marshal
647             * @param vcard the vcard the type belongs to
648             * @return the XML element
649             */
650            @SuppressWarnings({ "rawtypes", "unchecked" })
651            private Element marshalProperty(VCardProperty type, VCard vcard) {
652                    VCardPropertyScribe scribe = index.getPropertyScribe(type);
653                    VCardParameters parameters = scribe.prepareParameters(type, version4, vcard);
654    
655                    QName qname = scribe.getQName();
656                    Element propertyElement = createElement(qname.getLocalPart(), qname.getNamespaceURI());
657    
658                    //marshal the parameters
659                    if (!parameters.isEmpty()) {
660                            Element parametersElement = marshalParameters(parameters);
661                            propertyElement.appendChild(parametersElement);
662                    }
663    
664                    //marshal the value
665                    scribe.writeXml(type, propertyElement);
666    
667                    return propertyElement;
668            }

670            private Element marshalParameters(VCardParameters parameters) {
671                    Element parametersElement = createElement("parameters");
672    
673                    for (Map.Entry<String, List<String>> parameter : parameters) {
674                            String parameterName = parameter.getKey().toLowerCase();
675                            Element parameterElement = createElement(parameterName);
676    
677                            for (String parameterValue : parameter.getValue()) {
678                                    VCardDataType dataType = parameterDataTypes.get(parameterName);
679                                    String dataTypeElementName = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
680                                    Element dataTypeElement = createElement(dataTypeElementName);
681                                    dataTypeElement.setTextContent(parameterValue);
682                                    parameterElement.appendChild(dataTypeElement);
683                            }
684    
685                            parametersElement.appendChild(parameterElement);
686                    }
687    
688                    return parametersElement;
689            }

691            private List<Element> getVCardElements() {
692                    return getChildElements(root, "vcard");
693            }

695            private List<Element> getChildElements(Element parent, String localName) {
696                    List<Element> elements = new ArrayList<Element>();
697                    for (Element child : XmlUtils.toElementList(parent.getChildNodes())) {
698                            if (localName.equals(child.getLocalName()) && version4.getXmlNamespace().equals(child.getNamespaceURI())) {
699                                    elements.add(child);
700                            }
701                    }
702                    return elements;
703            }
