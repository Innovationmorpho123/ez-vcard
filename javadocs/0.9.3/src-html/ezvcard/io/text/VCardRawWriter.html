070            /**
071             * The characters that are not valid in parameter values and that should be
072             * removed.
073             */
074            private static final Map<VCardVersion, BitSet> invalidParamValueChars;
075            static {
076                    BitSet controlChars = new BitSet(128);
077                    controlChars.set(0, 31);
078                    controlChars.set(127);
079                    controlChars.set('\t', false); //allow
080                    controlChars.set('\n', false); //allow
081                    controlChars.set('\r', false); //allow
082    
083                    Map<VCardVersion, BitSet> map = new HashMap<VCardVersion, BitSet>();
084    
085                    //2.1
086                    {
087                            BitSet bitSet = new BitSet(128);
088                            bitSet.or(controlChars);
089    
090                            bitSet.set(',');
091                            bitSet.set('.');
092                            bitSet.set(':');
093                            bitSet.set('=');
094                            bitSet.set('[');
095                            bitSet.set(']');
096    
097                            map.put(VCardVersion.V2_1, bitSet);
098                    }
099    
100                    //3.0, 4.0
101                    {
102                            BitSet bitSet = new BitSet(128);
103                            bitSet.or(controlChars);
104    
105                            map.put(VCardVersion.V3_0, bitSet);
106                            map.put(VCardVersion.V4_0, bitSet);
107                    }
108    
109                    invalidParamValueChars = Collections.unmodifiableMap(map);
110            }

139            /**
140             * Creates a vCard raw writer.
141             * @param writer the writer to the data stream
142             * @param version the vCard version to adhere to
143             * @param foldingScheme the folding scheme to use or null not to fold at all
144             * @param newline the newline sequence to use
145             */
146            public VCardRawWriter(Writer writer, VCardVersion version, FoldingScheme foldingScheme, String newline) {
147                    if (foldingScheme == null) {
148                            this.writer = new FoldedLineWriter(writer, null, "", newline);
149                    } else {
150                            this.writer = new FoldedLineWriter(writer, foldingScheme.getLineLength(), foldingScheme.getIndent(), newline);
151                    }
152                    this.version = version;
153                    this.foldingScheme = foldingScheme;
154                    this.newline = newline;
155            }

354            /**
355             * Writes a property to the vCard data stream.
356             * @param group the group or null if there is no group
357             * @param propertyName the property name (e.g. "FN")
358             * @param parameters the property parameters
359             * @param value the property value (will be converted to "quoted-printable"
360             * encoding if the {@link Encoding#QUOTED_PRINTABLE} parameter is set)
361             * @throws IllegalArgumentException if the group or property name contains
362             * invalid characters
363             * @throws IOException if there's an I/O problem
364             */
365            public void writeProperty(String group, String propertyName, VCardParameters parameters, String value) throws IOException {
366                    //validate the group name
367                    if (group != null && !propertyNameRegex.matcher(group).matches()) {
368                            throw new IllegalArgumentException("Group contains invalid characters.  Valid characters are letters, numbers, and hyphens: " + group);
369                    }
370    
371                    //validate the property name
372                    if (!propertyNameRegex.matcher(propertyName).matches()) {
373                            throw new IllegalArgumentException("Property name contains invalid characters.  Valid characters are letters, numbers, and hyphens: " + propertyName);
374                    }
375    
376                    value = sanitizeValue(parameters, value);
377    
378                    //determine if the property value must be encoded in quoted printable
379                    //and determine the charset to use when encoding to quoted-printable
380                    boolean quotedPrintable = (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE);
381                    Charset charset = null;
382                    if (quotedPrintable) {
383                            String charsetParam = parameters.getCharset();
384                            if (charsetParam == null) {
385                                    charset = Charset.forName("UTF-8");
386                            } else {
387                                    try {
388                                            charset = Charset.forName(charsetParam);
389                                    } catch (Throwable t) {
390                                            charset = Charset.forName("UTF-8");
391                                    }
392                            }
393                            parameters.setCharset(charset.name());
394                    }
395    
396                    //write the group
397                    if (group != null) {
398                            writer.append(group);
399                            writer.append('.');
400                    }
401    
402                    //write the property name
403                    writer.append(propertyName);
404    
405                    //write the parameters
406                    for (Map.Entry<String, List<String>> parameter : parameters) {
407                            String parameterName = parameter.getKey();
408                            List<String> parameterValues = parameter.getValue();
409                            if (parameterValues.isEmpty()) {
410                                    continue;
411                            }
412    
413                            if (version == VCardVersion.V2_1) {
414                                    boolean isTypeParameter = VCardParameters.TYPE.equalsIgnoreCase(parameterName);
415                                    for (String parameterValue : parameterValues) {
416                                            parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
417    
418                                            if (isTypeParameter) {
419                                                    //e.g. ADR;HOME;WORK:
420                                                    writer.append(';').append(parameterValue.toUpperCase());
421                                            } else {
422                                                    //e.g. ADR;FOO=bar;FOO=car:
423                                                    writer.append(';').append(parameterName).append('=').append(parameterValue);
424                                            }
425                                    }
426                                    continue;
427                            }
428    
429                            //e.g. ADR;TYPE=home,work,"another,value":
430                            boolean first = true;
431                            writer.append(';').append(parameterName).append('=');
432                            for (String parameterValue : parameterValues) {
433                                    if (!first) {
434                                            writer.append(',');
435                                    }
436    
437                                    parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
438    
439                                    //surround with double quotes if contains special chars
440                                    if (quoteMeRegex.matcher(parameterValue).matches()) {
441                                            writer.append('"');
442                                            writer.append(parameterValue);
443                                            writer.append('"');
444                                    } else {
445                                            writer.append(parameterValue);
446                                    }
447    
448                                    first = false;
449                            }
450                    }
451    
452                    writer.append(':');
453                    writer.append(value, quotedPrintable, charset);
454                    writer.append(newline);
455            }

457            /**
458             * Sanitizes a property value for safe inclusion in a vCard.
459             * @param parameters the parameters
460             * @param value the value to sanitize
461             * @return the sanitized value
462             */
463            private String sanitizeValue(VCardParameters parameters, String value) {
464                    if (value == null) {
465                            return "";
466                    }
467    
468                    if (version == VCardVersion.V2_1 && containsNewlines(value)) {
469                            //2.1 does not support the "\n" escape sequence (see "Delimiters" sub-section in section 2 of the specs)
470                            parameters.setEncoding(Encoding.QUOTED_PRINTABLE);
471                            return value;
472                    }
473    
474                    return escapeNewlines(value);
475            }

477            /**
478             * Removes or escapes all invalid characters in a parameter value.
479             * @param parameterValue the parameter value
480             * @param parameterName the parameter name
481             * @param propertyName the name of the property to which the parameter
482             * belongs
483             * @return the sanitized parameter value
484             */
485            private String sanitizeParameterValue(String parameterValue, String parameterName, String propertyName) {
486                    String modifiedValue = null;
487                    boolean valueChanged = false;
488    
489                    //Note: String reference comparisons ("==") are used because the Pattern class returns the same instance if the String wasn't changed
490    
491                    switch (version) {
492                    case V2_1:
493                            //remove invalid characters
494                            modifiedValue = removeInvalidParameterValueChars(parameterValue);
495    
496                            //replace newlines with spaces
497                            modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll(" ");
498    
499                            //check to see if value was changed
500                            valueChanged = (parameterValue != modifiedValue);
501    
502                            //escape backslashes
503                            modifiedValue = modifiedValue.replace("\\", "\\\\");
504    
505                            //escape semi-colons (see section 2)
506                            modifiedValue = modifiedValue.replace(";", "\\;");
507    
508                            break;
509    
510                    case V3_0:
511                            //remove invalid characters
512                            modifiedValue = removeInvalidParameterValueChars(parameterValue);
513    
514                            if (caretEncodingEnabled) {
515                                    valueChanged = (modifiedValue != parameterValue);
516    
517                                    //apply caret encoding
518                                    modifiedValue = applyCaretEncoding(modifiedValue);
519                            } else {
520                                    //replace double quotes with single quotes
521                                    modifiedValue = modifiedValue.replace('"', '\'');
522    
523                                    //replace newlines with spaces
524                                    modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll(" ");
525    
526                                    valueChanged = (modifiedValue != parameterValue);
527                            }
528    
529                            break;
530    
531                    case V4_0:
532                            //remove invalid characters
533                            modifiedValue = removeInvalidParameterValueChars(parameterValue);
534    
535                            if (caretEncodingEnabled) {
536                                    valueChanged = (modifiedValue != parameterValue);
537    
538                                    //apply caret encoding
539                                    modifiedValue = applyCaretEncoding(modifiedValue);
540                            } else {
541                                    //replace double quotes with single quotes
542                                    modifiedValue = modifiedValue.replace('"', '\'');
543    
544                                    valueChanged = (modifiedValue != parameterValue);
545    
546                                    //backslash-escape newlines (for the "LABEL" parameter)
547                                    modifiedValue = newlineRegex.matcher(modifiedValue).replaceAll("\\\\\\n");
548                            }
549    
550                            break;
551                    }
552    
553                    if (valueChanged && problemsListener != null) {
554                            problemsListener.onParameterValueChanged(propertyName, parameterName, parameterValue, modifiedValue);
555                    }
556    
557                    return modifiedValue;
558            }

560            /**
561             * Removes invalid characters from a parameter value.
562             * @param value the parameter value
563             * @return the sanitized parameter value
564             */
565            private String removeInvalidParameterValueChars(String value) {
566                    BitSet invalidChars = invalidParamValueChars.get(version);
567                    StringBuilder sb = null;
568    
569                    for (int i = 0; i < value.length(); i++) {
570                            char ch = value.charAt(i);
571                            if (invalidChars.get(ch)) {
572                                    if (sb == null) {
573                                            sb = new StringBuilder(value.length());
574                                            sb.append(value.substring(0, i));
575                                    }
576                                    continue;
577                            }
578    
579                            if (sb != null) {
580                                    sb.append(ch);
581                            }
582                    }
583    
584                    return (sb == null) ? value : sb.toString();
585            }

587            /**
588             * Applies circumflex accent encoding to a string.
589             * @param value the string
590             * @return the encoded string
591             */
592            private String applyCaretEncoding(String value) {
593                    value = value.replace("^", "^^");
594                    value = newlineRegex.matcher(value).replaceAll("^n");
595                    value = value.replace("\"", "^'");
596                    return value;
597            }

599            /**
600             * <p>
601             * Escapes all newline character sequences. The newline character sequences
602             * are:
603             * </p>
604             * <ul>
605             * <li>{@code \r\n}</li>
606             * <li>{@code \r}</li>
607             * <li>{@code \n}</li>
608             * </ul>
609             * @param text the text to escape
610             * @return the escaped text
611             */
612            private String escapeNewlines(String text) {
613                    return newlineRegex.matcher(text).replaceAll("\\\\n");
614            }
615    
616            /**
617             * <p>
618             * Determines if a string has at least one newline character sequence. The
619             * newline character sequences are:
620             * </p>
621             * <ul>
622             * <li>{@code \r\n}</li>
623             * <li>{@code \r}</li>
624             * <li>{@code \n}</li>
625             * </ul>
626             * @param text the text to escape
627             * @return the escaped text
628             */
629            private boolean containsNewlines(String text) {
630                    return newlineRegex.matcher(text).find();
631            }

649            /**
650             * A listener whose methods are invoked when non-critical issues occur
651             * during the writing process.
652             * @author Michael Angstadt
653             */
654            public static interface ProblemsListener {
655                    /**
656                     * Called when a parameter value is changed in a lossy way, due to it
657                     * containing invalid characters. If a character can be escaped (such as
658                     * the "^" character when caret encoding is enabled), then this does not
659                     * count as the parameter being modified because it can be decoded
660                     * without losing any information.
661                     * @param propertyName the name of the property to which the parameter
662                     * belongs
663                     * @param parameterName the parameter name
664                     * @param originalValue the original parameter value
665                     * @param modifiedValue the modified parameter value
666                     */
667                    void onParameterValueChanged(String propertyName, String parameterName, String originalValue, String modifiedValue);
668            }
