063    /**
064     * <p>
065     * Writes {@link VCard} objects to a plain-text vCard data stream.
066     * </p>
067     * <p>
068     * <b>Example:</b>
069     * 
070     * <pre class="brush:java">
071     * VCard vcard1 = ...
072     * VCard vcard2 = ...
073     * 
074     * File file = new File("vcard.vcf");
075     * VCardWriter vcardWriter = new VCardWriter(file);
076     * vcardWriter.write(vcard1);
077     * vcardWriter.write(vcard2);
078     * vcardWriter.close();
079     * </pre>
080     * 
081     * </p>
082     * @author Michael Angstadt
083     */
084    public class VCardWriter implements Closeable, Flushable {
085            private ScribeIndex index = new ScribeIndex();
086            private boolean addProdId = true;
087            private boolean versionStrict = true;
088            private final VCardRawWriter writer;
089    
090            /**
091             * Creates a writer that writes vCards to an output stream (writes v3.0
092             * vCards and uses the standard folding scheme and newline sequence).
093             * @param out the output stream to write the vCard to
094             */
095            public VCardWriter(OutputStream out) {
096                    this(new OutputStreamWriter(out));
097            }

100             * Creates a writer that writes vCards to an output stream (uses the
101             * standard folding scheme and newline sequence).
102             * @param out the output stream to write the vCard to
103             * @param targetVersion the version that the vCards should conform to (if
104             * set to "4.0", vCards will be written in UTF-8 encoding)
105             */
106            public VCardWriter(OutputStream out, VCardVersion targetVersion) {
107                    this((targetVersion == VCardVersion.V4_0) ? utf8Writer(out) : new OutputStreamWriter(out), targetVersion);
108            }

346            /**
347             * Writes a vCard to the stream.
348             * @param vcard the vCard to write
349             * @throws IOException if there's a problem writing to the output stream
350             * @throws IllegalArgumentException if a scribe hasn't been registered for a
351             * custom property class (see: {@link #registerScribe})
352             */
353            public void write(VCard vcard) throws IOException {
354                    write(vcard, addProdId);
355            }
356    
357            @SuppressWarnings({ "rawtypes", "unchecked" })
358            private void write(VCard vcard, boolean addProdId) throws IOException {
359                    VCardVersion targetVersion = writer.getVersion();
360    
361                    List<VCardProperty> propertiesToAdd = new ArrayList<VCardProperty>();
362                    for (VCardProperty property : vcard) {
363                            if (addProdId && property instanceof ProductId) {
364                                    //do not add the PRODID in the vCard if "addProdId" is true
365                                    continue;
366                            }
367    
368                            if (versionStrict && !property.getSupportedVersions().contains(targetVersion)) {
369                                    //do not add the property to the vCard if it is not supported by the target version
370                                    continue;
371                            }
372    
373                            //check for scribes before writing anything to the stream
374                            if (!index.hasPropertyScribe(property)) {
375                                    throw new IllegalArgumentException("No scribe found for property class \"" + property.getClass().getName() + "\".");
376                            }
377    
378                            propertiesToAdd.add(property);
379    
380                            //add LABEL types for each ADR type if the target version is 2.1 or 3.0
381                            if (property instanceof Address && (targetVersion == VCardVersion.V2_1 || targetVersion == VCardVersion.V3_0)) {
382                                    Address adr = (Address) property;
383                                    String labelStr = adr.getLabel();
384                                    if (labelStr != null) {
385                                            Label label = new Label(labelStr);
386                                            for (AddressType adrType : adr.getTypes()) {
387                                                    label.addType(adrType);
388                                            }
389                                            propertiesToAdd.add(label);
390                                    }
391                            }
392                    }
393    
394                    //add an extended type saying it was generated by this library
395                    if (addProdId) {
396                            VCardProperty property;
397                            if (targetVersion == VCardVersion.V2_1) {
398                                    property = new RawProperty("X-PRODID", "ezvcard " + Ezvcard.VERSION);
399                            } else {
400                                    property = new ProductId("ezvcard " + Ezvcard.VERSION);
401                            }
402                            propertiesToAdd.add(property);
403                    }
404    
405                    writer.writeBeginComponent("VCARD");
406                    writer.writeVersion();
407    
408                    for (VCardProperty property : propertiesToAdd) {
409                            VCardPropertyScribe scribe = index.getPropertyScribe(property);
410    
411                            //marshal the value
412                            String value = null;
413                            VCard nestedVCard = null;
414                            try {
415                                    value = scribe.writeText(property, targetVersion);
416                            } catch (SkipMeException e) {
417                                    continue;
418                            } catch (EmbeddedVCardException e) {
419                                    nestedVCard = e.getVCard();
420                            }
421    
422                            //marshal the parameters
423                            VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
424    
425                            //is the value a nested vCard?
426                            if (nestedVCard != null) {
427                                    if (targetVersion == VCardVersion.V2_1) {
428                                            //write a nested vCard (2.1 style)
429                                            writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
430                                            write(nestedVCard, false);
431                                    } else {
432                                            //write an embedded vCard (3.0 style)
433                                            StringWriter sw = new StringWriter();
434                                            VCardWriter agentWriter = new VCardWriter(sw, targetVersion, null, "\n");
435                                            agentWriter.setAddProdId(false);
436                                            agentWriter.setVersionStrict(versionStrict);
437                                            try {
438                                                    agentWriter.write(nestedVCard);
439                                            } catch (IOException e) {
440                                                    //writing to a string
441                                            } finally {
442                                                    IOUtils.closeQuietly(agentWriter);
443                                            }
444    
445                                            String vCardStr = sw.toString();
446                                            vCardStr = VCardPropertyScribe.escape(vCardStr);
447                                            writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, vCardStr);
448                                    }
449                                    continue;
450                            }
451    
452                            if (value != null) {
453                                    //set the data type
454                                    //only add a VALUE parameter if the data type is (1) not "unknown" and (2) different from the property's default data type
455                                    VCardDataType dataType = scribe.dataType(property, targetVersion);
456                                    if (dataType != null) {
457                                            VCardDataType defaultDataType = scribe.defaultDataType(targetVersion);
458                                            if (dataType != defaultDataType) {
459                                                    if (defaultDataType == VCardDataType.DATE_AND_OR_TIME && (dataType == VCardDataType.DATE || dataType == VCardDataType.DATE_TIME || dataType == VCardDataType.TIME)) {
460                                                            //do not write VALUE if the default data type is "date-and-or-time" and the property's data type is time-based
461                                                    } else {
462                                                            parameters.setValue(dataType);
463                                                    }
464                                            }
465                                    }
466    
467                                    writer.writeProperty(property.getGroup(), scribe.getPropertyName(), parameters, value);
468                                    continue;
469                            }
470                    }
471    
472                    writer.writeEndComponent("VCARD");
473            }
