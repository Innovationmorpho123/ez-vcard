042/**
043 * <p>
044 * Defines the timezone that the person lives/works in.
045 * </p>
046 * 
047 * <p>
048 * <b>Code sample</b>
049 * </p>
050 * 
051 * <pre class="brush:java">
052 * VCard vcard = new VCard();
053 * 
054 * Timezone tz = new Timezone(-5, 0, &quot;America/New_York&quot;);
055 * vcard.addTimezone(tz);
056 * 
057 * //using a Java &quot;TimeZone&quot; object
058 * java.util.TimeZone javaTz = java.util.TimeZone.getTimeZone(&quot;America/New_York&quot;);
059 * tz = new Timezone(javaTz);
060 * vcard.addTimezone(tz);
061 * </pre>
062 * 
063 * <p>
064 * <b>Property name:</b> {@code TZ}
065 * </p>
066 * <p>
067 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
068 * </p>
069 * 
070 * @author Michael Angstadt
071 */
072
073//@formatter:off
074/* 
075 * Parsing===================
076 * 
077 * vCard 2.1:
078 * Parse as UTC offset.  If invalid, throw CannotParseException.
079 * 
080 * vCard 3.0, hCard:
081 * VALUE=text:                  Treat as text
082 * No VALUE param:              Parse as UTC offset.  If invalid, add warning and treat as text.
083 * 
084 * vCard 4.0, jCard:
085 * VALUE=text:                  Treat as text
086 * VALUE=utc-offset:    Parse as UTC offset.  If invalid, throw CannotParseException
087 * VALUE=uri:                   Not going to support this, as there is no description of what a timezone URI looks like
088 * No VALUE param:              Parse as UTC offset.  If invalid, treat as text
089 * 
090 * xCard:
091 * text | utc-offset    | result
092 * no   | no                    | throw CannotParseException
093 * yes  | no                    | OK
094 * no   | yes                   | OK
095 * no   | invalid               | throw CannotParseException
096 * yes  | yes                   | Parse text
097 * yes  | invalid               | Parse text
098 * 
099 * Writing===================
100 * 
101 * vCard 2.1:
102 * text | utc-offset    | result
103 * no   | no                    | empty string (validation warning)
104 * no   | yes                   | Write UTC offset
105 * yes  | no                    | empty string (validation warning)
106 * yes  | yes                   | Write UTC offset
107 * 
108 * vCard 3.0:
109 * text | utc-offset    | result
110 * no   | no                    | empty string (validation warning)
111 * no   | yes                   | Write UTC offset
112 * yes  | no                    | Write text, add "VALUE=text" parameter
113 * yes  | yes                   | Write UTC offset
114 * 
115 * vCard 4.0, xCard, jCard:
116 * text | utc-offset    | result
117 * no   | no                    | empty string (validation warning)
118 * no   | yes                   | Write UTC offset, add "VALUE=utc-offset" parameter
119 * yes  | no                    | Write text
120 * yes  | yes                   | Write text
121 */
122//@formatter:on
123public class Timezone extends VCardProperty implements HasAltId {
124        private UtcOffset offset;
125        private String text;
126
127        /**
128         * Creates a timezone property.
129         * @param text a free-form string representing the timezone, preferably a
130         * timezone ID from the <a
131         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
132         * Database</a> (e.g. "America/New_York")
133         */
134        public Timezone(String text) {
135                this(null, text);
136        }
137
138        /**
139         * Creates a timezone property.
140         * @param hourOffset the hour component of the UTC offset (e.g. -5)
141         * @param minuteOffset the minute component of the UTC offset (e.g. 0)
142         */
143        public Timezone(Integer hourOffset, Integer minuteOffset) {
144                this(new UtcOffset(hourOffset, minuteOffset));
145        }
146
147        /**
148         * Creates a timezone property.
149         * @param offset the UTC offset
150         */
151        public Timezone(UtcOffset offset) {
152                this(offset, null);
153        }
154
155        /**
156         * Creates a timezone property.
157         * @param hourOffset the hour component of the UTC offset (e.g. -5)
158         * @param minuteOffset the minute component of the UTC offset (e.g. 0)
159         * @param text a free-form string representing the timezone, preferably a
160         * timezone ID from the <a
161         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
162         * Database</a> (e.g. "America/New_York")
163         */
164        public Timezone(Integer hourOffset, Integer minuteOffset, String text) {
165                this(new UtcOffset(hourOffset, minuteOffset), text);
166        }
167
168        /**
169         * Creates a timezone property.
170         * @param offset the UTC offset
171         * @param text a free-form string representing the timezone, preferably a
172         * timezone ID from the <a
173         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
174         * Database</a> (e.g. "America/New_York")
175         */
176        public Timezone(UtcOffset offset, String text) {
177                setOffset(offset);
178                setText(text);
179        }
180
181        /**
182         * Creates a timezone property.
183         * @param timezone the timezone
184         */
185        public Timezone(TimeZone timezone) {
186                this(UtcOffset.parse(timezone), timezone.getID());
187        }
188
189        /**
190         * Gets the hour component of the UTC offset.
191         * @return the hour component of the UTC offset or null if not set
192         */
193        public Integer getHourOffset() {
194                return (offset == null) ? null : offset.getHour();
195        }
196
197        /**
198         * Gets the minute component of the UTC offset.
199         * @return the minute component of the UTC offset or null if not set
200         */
201        public Integer getMinuteOffset() {
202                return (offset == null) ? null : offset.getMinute();
203        }
204
205        /**
206         * Gets the UTC offset.
207         * @return the UTC offset or null if not set
208         */
209        public UtcOffset getOffset() {
210                return offset;
211        }
212
213        /**
214         * Sets the UTC offset.
215         * @param hourOffset the hour offset (e.g. -5)
216         * @param minuteOffset the minute offset (e.g. 0)
217         */
218        public void setOffset(int hourOffset, int minuteOffset) {
219                setOffset(new UtcOffset(hourOffset, minuteOffset));
220        }
221
222        /**
223         * Sets the UTC offset.
224         * @param offset the UTC offset
225         */
226        public void setOffset(UtcOffset offset) {
227                this.offset = offset;
228        }
229
230        /**
231         * Gets the text portion of the timezone.
232         * @return the free-form string representing the timezone, such as a
233         * timezone ID from the <a
234         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
235         * Database</a> (e.g. "America/New_York")
236         */
237        public String getText() {
238                return text;
239        }
240
241        /**
242         * Sets the text portion of the timezone.
243         * @param text a free-form string representing the timezone, preferably a
244         * timezone ID from the <a
245         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
246         * Database</a> (e.g. "America/New_York")
247         */
248        public void setText(String text) {
249                this.text = text;
250        }
251
252        /**
253         * Creates a {@link java.util.TimeZone} representation of this class.
254         * @return a {@link TimeZone} object or null if this object contains no
255         * offset data
256         */
257        public TimeZone toTimeZone() {
258                if (text != null) {
259                        TimeZone timezone = VCardDateFormat.parseTimeZoneId(text);
260                        if (timezone != null) {
261                                return timezone;
262                        }
263                }
264
265                if (offset != null) {
266                        int rawHourOffset = offset.getHour() * 60 * 60 * 1000;
267                        int rawMinuteOffset = offset.getMinute() * 60 * 1000;
268                        if (rawHourOffset < 0) {
269                                rawMinuteOffset *= -1;
270                        }
271                        int rawOffset = rawHourOffset + rawMinuteOffset;
272
273                        String id = (text == null) ? "" : text;
274
275                        return new SimpleTimeZone(rawOffset, id);
276                }
277
278                return null;
279        }
280
281        /**
282         * Gets the TYPE parameter.
283         * <p>
284         * <b>Supported versions:</b> {@code 4.0}
285         * </p>
286         * @return the TYPE value (typically, this will be either "work" or "home")
287         * or null if it doesn't exist
288         */
289        public String getType() {
290                return parameters.getType();
291        }
292
293        /**
294         * Sets the TYPE parameter.
295         * <p>
296         * <b>Supported versions:</b> {@code 4.0}
297         * </p>
298         * @param type the TYPE value (this should be either "work" or "home") or
299         * null to remove
300         */
301        public void setType(String type) {
302                parameters.setType(type);
303        }
304
305        /**
306         * Gets the MEDIATYPE parameter.
307         * <p>
308         * <b>Supported versions:</b> {@code 4.0}
309         * </p>
310         * @return the media type or null if not set
311         */
312        public String getMediaType() {
313                return parameters.getMediaType();
314        }
315
316        /**
317         * Sets the MEDIATYPE parameter.
318         * <p>
319         * <b>Supported versions:</b> {@code 4.0}
320         * </p>
321         * @param mediaType the media type or null to remove
322         */
323        public void setMediaType(String mediaType) {
324                parameters.setMediaType(mediaType);
325        }

344                return super.getPref();

349                super.setPref(pref);

354                return parameters.getAltId();

359                parameters.setAltId(altId);
360        }
361
362        @Override
363        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
364                if (offset == null && text == null) {
365                        warnings.add(new Warning(8));
366                }
367                if (offset == null && version == VCardVersion.V2_1) {
368                        warnings.add(new Warning(20));
369                }
370                if (offset != null && (offset.getMinute() < 0 || offset.getMinute() > 59)) {
371                        warnings.add(new Warning(21));
372                }
373        }
