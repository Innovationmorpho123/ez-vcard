059/**
060 * <p>
061 * Parses {@link VCard} objects from a JSON data stream (jCard format).
062 * </p>
063 * <p>
064 * <b>Example:</b>
065 * 
066 * <pre class="brush:java">
067 * File file = new File("vcards.json");
068 * JCardReader jcardReader = new JCardReader(file);
069 * VCard vcard;
070 * while ((vcard = jcardReader.readNext()) != null){
071 *   ...
072 * }
073 * jcardReader.close();
074 * </pre>
075 * 
076 * </p>
077 * @author Michael Angstadt
078 * @see <a href="http://tools.ietf.org/html/rfc7095">RFC 7095</a>
079 */
080public class JCardReader implements Closeable {
081        private ScribeIndex index = new ScribeIndex();
082        private final ParseWarnings warnings = new ParseWarnings();
083        private final JCardRawReader reader;
084
085        /**
086         * Creates a reader that parses jCards from a JSON string.
087         * @param json the JSON string
088         */
089        public JCardReader(String json) {
090                this(new StringReader(json));
091        }
092
093        /**
094         * Creates a reader that parses jCards from an input stream.
095         * @param in the input stream to read the vCards from
096         */
097        public JCardReader(InputStream in) {
098                this(utf8Reader(in));
099        }
100
101        /**
102         * Creates a reader that parses jCards from a file.
103         * @param file the file to read the vCards from
104         * @throws FileNotFoundException if the file doesn't exist
105         */
106        public JCardReader(File file) throws FileNotFoundException {
107                this(utf8Reader(file));
108        }
109
110        /**
111         * Creates a reader that parses jCards from a reader.
112         * @param reader the reader to read the vCards from
113         */
114        public JCardReader(Reader reader) {
115                this.reader = new JCardRawReader(reader);
116        }
117
118        /**
119         * Reads the next vCard from the data stream.
120         * @return the next vCard or null if there are no more
121         * @throws IOException if there's a problem reading from the stream
122         */
123        public VCard readNext() throws IOException {
124                if (reader.eof()) {
125                        return null;
126                }
127
128                warnings.clear();
129
130                JCardDataStreamListenerImpl listener = new JCardDataStreamListenerImpl();
131                reader.readNext(listener);
132                VCard vcard = listener.vcard;
133                if (vcard != null && !listener.versionFound) {
134                        warnings.add(reader.getLineNum(), null, 29);
135                }
136                return vcard;
137        }
138
139        /**
140         * <p>
141         * Registers a property scribe. This is the same as calling:
142         * </p>
143         * <p>
144         * {@code getScribeIndex().register(scribe)}
145         * </p>
146         * @param scribe the scribe to register
147         */
148        public void registerScribe(VCardPropertyScribe<? extends VCardProperty> scribe) {
149                index.register(scribe);
150        }
151
152        /**
153         * Gets the scribe index.
154         * @return the scribe index
155         */
156        public ScribeIndex getScribeIndex() {
157                return index;
158        }
159
160        /**
161         * Sets the scribe index.
162         * @param index the scribe index
163         */
164        public void setScribeIndex(ScribeIndex index) {
165                this.index = index;
166        }
167
168        /**
169         * Gets the warnings from the last vCard that was unmarshalled. This list is
170         * reset every time a new vCard is read.
171         * @return the warnings or empty list if there were no warnings
172         */
173        public List<String> getWarnings() {
174                return warnings.copy();
175        }
176
177        public void close() throws IOException {
178                reader.close();
179        }
180
181        private class JCardDataStreamListenerImpl implements JCardDataStreamListener {
182                private VCard vcard = null;
183                private boolean versionFound = false;
184
185                public void beginVCard() {
186                        vcard = new VCard();
187                        vcard.setVersion(VCardVersion.V4_0);
188                }
189
190                public void readProperty(String group, String propertyName, VCardParameters parameters, VCardDataType dataType, JCardValue value) {
191                        if ("version".equalsIgnoreCase(propertyName)) {
192                                //don't unmarshal "version" because we don't treat it as a property
193                                versionFound = true;
194
195                                VCardVersion version = VCardVersion.valueOfByStr(value.asSingle());
196                                if (version != VCardVersion.V4_0) {
197                                        warnings.add(reader.getLineNum(), propertyName, 30);
198                                }
199                                return;
200                        }
201
202                        VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(propertyName);
203                        if (scribe == null) {
204                                scribe = new RawPropertyScribe(propertyName);
205                        }
206
207                        Result<? extends VCardProperty> result;
208                        try {
209                                result = scribe.parseJson(value, dataType, parameters);
210                                for (String warning : result.getWarnings()) {
211                                        warnings.add(reader.getLineNum(), warning, propertyName);
212                                }
213                        } catch (SkipMeException e) {
214                                warnings.add(reader.getLineNum(), propertyName, 22, e.getMessage());
215                                return;
216                        } catch (CannotParseException e) {
217                                scribe = new RawPropertyScribe(propertyName);
218                                result = scribe.parseJson(value, dataType, parameters);
219
220                                VCardProperty property = result.getProperty();
221                                String valueStr = ((RawProperty) property).getValue();
222                                warnings.add(reader.getLineNum(), propertyName, 25, valueStr, e.getMessage());
223                        } catch (EmbeddedVCardException e) {
224                                warnings.add(reader.getLineNum(), propertyName, 31);
225                                return;
226                        }
227
228                        VCardProperty property = result.getProperty();
229                        property.setGroup(group);
230                        vcard.addProperty(property);
231                }
232        }
233}
