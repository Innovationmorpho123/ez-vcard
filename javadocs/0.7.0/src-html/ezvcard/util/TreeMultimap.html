037/**
038 * A multimap that uses {@link Set} objects to store its values.
039 * 
040 * <p>
041 * The internal {@link Map} implementation is a {@link TreeMap} that uses
042 * {@link TreeSet} for its values.
043 * </p>
044 * @author Michael Angstadt
045 * @param <K> the key
046 * @param <V> the value
047 */
048public class TreeMultimap<K, V> {
049        private final Map<K, Set<V>> map = new TreeMap<K, Set<V>>();
050
051        public TreeMultimap() {
052        }
053
054        /**
055         * Copy constructor.
056         * @param orig the multimap to copy from
057         */
058        public TreeMultimap(TreeMultimap<K, V> orig) {
059                for (Map.Entry<K, Set<V>> entry : orig.map.entrySet()) {
060                        Set<V> values = new TreeSet<V>(entry.getValue());
061                        map.put(entry.getKey(), values);
062                }
063        }
064
065        /**
066         * Adds a value to the multimap.
067         * @param key the key
068         * @param value the value
069         */
070        public void put(K key, V value) {
071                Set<V> values = map.get(key);
072                if (values == null) {
073                        values = new TreeSet<V>();
074                        map.put(key, values);
075                }
076                values.add(value);
077        }
078
079        /**
080         * Gets the values associated with the key.
081         * @param key the key
082         * @return the set of values or empty set if the key doesn't exist
083         */
084        public Set<V> get(K key) {
085                Set<V> values = map.get(key);
086                if (values == null) {
087                        values = new TreeSet<V>();
088                }
089                return values;
090        }
091
092        /**
093         * Removes all the values associated with a key
094         * @param key the key to remove
095         * @return the removed values
096         */
097        public Set<V> remove(K key) {
098                return map.remove(key);
099        }
100
101        /**
102         * Removes a particular value.
103         * @param key the key
104         * @param value the value to remove
105         * @return true if the multimap contained the value, false if not
106         */
107        public boolean remove(K key, V value) {
108                Set<V> values = map.get(key);
109                if (values != null) {
110                        return values.remove(value);
111                }
112                return false;
113        }
114
115        /**
116         * Returns all the keys.
117         * @return all the keys
118         */
119        public Set<K> keySet() {
120                return map.keySet();
121        }
122
123        /**
124         * Determines if the multimap is empty or not.
125         * @return true if it's empty, false if not
126         */
127        public boolean isEmpty() {
128                return size() == 0;
129        }
130
131        /**
132         * Returns the number of key/value pairs.
133         * @return the number of key/value pairs
134         */
135        public int size() {
136                int size = 0;
137                for (Map.Entry<K, Set<V>> entry : map.entrySet()) {
138                        size += entry.getValue().size();
139                }
140                return size;
141        }
142
143        /**
144         * Gets the underlying {@link Map} object.
145         * @return the underlying {@link Map} object
146         */
147        public Map<K, Set<V>> getMap() {
148                return map;
149        }
150}
