032/**
033 * Represents the TYPE parameter of the RELATED type.
034 * <p>
035 * vCard versions: 4.0
036 * </p>
037 * @author Michael Angstadt
038 */
039public class RelatedTypeParameter extends TypeParameter {
040        public static final RelatedTypeParameter ACQUAINTANCE = new RelatedTypeParameter("acquaintance");
041        public static final RelatedTypeParameter AGENT = new RelatedTypeParameter("agent");
042        public static final RelatedTypeParameter CHILD = new RelatedTypeParameter("child");
043        public static final RelatedTypeParameter CO_RESIDENT = new RelatedTypeParameter("co-resident");
044        public static final RelatedTypeParameter CO_WORKER = new RelatedTypeParameter("co-worker");
045        public static final RelatedTypeParameter COLLEAGUE = new RelatedTypeParameter("colleague");
046        public static final RelatedTypeParameter CONTACT = new RelatedTypeParameter("contact");
047        public static final RelatedTypeParameter CRUSH = new RelatedTypeParameter("crush");
048        public static final RelatedTypeParameter DATE = new RelatedTypeParameter("date");
049        public static final RelatedTypeParameter EMERGENCY = new RelatedTypeParameter("emergency");
050        public static final RelatedTypeParameter FRIEND = new RelatedTypeParameter("friend");
051        public static final RelatedTypeParameter KIN = new RelatedTypeParameter("kin");
052        public static final RelatedTypeParameter ME = new RelatedTypeParameter("me");
053        public static final RelatedTypeParameter MET = new RelatedTypeParameter("met");
054        public static final RelatedTypeParameter MUSE = new RelatedTypeParameter("muse");
055        public static final RelatedTypeParameter NEIGHBOR = new RelatedTypeParameter("neighbor");
056        public static final RelatedTypeParameter PARENT = new RelatedTypeParameter("parent");
057        public static final RelatedTypeParameter SIBLING = new RelatedTypeParameter("sibling");
058        public static final RelatedTypeParameter SPOUSE = new RelatedTypeParameter("spouse");
059        public static final RelatedTypeParameter SWEETHEART = new RelatedTypeParameter("sweetheart");
060
061        /**
062         * Use of this constructor is discouraged and should only be used for
063         * defining non-standard TYPEs. Please use one of the predefined static
064         * objects.
065         * @param value the type value (e.g. "spouse")
066         */
067        public RelatedTypeParameter(String value) {
068                super(value);
069        }
070
071        /**
072         * Searches the static objects in this class for one that has a certain type
073         * value.
074         * @param value the type value to search for (e.g. "spouse")
075         * @return the object or null if not found
076         */
077        public static RelatedTypeParameter valueOf(String value) {
078                return findByValue(value, RelatedTypeParameter.class);
079        }
080}
