046/**
047 * An email address associated with a person.
048 * 
049 * <pre>
050 * VCard vcard = new VCard();
051 * 
052 * EmailType email = new EmailType(&quot;superdude55@hotmail.com&quot;);
053 * email.addType(EmailTypeParameter.HOME);
054 * vcard.addEmail(email);
055 * 
056 * email = new EmailType(&quot;doe.john@company.com&quot;);
057 * email.addType(EmailTypeParameter.WORK);
058 * email.setPref(1); //the most preferred email
059 * vcard.addEmail(email);
060 * </pre>
061 * 
062 * <p>
063 * vCard property name: EMAIL
064 * </p>
065 * <p>
066 * vCard versions: 2.1, 3.0, 4.0
067 * </p>
068 * @author Michael Angstadt
069 */
070public class EmailType extends MultiValuedTypeParameterType<EmailTypeParameter> {
071        public static final String NAME = "EMAIL";
072
073        private String value;
074
075        public EmailType() {
076                this(null);
077        }
078
079        /**
080         * @param email the email
081         */
082        public EmailType(String email) {
083                super(NAME);
084                setValue(email);
085        }
086
087        @Override
088        protected EmailTypeParameter buildTypeObj(String type) {
089                EmailTypeParameter param = EmailTypeParameter.valueOf(type);
090                if (param == null) {
091                        param = new EmailTypeParameter(type);
092                }
093                return param;
094        }
095
096        /**
097         * Gets the email address.
098         * @return the email address
099         */
100        public String getValue() {
101                return value;
102        }
103
104        /**
105         * Sets the email address
106         * @param email the email address
107         */
108        public void setValue(String email) {
109                this.value = email;
110        }
111
112        /**
113         * Gets all PID parameter values.
114         * <p>
115         * vCard versions: 4.0
116         * </p>
117         * @return the PID values or empty set if there are none
118         * @see VCardSubTypes#getPids
119         */
120        public Set<Integer[]> getPids() {
121                return subTypes.getPids();
122        }
123
124        /**
125         * Adds a PID value.
126         * <p>
127         * vCard versions: 4.0
128         * </p>
129         * @param localId the local ID
130         * @param clientPidMapRef the ID used to reference the property's globally
131         * unique identifier in the CLIENTPIDMAP property.
132         * @see VCardSubTypes#addPid(int, int)
133         */
134        public void addPid(int localId, int clientPidMapRef) {
135                subTypes.addPid(localId, clientPidMapRef);
136        }
137
138        /**
139         * Removes all PID values.
140         * <p>
141         * vCard versions: 4.0
142         * </p>
143         * @see VCardSubTypes#removePids
144         */
145        public void removePids() {
146                subTypes.removePids();
147        }
148
149        /**
150         * Gets the preference value.
151         * <p>
152         * vCard versions: 4.0
153         * </p>
154         * @return the preference value or null if it doesn't exist
155         * @see VCardSubTypes#getPref
156         */
157        public Integer getPref() {
158                return subTypes.getPref();
159        }
160
161        /**
162         * Sets the preference value.
163         * <p>
164         * vCard versions: 4.0
165         * </p>
166         * @param pref the preference value or null to remove
167         * @see VCardSubTypes#setPref
168         */
169        public void setPref(Integer pref) {
170                subTypes.setPref(pref);
171        }
172
173        /**
174         * Gets the ALTID.
175         * <p>
176         * vCard versions: 4.0
177         * </p>
178         * @return the ALTID or null if it doesn't exist
179         * @see VCardSubTypes#getAltId
180         */
181        public String getAltId() {
182                return subTypes.getAltId();
183        }
184
185        /**
186         * Sets the ALTID.
187         * <p>
188         * vCard versions: 4.0
189         * </p>
190         * @param altId the ALTID or null to remove
191         * @see VCardSubTypes#setAltId
192         */
193        public void setAltId(String altId) {
194                subTypes.setAltId(altId);
195        }
196
197        @Override
198        protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
199                //replace "TYPE=pref" with "PREF=1"
200                if (version == VCardVersion.V4_0) {
201                        if (getTypes().contains(EmailTypeParameter.PREF)) {
202                                copy.removeType(EmailTypeParameter.PREF.getValue());
203                                copy.setPref(1);
204                        }
205                } else {
206                        copy.setPref(null);
207
208                        //find the EMAIL with the lowest PREF value in the vCard
209                        EmailType mostPreferred = null;
210                        for (EmailType email : vcard.getEmails()) {
211                                Integer pref = email.getPref();
212                                if (pref != null) {
213                                        if (mostPreferred == null || pref < mostPreferred.getPref()) {
214                                                mostPreferred = email;
215                                        }
216                                }
217                        }
218                        if (this == mostPreferred) {
219                                copy.addType(EmailTypeParameter.PREF.getValue());
220                        }
221                }
222        }
223
224        @Override
225        protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
226                sb.append(VCardStringUtils.escape(value));
227        }
228
229        @Override
230        protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
231                this.value = VCardStringUtils.unescape(value);
232        }
233
234        @Override
235        protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
236                parent.appendText(getValue());
237        }
238
239        @Override
240        protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
241                setValue(element.getText());
242        }
243
244        @Override
245        protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
246                List<String> types = element.types();
247                for (String type : types) {
248                        subTypes.addType(type);
249                }
250
251                //check to see if the email address is within in "mailto:" link
252                String email = null;
253                String href = element.attr("href");
254                if (href.length() > 0) {
255                        Pattern p = Pattern.compile("^mailto:(.*)$", Pattern.CASE_INSENSITIVE);
256                        Matcher m = p.matcher(href);
257                        if (m.find()) {
258                                email = m.group(1);
259                        }
260                }
261                if (email == null) {
262                        email = element.value();
263                }
264                setValue(email);
265        }
