050/**
051 * Contains the timezone that the person lives/works in.
052 * 
053 * <pre>
054 * VCard vcard = new VCard();
055 * TimezoneType tz = new TimezoneType(-5, 0, &quot;America/New_York&quot;);
056 * vcard.addTimezone(tz);
057 * </pre>
058 * 
059 * <p>
060 * vCard property name: TZ
061 * </p>
062 * <p>
063 * vCard versions: 2.1, 3.0, 4.0
064 * </p>
065 * 
066 * @author Michael Angstadt
067 */
068public class TimezoneType extends VCardType {
069        public static final String NAME = "TZ";
070
071        private Integer hourOffset;
072        private Integer minuteOffset;
073        private String text;
074
075        public TimezoneType() {
076                super(NAME);
077        }
078
079        /**
080         * This is the recommended constructor for version 4.0 vCards.
081         * @param text string representing the timezone from the <a
082         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
083         * Database</a> (e.g. "America/New_York")
084         */
085        public TimezoneType(String text) {
086                this(null, null, text);
087        }
088
089        /**
090         * This is the recommended constructor for version 2.1 and 3.0 vCards.
091         * @param hourOffset the hour offset
092         * @param minuteOffset the minute offset
093         */
094        public TimezoneType(Integer hourOffset, Integer minuteOffset) {
095                this(hourOffset, minuteOffset, null);
096        }
097
098        /**
099         * This constructor can be used for all vCard versions.
100         * @param hourOffset the hour offset
101         * @param minuteOffset the minute offset
102         * @param text can be anything, but should be a string representing the
103         * timezone from the <a
104         * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
105         * Database</a> (e.g. "America/New_York")
106         */
107        public TimezoneType(Integer hourOffset, Integer minuteOffset, String text) {
108                super(NAME);
109                setHourOffset(hourOffset);
110                setMinuteOffset(minuteOffset);
111                setText(text);
112        }
113
114        /**
115         * Gets the hour offset.
116         * @return the hour offset or null if not set
117         */
118        public Integer getHourOffset() {
119                return hourOffset;
120        }
121
122        /**
123         * Sets the hour offset.
124         * @param hourOffset the hour offset or null to remove
125         */
126        public void setHourOffset(Integer hourOffset) {
127                this.hourOffset = hourOffset;
128        }
129
130        /**
131         * Gets the minute offset.
132         * @return the minute offset or null if not set
133         */
134        public Integer getMinuteOffset() {
135                return minuteOffset;
136        }
137
138        /**
139         * Sets the minute offset.
140         * @param minuteOffset the minute offset or null to remove
141         * @throws IllegalArgumentException if the minute offset is not between 0
142         * and 59
143         */
144        public void setMinuteOffset(Integer minuteOffset) {
145                if (minuteOffset != null && (minuteOffset < 0 || minuteOffset > 59)) {
146                        throw new IllegalArgumentException("Minute offset must be between 0 and 59.");
147                }
148                this.minuteOffset = minuteOffset;
149        }
150
151        /**
152         * Gets the text portion of the timezone.
153         * @return the text portion (e.g. "America/New_York")
154         */
155        public String getText() {
156                return text;
157        }
158
159        /**
160         * Sets the text portion of the timezone.
161         * @param text the text portion (e.g. "America/New_York")
162         */
163        public void setText(String text) {
164                this.text = text;
165        }
166
167        /**
168         * Creates a {@link java.util.TimeZone} representation of this class.
169         * @return a {@link TimeZone} object or null if this object contains no
170         * offset data
171         */
172        public TimeZone toTimeZone() {
173                if (hourOffset == null || minuteOffset == null) {
174                        return null;
175                }
176
177                int rawHourOffset = hourOffset * 60 * 60 * 1000;
178                int rawMinuteOffset = minuteOffset * 60 * 1000;
179                if (rawHourOffset < 0) {
180                        rawMinuteOffset *= -1;
181                }
182                int rawOffset = rawHourOffset + rawMinuteOffset;
183                return new SimpleTimeZone(rawOffset, "");
184        }

317        @Override
318        protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
319                if (text != null) {
320                        copy.setValue(ValueParameter.TEXT);
321                } else if (hourOffset != null && minuteOffset != null && version == VCardVersion.V4_0) {
322                        copy.setValue(ValueParameter.UTC_OFFSET);
323                }
324        }
325
326        @Override
327        protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
328                if (text != null) {
329                        if ((version == VCardVersion.V2_1 || version == VCardVersion.V3_0) && hourOffset != null && minuteOffset != null) {
330                                sb.append(VCardDateFormatter.formatTimeZone(hourOffset, minuteOffset, true));
331                                sb.append(';');
332                        }
333                        sb.append(VCardStringUtils.escape(text));
334                } else if (hourOffset != null && minuteOffset != null) {
335                        sb.append(VCardDateFormatter.formatTimeZone(hourOffset, minuteOffset, true));
336                } else {
337                        throw new SkipMeException("Property does not have text or a UTC offset associated with it.");
338                }
339        }
340
341        @Override
342        protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
343                parseValue(value);
344                if (text != null) {
345                        text = VCardStringUtils.unescape(text);
346                }
347        }
348
349        @Override
350        protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
351                if (text != null) {
352                        parent.appendText(text);
353                } else if (hourOffset != null && minuteOffset != null) {
354                        String offset = VCardDateFormatter.formatTimeZone(hourOffset, minuteOffset, true);
355                        parent.append("utc-offset", offset);
356                } else {
357                        throw new SkipMeException("Property does not have text or a UTC offset associated with it.");
358                }
359        }
360
361        @Override
362        protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
363                String value = element.get("text", "uri", "utc-offset");
364                if (value != null) {
365                        parseValue(value);
366                }
367        }
368
369        private void parseValue(String value) {
370                Pattern p = Pattern.compile("^([-\\+]?\\d{1,2}(:?\\d{2})?)(.*)");
371                Matcher m = p.matcher(value);
372                if (m.find()) {
373                        //do some smart parsing--if the value starts with a timezone, then parse it
374                        int offsets[] = VCardDateFormatter.parseTimeZone(m.group(1));
375                        hourOffset = offsets[0];
376                        minuteOffset = offsets[1];
377
378                        String text = m.group(3);
379                        if (text != null && text.length() == 0) {
380                                text = null;
381                        }
382                        this.text = text;
383                } else {
384                        hourOffset = null;
385                        minuteOffset = null;
386                        text = value;
387                }
388        }
389
390        @Override
391        protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
392                String value = element.value();
393                doUnmarshalValue(value, VCardVersion.V3_0, warnings, CompatibilityMode.RFC);
394        }
