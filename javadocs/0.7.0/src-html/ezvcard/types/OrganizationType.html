038/**
039 * A list of organizations the person belongs to. The list is ordered. It starts
040 * with the broadest organization and ends with the most specific.
041 * 
042 * <pre>
043 * VCard vcard = new VCard();
044 * OrganizationType org = new OrganizationType();
045 * org.addValue(&quot;Google&quot;);
046 * org.addValue(&quot;GMail Team&quot;);
047 * org.addValue(&quot;Spam Detection Team&quot;);
048 * vcard.setOrganizations(org);
049 * </pre>
050 * 
051 * <p>
052 * vCard property name: ORG
053 * </p>
054 * <p>
055 * vCard versions: 2.1, 3.0, 4.0
056 * </p>
057 * @author Michael Angstadt
058 */
059public class OrganizationType extends TextListType {
060        public static final String NAME = "ORG";
061
062        public OrganizationType() {
063                super(NAME, ';');
064        }
065
066        /**
067         * Gets the language that the address is written in.
068         * @return the language or null if not set
069         * @see VCardSubTypes#getLanguage
070         */
071        public String getLanguage() {
072                return subTypes.getLanguage();
073        }
074
075        /**
076         * Sets the language that the address is written in.
077         * @param language the language or null to remove
078         * @see VCardSubTypes#setLanguage
079         */
080        public void setLanguage(String language) {
081                subTypes.setLanguage(language);
082        }
083
084        /**
085         * Gets the TYPE parameter.
086         * <p>
087         * vCard versions: 4.0
088         * </p>
089         * @return the TYPE value (typically, this will be either "work" or "home")
090         * or null if it doesn't exist
091         */
092        public String getType() {
093                return subTypes.getType();
094        }
095
096        /**
097         * Sets the TYPE parameter.
098         * <p>
099         * vCard versions: 4.0
100         * </p>
101         * @param type the TYPE value (this should be either "work" or "home") or
102         * null to remove
103         */
104        public void setType(String type) {
105                subTypes.setType(type);
106        }
107
108        /**
109         * Gets all PID parameter values.
110         * <p>
111         * vCard versions: 4.0
112         * </p>
113         * @return the PID values or empty set if there are none
114         * @see VCardSubTypes#getPids
115         */
116        public Set<Integer[]> getPids() {
117                return subTypes.getPids();
118        }
119
120        /**
121         * Adds a PID value.
122         * <p>
123         * vCard versions: 4.0
124         * </p>
125         * @param localId the local ID
126         * @param clientPidMapRef the ID used to reference the property's globally
127         * unique identifier in the CLIENTPIDMAP property.
128         * @see VCardSubTypes#addPid(int, int)
129         */
130        public void addPid(int localId, int clientPidMapRef) {
131                subTypes.addPid(localId, clientPidMapRef);
132        }
133
134        /**
135         * Removes all PID values.
136         * <p>
137         * vCard versions: 4.0
138         * </p>
139         * @see VCardSubTypes#removePids
140         */
141        public void removePids() {
142                subTypes.removePids();
143        }
144
145        /**
146         * Gets the preference value.
147         * <p>
148         * vCard versions: 4.0
149         * </p>
150         * @return the preference value or null if it doesn't exist
151         * @see VCardSubTypes#getPref
152         */
153        public Integer getPref() {
154                return subTypes.getPref();
155        }
156
157        /**
158         * Sets the preference value.
159         * <p>
160         * vCard versions: 4.0
161         * </p>
162         * @param pref the preference value or null to remove
163         * @see VCardSubTypes#setPref
164         */
165        public void setPref(Integer pref) {
166                subTypes.setPref(pref);
167        }
168
169        /**
170         * Gets the ALTID.
171         * <p>
172         * vCard versions: 4.0
173         * </p>
174         * @return the ALTID or null if it doesn't exist
175         * @see VCardSubTypes#getAltId
176         */
177        public String getAltId() {
178                return subTypes.getAltId();
179        }
180
181        /**
182         * Sets the ALTID.
183         * <p>
184         * vCard versions: 4.0
185         * </p>
186         * @param altId the ALTID or null to remove
187         * @see VCardSubTypes#setAltId
188         */
189        public void setAltId(String altId) {
190                subTypes.setAltId(altId);
191        }
192
193        /**
194         * Gets the string that determines how this org name should be sorted.
195         * <p>
196         * vCard versions: 4.0
197         * </p>
198         * @return the sort string or null if not set
199         * @see VCardSubTypes#getSortAs
200         */
201        public String getSortAs() {
202                return subTypes.getSortAs();
203        }
204
205        /**
206         * Sets the string that determines how this org name should be sorted.
207         * <p>
208         * vCard versions: 4.0
209         * </p>
210         * @param sortAs the sort string or null to remove
211         * @see VCardSubTypes#setSortAs
212         */
213        public void setSortAs(String sortAs) {
214                subTypes.setSortAs(sortAs);
215        }
216
217        @Override
218        protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
219                String orgName = element.firstValue("organization-name");
220                if (orgName != null) {
221                        addValue(orgName);
222                }
223
224                String orgUnit = element.firstValue("organization-unit");
225                if (orgUnit != null) {
226                        addValue(orgUnit);
227                }
228
229                if (getValues().isEmpty()) {
230                        addValue(element.value());
231                }
232        }
233}
