061/**
062 * Any XML data attached to the vCard. This is used if the vCard was encoded in
063 * XML (xCard standard) and it contained some non-standard elements.
064 * 
065 * <pre>
066 * VCard vcard = new VCard();
067 * XmlType xml = new XmlType(&quot;&lt;b&gt;Some xml&lt;/b&gt;&quot;);
068 * vcard.addXml(xml);
069 * </pre>
070 * 
071 * <p>
072 * vCard property name: XML
073 * </p>
074 * <p>
075 * vCard versions: 4.0
076 * </p>
077 * @author Michael Angstadt
078 */
079public class XmlType extends TextType {
080        public static final String NAME = "XML";
081
082        public XmlType() {
083                super(NAME);
084        }
085
086        /**
087         * @param xml the XML element
088         */
089        public XmlType(String xml) {
090                super(NAME, xml);
091        }
092
093        /**
094         * Gets the ALTID.
095         * <p>
096         * vCard versions: 4.0
097         * </p>
098         * @return the ALTID or null if it doesn't exist
099         * @see VCardSubTypes#getAltId
100         */
101        public String getAltId() {
102                return subTypes.getAltId();
103        }
104
105        /**
106         * Sets the ALTID.
107         * <p>
108         * vCard versions: 4.0
109         * </p>
110         * @param altId the ALTID or null to remove
111         * @see VCardSubTypes#setAltId
112         */
113        public void setAltId(String altId) {
114                subTypes.setAltId(altId);
115        }
116
117        @Override
118        public VCardVersion[] getSupportedVersions() {
119                return new VCardVersion[] { VCardVersion.V4_0 };
120        }
121
122        /**
123         * Converts the text value of this property to an XML {@link Element}
124         * object.
125         * @return the element object or null if this property's value is null
126         * @throws SAXException if there's a problem parsing the XML
127         */
128        public Element asElement() throws SAXException {
129                if (value == null) {
130                        return null;
131                }
132
133                try {
134                        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
135                        dbf.setNamespaceAware(true);
136                        DocumentBuilder db = dbf.newDocumentBuilder();
137                        Document document = db.parse(new InputSource(new StringReader(value)));
138                        return document.getDocumentElement();
139                } catch (IOException e) {
140                        //never thrown because we're reading from a string
141                        return null;
142                } catch (ParserConfigurationException e) {
143                        //never thrown because we're not doing anything fancy with the configuration
144                        return null;
145                }
146        }
147
148        @Override
149        protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
150                if (value == null) {
151                        throw new SkipMeException("Property does not have a value associated with it.");
152                }
153
154                //parse the XML string
155                Element root = null;
156                try {
157                        DocumentBuilderFactory fact = DocumentBuilderFactory.newInstance();
158                        fact.setNamespaceAware(true);
159                        DocumentBuilder builder = fact.newDocumentBuilder();
160                        Document document = builder.parse(new ByteArrayInputStream(value.getBytes()));
161                        root = XmlUtils.getRootElement(document);
162                } catch (IOException e) {
163                        //never thrown because reading from a string
164                } catch (ParserConfigurationException e) {
165                        //should never be thrown
166                } catch (SAXException e) {
167                        throw new SkipMeException("Property value is not valid XML.");
168                }
169
170                //add XML element to marshalled document
171                Node imported = parent.getElement().getOwnerDocument().importNode(root, true);
172                parent.getElement().appendChild(imported);
173        }
174
175        @Override
176        protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
177                try {
178                        StringWriter writer = new StringWriter();
179                        DOMSource source = new DOMSource(element.getElement());
180                        StreamResult result = new StreamResult(writer);
181                        Transformer t = TransformerFactory.newInstance().newTransformer();
182                        t.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
183                        t.transform(source, result);
184                        value = writer.toString();
185                } catch (TransformerException e) {
186                        warnings.add("Problem transforming XML element to string for " + NAME + " property: " + e.getMessage());
187                }
188        }
189}
