041/**
042 * Represents a type whose value is a list of values.
043 * @author Michael Angstadt
044 */
045public class TextListType extends VCardType {
046        protected List<String> values = new ArrayList<String>();
047        private final char separator;
048
049        /**
050         * @param name the type name (e.g. "NICKNAME")
051         * @param separator the delimiter to use (e.g. ",")
052         */
053        public TextListType(String name, char separator) {
054                super(name);
055                this.separator = separator;
056        }
057
058        /**
059         * Gest the list of values.
060         * @return the list of values
061         */
062        public List<String> getValues() {
063                return values;
064        }
065
066        /**
067         * Adds a value to the list.
068         * @param value the value to add
069         */
070        public void addValue(String value) {
071                values.add(value);
072        }
073
074        /**
075         * Removes a value from the list.
076         * @param value the value to remove
077         */
078        public void removeValue(String value) {
079                values.remove(value);
080        }
081
082        @Override
083        protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
084                if (!values.isEmpty()) {
085                        for (String value : values) {
086                                value = VCardStringUtils.escape(value);
087                                sb.append(value);
088                                sb.append(separator);
089                        }
090                        sb.deleteCharAt(sb.length() - 1);
091                }
092        }
093
094        @Override
095        protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
096                String split[] = VCardStringUtils.splitBy(value, separator, true, true);
097                values = new ArrayList<String>(Arrays.asList(split));
098        }
099
100        @Override
101        protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
102                parent.append("text", values);
103        }
104
105        @Override
106        protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
107                values.clear();
108                values.addAll(element.getAll("text"));
109        }
110}
