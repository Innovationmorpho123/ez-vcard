037/**
038 * This just says that the vCard is a vCard, it's not really necessary.
039 * 
040 * <pre>
041 * VCard vcard = new VCard();
042 * ProfileType profile = new ProfileType();
043 * vcard.setProfile(profile);
044 * </pre>
045 * 
046 * <p>
047 * vCard property name: PROFILE
048 * </p>
049 * <p>
050 * vCard versions: 3.0
051 * </p>
052 * @author Michael Angstadt
053 */
054public class ProfileType extends TextType {
055        public static final String NAME = "PROFILE";
056
057        public ProfileType() {
058                super(NAME, "VCARD");
059        }
060
061        @Override
062        protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
063                if ("VCARD".equalsIgnoreCase(value)) {
064                        //see RFC 2426 p.5
065                        warnings.add("The value of the " + NAME + " property should be \"VCARD\", but it is \"" + value + "\"");
066                }
067                super.doMarshalValue(sb, version, warnings, compatibilityMode);
068        }
069
070        @Override
071        protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
072                if ("VCARD".equalsIgnoreCase(value)) {
073                        //see RFC 2426 p.5
074                        warnings.add("The value of the " + ProfileType.NAME + " type should be \"VCARD\", but it is \"" + value + "\"");
075                }
076                super.doUnmarshalValue(value, version, warnings, compatibilityMode);
077        }
078        
079        @Override
080        public VCardVersion[] getSupportedVersions() {
081                return new VCardVersion[] { VCardVersion.V3_0 };
082        }
083}
