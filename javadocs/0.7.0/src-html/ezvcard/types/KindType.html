034/**
035 * Defines the type of entity that this vCard represents, such as an individual
036 * or an organization.
037 * 
038 * <p>
039 * <b>Setting the KIND</b>
040 * </p>
041 * 
042 * <pre>
043 * //use static methods to create a KindType object
044 * VCard vcard = new VCard();
045 * KindType kind = KindType.individual();
046 * vcard.setKind(kind);
047 * </pre>
048 * 
049 * <p>
050 * <b>Getting the KIND</b>
051 * </p>
052 * 
053 * <pre>
054 * //use "is*" methods to determine the KindType value
055 * VCard vcard = ...
056 * KindType kind = vcard.getKind();
057 * if (kind != null){
058 *   if (kind.isIndividual()){
059 *     ...
060 *   } else if (kind.isGroup()){
061 *     ...
062 *   }
063 *   ...
064 * }
065 * </pre>
066 * 
067 * <p>
068 * vCard property name: KIND
069 * </p>
070 * <p>
071 * vCard versions: 4.0
072 * </p>
073 * @author Michael Angstadt
074 */
075public class KindType extends TextType {
076        public static final String NAME = "KIND";
077
078        public static final String INDIVIDUAL = "individual";
079        public static final String GROUP = "group";
080        public static final String ORG = "org";
081        public static final String LOCATION = "location";
082        public static final String APPLICATION = "application";
083
084        public KindType() {
085                super(NAME);
086        }
087
088        /**
089         * Use of this constructor is discouraged. Please use one of the static
090         * methods to create a new KIND type.
091         * @param kind the kind value (e.g. "group")
092         */
093        public KindType(String kind) {
094                super(NAME, kind);
095        }
096
097        /**
098         * Determines if the value is set to "individual".
099         * @return true if the value is "individual", false if not
100         */
101        public boolean isIndividual() {
102                return INDIVIDUAL.equals(value);
103        }
104
105        /**
106         * Determines if the value is set to "group".
107         * @return true if the value is "group", false if not
108         */
109        public boolean isGroup() {
110                return GROUP.equals(value);
111        }
112
113        /**
114         * Determines if the value is set to "org".
115         * @return true if the value is "org", false if not
116         */
117        public boolean isOrg() {
118                return ORG.equals(value);
119        }
120
121        /**
122         * Determines if the value is set to "location".
123         * @return true if the value is "location", false if not
124         */
125        public boolean isLocation() {
126                return LOCATION.equals(value);
127        }
128
129        /**
130         * Determines if the value is set to "application".
131         * @return true if the value is "application", false if not
132         * @see <a href="http://tools.ietf.org/html/rfc6473">RFC 6473</a>
133         */
134        public boolean isApplication() {
135                return APPLICATION.equals(value);
136        }
137
138        /**
139         * Creates a new KIND type whose value is set to "individual".
140         * @return the new KIND type
141         */
142        public static KindType individual() {
143                return new KindType(INDIVIDUAL);
144        }
145
146        /**
147         * Creates a new KIND type whose value is set to "group".
148         * @return the new KIND type
149         */
150        public static KindType group() {
151                return new KindType(GROUP);
152        }
153
154        /**
155         * Creates a new KIND type whose value is set to "org".
156         * @return the new KIND type
157         */
158        public static KindType org() {
159                return new KindType(ORG);
160        }
161
162        /**
163         * Creates a new KIND type whose value is set to "location".
164         * @return the new KIND type
165         */
166        public static KindType location() {
167                return new KindType(LOCATION);
168        }
169
170        /**
171         * Creates a new KIND type whose value is set to "application".
172         * @return the new KIND type
173         * @see <a href="http://tools.ietf.org/html/rfc6473">RFC 6473</a>
174         */
175        public static KindType application() {
176                return new KindType(APPLICATION);
177        }
178
179        @Override
180        public VCardVersion[] getSupportedVersions() {
181                return new VCardVersion[] { VCardVersion.V4_0 };
182        }
183}
