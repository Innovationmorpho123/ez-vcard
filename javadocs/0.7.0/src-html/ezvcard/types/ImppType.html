049/**
050 * An instant message handle. The handle is represented as a URI in the format "
051 * <code>&lt;IM-PROTOCOL&gt;:&lt;IM-HANDLE&gt;</code>". For example, someone
052 * with a Yahoo! Messenger handle of "johndoe@yahoo.com" would have an IMPP
053 * vCard property value of "ymsgr:johndoe@yahoo.com".
054 * 
055 * <pre>
056 * VCard vcard = new VCard();
057 * 
058 * //URI
059 * ImppType impp = new ImppType(&quot;aim:johndoe@aol.com&quot;);
060 * vcard.addImpp(impp);
061 * 
062 * //static helper constructors
063 * impp = ImppType.msn(&quot;janedoe@msn.com&quot;);
064 * vcard.addImpp(impp);
065 * </pre>
066 * 
067 * <p>
068 * vCard property name: IMPP
069 * </p>
070 * <p>
071 * vCard versions: 3.0, 4.0
072 * </p>
073 * @author Michael Angstadt
074 */
075public class ImppType extends MultiValuedTypeParameterType<ImppTypeParameter> {
076        public static final String NAME = "IMPP";

082        private static final String SIP = "sip";
083        private static final String SKYPE = "skype";
084        private static final String XMPP = "xmpp";
085        private static final String YAHOO = "ymsgr";

087        /**
088         * List of recognized IM protocols that can be parsed from an HTML link
089         * (hCard).
090         */
091        private static final List<ImHtmlLink> htmlParseableProtocols = new ArrayList<ImHtmlLink>();
092        static {
093                //http://en.wikipedia.org/wiki/AOL_Instant_Messenger#URI_scheme
094                htmlParseableProtocols.add(new ImHtmlLink(AIM, "(goim|addbuddy)\\?.*?\\bscreenname=(.*?)(&|$)", 2, "goim?screenname=%s"));
095
096                //http://en.wikipedia.org/wiki/Yahoo!_Messenger#URI_scheme
097                htmlParseableProtocols.add(new ImHtmlLink(YAHOO, "(sendim|addfriend|sendfile|call)\\?(.*)", 2, "sendim?%s"));
098
099                //http://developer.skype.com/skype-uri/skype-uri-ref-api
100                htmlParseableProtocols.add(new ImHtmlLink(SKYPE, "(.*?)(\\?|$)", 1, "%s"));
101
102                //http://www.tech-recipes.com/rx/1157/msn-messenger-msnim-hyperlink-command-codes/
103                htmlParseableProtocols.add(new ImHtmlLink(MSN, "(chat|add|voice|video)\\?contact=(.*?)(&|$)", 2, "chat?contact=%s"));
104
105                //http://www.tech-recipes.com/rx/1157/msn-messenger-msnim-hyperlink-command-codes/
106                htmlParseableProtocols.add(new ImHtmlLink(XMPP, "(.*?)(\\?|$)", 1, "%s?message"));
107
108                //http://forums.miranda-im.org/showthread.php?26589-Add-support-to-quot-icq-message-uin-12345-quot-web-links
109                htmlParseableProtocols.add(new ImHtmlLink(ICQ, "message\\?uin=(\\d+)", 1, "message?uin=%s"));
110
111                //SIP: http://en.wikipedia.org/wiki/Session_Initiation_Protocol
112                //leave as-is
113                htmlParseableProtocols.add(new ImHtmlLink(SIP));
114
115                //IRC: http://stackoverflow.com/questions/11970897/how-do-i-open-a-query-window-using-the-irc-uri-scheme
116                //IRC handles are not globally unique, so leave as-is
117                htmlParseableProtocols.add(new ImHtmlLink(IRC));
118        }

520        @Override
521        protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
522                String href = element.attr("href");
523                if (href.length() == 0) {
524                        href = element.value();
525                }
526
527                try {
528                        URI uri = parseUriFromLink(href);
529                        if (uri == null) {
530                                throw new IllegalArgumentException();
531                        }
532                        setUri(uri);
533                } catch (IllegalArgumentException e) {
534                        throw new SkipMeException("Could not parse instant messenger information out of link: " + href);
535                }
536        }

538        /**
539         * Parses an IM URI from an HTML link.
540         * @param linkUri the HTML link (e.g. "aim:goim?screenname=theuser")
541         * @return the IM URI or null if not recognized
542         */
543        protected static URI parseUriFromLink(String linkUri) {
544                for (ImHtmlLink imLink : htmlParseableProtocols) {
545                        String handle = imLink.parseHandle(linkUri);
546                        if (handle != null) {
547                                try {
548                                        return new URI(imLink.getProtocol(), handle, null);
549                                } catch (URISyntaxException e) {
550                                        throw new IllegalArgumentException(e);
551                                }
552                        }
553                }
554                return null;
555        }
556
557        /**
558         * Builds a URI suitable for use as a link on a webpage.
559         * @return the link URI (e.g. "aim:goim?screenname=theuser") or null if the
560         * IMPP URI was never set
561         */
562        public String buildLink() {
563                if (uri == null) {
564                        return null;
565                }
566
567                String protocol = uri.getScheme();
568                String handle = uri.getSchemeSpecificPart();
569
570                for (ImHtmlLink imLink : htmlParseableProtocols) {
571                        if (protocol.equals(imLink.getProtocol())) {
572                                return imLink.buildLink(handle);
573                        }
574                }
575                return uri.toString();
576        }

578        /**
579         * Helper class for parsing and building instant messenger links for
580         * webpages.
581         */
582        protected static class ImHtmlLink {
583                private final Pattern linkRegex;
584                private final String protocol;
585                private final int handleGroup;
586                private final String linkFormat;
587
588                /**
589                 * @param protocol the IM protocol (e.g. "aim")
590                 */
591                public ImHtmlLink(String protocol) {
592                        this(protocol, "(.*)", 1, "%s");
593                }
594
595                /**
596                 * @param protocol the IM protocol (e.g. "aim")
597                 * @param linkRegex the regular expression used to parse a link
598                 * @param handleGroup the group number from the regular expression that
599                 * contains the IM handle
600                 * @param linkFormat the format string for building a link
601                 */
602                public ImHtmlLink(String protocol, String linkRegex, int handleGroup, String linkFormat) {
603                        this.linkRegex = Pattern.compile('^' + protocol + ':' + linkRegex, Pattern.CASE_INSENSITIVE);
604                        this.protocol = protocol;
605                        this.handleGroup = handleGroup;
606                        this.linkFormat = protocol + ':' + linkFormat;
607                }
608
609                /**
610                 * Parses the IM handle out of a link.
611                 * @param linkUri the link
612                 * @return the IM handle or null if it can't be found
613                 */
614                public String parseHandle(String linkUri) {
615                        Matcher m = linkRegex.matcher(linkUri);
616                        if (m.find()) {
617                                return m.group(handleGroup);
618                        }
619                        return null;
620                }
621
622                /**
623                 * Builds a link for inclusion in a webpage.
624                 * @param handle the IM handle
625                 * @return the link
626                 */
627                public String buildLink(String handle) {
628                        return String.format(linkFormat, handle);
629                }
630
631                public String getProtocol() {
632                        return protocol;
633                }
634        }
