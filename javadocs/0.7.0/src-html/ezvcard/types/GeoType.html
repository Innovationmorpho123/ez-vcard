047/**
048 * A set of latitude/longitude coordinates. There is no rule for what these
049 * coordinates must represent, but the meaning could vary depending on the value
050 * of the vCard KIND type.
051 * 
052 * <ul>
053 * <li>"individual": the location of the person's home or workplace.</li>
054 * <li>"group": the location of the group's meeting place.</li>
055 * <li>"org": the coordinates of the organization's headquarters.</li>
056 * <li>"location": the coordinates of the location itself.</li>
057 * </ul>
058 * 
059 * <pre>
060 * VCard vcard = new VCard();
061 * GeoType geo = new GeoType(-123.456, 12.54);
062 * vcard.setGeo(geo);
063 * </pre>
064 * 
065 * <p>
066 * vCard property name: GEO
067 * </p>
068 * <p>
069 * vCard versions: 2.1, 3.0, 4.0
070 * </p>
071 * @author Michael Angstadt
072 */
073public class GeoType extends VCardType {
074        public static final String NAME = "GEO";
075        private GeoUri uri = new GeoUri();
076
077        public GeoType() {
078                this(null, null);
079        }
080
081        /**
082         * @param latitude the latitude
083         * @param longitude the longitude
084         */
085        public GeoType(Double latitude, Double longitude) {
086                super(NAME);
087                setLatitude(latitude);
088                setLongitude(longitude);
089        }
090
091        /**
092         * Gets the latitude.
093         * @return the latitude
094         */
095        public Double getLatitude() {
096                return uri.getCoordA();
097        }
098
099        /**
100         * Sets the latitude.
101         * @param latitude the latitude
102         */
103        public void setLatitude(Double latitude) {
104                uri.setCoordA(latitude);
105        }
106
107        /**
108         * Gets the longitude.
109         * @return the longitude
110         */
111        public Double getLongitude() {
112                return uri.getCoordB();
113        }
114
115        /**
116         * Sets the longitude.
117         * @param longitude the longitude
118         */
119        public void setLongitude(Double longitude) {
120                uri.setCoordB(longitude);
121        }
122
123        /**
124         * Gets the raw object used for storing the GEO information. This can be
125         * used to supplement the GEO value with additional information (such as the
126         * altitude). Geo URIs are only supported by vCard version 4.0. Everything
127         * but latitude and longitude will be lost when marshalling to an earlier
128         * vCard version.
129         * @return the geo URI object
130         * @see <a href="http://tools.ietf.org/html/rfc5870">RFC 5870</a>
131         */
132        public GeoUri getGeoUri() {
133                return uri;
134        }

267        @Override
268        protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
269                if (version == VCardVersion.V4_0) {
270                        copy.setValue(ValueParameter.URI);
271                }
272        }
273
274        @Override
275        protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
276                if (getLatitude() == null || getLongitude() == null) {
277                        throw new SkipMeException("Latitude and/or longitude is missing.");
278                }
279
280                if (version == VCardVersion.V4_0) {
281                        sb.append(uri.toString(6));
282                } else {
283                        NumberFormat nf = new DecimalFormat("0.######");
284                        sb.append(nf.format(getLatitude()));
285                        sb.append(';');
286                        sb.append(nf.format(getLongitude()));
287                }
288        }
289
290        @Override
291        protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
292                try {
293                        //4.0 syntax: GEO;VALUE=uri:geo:12,23
294                        uri = new GeoUri(value);
295                } catch (IllegalArgumentException e) {
296                        //2.1/3.0 syntax: GEO:12;34
297                        String split[] = value.split(";");
298                        uri = new GeoUri();
299
300                        try {
301                                setLatitude(Double.valueOf(split[0]));
302                        } catch (NumberFormatException e2) {
303                                //do nothing (handled below)
304                        }
305
306                        boolean longMissing = false;
307                        if (split.length > 1) {
308                                try {
309                                        setLongitude(Double.valueOf(split[1]));
310                                } catch (NumberFormatException e2) {
311                                        //do nothing (handled below)
312                                }
313                        } else {
314                                longMissing = true;
315                        }
316
317                        if (getLatitude() == null && getLongitude() == null) {
318                                throw new SkipMeException("Unparseable value: \"" + value + "\"");
319                        } else if (longMissing) {
320                                warnings.add("Longitude missing from " + NAME + " type value: \"" + value + "\"");
321                        } else if (getLatitude() == null) {
322                                warnings.add("Could not parse latitude from " + NAME + " type value: \"" + value + "\"");
323                        } else if (getLongitude() == null) {
324                                warnings.add("Could not parse longitude from " + NAME + " type value: \"" + value + "\"");
325                        }
326                }
327        }
328
329        @Override
330        protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
331                StringBuilder sb = new StringBuilder();
332                doMarshalValue(sb, parent.getVersion(), warnings, compatibilityMode);
333                parent.appendUri(sb.toString());
334        }
335
336        @Override
337        protected void doUnmarshalValue(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
338                String value = element.getUri();
339                if (value != null) {
340                        doUnmarshalValue(value, element.getVersion(), warnings, compatibilityMode);
341                }
342        }
343
344        @Override
345        protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
346                String latitude = element.firstValue("latitude");
347                if (latitude == null) {
348                        warnings.add("Latitude missing from " + NAME + " type.");
349                } else {
350                        try {
351                                setLatitude(Double.parseDouble(latitude));
352                        } catch (NumberFormatException e) {
353                                warnings.add("Could not parse latitude from " + NAME + " type.");
354                        }
355                }
356
357                String longitude = element.firstValue("longitude");
358                if (longitude == null) {
359                        warnings.add("Longitude missing from " + NAME + " type.");
360                } else {
361                        try {
362                                setLongitude(Double.parseDouble(longitude));
363                        } catch (NumberFormatException e) {
364                                warnings.add("Could not parse longitude from " + NAME + " type.");
365                        }
366                }
367
368                if (getLatitude() == null && getLongitude() == null) {
369                        throw new SkipMeException("Unparsable value.");
370                }
371        }
