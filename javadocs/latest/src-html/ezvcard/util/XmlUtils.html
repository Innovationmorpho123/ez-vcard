061/**
062 * Generic XML utility methods.
063 * @author Michael Angstadt
064 */
065public class XmlUtils {
066        /**
067         * Creates a new XML document.
068         * @return the XML document
069         */
070        public static Document createDocument() {
071                try {
072                        DocumentBuilderFactory fact = DocumentBuilderFactory.newInstance();
073                        fact.setNamespaceAware(true);
074                        DocumentBuilder db = fact.newDocumentBuilder();
075                        return db.newDocument();
076                } catch (ParserConfigurationException e) {
077                        //will probably never be thrown because we're not doing anything fancy with the configuration
078                        throw new RuntimeException(e);
079                }
080        }
081
082        /**
083         * Parses an XML string into a DOM.
084         * @param xml the XML string
085         * @return the parsed DOM
086         * @throws SAXException if the string is not valid XML
087         */
088        public static Document toDocument(String xml) throws SAXException {
089                try {
090                        return toDocument(new StringReader(xml));
091                } catch (IOException e) {
092                        //reading from string
093                        throw new RuntimeException(e);
094                }
095        }
096
097        /**
098         * Parses an XML document from an input stream.
099         * @param in the input stream
100         * @return the parsed DOM
101         * @throws SAXException if the XML is not valid
102         * @throws IOException if there is a problem reading from the input stream
103         */
104        public static Document toDocument(InputStream in) throws SAXException, IOException {
105                return toDocument(new InputSource(in));
106        }
107
108        /**
109         * <p>
110         * Parses an XML document from a reader.
111         * </p>
112         * <p>
113         * Note that use of this method is discouraged. It ignores the character
114         * encoding that is defined within the XML document itself, and should only
115         * be used if the encoding is undefined or if the encoding needs to be
116         * ignored for whatever reason. The {@link #toDocument(InputStream)} method
117         * should be used instead, since it takes the XML document's character
118         * encoding into account when parsing.
119         * </p>
120         * @param reader the reader
121         * @return the parsed DOM
122         * @throws SAXException if the XML is not valid
123         * @throws IOException if there is a problem reading from the reader
124         * @see <a
125         * href="http://stackoverflow.com/q/3482494/13379">http://stackoverflow.com/q/3482494/13379</a>
126         */
127        public static Document toDocument(Reader reader) throws SAXException, IOException {
128                return toDocument(new InputSource(reader));
129        }
130
131        private static Document toDocument(InputSource in) throws SAXException, IOException {
132                DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
133                factory.setNamespaceAware(true);
134                factory.setIgnoringComments(true);
135
136                DocumentBuilder builder;
137                try {
138                        builder = factory.newDocumentBuilder();
139                } catch (ParserConfigurationException e) {
140                        //will probably never be thrown because we're not doing anything fancy with the configuration
141                        throw new RuntimeException(e);
142                }
143
144                return builder.parse(in);
145        }
146
147        /**
148         * Converts an XML node to a string.
149         * @param node the XML node
150         * @return the string
151         */
152        public static String toString(Node node) {
153                return toString(node, new HashMap<String, String>());
154        }
155
156        /**
157         * Converts an XML node to a string.
158         * @param node the XML node
159         * @param outputProperties the output properties
160         * @return the string
161         */
162        public static String toString(Node node, Map<String, String> outputProperties) {
163                try {
164                        StringWriter writer = new StringWriter();
165                        toWriter(node, writer, outputProperties);
166                        return writer.toString();
167                } catch (TransformerException e) {
168                        //should never be thrown because we're writing to a string
169                        throw new RuntimeException(e);
170                }
171        }
172
173        /**
174         * Writes an XML node to a writer.
175         * @param node the XML node
176         * @param writer the writer
177         * @throws TransformerException if there's a problem writing to the writer
178         */
179        public static void toWriter(Node node, Writer writer) throws TransformerException {
180                toWriter(node, writer, new HashMap<String, String>());
181        }
182
183        /**
184         * Writes an XML node to a writer.
185         * @param node the XML node
186         * @param writer the writer
187         * @param outputProperties the output properties
188         * @throws TransformerException if there's a problem writing to the writer
189         */
190        public static void toWriter(Node node, Writer writer, Map<String, String> outputProperties) throws TransformerException {
191                Transformer transformer;
192                try {
193                        transformer = TransformerFactory.newInstance().newTransformer();
194                } catch (TransformerConfigurationException e) {
195                        //no complex configurations
196                        throw new RuntimeException(e);
197                } catch (TransformerFactoryConfigurationError e) {
198                        //no complex configurations
199                        throw new RuntimeException(e);
200                }
201
202                for (Map.Entry<String, String> property : outputProperties.entrySet()) {
203                        try {
204                                transformer.setOutputProperty(property.getKey(), property.getValue());
205                        } catch (IllegalArgumentException e) {
206                                //ignore invalid output properties
207                        }
208                }
209
210                DOMSource source = new DOMSource(node);
211                StreamResult result = new StreamResult(writer);
212                transformer.transform(source, result);
213        }
214
215        /**
216         * Gets all the elements out of a {@link NodeList}.
217         * @param nodeList the node list
218         * @return the elements
219         */
220        public static List<Element> toElementList(NodeList nodeList) {
221                List<Element> elements = new ArrayList<Element>();
222                for (int i = 0; i < nodeList.getLength(); i++) {
223                        Node node = nodeList.item(i);
224                        if (node instanceof Element) {
225                                elements.add((Element) node);
226                        }
227                }
228                return elements;
229        }
230
231        /**
232         * Gets the root element of a document.
233         * @param parent the document
234         * @return the root element
235         */
236        public static Element getRootElement(Document parent) {
237                return getFirstChildElement((Node) parent);
238        }
239
240        /**
241         * Gets the first child element of an element.
242         * @param parent the parent element
243         * @return the first child element or null if there are no child elements
244         */
245        public static Element getFirstChildElement(Element parent) {
246                return getFirstChildElement((Node) parent);
247        }
248
249        /**
250         * Gets the first child element of a node.
251         * @param parent the node
252         * @return the first child element or null if there are no child elements
253         */
254        private static Element getFirstChildElement(Node parent) {
255                NodeList nodeList = parent.getChildNodes();
256                for (int i = 0; i < nodeList.getLength(); i++) {
257                        Node node = nodeList.item(i);
258                        if (node instanceof Element) {
259                                return (Element) node;
260                        }
261                }
262                return null;
263        }

266                //hide
267        }
