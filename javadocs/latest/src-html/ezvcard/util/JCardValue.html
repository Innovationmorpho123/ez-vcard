042    /**
043     * Holds the data type and value of a jCard property.
044     * @author Michael Angstadt
045     */
046    public class JCardValue {
047            private final VCardDataType dataType;
048            private final List<JsonValue> values;
049    
050            /**
051             * Creates a new jCard value.
052             * @param dataType the data type or null for "unknown"
053             * @param values the values
054             */
055            public JCardValue(VCardDataType dataType, List<JsonValue> values) {
056                    this.dataType = dataType;
057                    this.values = Collections.unmodifiableList(values);
058            }
059    
060            /**
061             * Creates a new jCard value.
062             * @param dataType the data type or null for "unknown"
063             * @param values the values
064             */
065            public JCardValue(VCardDataType dataType, JsonValue... values) {
066                    this.dataType = dataType;
067                    this.values = Arrays.asList(values); //unmodifiable
068            }
069    
070            /**
071             * Creates a single-valued value.
072             * @param dataType the data type or null for "unknown"
073             * @param value the value
074             * @return the jCard value
075             */
076            public static JCardValue single(VCardDataType dataType, Object value) {
077                    return new JCardValue(dataType, new JsonValue(value));
078            }
079    
080            /**
081             * Creates a multi-valued value.
082             * @param dataType the data type or null for "unknown"
083             * @param values the values
084             * @return the jCard value
085             */
086            public static JCardValue multi(VCardDataType dataType, Object... values) {
087                    return multi(dataType, Arrays.asList(values));
088            }
089    
090            /**
091             * Creates a multi-valued value.
092             * @param dataType the data type or null for "unknown"
093             * @param values the values
094             * @return the jCard value
095             */
096            public static JCardValue multi(VCardDataType dataType, List<?> values) {
097                    List<JsonValue> multiValues = new ArrayList<JsonValue>(values.size());
098                    for (Object value : values) {
099                            multiValues.add(new JsonValue(value));
100                    }
101                    return new JCardValue(dataType, multiValues);
102            }
103    
104            /**
105             * <p>
106             * Creates a structured value.
107             * </p>
108             * <p>
109             * This method accepts a vararg of {@link Object} instances. {@link List}
110             * objects will be treated as multi-valued components. Null objects will be
111             * treated as empty components.
112             * </p>
113             * @param dataType the data type or null for "unknown"
114             * @param values the values
115             * @return the jCard value
116             */
117            public static JCardValue structured(VCardDataType dataType, Object... values) {
118                    List<List<?>> valuesList = new ArrayList<List<?>>(values.length);
119                    for (Object value : values) {
120                            List<?> list = (value instanceof List) ? (List<?>) value : Arrays.asList(value);
121                            valuesList.add(list);
122                    }
123                    return structured(dataType, valuesList);
124            }
125    
126            /**
127             * Creates a structured value.
128             * @param dataType the data type or null for "unknown"
129             * @param values the values
130             * @return the jCard value
131             */
132            public static JCardValue structured(VCardDataType dataType, List<List<?>> values) {
133                    List<JsonValue> array = new ArrayList<JsonValue>(values.size());
134    
135                    for (List<?> list : values) {
136                            if (list.isEmpty()) {
137                                    array.add(new JsonValue(""));
138                                    continue;
139                            }
140    
141                            if (list.size() == 1) {
142                                    Object value = list.get(0);
143                                    if (value == null) {
144                                            value = "";
145                                    }
146                                    array.add(new JsonValue(value));
147                                    continue;
148                            }
149    
150                            List<JsonValue> subArray = new ArrayList<JsonValue>(list.size());
151                            for (Object value : list) {
152                                    if (value == null) {
153                                            value = "";
154                                    }
155                                    subArray.add(new JsonValue(value));
156                            }
157                            array.add(new JsonValue(subArray));
158                    }
159    
160                    return new JCardValue(dataType, new JsonValue(array));
161            }
162    
163            /**
164             * Gets the jCard data type
165             * @return the data type or null for "unknown"
166             */
167            public VCardDataType getDataType() {
168                    return dataType;
169            }
170    
171            /**
172             * Gets all the JSON values.
173             * @return the JSON values
174             */
175            public List<JsonValue> getValues() {
176                    return values;
177            }
178    
179            /**
180             * Gets the value of a single-valued property (such as {@link NoteType}).
181             * @return the value or empty string if not found
182             */
183            public String asSingle() {
184                    if (values.isEmpty()) {
185                            return "";
186                    }
187    
188                    JsonValue first = values.get(0);
189                    if (first.isNull()) {
190                            return "";
191                    }
192    
193                    Object obj = first.getValue();
194                    if (obj != null) {
195                            return obj.toString();
196                    }
197    
198                    //get the first element of the array
199                    List<JsonValue> array = first.getArray();
200                    if (array != null && !array.isEmpty()) {
201                            obj = array.get(0).getValue();
202                            if (obj != null) {
203                                    return obj.toString();
204                            }
205                    }
206    
207                    return "";
208            }
209    
210            /**
211             * Gets the value of a structured property (such as
212             * {@link StructuredNameType}).
213             * @return the values or empty list if not found
214             */
215            public List<List<String>> asStructured() {
216                    if (values.isEmpty()) {
217                            return Collections.emptyList();
218                    }
219    
220                    JsonValue first = values.get(0);
221    
222                    //["gender", {}, "text", ["M", "text"] ]
223                    List<JsonValue> array = first.getArray();
224                    if (array != null) {
225                            List<List<String>> valuesStr = new ArrayList<List<String>>(array.size());
226                            for (JsonValue value : array) {
227                                    if (value.isNull()) {
228                                            valuesStr.add(Arrays.asList(""));
229                                            continue;
230                                    }
231    
232                                    Object obj = value.getValue();
233                                    if (obj != null) {
234                                            valuesStr.add(Arrays.asList(obj.toString()));
235                                            continue;
236                                    }
237    
238                                    List<JsonValue> subArray = value.getArray();
239                                    if (subArray != null) {
240                                            List<String> subValuesStr = new ArrayList<String>(subArray.size());
241                                            for (JsonValue subArrayValue : subArray) {
242                                                    if (subArrayValue.isNull()) {
243                                                            subValuesStr.add("");
244                                                            continue;
245                                                    }
246    
247                                                    obj = subArrayValue.getValue();
248                                                    if (obj != null) {
249                                                            subValuesStr.add(obj.toString());
250                                                            continue;
251                                                    }
252                                            }
253                                            valuesStr.add(subValuesStr);
254                                    }
255                            }
256                            return valuesStr;
257                    }
258    
259                    //get the first value if it's not enclosed in an array
260                    //["gender", {}, "text", "M"]
261                    Object obj = first.getValue();
262                    if (obj != null) {
263                            List<List<String>> values = new ArrayList<List<String>>(1);
264                            values.add(Arrays.asList(obj.toString()));
265                            return values;
266                    }
267    
268                    //["gender", {}, "text", null]
269                    if (first.isNull()) {
270                            List<List<String>> values = new ArrayList<List<String>>(1);
271                            values.add(Arrays.asList(""));
272                            return values;
273                    }
274    
275                    return Collections.emptyList();
276            }
277    
278            /**
279             * Gets the value of a multi-valued property (such as {@link CategoriesType}
280             * ).
281             * @return the values or empty list if not found
282             */
283            public List<String> asMulti() {
284                    if (values.isEmpty()) {
285                            return Collections.emptyList();
286                    }
287    
288                    List<String> multi = new ArrayList<String>(values.size());
289                    for (JsonValue value : values) {
290                            if (value.isNull()) {
291                                    multi.add("");
292                                    continue;
293                            }
294    
295                            Object obj = value.getValue();
296                            if (obj != null) {
297                                    multi.add(obj.toString());
298                                    continue;
299                            }
300                    }
301                    return multi;
302            }
