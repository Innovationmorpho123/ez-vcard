041    /**
042     * Holds the data type and value of a jCal property.
043     * @author Michael Angstadt
044     */
045    public class JCardValue {
046            private final JCardDataType dataType;
047            private final List<JsonValue> values;
048    
049            /**
050             * Creates a new jCal value.
051             * @param dataType the data type or null for "unknown"
052             * @param values the values
053             */
054            public JCardValue(JCardDataType dataType, List<JsonValue> values) {
055                    this.dataType = dataType;
056                    this.values = Collections.unmodifiableList(values);
057            }
058    
059            /**
060             * Creates a new jCal value.
061             * @param dataType the data type or null for "unknown"
062             * @param values the values
063             */
064            public JCardValue(JCardDataType dataType, JsonValue... values) {
065                    this.dataType = dataType;
066                    this.values = Arrays.asList(values); //unmodifiable
067            }
068    
069            /**
070             * Creates a single-valued value.
071             * @param dataType the data type or null for "unknown"
072             * @param value the value
073             * @return the jCal value
074             */
075            public static JCardValue single(JCardDataType dataType, Object value) {
076                    return new JCardValue(dataType, new JsonValue(value));
077            }
078    
079            /**
080             * Creates a multi-valued value.
081             * @param dataType the data type or null for "unknown"
082             * @param values the values
083             * @return the jCal value
084             */
085            public static JCardValue multi(JCardDataType dataType, Object... values) {
086                    return multi(dataType, Arrays.asList(values));
087            }
088    
089            /**
090             * Creates a multi-valued value.
091             * @param dataType the data type or null for "unknown"
092             * @param values the values
093             * @return the jCal value
094             */
095            public static JCardValue multi(JCardDataType dataType, List<?> values) {
096                    List<JsonValue> multiValues = new ArrayList<JsonValue>(values.size());
097                    for (Object value : values) {
098                            multiValues.add(new JsonValue(value));
099                    }
100                    return new JCardValue(dataType, multiValues);
101            }
102    
103            /**
104             * Creates a structured value.
105             * @param dataType the data type or null for "unknown"
106             * @param values the values
107             * @return the jCal value
108             */
109            public static JCardValue structured(JCardDataType dataType, Object... values) {
110                    List<List<?>> valuesList = new ArrayList<List<?>>(values.length);
111                    for (Object value : values) {
112                            valuesList.add(Arrays.asList(value));
113                    }
114                    return structured(dataType, valuesList);
115            }
116    
117            /**
118             * Creates a structured value.
119             * @param dataType the data type or null for "unknown"
120             * @param values the values
121             * @return the jCal value
122             */
123            public static JCardValue structured(JCardDataType dataType, List<List<?>> values) {
124                    List<JsonValue> array = new ArrayList<JsonValue>(values.size());
125                    for (List<?> list : values) {
126                            if (list.isEmpty()) {
127                                    array.add(new JsonValue(""));
128                            } else if (list.size() == 1) {
129                                    Object value = list.get(0);
130                                    if (value == null) {
131                                            value = "";
132                                    }
133                                    array.add(new JsonValue(value));
134                            } else {
135                                    List<JsonValue> subArray = new ArrayList<JsonValue>(list.size());
136                                    for (Object value : list) {
137                                            if (value == null) {
138                                                    value = "";
139                                            }
140                                            subArray.add(new JsonValue(value));
141                                    }
142                                    array.add(new JsonValue(subArray));
143                            }
144                    }
145                    return new JCardValue(dataType, new JsonValue(array));
146            }
147    
148            /**
149             * Gets the jCard data type
150             * @return the data type or null for "unknown"
151             */
152            public JCardDataType getDataType() {
153                    return dataType;
154            }
155    
156            /**
157             * Gets all the JSON values.
158             * @return the JSON values
159             */
160            public List<JsonValue> getValues() {
161                    return values;
162            }
163    
164            /**
165             * Gets the value of a single-valued property (such as {@link NoteType}).
166             * @return the value or null if not found
167             */
168            public String getSingleValued() {
169                    if (values.isEmpty()) {
170                            return null;
171                    }
172    
173                    JsonValue first = values.get(0);
174    
175                    if (first.isNull()) {
176                            return null;
177                    }
178    
179                    Object obj = first.getValue();
180                    if (obj != null) {
181                            return obj.toString();
182                    }
183    
184                    //get the first element of the array
185                    List<JsonValue> array = first.getArray();
186                    if (array != null && !array.isEmpty()) {
187                            obj = array.get(0).getValue();
188                            if (obj != null) {
189                                    return obj.toString();
190                            }
191                    }
192    
193                    return null;
194            }
195    
196            /**
197             * Gets the value of a structured property (such as
198             * {@link StructuredNameType}).
199             * @return the values or empty list if not found
200             */
201            public List<List<String>> getStructured() {
202                    if (values.isEmpty()) {
203                            return Collections.emptyList();
204                    }
205    
206                    JsonValue first = values.get(0);
207    
208                    //["gender", {}, "text", ["M", "text"] ]
209                    List<JsonValue> array = first.getArray();
210                    if (array != null) {
211                            List<List<String>> valuesStr = new ArrayList<List<String>>(array.size());
212                            for (JsonValue value : array) {
213                                    if (value.isNull()) {
214                                            valuesStr.add(Arrays.asList((String) null));
215                                            continue;
216                                    }
217    
218                                    Object obj = value.getValue();
219                                    if (obj != null) {
220                                            valuesStr.add(Arrays.asList(obj.toString()));
221                                            continue;
222                                    }
223    
224                                    List<JsonValue> subArray = value.getArray();
225                                    if (subArray != null) {
226                                            List<String> subValuesStr = new ArrayList<String>(subArray.size());
227                                            for (JsonValue subArrayValue : subArray) {
228                                                    if (subArrayValue.isNull()) {
229                                                            subValuesStr.add(null);
230                                                            continue;
231                                                    }
232    
233                                                    obj = subArrayValue.getValue();
234                                                    if (obj != null) {
235                                                            subValuesStr.add(obj.toString());
236                                                            continue;
237                                                    }
238                                            }
239                                            valuesStr.add(subValuesStr);
240                                    }
241                            }
242                            return valuesStr;
243                    }
244    
245                    //get the first value if it's not enclosed in an array
246                    //["gender", {}, "text", "M"]
247                    Object obj = first.getValue();
248                    if (obj != null) {
249                            List<List<String>> values = new ArrayList<List<String>>(1);
250                            values.add(Arrays.asList(obj.toString()));
251                            return values;
252                    }
253    
254                    //["gender", {}, "text", null]
255                    if (first.isNull()) {
256                            List<List<String>> values = new ArrayList<List<String>>(1);
257                            values.add(Arrays.asList((String) null));
258                            return values;
259                    }
260    
261                    return Collections.emptyList();
262            }
263    
264            /**
265             * Gets the value of a multi-valued property (such as {@link CategoriesType}
266             * ).
267             * @return the values or empty list if not found
268             */
269            public List<String> getMultivalued() {
270                    if (values.isEmpty()) {
271                            return Collections.emptyList();
272                    }
273    
274                    List<String> multi = new ArrayList<String>(values.size());
275                    for (JsonValue value : values) {
276                            if (value.isNull()) {
277                                    multi.add(null);
278                                    continue;
279                            }
280    
281                            Object obj = value.getValue();
282                            if (obj != null) {
283                                    multi.add(obj.toString());
284                            }
285                    }
286                    return multi;
287            }
