594        static abstract class ParserChain<T> {
595                final ScribeIndex index = new ScribeIndex();
596                List<List<String>> warnings;
597
598                @SuppressWarnings("unchecked")
599                final T this_ = (T) this;
600
601                /**
602                 * Registers a property scribe.
603                 * @param scribe the scribe
604                 * @return this
605                 */
606                public T register(VCardPropertyScribe<? extends VCardProperty> scribe) {
607                        index.register(scribe);
608                        return this_;
609                }
610
611                /**
612                 * Provides a list object that any unmarshal warnings will be put into.
613                 * @param warnings the list object that will be populated with the
614                 * warnings of each unmarshalled vCard. Each element of the list is the
615                 * list of warnings for one of the unmarshalled vCards. Therefore, the
616                 * size of this list will be equal to the number of parsed vCards. If a
617                 * vCard does not have any warnings, then its warning list will be
618                 * empty.
619                 * @return this
620                 */
621                public T warnings(List<List<String>> warnings) {
622                        this.warnings = warnings;
623                        return this_;
624                }
625
626                /**
627                 * Reads the first vCard from the stream.
628                 * @return the vCard or null if there are no vCards
629                 * @throws IOException if there's an I/O problem
630                 * @throws SAXException if there's a problem parsing the XML
631                 */
632                public abstract VCard first() throws IOException, SAXException;
633
634                /**
635                 * Reads all vCards from the stream.
636                 * @return the parsed vCards
637                 * @throws IOException if there's an I/O problem
638                 * @throws SAXException if there's a problem parsing the XML
639                 */
640                public abstract List<VCard> all() throws IOException, SAXException;
641        }
642
643        static abstract class ParserChainText<T> extends ParserChain<T> {
644                boolean caretDecoding = true;
645                final boolean closeWhenDone;
646
647                private ParserChainText(boolean closeWhenDone) {
648                        this.closeWhenDone = closeWhenDone;
649                }
650
651                /**
652                 * Sets whether the reader will decode characters in parameter values
653                 * that use circumflex accent encoding (enabled by default).
654                 * 
655                 * @param enable true to use circumflex accent decoding, false not to
656                 * @return this
657                 * @see VCardReader#setCaretDecodingEnabled(boolean)
658                 * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
659                 */
660                public T caretDecoding(boolean enable) {
661                        caretDecoding = enable;
662                        return this_;
663                }
664
665                @Override
666                public VCard first() throws IOException {
667                        VCardReader parser = constructReader();
668
669                        try {
670                                VCard vcard = parser.readNext();
671                                if (warnings != null) {
