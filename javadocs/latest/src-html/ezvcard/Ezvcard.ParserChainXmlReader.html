643            static abstract class ParserChain<T> {
644                    final List<Class<? extends VCardType>> extendedTypes = new ArrayList<Class<? extends VCardType>>();
645                    List<List<String>> warnings;
646    
647                    @SuppressWarnings("unchecked")
648                    final T this_ = (T) this;
649    
650                    /**
651                     * Registers an extended type class.
652                     * @param typeClass the extended type class
653                     * @return this
