032    /**
033     * Manages the list of pre-defined values for a media type parameter.
034     * @author Michael Angstadt
035     * @param <T> the parameter class
036     */
037    public class MediaTypeCaseClasses<T extends MediaTypeParameter> extends CaseClasses<T, String[]> {
038            public MediaTypeCaseClasses(Class<T> clazz) {
039                    super(clazz);
040            }
041    
042            @Override
043            protected T create(String[] value) {
044                    try {
045                            //reflection: return new ClassName(value, mediaType, extension);
046                            Constructor<T> constructor = clazz.getDeclaredConstructor(String.class, String.class, String.class);
047                            constructor.setAccessible(true);
048                            return constructor.newInstance(value[0], value[1], value[2]);
049                    } catch (Exception e) {
050                            throw new RuntimeException(e);
051                    }
052            }
053    
054            @Override
055            protected boolean matches(T object, String[] value) {
056                    String objectValues[] = new String[] { object.getValue(), object.getMediaType(), object.getExtension() };
057                    for (int i = 0; i < value.length; i++) {
058                            String v = value[i];
059                            if (v != null && !v.equalsIgnoreCase(objectValues[i])) {
060                                    return false;
061                            }
062                    }
063                    return true;
064            }
065    }
