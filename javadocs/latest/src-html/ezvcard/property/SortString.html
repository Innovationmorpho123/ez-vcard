037    /**
038     * <p>
039     * Defines a string that should be used when an application sorts this vCard in
040     * some way.
041     * </p>
042     * 
043     * <p>
044     * The SORT-STRING property is not supported in 4.0. Instead, use the
045     * {@link StructuredName#setSortAs} and/or {@link Organization#setSortAs}
046     * methods.
047     * </p>
048     * 
049     * <p>
050     * <b>Code sample (2.1, 3.0)</b>
051     * </p>
052     * 
053     * <pre class="brush:java">
054     * VCard vcard = new VCard();
055     * StructuredName n = new StructuredName();
056     * n.setFamily(&quot;d'Armour&quot;);
057     * n.setGiven(&quot;Miles&quot;);
058     * vcard.setStructuredName(n);
059     * SortString sortString = new SortString(&quot;Armour&quot;);
060     * vcard.setSortString(sortString);
061     * </pre>
062     * 
063     * <p>
064     * <b>Code sample (4.0)</b>
065     * </p>
066     * 
067     * <pre class="brush:java">
068     * VCard vcard = new VCard();
069     * StructuredName n = new StructuredName();
070     * n.setFamily(&quot;d'Armour&quot;);
071     * n.setGiven(&quot;Miles&quot;);
072     * n.setSortAs(&quot;Armour&quot;);
073     * vcard.setStructuredName(n);
074     * </pre>
075     * 
076     * <p>
077     * <b>Property name:</b> {@code SORT-STRING}
078     * </p>
079     * <p>
080     * <b>Supported versions:</b> {@code 2.1, 3.0}
081     * </p>
082     * @author Michael Angstadt
083     */
084    public class SortString extends TextProperty {
085            /**
086             * Creates a sort-string property.
087             * @param sortString the sort string
088             */
089            public SortString(String sortString) {
090                    super(sortString);
091            }
092    
093            @Override
094            public Set<VCardVersion> _supportedVersions() {
095                    return EnumSet.of(VCardVersion.V2_1, VCardVersion.V3_0);
096            }
097    }
