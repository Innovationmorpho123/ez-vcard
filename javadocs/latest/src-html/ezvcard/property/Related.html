043/**
044 * <p>
045 * Defines someone that the person is related to.
046 * </p>
047 * 
048 * <p>
049 * <b>Code sample</b>
050 * </p>
051 * 
052 * <pre class="brush:java">
053 * VCard vcard = new VCard();
054 * 
055 * //static factory methods
056 * Related related = Related.email(&quot;bob.smith@example.com&quot;);
057 * related.addType(RelatedType.CO_WORKER);
058 * related.addType(RelatedType.FRIEND);
059 * vcard.addRelated(related);
060 * 
061 * //free-form text
062 * related = new Related();
063 * related.setText(&quot;Edna Smith&quot;);
064 * related.addType(RelatedType.SPOUSE);
065 * vcard.addRelated(related);
066 * 
067 * //reference another vCard by putting its UID property here
068 * related = new Related(&quot;urn:uuid:03a0e51f-d1aa-4385-8a53-e29025acd8af&quot;);
069 * related.addType(RelatedType.SIBLING);
070 * vcard.addRelated(related);
071 * </pre>
072 * 
073 * <p>
074 * <b>Property name:</b> {@code RELATED}
075 * </p>
076 * <p>
077 * <b>Supported versions:</b> {@code 4.0}
078 * </p>
079 * @author Michael Angstadt
080 */
081public class Related extends VCardProperty implements HasAltId {
082        private String uri;
083        private String text;
084
085        /**
086         * Creates a related property
087         */
088        public Related() {
089                //empty
090        }
091
092        /**
093         * Creates a related property.
094         * @param uri the URI representing the person
095         */
096        public Related(String uri) {
097                setUri(uri);
098        }
099
100        /**
101         * Creates a related property whose value is an email address.
102         * @param email the email address
103         * @return the property
104         */
105        public static Related email(String email) {
106                return new Related("mailto:" + email);
107        }
108
109        /**
110         * Creates a related property whose value is an instant messenger handle.
111         * @param protocol the instant messenger protocol (e.g. "aim")
112         * @param handle the instant messenger handle (e.g. "johndoe")
113         * @return the property
114         */
115        public static Related im(String protocol, String handle) {
116                return new Related(protocol + ":" + handle);
117        }
118
119        /**
120         * Creates a related property whose value is a telephone number.
121         * @param telUri the telephone number
122         * @return the property
123         */
124        public static Related telephone(TelUri telUri) {
125                return new Related(telUri.toString());
126        }
127
128        @Override
129        public Set<VCardVersion> _supportedVersions() {
130                return EnumSet.of(VCardVersion.V4_0);
131        }
132
133        /**
134         * Gets the URI value.
135         * @return the URI value or null if no URI value is set
136         */
137        public String getUri() {
138                return uri;
139        }
140
141        /**
142         * Sets the URI.
143         * @param uri the URI
144         */
145        public void setUri(String uri) {
146                this.uri = uri;
147                text = null;
148        }
149
150        /**
151         * Gets the text value.
152         * @return the text value or null if no text value is set
153         */
154        public String getText() {
155                return text;
156        }
157
158        /**
159         * Sets the value to free-form text instead of a URI.
160         * @param text the text
161         */
162        public void setText(String text) {
163                this.text = text;
164                uri = null;
165        }
166
167        /**
168         * Gets all the TYPE parameters.
169         * @return the TYPE parameters or empty set if there are none
170         */
171        public Set<RelatedType> getTypes() {
172                Set<String> values = parameters.getTypes();
173                Set<RelatedType> types = new HashSet<RelatedType>(values.size());
174                for (String value : values) {
175                        types.add(RelatedType.get(value));
176                }
177                return types;
178        }
179
180        /**
181         * Adds a TYPE parameter.
182         * @param type the TYPE parameter to add
183         */
184        public void addType(RelatedType type) {
185                parameters.addType(type.getValue());
186        }
187
188        /**
189         * Removes a TYPE parameter.
190         * @param type the TYPE parameter to remove
191         */
192        public void removeType(RelatedType type) {
193                parameters.removeType(type.getValue());
194        }
195
196        @Override
197        public List<Integer[]> getPids() {
198                return super.getPids();
199        }
200
201        @Override
202        public void addPid(int localId, int clientPidMapRef) {
203                super.addPid(localId, clientPidMapRef);
204        }
205
206        @Override
207        public void removePids() {
208                super.removePids();
209        }
210
211        @Override
212        public Integer getPref() {
213                return super.getPref();
214        }
215
216        @Override
217        public void setPref(Integer pref) {
218                super.setPref(pref);
219        }
220
221        //@Override
222        public String getAltId() {
223                return parameters.getAltId();
224        }
225
226        //@Override
227        public void setAltId(String altId) {
228                parameters.setAltId(altId);
229        }
230
231        @Override
232        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
233                if (uri == null && text == null) {
234                        warnings.add(new Warning(8));
235                }
236        }
237}
