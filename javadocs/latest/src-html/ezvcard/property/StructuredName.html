037/**
038 * <p>
039 * Defines the individual components of the person's name.
040 * </p>
041 * 
042 * <p>
043 * <b>Code sample</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * VCard vcard = new VCard();
048 * 
049 * StructuredName n = new StructuredName();
050 * n.setFamily(&quot;House&quot;);
051 * n.setGiven(&quot;Gregory&quot;);
052 * n.addPrefix(&quot;Dr&quot;);
053 * n.addSuffix(&quot;MD&quot;);
054 * vcard.setStructuredName(n);
055 * </pre>
056 * 
057 * <p>
058 * <b>Property name:</b> {@code N}
059 * </p>
060 * <p>
061 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
062 * </p>
063 * @author Michael Angstadt
064 */
065public class StructuredName extends VCardProperty implements HasAltId {
066        private String family;
067        private String given;
068        private List<String> additional = new ArrayList<String>();
069        private List<String> prefixes = new ArrayList<String>();
070        private List<String> suffixes = new ArrayList<String>();
071
072        /**
073         * Gets the family name (aka "last name").
074         * @return the family name or null if not set
075         */
076        public String getFamily() {
077                return family;
078        }
079
080        /**
081         * Sets the family name (aka "last name").
082         * @param family the family name or null to remove
083         */
084        public void setFamily(String family) {
085                this.family = family;
086        }
087
088        /**
089         * Gets the given name (aka "first name").
090         * @return the given name or null if not set
091         */
092        public String getGiven() {
093                return given;
094        }
095
096        /**
097         * Sets the given name (aka "first name").
098         * @param given the given name or null to remove
099         */
100        public void setGiven(String given) {
101                this.given = given;
102        }
103
104        /**
105         * Gets any additional names the person goes by.
106         * @return the additional names or empty list if there are none
107         */
108        public List<String> getAdditional() {
109                return additional;
110        }
111
112        /**
113         * Adds an additional name the person goes by.
114         * @param additional the additional name to add
115         */
116        public void addAdditional(String additional) {
117                this.additional.add(additional);
118        }
119
120        /**
121         * Gets the prefixes.
122         * @return the prefixes (e.g. "Mr.") or empty list if there are none
123         */
124        public List<String> getPrefixes() {
125                return prefixes;
126        }
127
128        /**
129         * Adds a prefix.
130         * @param prefix the prefix to add (e.g. "Mr.")
131         */
132        public void addPrefix(String prefix) {
133                this.prefixes.add(prefix);
134        }
135
136        /**
137         * Gets the suffixes.
138         * @return the suffixes (e.g. "Jr.") or empty list if there are none
139         */
140        public List<String> getSuffixes() {
141                return suffixes;
142        }
143
144        /**
145         * Adds a suffix.
146         * @param suffix the suffix to add (e.g. "Jr.")
147         */
148        public void addSuffix(String suffix) {
149                this.suffixes.add(suffix);
150        }
151
152        /**
153         * Gets the string(s) that define how to sort the vCard.
154         * <p>
155         * 2.1 and 3.0 vCards should use the {@link SortString SORT-STRING} property
156         * instead.
157         * </p>
158         * <p>
159         * <b>Supported versions:</b> {@code 4.0}
160         * </p>
161         * @return the sort string(s) (e.g. ["Aboville", "Christine"] if the family
162         * name is "d'Aboville" and the given name is "Christine") or empty list if
163         * there are none
164         * @see VCardParameters#getSortAs
165         */
166        public List<String> getSortAs() {
167                return parameters.getSortAs();
168        }
169
170        /**
171         * Sets the string that defines how to sort the vCard.
172         * <p>
173         * 2.1 and 3.0 vCards should use the {@link SortString SORT-STRING} property
174         * instead.
175         * </p>
176         * <p>
177         * <b>Supported versions:</b> {@code 4.0}
178         * </p>
179         * @param family the sorttable family name (e.g. "Adboville" if the family
180         * name is "d'Aboville") or null to remove
181         */
182        public void setSortAs(String family) {
183                if (family == null) {
184                        parameters.setSortAs();
185                } else {
186                        parameters.setSortAs(family);
187                }
188        }
189
190        /**
191         * Sets the strings that define how to sort the vCard.
192         * <p>
193         * 2.1 and 3.0 vCards should use the {@link SortString SORT-STRING} property
194         * instead.
195         * </p>
196         * <p>
197         * <b>Supported versions:</b> {@code 4.0}
198         * </p>
199         * @param family the sortable family name (e.g. "Adboville" if the family
200         * name is "d'Aboville")
201         * @param given the sortable given name
202         */
203        public void setSortAs(String family, String given) {
204                parameters.setSortAs(family, given);
205        }
206
207        @Override
208        public String getLanguage() {
209                return super.getLanguage();
210        }
211
212        @Override
213        public void setLanguage(String language) {
214                super.setLanguage(language);
215        }
216
217        //@Override
218        public String getAltId() {
219                return parameters.getAltId();
220        }
221
222        //@Override
223        public void setAltId(String altId) {
224                parameters.setAltId(altId);
225        }
226}
