041    /**
042     * A telephone number.
043     * 
044     * <p>
045     * <b>Code sample</b>
046     * </p>
047     * 
048     * <pre class="brush:java">
049     * VCard vcard = new VCard();
050     * Telephone tel = new Telephone(&quot;(123) 555-6789&quot;);
051     * tel.addType(TelephoneType.HOME);
052     * tel.setPref(2); //the second-most preferred
053     * vcard.addTelephoneNumber(tel);
054     * 
055     * TelUri uri = new TelUri.Builder(&quot;+1-800-555-9876&quot;).extension(&quot;111&quot;).build();
056     * tel = new Telephone(uri);
057     * tel.addType(TelephoneType.WORK);
058     * tel.setPref(1); //the most preferred
059     * vcard.addTelephoneNumber(tel);
060     * </pre>
061     * 
062     * <p>
063     * <b>Property name:</b> {@code TEL}
064     * </p>
065     * <p>
066     * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
067     * </p>
068     * @author Michael Angstadt
069     */
070    public class Telephone extends VCardProperty implements HasAltId {
071            private String text;
072            private TelUri uri;
073    
074            /**
075             * Creates a telephone property.
076             * @param text the telephone number (e.g. "(123) 555-6789")
077             */
078            public Telephone(String text) {
079                    setText(text);
080            }
081    
082            /**
083             * Creates a telephone property.
084             * @param uri a "tel" URI representing the telephone number (vCard 4.0 only)
085             */
086            public Telephone(TelUri uri) {
087                    setUri(uri);
088            }
089    
090            /**
091             * Gets the telephone number as a text value.
092             * @return the telephone number or null if the text value is not set
093             */
094            public String getText() {
095                    return text;
096            }
097    
098            /**
099             * Sets the telephone number as a text value.
100             * @param text the telephone number
101             */
102            public void setText(String text) {
103                    this.text = text;
104                    uri = null;
105            }
106    
107            /**
108             * Gets a "tel" URI representing the phone number.
109             * <p>
110             * <b>Supported versions:</b> {@code 4.0}
111             * </p>
112             * @return the "tel" URI or null if it is not set
113             */
114            public TelUri getUri() {
115                    return uri;
116            }
117    
118            /**
119             * Sets a "tel" URI representing the phone number.
120             * <p>
121             * <b>Supported versions:</b> {@code 4.0}
122             * </p>
123             * @param uri the "tel" URI
124             */
125            public void setUri(TelUri uri) {
126                    text = null;
127                    this.uri = uri;
128            }
129    
130            /**
131             * Gets all the TYPE parameters.
132             * @return the TYPE parameters or empty set if there are none
133             */
134            public Set<TelephoneType> getTypes() {
135                    Set<String> values = parameters.getTypes();
136                    Set<TelephoneType> types = new HashSet<TelephoneType>(values.size());
137                    for (String value : values) {
138                            types.add(TelephoneType.get(value));
139                    }
140                    return types;
141            }
142    
143            /**
144             * Adds a TYPE parameter.
145             * @param type the TYPE parameter to add
146             */
147            public void addType(TelephoneType type) {
148                    parameters.addType(type.getValue());
149            }
150    
151            /**
152             * Removes a TYPE parameter.
153             * @param type the TYPE parameter to remove
154             */
155            public void removeType(TelephoneType type) {
156                    parameters.removeType(type.getValue());
157            }
158    
159            @Override
160            public List<Integer[]> getPids() {
161                    return super.getPids();
162            }
163    
164            @Override
165            public void addPid(int localId, int clientPidMapRef) {
166                    super.addPid(localId, clientPidMapRef);
167            }
168    
169            @Override
170            public void removePids() {
171                    super.removePids();
172            }
173    
174            @Override
175            public Integer getPref() {
176                    return super.getPref();
177            }
178    
179            @Override
180            public void setPref(Integer pref) {
181                    super.setPref(pref);
182            }
183    
184            //@Override
185            public String getAltId() {
186                    return parameters.getAltId();
187            }
188    
189            //@Override
190            public void setAltId(String altId) {
191                    parameters.setAltId(altId);
192            }
193    
194            @Override
195            protected void _validate(List<String> warnings, VCardVersion version, VCard vcard) {
196                    if (uri == null && text == null) {
197                            warnings.add("Property has neither a URI nor a text value associated with it.");
198                    }
199    
200                    if (uri != null && (version == VCardVersion.V2_1 || version == VCardVersion.V3_0)) {
201                            warnings.add("\"tel\" URIs are not supported by vCard version " + version.getVersion() + ".  The URI will be converted to a string.  Some data may be lost.");
202                    }
203    
204                    for (TelephoneType type : getTypes()) {
205                            if (type == TelephoneType.PREF) {
206                                    //ignore because it is converted to a PREF parameter for 4.0 vCards
207                                    continue;
208                            }
209    
210                            if (!type.isSupported(version)) {
211                                    warnings.add("Type value \"" + type.getValue() + "\" is not supported in version " + version.getVersion() + ".");
212                            }
213                    }
214            }
215    }
