047    /**
048     * An email address associated with a person.
049     * 
050     * <p>
051     * <b>Code sample</b>
052     * </p>
053     * 
054     * <pre class="brush:java">
055     * VCard vcard = new VCard();
056     * 
057     * EmailType email = new EmailType(&quot;superdude55@hotmail.com&quot;);
058     * email.addType(EmailTypeParameter.HOME);
059     * vcard.addEmail(email);
060     * 
061     * email = new EmailType(&quot;doe.john@company.com&quot;);
062     * email.addType(EmailTypeParameter.WORK);
063     * email.setPref(1); //the most preferred email
064     * vcard.addEmail(email);
065     * </pre>
066     * 
067     * <p>
068     * <b>Property name:</b> {@code EMAIL}
069     * </p>
070     * <p>
071     * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
072     * </p>
073     * @author Michael Angstadt
074     */
075    public class EmailType extends MultiValuedTypeParameterType<EmailTypeParameter> implements HasAltId {
076            public static final String NAME = "EMAIL";
077    
078            private String value;
079    
080            /**
081             * Creates an empty email property.
082             */
083            public EmailType() {
084                    this(null);
085            }
086    
087            /**
088             * Creates an email property.
089             * @param email the email (e.g. "johndoe@example.com")
090             */
091            public EmailType(String email) {
092                    super(NAME);
093                    setValue(email);
094            }
095    
096            @Override
097            protected EmailTypeParameter buildTypeObj(String type) {
098                    return EmailTypeParameter.get(type);
099            }
100    
101            /**
102             * Gets the email address.
103             * @return the email address (e.g. "johndoe@example.com")
104             */
105            public String getValue() {
106                    return value;
107            }
108    
109            /**
110             * Sets the email address
111             * @param email the email address (e.g. "johndoe@example.com")
112             */
113            public void setValue(String email) {
114                    this.value = email;
115            }
116    
117            @Override
118            public List<Integer[]> getPids() {
119                    return super.getPids();
120            }
121    
122            @Override
123            public void addPid(int localId, int clientPidMapRef) {
124                    super.addPid(localId, clientPidMapRef);
125            }
126    
127            @Override
128            public void removePids() {
129                    super.removePids();
130            }
131    
132            @Override
133            public Integer getPref() {
134                    return super.getPref();
135            }
136    
137            @Override
138            public void setPref(Integer pref) {
139                    super.setPref(pref);
140            }
141    
142            //@Override
143            public String getAltId() {
144                    return subTypes.getAltId();
145            }
146    
147            //@Override
148            public void setAltId(String altId) {
149                    subTypes.setAltId(altId);
150            }
151    
152            @Override
153            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, CompatibilityMode compatibilityMode, VCard vcard) {
154                    switch (version) {
155                    case V2_1:
156                    case V3_0:
157                            copy.setPref(null);
158    
159                            //find the EMAIL with the lowest PREF value in the vCard
160                            EmailType mostPreferred = null;
161                            for (EmailType email : vcard.getEmails()) {
162                                    Integer pref = email.getPref();
163                                    if (pref == null) {
164                                            continue;
165                                    }
166    
167                                    if (mostPreferred == null || pref < mostPreferred.getPref()) {
168                                            mostPreferred = email;
169                                    }
170                            }
171                            if (this == mostPreferred) {
172                                    copy.addType(EmailTypeParameter.PREF.getValue());
173                            }
174                            break;
175                    case V4_0:
176                            //replace "TYPE=pref" with "PREF=1"
177                            if (getTypes().contains(EmailTypeParameter.PREF)) {
178                                    copy.removeType(EmailTypeParameter.PREF.getValue());
179                                    copy.setPref(1);
180                            }
181                            break;
182                    }
183            }
184    
185            @Override
186            protected void doMarshalText(StringBuilder sb, VCardVersion version, CompatibilityMode compatibilityMode) {
187                    if (value != null) {
188                            sb.append(VCardStringUtils.escape(value));
189                            return;
190                    }
191            }
192    
193            @Override
194            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
195                    setValue(VCardStringUtils.unescape(value));
196            }
197    
198            @Override
199            protected void doMarshalXml(XCardElement parent, CompatibilityMode compatibilityMode) {
200                    parent.append(VCardDataType.TEXT, value);
201            }
202    
203            @Override
204            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
205                    String value = element.first(VCardDataType.TEXT);
206                    if (value != null) {
207                            setValue(value);
208                            return;
209                    }
210    
211                    throw missingXmlElements(VCardDataType.TEXT);
212            }
213    
214            @Override
215            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
216                    List<String> types = element.types();
217                    for (String type : types) {
218                            subTypes.addType(type);
219                    }
220    
221                    //check to see if the email address is within in "mailto:" link
222                    String email = null;
223                    String href = element.attr("href");
224                    if (href.length() > 0) {
225                            Pattern p = Pattern.compile("^mailto:(.*)$", Pattern.CASE_INSENSITIVE);
226                            Matcher m = p.matcher(href);
227                            if (m.find()) {
228                                    email = m.group(1);
229                            }
230                    }
231                    if (email == null) {
232                            email = element.value();
233                    }
234                    setValue(email);
235            }
236    
237            @Override
238            protected JCardValue doMarshalJson(VCardVersion version) {
239                    return JCardValue.single(VCardDataType.TEXT, value);
240            }
241    
242            @Override
243            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
244                    setValue(value.asSingle());
245            }
246    
247            @Override
248            protected void _validate(List<String> warnings, VCardVersion version, VCard vcard) {
249                    if (value == null) {
250                            warnings.add("Property value is null.");
251                    }
252    
253                    for (EmailTypeParameter type : getTypes()) {
254                            if (type == EmailTypeParameter.PREF) {
255                                    //ignore because it is converted to a PREF parameter for 4.0 vCards
256                                    continue;
257                            }
258                            if (!type.isSupported(version)) {
259                                    warnings.add("Type value \"" + type.getValue() + "\" is not supported in version " + version.getVersion() + ".");
260                            }
261                    }
262            }
263    }
