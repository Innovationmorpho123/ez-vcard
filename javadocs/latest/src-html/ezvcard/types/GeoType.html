046    /**
047     * <p>
048     * A set of latitude/longitude coordinates. There is no rule for what these
049     * coordinates must represent, but the meaning could vary depending on the value
050     * of {@link KindType}:
051     * </p>
052     * 
053     * <table border="1">
054     * <tr>
055     * <th>KIND value</th>
056     * <th>GEO meaning</th>
057     * </tr>
058     * <tr>
059     * <td>"individual"</td>
060     * <td>the location of the person's home or workplace.</td>
061     * </tr>
062     * <tr>
063     * <td>"group"</td>
064     * <td>the location of the group's meeting place.</td>
065     * </tr>
066     * <tr>
067     * <td>"org"</td>
068     * <td>the coordinates of the organization's headquarters.</td>
069     * </tr>
070     * <tr>
071     * <td>"location"</td>
072     * <td>the coordinates of the location itself.</td>
073     * </tr>
074     * </table>
075     * 
076     * <p>
077     * <b>Code sample</b>
078     * </p>
079     * 
080     * <pre class="brush:java">
081     * VCard vcard = new VCard();
082     * GeoType geo = new GeoType(-123.456, 12.54);
083     * vcard.setGeo(geo);
084     * </pre>
085     * 
086     * <p>
087     * <b>Property name:</b> {@code GEO}
088     * </p>
089     * <p>
090     * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
091     * </p>
092     * @author Michael Angstadt
093     */
094    public class GeoType extends VCardType implements HasAltId {
095            public static final String NAME = "GEO";
096            private GeoUri uri;
097    
098            /**
099             * Creates an empty geo property.
100             */
101            public GeoType() {
102                    this(null);
103            }
104    
105            /**
106             * Creates a geo property.
107             * @param latitude the latitude
108             * @param longitude the longitude
109             */
110            public GeoType(Double latitude, Double longitude) {
111                    this(new GeoUri.Builder(latitude, longitude).build());
112            }
113    
114            /**
115             * Creates a geo property.
116             * @param uri the geo URI
117             */
118            public GeoType(GeoUri uri) {
119                    super(NAME);
120                    this.uri = uri;
121            }
122    
123            /**
124             * Gets the latitude.
125             * @return the latitude
126             */
127            public Double getLatitude() {
128                    return (uri == null) ? null : uri.getCoordA();
129            }
130    
131            /**
132             * Sets the latitude.
133             * @param latitude the latitude
134             */
135            public void setLatitude(Double latitude) {
136                    if (uri == null) {
137                            uri = new GeoUri.Builder(latitude, null).build();
138                    } else {
139                            uri = new GeoUri.Builder(uri).coordA(latitude).build();
140                    }
141            }
142    
143            /**
144             * Gets the longitude.
145             * @return the longitude
146             */
147            public Double getLongitude() {
148                    return (uri == null) ? null : uri.getCoordB();
149            }
150    
151            /**
152             * Sets the longitude.
153             * @param longitude the longitude
154             */
155            public void setLongitude(Double longitude) {
156                    if (uri == null) {
157                            uri = new GeoUri.Builder(null, longitude).build();
158                    } else {
159                            uri = new GeoUri.Builder(uri).coordB(longitude).build();
160                    }
161            }
162    
163            /**
164             * Gets the raw object used for storing the GEO information. This can be
165             * used to supplement the GEO value with additional information (such as
166             * altitude or level of accuracy). Geo URIs are only supported by vCard
167             * version 4.0. Only latitude and longitude values are used when marshalling
168             * to earlier vCard versions.
169             * @return the geo URI object or null if not set
170             * @see <a href="http://tools.ietf.org/html/rfc5870">RFC 5870</a>
171             */
172            public GeoUri getGeoUri() {
173                    return uri;
174            }
175    
176            /**
177             * Sets the raw object used for storing the GEO information. This can be
178             * used to supplement the GEO value with additional information (such as
179             * altitude or level of accuracy). Geo URIs are only supported by vCard
180             * version 4.0. Only latitude and longitude values are used when marshalling
181             * to earlier vCard versions.
182             * @param uri the geo URI object
183             * @see <a href="http://tools.ietf.org/html/rfc5870">RFC 5870</a>
184             */
185            public void setGeoUri(GeoUri uri) {
186                    this.uri = uri;
187            }
188    
189            /**
190             * Gets the TYPE parameter.
191             * <p>
192             * <b>Supported versions:</b> {@code 4.0}
193             * </p>
194             * @return the TYPE value (typically, this will be either "work" or "home")
195             * or null if it doesn't exist
196             */

198                    return subTypes.getType();

207             * null to remove
208             */

232                    subTypes.setMediaType(mediaType);

270            @Override
271            protected void doMarshalText(StringBuilder sb, VCardVersion version, CompatibilityMode compatibilityMode) {
272                    sb.append(write(version));
273            }
274    
275            @Override
276            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
277                    value = VCardStringUtils.unescape(value);
278                    parse(value, version, warnings);
279            }
280    
281            @Override
282            protected void doMarshalXml(XCardElement parent, CompatibilityMode compatibilityMode) {
283                    parent.append(VCardDataType.URI, write(parent.version()));
284            }
285    
286            @Override
287            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
288                    String value = element.first(VCardDataType.URI);
289                    if (value != null) {
290                            parse(value, element.version(), warnings);
291                            return;
292                    }
293    
294                    throw missingXmlElements(VCardDataType.URI);
295            }
296    
297            @Override
298            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
299                    String latitudeStr = element.firstValue("latitude");
300                    if (latitudeStr == null) {
301                            throw new CannotParseException("Latitude missing.");
302                    }
303    
304                    Double latitude;
305                    try {
306                            latitude = Double.parseDouble(latitudeStr);
307                    } catch (NumberFormatException e) {
308                            throw new CannotParseException("Could not parse latitude: " + latitudeStr);
309                    }
310    
311                    String longitudeStr = element.firstValue("longitude");
312                    if (longitudeStr == null) {
313                            throw new CannotParseException("Longitude missing.");
314                    }
315    
316                    Double longitude;
317                    try {
318                            longitude = Double.parseDouble(longitudeStr);
319                    } catch (NumberFormatException e) {
320                            throw new CannotParseException("Could not parse longitude: " + longitudeStr);
321                    }
322    
323                    uri = new GeoUri.Builder(latitude, longitude).build();
324            }
325    
326            @Override
327            protected JCardValue doMarshalJson(VCardVersion version) {
328                    return JCardValue.single(VCardDataType.URI, write(version));
329            }
330    
331            @Override
332            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
333                    parse(value.asSingle(), version, warnings);
334            }
335    
336            @Override
337            protected void _validate(List<String> warnings, VCardVersion version, VCard vcard) {
338                    if (getLatitude() == null) {
339                            warnings.add("Latitude is missing.");
340                    }
341                    if (getLongitude() == null) {
342                            warnings.add("Longitude is missing.");
343                    }
344            }
345    
346            private void parse(String value, VCardVersion version, List<String> warnings) {
347                    if (value == null || value.length() == 0) {
348                            return;
349                    }
350    
351                    switch (version) {
352                    case V2_1:
353                    case V3_0:
354                            String split[] = value.split(";");
355                            if (split.length != 2) {
356                                    throw new CannotParseException("Incorrect data format.  Value must contain a latitude and longitude, separated by a semi-colon.");
357                            }
358    
359                            String latitudeStr = split[0];
360                            Double latitude;
361                            try {
362                                    latitude = Double.valueOf(latitudeStr);
363                            } catch (NumberFormatException e) {
364                                    throw new CannotParseException("Could not parse latitude: " + latitudeStr);
365                            }
366    
367                            String longitudeStr = split[1];
368                            Double longitude;
369                            try {
370                                    longitude = Double.valueOf(longitudeStr);
371                            } catch (NumberFormatException e) {
372                                    throw new CannotParseException("Could not parse longtude: " + longitudeStr);
373                            }
374    
375                            uri = new GeoUri.Builder(latitude, longitude).build();
376                            break;
377                    case V4_0:
378                            try {
379                                    uri = GeoUri.parse(value);
380                            } catch (IllegalArgumentException e) {
381                                    throw new CannotParseException("Invalid geo URI.");
382                            }
383                            break;
384                    }
385            }
386    
387            private String write(VCardVersion version) {
388                    if (uri == null) {
389                            return "";
390                    }
391    
392                    switch (version) {
393                    case V2_1:
394                    case V3_0:
395                            VCardFloatFormatter formatter = new VCardFloatFormatter(6);
396                            StringBuilder sb = new StringBuilder();
397    
398                            sb.append(formatter.format(getLatitude()));
399                            sb.append(';');
400                            sb.append(formatter.format(getLongitude()));
401    
402                            return sb.toString();
403                    case V4_0:
404                            return uri.toString(6);
405                    }
406                    return null; //needed to prevent compilation error
407            }
