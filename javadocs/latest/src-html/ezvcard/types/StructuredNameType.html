047    /**
048     * Contains the separated components of the person's name.
049     * 
050     * <p>
051     * <b>Code sample</b>
052     * </p>
053     * 
054     * <pre>
055     * VCard vcard = new VCard();
056     * StructuredNameType n = new StructuredNameType();
057     * n.setFamily(&quot;House&quot;);
058     * n.setGiven(&quot;Gregory&quot;);
059     * n.addPrefix(&quot;Dr&quot;);
060     * n.addSuffix(&quot;MD&quot;);
061     * vcard.setStructuredName(n);
062     * </pre>
063     * 
064     * <p>
065     * <b>Property name:</b> <code>N</code>
066     * </p>
067     * <p>
068     * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
069     * </p>
070     * @author Michael Angstadt
071     */
072    public class StructuredNameType extends VCardType implements HasAltId {
073            public static final String NAME = "N";
074    
075            private String family;
076            private String given;
077            private List<String> additional = new ArrayList<String>();
078            private List<String> prefixes = new ArrayList<String>();
079            private List<String> suffixes = new ArrayList<String>();
080    
081            /**
082             * Creates a structured name property.
083             */
084            public StructuredNameType() {
085                    super(NAME);
086            }
087    
088            /**
089             * Gets the family name (aka "last name").
090             * @return the family name or null if not set
091             */
092            public String getFamily() {

094            }

096            /**
097             * Sets the family name (aka "last name").
098             * @param family the family name or null to remove

101                    this.family = family;
102            }
103    
104            /**
105             * Gets the given name (aka "first name").
106             * @return the given name or null if not set
107             */
108            public String getGiven() {
109                    return given;
110            }
111    
112            /**
113             * Sets the given name (aka "first name").
114             * @param given the given name or null to remove
115             */
116            public void setGiven(String given) {
117                    this.given = given;
118            }
119    
120            /**
121             * Gets any additional names the person goes by.
122             * @return the additional names or empty list if there are none
123             */
124            public List<String> getAdditional() {
125                    return additional;
126            }
127    
128            /**
129             * Adds an additional name the person goes by.

131             */
132            public void addAdditional(String additional) {
133                    this.additional.add(additional);
134            }
135    
136            /**
137             * Gets the prefixes.
138             * @return the prefixes (e.g. "Mr.") or empty list if there are none
139             */
140            public List<String> getPrefixes() {
141                    return prefixes;
142            }
143    
144            /**
145             * Adds a prefix.
146             * @param prefix the prefix to add (e.g. "Mr.")
147             */
148            public void addPrefix(String prefix) {
149                    this.prefixes.add(prefix);
150            }
151    
152            /**
153             * Gets the suffixes.
154             * @return the suffixes (e.g. "Jr.") or empty list if there are none
155             */
156            public List<String> getSuffixes() {
157                    return suffixes;
158            }
159    
160            /**
161             * Adds a suffix.
162             * @param suffix the suffix to add (e.g. "Jr.")
163             */
164            public void addSuffix(String suffix) {
165                    this.suffixes.add(suffix);
166            }
167    
168            /**
169             * Gets the string(s) that define how to sort the vCard.
170             * <p>
171             * 2.1 and 3.0 vCards should use the {@link SortStringType SORT-STRING}
172             * property instead.
173             * </p>
174             * <p>
175             * <b>Supported versions:</b> <code>4.0</code>
176             * </p>
177             * @return the sort string(s) (e.g. ["Aboville", "Christine"] if the family
178             * name is "d'Aboville" and the given name is "Christine") or empty list if
179             * there are none
180             * @see VCardSubTypes#getSortAs
181             */
182            public List<String> getSortAs() {
183                    return subTypes.getSortAs();
184            }
185    
186            /**
187             * Sets the string that defines how to sort the vCard.
188             * <p>
189             * 2.1 and 3.0 vCards should use the {@link SortStringType SORT-STRING}
190             * property instead.
191             * </p>
192             * <p>
193             * <b>Supported versions:</b> <code>4.0</code>
194             * </p>
195             * @param family the sorttable family name (e.g. "Adboville" if the family
196             * name is "d'Aboville") or null to remove
197             */
198            public void setSortAs(String family) {
199                    if (family == null) {
200                            subTypes.setSortAs();
201                    } else {
202                            subTypes.setSortAs(family);
203                    }
204            }
205    
206            /**
207             * Sets the strings that define how to sort the vCard.
208             * <p>
209             * 2.1 and 3.0 vCards should use the {@link SortStringType SORT-STRING}
210             * property instead.
211             * </p>
212             * <p>
213             * <b>Supported versions:</b> <code>4.0</code>
214             * </p>
215             * @param family the sortable family name (e.g. "Adboville" if the family
216             * name is "d'Aboville")
217             * @param given the sortable given name
218             */
219            public void setSortAs(String family, String given) {
220                    subTypes.setSortAs(family, given);
221            }

243            @Override
244            protected void doMarshalText(final StringBuilder value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
245                    List<List<String>> values = new ArrayList<List<String>>();
246                    values.add(Arrays.asList(family));
247                    values.add(Arrays.asList(given));
248                    values.add(additional);
249                    values.add(prefixes);
250                    values.add(suffixes);
251    
252                    VCardStringUtils.join(values, ";", value, new JoinCallback<List<String>>() {
253                            public void handle(StringBuilder sb, List<String> v) {
254                                    VCardStringUtils.join(v, ",", value, new JoinCallback<String>() {
255                                            public void handle(StringBuilder sb, String v) {
256                                                    if (v != null) {
257                                                            sb.append(VCardStringUtils.escape(v));
258                                                    }
259                                            }
260                                    });
261                            }
262                    });
263            }
264    
265            @Override
266            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
267                    //preserve empty items and don't unescape escaped characters (e.g. "additional" might have escaped commas)
268                    Iterator<String> it = VCardStringUtils.splitBy(value, ';', false, false).iterator();
269    
270                    family = nextTextComponent(it);
271                    given = nextTextComponent(it);
272                    additional = nextTextListComponent(it);
273                    prefixes = nextTextListComponent(it);
274                    suffixes = nextTextListComponent(it);
275            }
276    
277            private String nextTextComponent(Iterator<String> it) {
278                    if (!it.hasNext()) {
279                            return null;
280                    }
281    
282                    String value = it.next();
283                    return (value.length() == 0) ? null : VCardStringUtils.unescape(value);
284            }
285    
286            private List<String> nextTextListComponent(Iterator<String> it) {
287                    if (!it.hasNext()) {
288                            return new ArrayList<String>(0);
289                    }
290    
291                    String value = it.next();
292                    return VCardStringUtils.splitBy(value, ',', true, true);
293            }

295            @Override
296            protected void doMarshalXml(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
297                    parent.append("surname", family); //the XML element still needs to be printed if value == null
298                    parent.append("given", given);
299                    parent.append("additional", additional);
300                    parent.append("prefix", prefixes);
301                    parent.append("suffix", suffixes);
302            }
303    
304            @Override
305            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
306                    family = sanitizeXml(element, "surname");
307                    given = sanitizeXml(element, "given");
308                    additional = element.getAll("additional");
309                    prefixes = element.getAll("prefix");
310                    suffixes = element.getAll("suffix");
311            }
312    
313            private String sanitizeXml(XCardElement element, String name) {
314                    String value = element.get(name);
315                    return (value != null && value.length() == 0) ? null : value;
316            }
317    
318            @Override
319            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
320                    family = element.firstValue("family-name");
321                    given = element.firstValue("given-name");
322                    additional = element.allValues("additional-name");
323                    prefixes = element.allValues("honorific-prefix");
324                    suffixes = element.allValues("honorific-suffix");
325            }
326    
327            @Override
328            protected JCardValue doMarshalJson(VCardVersion version, List<String> warnings) {
329                    List<List<?>> values = new ArrayList<List<?>>();
330                    values.add(Arrays.asList(family));
331                    values.add(Arrays.asList(given));
332                    values.add(additional);
333                    values.add(prefixes);
334                    values.add(suffixes);
335                    return JCardValue.structured(JCardDataType.TEXT, values);
336            }
337    
338            @Override
339            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
340                    Iterator<List<String>> it = value.getStructured().iterator();
341    
342                    family = nextJsonComponent(it);
343                    given = nextJsonComponent(it);
344                    additional = nextJsonListComponent(it);
345                    prefixes = nextJsonListComponent(it);
346                    suffixes = nextJsonListComponent(it);
347            }
348    
349            private String nextJsonComponent(Iterator<List<String>> it) {
350                    List<String> values = nextJsonListComponent(it);
351                    return values.isEmpty() ? null : values.get(0);
352            }
353    
354            private List<String> nextJsonListComponent(Iterator<List<String>> it) {
355                    if (!it.hasNext()) {
356                            return new ArrayList<String>(0);
357                    }
358    
359                    List<String> values = it.next();
360                    List<String> nonEmpty = new ArrayList<String>(values.size());
361                    for (String valueStr : values) {
362                            if (valueStr.length() > 0) {
363                                    nonEmpty.add(valueStr);
364                            }
365                    }
366                    return nonEmpty;
367            }
