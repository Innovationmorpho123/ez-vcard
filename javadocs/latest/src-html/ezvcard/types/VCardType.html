052    /**
053     * Represents a vCard key/value pair entry (called a "type" or "property").
054     * @author Michael Angstadt
055     */
056    public abstract class VCardType implements Comparable<VCardType> {
057            /**
058             * The name of the type.
059             */
060            protected final String typeName;
061    
062            /**
063             * The group that this type belongs to or null if it doesn't belong to a
064             * group.
065             */
066            protected String group;
067    
068            /**
069             * The list of attributes that are associated with this type (called
070             * "sub types" or "parameters").
071             */
072            protected VCardSubTypes subTypes = new VCardSubTypes();

098            /**
099             * Converts this type object to a string for sending over the wire. It is
100             * NOT responsible for folding.
101             * @param version the version vCard that is being generated
102             * @param warnings allows the programmer to alert the user to any
103             * note-worthy (but non-critical) issues that occurred during the
104             * marshalling process
105             * @param compatibilityMode allows the programmer to customize the
106             * marshalling process depending on the expected consumer of the vCard
107             * @return the string for sending over the wire
108             * @throws SkipMeException if this type should NOT be marshalled into the
109             * vCard
110             * @throws EmbeddedVCardException if the value of this type is an embedded
111             * vCard (i.e. the AGENT type)

114                    StringBuilder sb = new StringBuilder();
115                    doMarshalText(sb, version, warnings, compatibilityMode);

117            }

158             * @param parent the XML element that the type's value will be inserted

188            /**
189             * Marshals this type for inclusion in a jCard (JSON document).
190             * @param version the version vCard that is being generated
191             * @param warnings allows the programmer to alert the user to any
192             * note-worthy (but non-critical) issues that occurred during the
193             * marshalling process
194             * @return the marshalled jCard value
195             * @throws SkipMeException if this type should NOT be marshalled into the
196             * vCard
197             */
198            protected JCardValue doMarshalJson(VCardVersion version, List<String> warnings) {
199                    String valueStr = marshalText(version, warnings, CompatibilityMode.RFC);
200    
201                    //determine the data type based on the VALUE parameter
202                    ValueParameter valueParam = subTypes.getValue();
203                    JCardDataType dataType = (valueParam == null) ? null : JCardDataType.get(valueParam.getValue());
204                    return JCardValue.single(dataType, valueStr);
205            }

400            private String jcardValueToString(JCardValue value) {
401                    if (value.getValues().size() > 1) {
402                            List<String> multi = value.getMultivalued();
403                            if (multi != null) {
404                                    return VCardStringUtils.join(multi, ",", new JoinCallback<String>() {
405                                            public void handle(StringBuilder sb, String value) {
406                                                    sb.append(VCardStringUtils.escape(value));
407                                            }
408                                    });
409                            }
410                    }
411    
412                    if (!value.getValues().isEmpty() && value.getValues().get(0).getArray() != null) {
413                            List<List<String>> structured = value.getStructured();
414                            if (structured != null) {
415                                    return VCardStringUtils.join(structured, ";", new JoinCallback<List<String>>() {
416                                            public void handle(StringBuilder sb, List<String> value) {
417                                                    VCardStringUtils.join(value, ",", sb, new JoinCallback<String>() {
418                                                            public void handle(StringBuilder sb, String value) {
419                                                                    sb.append(VCardStringUtils.escape(value));
420                                                            }
421                                                    });
422                                            }
423                                    });
424                            }
425                    }
426    
427                    return value.getSingleValued();
428            }

488            /**
489             * Sorts by PREF parameter ascending. Types that do not have a PREF
490             * parameter are pushed to the end of the list.
491             */
492            public int compareTo(VCardType that) {
493                    Integer pref0 = this.getSubTypes().getPref();
494                    Integer pref1 = that.getSubTypes().getPref();
495                    if (pref0 == null && pref1 == null) {
496                            return 0;
497                    }
498                    if (pref0 == null) {
499                            return 1;
500                    }
501                    if (pref1 == null) {
502                            return -1;
503                    }
504                    return pref1.compareTo(pref0);
505            }
