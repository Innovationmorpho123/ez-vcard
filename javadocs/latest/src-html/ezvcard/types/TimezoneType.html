049    /**
050     * Contains the timezone that the person lives/works in.
051     * 
052     * <p>
053     * <b>Code sample</b>
054     * </p>
055     * 
056     * <pre>
057     * VCard vcard = new VCard();
058     * TimezoneType tz = new TimezoneType(-5, 0, &quot;America/New_York&quot;);
059     * vcard.addTimezone(tz);
060     * </pre>
061     * 
062     * <p>
063     * <b>Property name:</b> <code>TZ</code>
064     * </p>
065     * <p>
066     * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
067     * </p>
068     * 
069     * @author Michael Angstadt
070     */
071    
072    //@formatter:off
073    /* 
074     * Parsing===================
075     * 
076     * vCard 2.1:
077     * Parse as UTC offset.  If invalid, throw SkipMeException.
078     * 
079     * vCard 3.0, hCard:
080     * VALUE=text:                  Treat as text
081     * No VALUE param:              Parse as UTC offset.  If invalid, add warning and treat as text.
082     * 
083     * vCard 4.0, jCard:
084     * VALUE=text:                  Treat as text
085     * VALUE=utc-offset:    Parse as UTC offset.  If invalid, throw SkipMeException
086     * VALUE=uri:                   Not going to support this, as there is no description of what a timezone URI looks like
087     * No VALUE param:              Parse as UTC offset.  If invalid, treat as text
088     * 
089     * xCard:
090     * text | utc-offset    | result
091     * no   | no                    | throw SkipMeException
092     * yes  | no                    | OK
093     * no   | yes                   | OK
094     * no   | invalid               | throw SkipMeException
095     * yes  | yes                   | Parse both
096     * yes  | invalid               | Add warning, ignore utc-offset
097     * 
098     * Writing===================
099     * 
100     * vCard 2.1:
101     * text | utc-offset    | result
102     * no   | no                    | SkipMeException
103     * no   | yes                   | Write UTC offset
104     * yes  | no                    | SkipMeException
105     * yes  | yes                   | Write UTC offset
106     * 
107     * vCard 3.0:
108     * text | utc-offset    | result
109     * no   | no                    | SkipMeException
110     * no   | yes                   | Write UTC offset
111     * yes  | no                    | Write text, add "VALUE=text" parameter
112     * yes  | yes                   | Write UTC offset
113     * 
114     * vCard 4.0, xCard, jCard:
115     * text | utc-offset    | result
116     * no   | no                    | SkipMeException
117     * no   | yes                   | Write UTC offset, add "VALUE=utc-offset" parameter
118     * yes  | no                    | Write text
119     * yes  | yes                   | Write text
120     */
121    //@formatter:on
122    public class TimezoneType extends VCardType implements HasAltId {
123            public static final String NAME = "TZ";
124    
125            private Integer hourOffset;
126            private Integer minuteOffset;
127            private String text;
128    
129            /**
130             * Creates an empty timezone property.
131             */
132            public TimezoneType() {
133                    super(NAME);
134            }
135    
136            /**
137             * Creates a timezone property (this is the recommended constructor for
138             * version 4.0 vCards).
139             * @param text string representing the timezone from the <a
140             * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
141             * Database</a> (e.g. "America/New_York")
142             */
143            public TimezoneType(String text) {
144                    this(null, null, text);
145            }
146    
147            /**
148             * Creates a timezone property (this is the recommended constructor for
149             * version 2.1 and 3.0 vCards).
150             * @param hourOffset the hour offset
151             * @param minuteOffset the minute offset
152             */
153            public TimezoneType(Integer hourOffset, Integer minuteOffset) {
154                    this(hourOffset, minuteOffset, null);
155            }
156    
157            /**
158             * This constructor can be used for all vCard versions.
159             * @param hourOffset the hour offset
160             * @param minuteOffset the minute offset
161             * @param text can be anything, but should be a string representing the
162             * timezone from the <a
163             * href="http://en.wikipedia.org/wiki/List_of_tz_database_time_zones">Olson
164             * Database</a> (e.g. "America/New_York")
165             */
166            public TimezoneType(Integer hourOffset, Integer minuteOffset, String text) {
167                    super(NAME);
168                    setOffset(hourOffset, minuteOffset);
169                    setText(text);
170            }
171    
172            /**
173             * Gets the hour offset.
174             * @return the hour offset or null if not set
175             */
176            public Integer getHourOffset() {
177                    return hourOffset;
178            }
179    
180            /**
181             * Gets the minute offset.
182             * @return the minute offset or null if not set
183             */
184            public Integer getMinuteOffset() {
185                    return minuteOffset;
186            }
187    
188            /**
189             * Sets the UTC offset. Both parameters can be set to null to remove the UTC
190             * offset from this object.
191             * @param hourOffset the hour offset (e.g. -5)
192             * @param minuteOffset the minute offset (e.g. 0)
193             * @throws IllegalArgumentException if the minute offset is not between 0
194             * and 59
195             */
196            public void setOffset(Integer hourOffset, Integer minuteOffset) {
197                    if (minuteOffset != null && (minuteOffset < 0 || minuteOffset > 59)) {
198                            throw new IllegalArgumentException("Minute offset must be between 0 and 59.");
199                    }
200    
201                    if (hourOffset != null && minuteOffset == null) {
202                            minuteOffset = 0;
203                    } else if (hourOffset == null && minuteOffset != null) {
204                            hourOffset = 0;
205                    }
206                    this.hourOffset = hourOffset;
207                    this.minuteOffset = minuteOffset;
208            }
209    
210            private boolean hasOffset() {
211                    return hourOffset != null && minuteOffset != null;
212            }
213    
214            private boolean hasText() {
215                    return text != null;
216            }
217    
218            /**
219             * Gets the text portion of the timezone.
220             * @return the text portion (e.g. "America/New_York")
221             */
222            public String getText() {
223                    return text;
224            }
225    
226            /**
227             * Sets the text portion of the timezone.
228             * @param text the text portion (e.g. "America/New_York")
229             */
230            public void setText(String text) {
231                    this.text = text;
232            }
233    
234            /**
235             * Creates a {@link java.util.TimeZone} representation of this class.
236             * @return a {@link TimeZone} object or null if this object contains no
237             * offset data
238             */
239            public TimeZone toTimeZone() {
240                    if (!hasOffset()) {
241                            return null;
242                    }
243    
244                    int rawHourOffset = hourOffset * 60 * 60 * 1000;
245                    int rawMinuteOffset = minuteOffset * 60 * 1000;
246                    if (rawHourOffset < 0) {
247                            rawMinuteOffset *= -1;
248                    }
249                    int rawOffset = rawHourOffset + rawMinuteOffset;
250                    return new SimpleTimeZone(rawOffset, "");
251            }
252    
253            /**
254             * Gets the TYPE parameter.
255             * <p>
256             * <b>Supported versions:</b> <code>4.0</code>
257             * </p>
258             * @return the TYPE value (typically, this will be either "work" or "home")
259             * or null if it doesn't exist
260             */
261            public String getType() {
262                    return subTypes.getType();
263            }
264    
265            /**
266             * Sets the TYPE parameter.
267             * <p>
268             * <b>Supported versions:</b> <code>4.0</code>
269             * </p>
270             * @param type the TYPE value (this should be either "work" or "home") or
271             * null to remove
272             */
273            public void setType(String type) {
274                    subTypes.setType(type);
275            }
276

278             * Gets the MEDIATYPE parameter.

296                    subTypes.setMediaType(mediaType);
297            }

334            @Override
335            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
336                    switch (version) {
337                    case V3_0:
338                            if (!hasOffset() && hasText()) {
339                                    copy.setValue(ValueParameter.TEXT);
340                            }
341                            break;
342                    case V4_0:
343                            if (hasOffset() && !hasText()) {
344                                    copy.setValue(ValueParameter.UTC_OFFSET);
345                            }
346                            break;
347                    }
348            }
349    
350            @Override
351            protected void doMarshalText(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
352                    checkForValue();
353    
354                    if (version == null) {
355                            version = VCardVersion.V2_1;
356                    }
357    
358                    boolean writeText = true;
359                    switch (version) {
360                    case V2_1:
361                            if (!hasOffset()) {
362                                    throw new SkipMeException("Version " + VCardVersion.V2_1 + " requires a UTC offset.");
363                            }
364                            writeText = false;
365                            break;
366                    case V3_0:
367                            if (hasOffset()) {
368                                    writeText = false;
369                            } else {
370                                    writeText = true;
371                            }
372                            break;
373                    case V4_0:
374                            if (hasText()) {
375                                    writeText = true;
376                            } else {
377                                    writeText = false;
378                            }
379                            break;
380                    }
381    
382                    if (writeText) {
383                            sb.append(VCardStringUtils.escape(text));
384                    } else {
385                            //2.1 - either extended or basic
386                            //3.0 - extended
387                            //4.0 - basic
388                            boolean extended = (version == VCardVersion.V3_0);
389                            sb.append(VCardDateFormatter.formatTimeZone(hourOffset, minuteOffset, extended));
390                    }
391            }
392    
393            @Override
394            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
395                    value = VCardStringUtils.unescape(value);
396                    ValueParameter valueParam = subTypes.getValue();
397                    parse(value, valueParam == ValueParameter.TEXT, valueParam == ValueParameter.UTC_OFFSET, version, warnings);
398            }
399    
400            @Override
401            protected void doMarshalXml(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
402                    checkForValue();
403    
404                    if (hasText()) {
405                            parent.text(text);
406                    } else {
407                            String offset = VCardDateFormatter.formatTimeZone(hourOffset, minuteOffset, false);
408                            parent.utcOffset(offset);
409                    }
410            }
411    
412            @Override
413            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
414                    String text = element.get("text");
415                    String utcOffset = element.utcOffset();
416    
417                    if (text == null && utcOffset == null) {
418                            throw new SkipMeException("No timezone data found.");
419                    }
420    
421                    if (text != null) {
422                            setText(text);
423                    }
424    
425                    if (utcOffset != null) {
426                            try {
427                                    int offsets[] = VCardDateFormatter.parseTimeZone(utcOffset);
428                                    setOffset(offsets[0], offsets[1]);
429                            } catch (IllegalArgumentException e) {
430                                    if (text == null) {
431                                            throw new SkipMeException("Unable to parse UTC offset: " + utcOffset);
432                                    } else {
433                                            warnings.add("Ignoring invalid UTC offset: " + utcOffset);
434                                    }
435                            }
436                    }
437            }
438    
439            @Override
440            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
441                    parse(element.value(), false, false, VCardVersion.V3_0, warnings);
442            }
443    
444            @Override
445            protected JCardValue doMarshalJson(VCardVersion version, List<String> warnings) {
446                    checkForValue();
447    
448                    JCardDataType dataType;
449                    String value;
450                    if (hasText()) {
451                            dataType = JCardDataType.TEXT;
452                            value = text;
453                    } else {
454                            dataType = JCardDataType.UTC_OFFSET;
455                            value = VCardDateFormatter.formatTimeZone(hourOffset, minuteOffset, true);
456                    }
457    
458                    return JCardValue.single(dataType, value);
459            }
460    
461            @Override
462            protected void doUnmarshalJson(JCardValue value, VCardVersion version, List<String> warnings) {
463                    String valueStr = value.getSingleValued();
464                    JCardDataType dataType = value.getDataType();
465                    parse(valueStr, dataType == JCardDataType.TEXT, dataType == JCardDataType.UTC_OFFSET, version, warnings);
466            }
467    
468            private void checkForValue() {
469                    if (!hasText() && !hasOffset()) {
470                            throw new SkipMeException("Property does not have text or a UTC offset associated with it.");
471                    }
472            }
473    
474            private void parse(String value, boolean isTextDataType, boolean isUtcOffsetDataType, VCardVersion version, List<String> warnings) {
475                    if (version == null) {
476                            version = VCardVersion.V2_1;
477                    }
478    
479                    switch (version) {
480                    case V2_1:
481                            //e.g. "-05:00"
482                            try {
483                                    int offsets[] = VCardDateFormatter.parseTimeZone(value);
484                                    setOffset(offsets[0], offsets[1]);
485                            } catch (IllegalArgumentException e) {
486                                    throw new SkipMeException("Unable to parse UTC offset: " + value);
487                            }
488                            break;
489                    case V3_0:
490                            //e.g. "-05:00"
491                            //e.g. "-05:00;EDT;America/New_York"
492                            if (isTextDataType) {
493                                    setText(value);
494                            } else {
495                                    try {
496                                            int offsets[] = VCardDateFormatter.parseTimeZone(value);
497                                            setOffset(offsets[0], offsets[1]);
498                                    } catch (IllegalArgumentException e) {
499                                            warnings.add("Unable to parse UTC offset.  Treating as text: " + value);
500                                            setText(value);
501                                    }
502                            }
503                            break;
504                    case V4_0:
505                            //e.g. "-0500"
506                            //e.g. "America/New_York"
507                            if (isTextDataType) {
508                                    setText(value);
509                            } else {
510                                    try {
511                                            int offsets[] = VCardDateFormatter.parseTimeZone(value);
512                                            setOffset(offsets[0], offsets[1]);
513                                    } catch (IllegalArgumentException e) {
514                                            if (isUtcOffsetDataType) {
515                                                    throw new SkipMeException("Unable to parse UTC offset: " + value);
516                                            }
517                                            setText(value);
518                                    }
519                            }
520                            break;
521                    }
522            }
