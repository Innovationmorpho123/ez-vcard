048    /**
049     * Contains the timezone that the person lives/works in.
050     * 
051     * <p>
052     * <b>Code sample</b>
053     * </p>
054     * 
055     * <pre class="brush:java">
056     * VCard vcard = new VCard();
057     * TimezoneType tz = new TimezoneType(-5, 0, &quot;America/New_York&quot;);
058     * vcard.addTimezone(tz);
059     * </pre>
060     * 
061     * <p>
062     * <b>Property name:</b> {@code TZ}
063     * </p>
064     * <p>
065     * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
066     * </p>
067     * 
068     * @author Michael Angstadt
069     */
070    
071    //@formatter:off
072    /* 
073     * Parsing===================
074     * 
075     * vCard 2.1:
076     * Parse as UTC offset.  If invalid, throw CannotParseException.
077     * 
078     * vCard 3.0, hCard:
079     * VALUE=text:                  Treat as text
080     * No VALUE param:              Parse as UTC offset.  If invalid, add warning and treat as text.
081     * 
082     * vCard 4.0, jCard:
083     * VALUE=text:                  Treat as text
084     * VALUE=utc-offset:    Parse as UTC offset.  If invalid, throw CannotParseException
085     * VALUE=uri:                   Not going to support this, as there is no description of what a timezone URI looks like
086     * No VALUE param:              Parse as UTC offset.  If invalid, treat as text
087     * 
088     * xCard:
089     * text | utc-offset    | result
090     * no   | no                    | throw CannotParseException
091     * yes  | no                    | OK
092     * no   | yes                   | OK
093     * no   | invalid               | throw CannotParseException
094     * yes  | yes                   | Parse text
095     * yes  | invalid               | Parse text
096     * 
097     * Writing===================
098     * 
099     * vCard 2.1:
100     * text | utc-offset    | result
101     * no   | no                    | empty string (validation warning)
102     * no   | yes                   | Write UTC offset
103     * yes  | no                    | empty string (validation warning)
104     * yes  | yes                   | Write UTC offset
105     * 
106     * vCard 3.0:
107     * text | utc-offset    | result
108     * no   | no                    | empty string (validation warning)
109     * no   | yes                   | Write UTC offset
110     * yes  | no                    | Write text, add "VALUE=text" parameter
111     * yes  | yes                   | Write UTC offset
112     * 
113     * vCard 4.0, xCard, jCard:
114     * text | utc-offset    | result
115     * no   | no                    | empty string (validation warning)
116     * no   | yes                   | Write UTC offset, add "VALUE=utc-offset" parameter
117     * yes  | no                    | Write text
118     * yes  | yes                   | Write text
119     */
120    //@formatter:on

260            /**
261             * Creates a {@link java.util.TimeZone} representation of this class.
262             * @return a {@link TimeZone} object or null if this object contains no
263             * offset data
264             */
265            public TimeZone toTimeZone() {
266                    if (text != null) {
267                            TimeZone timezone = timezoneFromId(text);
268                            if (timezone != null) {
269                                    return timezone;
270                            }
271                    }
272    
273                    if (offset != null) {
274                            int rawHourOffset = offset.getHour() * 60 * 60 * 1000;
275                            int rawMinuteOffset = offset.getMinute() * 60 * 1000;
276                            if (rawHourOffset < 0) {
277                                    rawMinuteOffset *= -1;
278                            }
279                            int rawOffset = rawHourOffset + rawMinuteOffset;
280    
281                            String id = (text == null) ? "" : text;
282    
283                            return new SimpleTimeZone(rawOffset, id);
284                    }
285    
286                    return null;
287            }

370            @Override
371            protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, CompatibilityMode compatibilityMode, VCard vcard) {
372                    VCardDataType dataType = null;
373    
374                    switch (version) {
375                    case V2_1:
376                            dataType = null;
377                            break;
378                    case V3_0:
379                            if (offset == null && text != null) {
380                                    dataType = VCardDataType.TEXT;
381                            }
382                            break;
383                    case V4_0:
384                            if (offset != null && text == null) {
385                                    dataType = VCardDataType.UTC_OFFSET;
386                            }
387                            break;
388                    }
389    
390                    copy.setValue(dataType);
391            }
392    
393            @Override
394            protected void doMarshalText(StringBuilder sb, VCardVersion version, CompatibilityMode compatibilityMode) {
395                    switch (version) {
396                    case V2_1:
397                            if (offset != null) {
398                                    sb.append(offset.toString(false)); //2.1 allows either basic or extended
399                                    return;
400                            }
401    
402                            if (text != null) {
403                                    //attempt to find the offset by treating the text as a timezone ID, like "America/New_York"
404                                    TimeZone timezone = timezoneFromId(text);
405                                    if (timezone != null) {
406                                            UtcOffset offset = offsetFromTimezone(timezone);
407                                            sb.append(offset.toString(false));
408                                    }
409                                    return;
410                            }
411                    case V3_0:
412                            if (offset != null) {
413                                    sb.append(offset.toString(true)); //3.0 only allows extended
414                            } else if (text != null) {
415                                    sb.append(VCardStringUtils.escape(text));
416                            }
417                            return;
418                    case V4_0:
419                            if (text != null) {
420                                    sb.append(VCardStringUtils.escape(text));
421                            } else if (offset != null) {
422                                    sb.append(offset.toString(false)); //4.0 only allows basic
423                            }
424                            return;
425                    }
426            }

428            @Override
429            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
430                    value = VCardStringUtils.unescape(value);
431                    VCardDataType dataType = subTypes.getValue();
432                    parse(value, dataType, version, warnings);
433            }

435            @Override
436            protected void doMarshalXml(XCardElement parent, CompatibilityMode compatibilityMode) {
437                    if (text != null) {
438                            parent.append(VCardDataType.TEXT, text);
439                            return;
440                    }
441    
442                    if (offset != null) {
443                            parent.append(VCardDataType.UTC_OFFSET, offset.toString(false));
444                            return;
445                    }
446    
447                    parent.append(VCardDataType.TEXT, "");
448            }

450            @Override
451            protected void doUnmarshalXml(XCardElement element, List<String> warnings, CompatibilityMode compatibilityMode) {
452                    String text = element.first(VCardDataType.TEXT);
453                    if (text != null) {
454                            setText(text);
455                            return;
456                    }
457    
458                    String utcOffset = element.first(VCardDataType.UTC_OFFSET);
459                    if (utcOffset != null) {
460                            try {
461                                    setOffset(UtcOffset.parse(utcOffset));
462                            } catch (IllegalArgumentException e) {
463                                    throw new CannotParseException("Unable to parse UTC offset.");
464                            }
465                            return;
466                    }
467    
468                    throw missingXmlElements(VCardDataType.TEXT, VCardDataType.UTC_OFFSET);
469            }

472            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
473                    parse(element.value(), null, VCardVersion.V3_0, warnings);
474            }

496            @Override
497            protected void _validate(List<String> warnings, VCardVersion version, VCard vcard) {
498                    if (offset == null && text == null) {
499                            warnings.add("Property does not have text or UTC offset values associated with it.");
500                    }
501                    if (offset == null && version == VCardVersion.V2_1) {
502                            warnings.add("Property requires a UTC offset for its value in version " + version.getVersion() + ".");
503                    }
504                    if (offset != null && (offset.getMinute() < 0 || offset.getMinute() > 59)) {
505                            warnings.add("Minute offset must be between 0 and 59.");
506                    }
507            }
508    
509            private void parse(String value, VCardDataType dataType, VCardVersion version, List<String> warnings) {
510                    if (value == null || value.length() == 0) {
511                            return;
512                    }
513    
514                    switch (version) {
515                    case V2_1:
516                            //e.g. "-05:00"
517                            try {
518                                    setOffset(UtcOffset.parse(value));
519                            } catch (IllegalArgumentException e) {
520                                    throw new CannotParseException("Unable to parse UTC offset.");
521                            }
522                            break;
523                    case V3_0:
524                            //e.g. "-05:00"
525                            //e.g. "-05:00;EDT;America/New_York"
526                            if (dataType == VCardDataType.TEXT) {
527                                    setText(value);
528                                    return;
529                            }
530    
531                            try {
532                                    setOffset(UtcOffset.parse(value));
533                            } catch (IllegalArgumentException e) {
534                                    warnings.add("Unable to parse UTC offset.  Treating as text: " + value);
535                                    setText(value);
536                            }
537                            break;
538                    case V4_0:
539                            //e.g. "-0500"
540                            //e.g. "America/New_York"
541                            if (dataType == VCardDataType.TEXT) {
542                                    setText(value);
543                                    return;
544                            }
545    
546                            try {
547                                    setOffset(UtcOffset.parse(value));
548                            } catch (IllegalArgumentException e) {
549                                    if (dataType == VCardDataType.UTC_OFFSET) {
550                                            throw new CannotParseException("Unable to parse UTC offset.");
551                                    }
552                                    setText(value);
553                            }
554                            break;
555                    }
556            }
557    
558            private static UtcOffset offsetFromTimezone(TimeZone timezone) {
559                    long offsetMs = timezone.getOffset(System.currentTimeMillis());
560                    int hours = (int) (offsetMs / 1000 / 60 / 60);
561                    int minutes = (int) ((offsetMs / 1000 / 60) % 60);
562                    if (minutes < 0) {
563                            minutes *= -1;
564                    }
565                    return new UtcOffset(hours, minutes);
566            }
567    
568            private static TimeZone timezoneFromId(String id) {
569                    TimeZone timezone = TimeZone.getTimeZone(id);
570                    return "GMT".equals(timezone.getID()) ? null : timezone;
571            }
