066    /**
067     * Parses {@link VCard} objects from an HTML page (hCard format).

071     */
072    public class HCardReader implements IParser {
073            private String pageUrl;
074            private List<String> warnings = new ArrayList<String>();
075            private Map<String, Class<? extends VCardType>> extendedTypeClasses = new HashMap<String, Class<? extends VCardType>>();
076            private Elements vcardElements;
077            private Iterator<Element> it;
078            private List<LabelType> labels = new ArrayList<LabelType>();
079            private List<String> warningsBuffer = new ArrayList<String>();

082            private NicknameType nickname;

086             * Creates an hCard reader.
087             * @param url the URL of the webpage
088             * @throws IOException if there's a problem loading the webpage

091                    pageUrl = url.toString();
092                    Document document = Jsoup.parse(url, 30000);
093                    init(document, url.getRef());
094            }

149            /**
150             * Creates an hCard reader.
151             * @param reader the input stream to the HTML page
152             * @param pageUrl the original URL of the HTML page
153             * @throws IOException if there's a problem reading the HTML page
154             */
155            public HCardReader(Reader reader, String pageUrl) throws IOException {
156                    this.pageUrl = pageUrl;
157    
158                    StringBuilder sb = new StringBuilder();
159                    char buffer[] = new char[4096];
160                    int read;
161                    while ((read = reader.read(buffer)) != -1) {
162                            sb.append(buffer, 0, read);
163                    }
164                    String html = sb.toString();
165    
166                    Document document = (pageUrl == null) ? Jsoup.parse(html) : Jsoup.parse(html, pageUrl);
167                    String anchor = getAnchor(pageUrl);
168                    init(document, anchor);
169            }

192            /**
193             * Constructor for reading embedded vCards.
194             * @param embeddedVCard the HTML element of the embedded vCard
195             * @param pageUrl the original URL of the HTML page
196             */
197            private HCardReader(Element embeddedVCard, String pageUrl) {
198                    this.pageUrl = pageUrl;
199                    vcardElements = new Elements(embeddedVCard);
200                    it = vcardElements.iterator();
201            }
202    
203            private void init(Document document, String anchor) {
204                    Element searchIn = null;
205                    if (anchor != null) {
206                            searchIn = document.getElementById(anchor);
207                    }
208                    if (searchIn == null) {
209                            searchIn = document;
210                    }
211    
212                    vcardElements = searchIn.getElementsByClass("vcard");
213                    it = vcardElements.iterator();
214            }
215    
216            /**
217             * Gets the anchor part of a URL.
218             * @param urlStr the URL
219             * @return the anchor (e.g. "foo" from the URL
220             * "http://example.com/index.php#foo")
221             */
222            private String getAnchor(String urlStr) {
223                    if (urlStr == null) {
224                            return null;
225                    }
226    
227                    try {
228                            URL url = new URL(urlStr);
229                            return url.getRef();
230                    } catch (MalformedURLException e) {
231                            return null;
232                    }
233            }

250            //@Override
251            public VCard readNext() {
252                    Element vcardElement = null;
253                    while (it.hasNext() && vcardElement == null) {
254                            vcardElement = it.next();
255    
256                            //if this element is a child of another "vcard" element, then ignore it because it's an embedded vcard
257                            if (HtmlUtils.isChildOf(vcardElement, vcardElements)) {
258                                    vcardElement = null;
259                            }
260                    }
261                    if (vcardElement == null) {
262                            return null;
263                    }
264    
265                    warnings.clear();
266                    warningsBuffer.clear();
267                    labels.clear();
268                    nickname = null;
269                    categories = null;
270    
271                    curVCard = new VCard();
272                    curVCard.setVersion(VCardVersion.V3_0);
273                    if (pageUrl != null) {
274                            curVCard.addSource(new SourceType(pageUrl));
275                    }
276    
277                    //visit all descendant nodes, depth-first
278                    for (Element child : vcardElement.children()) {
279                            visit(child);
280                    }
281    
282                    //assign labels to their addresses
283                    for (LabelType label : labels) {
284                            boolean orphaned = true;
285                            for (AddressType adr : curVCard.getAddresses()) {
286                                    if (adr.getLabel() == null && adr.getTypes().equals(label.getTypes())) {
287                                            adr.setLabel(label.getValue());
288                                            orphaned = false;
289                                            break;
290                                    }
291                            }
292                            if (orphaned) {
293                                    curVCard.addOrphanedLabel(label);
294                            }
295                    }
296    
297                    return curVCard;
298            }
299    
300            private void visit(Element element) {
301                    Set<String> classNames = element.classNames();
302                    for (String className : classNames) {
303                            if (UrlType.NAME.equalsIgnoreCase(className)) {
304                                    String href = element.attr("href");
305                                    if (href.length() > 0) {
306                                            if (!classNames.contains(EmailType.NAME.toLowerCase()) && href.matches("(?i)mailto:.*")) {
307                                                    className = EmailType.NAME;
308                                            } else if (!classNames.contains(TelephoneType.NAME.toLowerCase()) && href.matches("(?i)tel:.*")) {
309                                                    className = TelephoneType.NAME;
310                                            } else {
311                                                    //try parsing as IMPP
312                                                    warningsBuffer.clear();
313                                                    ImppType impp = new ImppType();
314                                                    try {
315                                                            impp.unmarshalHtml(element, warningsBuffer);
316                                                            curVCard.addType(impp);
317                                                            for (String warning : warningsBuffer) {
318                                                                    addWarning(warning, impp.getTypeName());
319                                                            }
320                                                            continue;
321                                                    } catch (SkipMeException e) {
322                                                            //URL is not an instant messenger URL
323                                                    }
324                                            }
325                                    }
326                            }
327    
328                            VCardType type = createTypeObject(className);
329                            if (type == null) {
330                                    //if no type class is found, then it must be an arbitrary CSS class that has nothing to do with vCard
331                                    continue;
332                            }
333    
334                            warningsBuffer.clear();
335                            try {
336                                    type.unmarshalHtml(element, warningsBuffer);
337    
338                                    //add to vcard
339                                    if (type instanceof LabelType) {
340                                            //LABELs must be treated specially so they can be matched up with their ADRs
341                                            labels.add((LabelType) type);
342                                    } else if (type instanceof NicknameType) {
343                                            //add all NICKNAMEs to the same type object
344                                            NicknameType nn = (NicknameType) type;
345                                            if (nickname == null) {
346                                                    nickname = nn;
347                                                    curVCard.addType(nickname);
348                                            } else {
349                                                    nickname.getValues().addAll(nn.getValues());
350                                            }
351                                    } else if (type instanceof CategoriesType) {
352                                            //add all CATEGORIES to the same type object
353                                            CategoriesType c = (CategoriesType) type;
354                                            if (categories == null) {
355                                                    categories = c;
356                                                    curVCard.addType(categories);
357                                            } else {
358                                                    categories.getValues().addAll(c.getValues());
359                                            }
360                                    } else {
361                                            curVCard.addType(type);
362                                    }
363                            } catch (SkipMeException e) {
364                                    warningsBuffer.add("Property has requested that it be skipped: " + e.getMessage());
365                            } catch (EmbeddedVCardException e) {
366                                    if (HtmlUtils.isChildOf(element, embeddedVCards)) {
367                                            //prevents multiple-nested embedded elements from overwriting each other
368                                            continue;
369                                    }
370    
371                                    embeddedVCards.add(element);
372                                    HCardReader embeddedReader = new HCardReader(element, pageUrl);
373                                    try {
374                                            VCard embeddedVCard = embeddedReader.readNext();
375                                            e.injectVCard(embeddedVCard);
376                                    } finally {
377                                            for (String w : embeddedReader.getWarnings()) {
378                                                    warningsBuffer.add("Problem unmarshalling nested vCard value: " + w);
379                                            }
380                                    }
381                                    curVCard.addType(type);
382                            } catch (UnsupportedOperationException e) {
383                                    //type class does not support hCard
384                                    warningsBuffer.add("Property class \"" + type.getClass().getName() + "\" does not support hCard unmarshalling.");
385                            } finally {
386                                    for (String warning : warningsBuffer) {
387                                            addWarning(warning, type.getTypeName());
388                                    }
389                            }
390                    }
391    
392                    for (Element child : element.children()) {
393                            visit(child);
394                    }
395            }

397            /**
398             * Creates the appropriate {@link VCardType} instance, given the type name.
399             * This method does not unmarshal the type, it just creates the type object.
400             * @param typeName the type name (e.g. "fn")
401             * @return the type object or null if the type name was not recognized
402             */
403            private VCardType createTypeObject(String typeName) {
404                    typeName = typeName.toLowerCase();
405                    VCardType t = null;
406                    Class<? extends VCardType> clazz = TypeList.getTypeClassByHCardTypeName(typeName);
407                    if (clazz != null) {
408                            try {
409                                    //create a new instance of the class
410                                    t = clazz.newInstance();
411                            } catch (Exception e) {
412                                    //it is the responsibility of the ez-vcard developer to ensure that this exception is never thrown
413                                    //all type classes defined in the ez-vcard library MUST have public, no-arg constructors
414                                    throw new RuntimeException(e);
415                            }
416                    } else {
417                            Class<? extends VCardType> extendedTypeClass = extendedTypeClasses.get(typeName);
418                            if (extendedTypeClass != null) {
419                                    try {
420                                            t = extendedTypeClass.newInstance();
421                                    } catch (Exception e) {
422                                            //this should never happen because the type class is checked to see if it has a public, no-arg constructor in the "registerExtendedType" method
423                                            throw new RuntimeException("Extended type class \"" + extendedTypeClass.getName() + "\" must have a public, no-arg constructor.");
424                                    }
425                            } else if (typeName.startsWith("x-")) {
426                                    t = new RawType(typeName); //use RawType instead of TextType because we don't want to unescape any characters that might be meaningful to this type
427                            }
428                    }
429                    return t;
430            }

432            /**
433             * Gets the type name from a type class.
434             * @param clazz the type class
435             * @return the type name
436             */
437            private String getTypeNameFromTypeClass(Class<? extends VCardType> clazz) {
438                    try {
439                            VCardType t = clazz.newInstance();
440                            return t.getTypeName().toLowerCase();
441                    } catch (Exception e) {
442                            //there is no public, no-arg constructor
443                            throw new RuntimeException(e);
444                    }
445            }
446    
447            private void addWarning(String message, String propertyName) {
448                    warnings.add(propertyName + " property: " + message);
449            }
