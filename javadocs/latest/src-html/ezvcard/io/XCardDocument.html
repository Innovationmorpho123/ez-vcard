075    /**
076    * <p>
077    * Represents an XML document that contains vCard objects ("xCard" standard).
078    * This class can be used to read and write xCard documents.
079    * </p>
080    * <p>
081    * <b>Examples:</b>
082    * 
083    * <pre class="brush:java">
084    * String xml =
085    * "&lt;vcards xmlns=\"urn:ietf:params:xml:ns:vcard-4.0\"&gt;" +
086    *   "&lt;vcard&gt;" +
087    *     "&lt;fn&gt;" +
088    *       "&lt;text&gt;John Doe&lt;/text&gt;" +
089    *     "&lt;/fn&gt;" +
090    *     "&lt;n&gt;" +
091    *       "&lt;surname&gt;Doe&lt;/surname&gt;" +
092    *        "&lt;given&gt;Johnathan&lt;/given&gt;" +
093    *        "&lt;additional&gt;Jonny&lt;/additional&gt;" +
094    *        "&lt;additional&gt;John&lt;/additional&gt;" +
095    *        "&lt;prefix&gt;Mr.&lt;/prefix&gt;" +
096    *        "&lt;suffix /&gt;" +
097    *      "&lt;/n&gt;" +
098    *    "&lt;/vcard&gt;" +
099    * "&lt;/vcards&gt;";
100    *     
101    * //parsing an existing xCard document
102    * XCardDocument xcard = new XCardDocument(xml);
103    * List&lt;VCard&gt; vcards = xcard.parseAll();
104    * 
105    * //creating an empty xCard document
106    * XCardDocument xcard = new XCardDocument();
107    * 
108    * //VCard objects can be added at any time
109    * VCard vcard = ...
110    * xcard.add(vcard);
111    * 
112    * //retrieving the raw XML DOM
113    * Document document = xcard.getDocument();
114    * 
115    * //call one of the "write()" methods to output the xCard document
116    * File file = new File("johndoe.xml");
117    * xcard.write(file);
118    * </pre>
119    * 
120    * </p>
121    * @author Michael Angstadt
122    * @see <a href="http://tools.ietf.org/html/rfc6351">RFC 6351</a>
123    */
124    //@formatter:on

228            private void init(Document document) {
229                    XPath xpath = XPathFactory.newInstance().newXPath();
230                    xpath.setNamespaceContext(nsContext);
231    
232                    try {
233                            //find the <vcards> element
234                            String prefix = nsContext.getPrefix();
235                            root = (Element) xpath.evaluate("//" + prefix + ":vcards", document, XPathConstants.NODE);
236                    } catch (XPathExpressionException e) {
237                            //never thrown, xpath expression is hard coded
238                    }
239            }

352            /**
353             * Parses all the {@link VCard} objects from the xCard document.
354             * @return the vCard objects
355             */
356            public List<VCard> parseAll() {
357                    parseWarnings.clear();
358    
359                    if (root == null) {
360                            return Collections.emptyList();
361                    }
362    
363                    List<VCard> vcards = new ArrayList<VCard>();
364                    for (Element vcardElement : getVCardElements()) {
365                            List<String> warnings = new ArrayList<String>();
366                            parseWarnings.add(warnings);
367                            VCard vcard = parseVCardElement(vcardElement, warnings);
368                            vcards.add(vcard);
369                    }
370    
371                    return vcards;
372            }
373    
374            /**
375             * Parses the first the {@link VCard} object from the xCard document.
376             * @return the vCard object
377             */
378            public VCard parseFirst() {
379                    parseWarnings.clear();
380    
381                    if (root == null) {
382                            return null;
383                    }
384    
385                    List<Element> vcardElements = getVCardElements();
386                    if (vcardElements.isEmpty()) {
387                            return null;
388                    }
389    
390                    List<String> warnings = new ArrayList<String>();
391                    parseWarnings.add(warnings);
392                    return parseVCardElement(vcardElements.get(0), warnings);
393            }
394    
395            private VCard parseVCardElement(Element vcardElement, List<String> warnings) {
396                    VCard vcard = new VCard();
397                    vcard.setVersion(version4);
398    
399                    String ns = version4.getXmlNamespace();
400                    List<Element> children = XmlUtils.toElementList(vcardElement.getChildNodes());
401                    for (Element child : children) {
402                            if ("group".equals(child.getLocalName()) && ns.equals(child.getNamespaceURI())) {
403                                    String group = child.getAttribute("name");
404                                    if (group.length() == 0) {
405                                            group = null;
406                                    }
407                                    List<Element> propElements = XmlUtils.toElementList(child.getChildNodes());
408                                    for (Element propElement : propElements) {
409                                            parseAndAddElement(propElement, group, vcard, warnings);
410                                    }
411                            } else {
412                                    parseAndAddElement(child, null, vcard, warnings);
413                            }
414                    }
415    
416                    return vcard;
417            }

419            /**
420             * Parses a property element from the XML document and adds the property to
421             * the vCard.
422             * @param element the element to parse
423             * @param group the group name or null if the property does not belong to a
424             * group
425             * @param vcard the vCard object
426             * @param warningsBuf the list to add the warnings to
427             */
428            private void parseAndAddElement(Element element, String group, VCard vcard, List<String> warnings) {
429                    List<String> warningsBuf = new ArrayList<String>();
430    
431                    VCardSubTypes subTypes = parseSubTypes(element);
432                    String propertyName = element.getLocalName();
433                    VCardType type = createTypeObject(propertyName, element.getNamespaceURI());
434                    type.setGroup(group);
435                    try {
436                            type.unmarshalXml(subTypes, element, version4, warningsBuf, compatibilityMode);
437                    } catch (SkipMeException e) {
438                            warningsBuf.add("Property has requested that it be skipped: " + e.getMessage());
439                            return;
440                    } catch (CannotParseException e) {
441                            String xml = XmlUtils.toString(element);
442                            warningsBuf.add("Property value could not be parsed.  It will be unmarshalled as a " + XmlType.NAME + " property instead." + NEWLINE + "  XML: " + xml + NEWLINE + "  Reason: " + e.getMessage());
443                            type = new XmlType();
444                            type.setGroup(group);
445                            type.unmarshalXml(subTypes, element, version4, warningsBuf, compatibilityMode);
446                    } catch (UnsupportedOperationException e) {
447                            warningsBuf.add("Property class \"" + type.getClass().getName() + "\" does not support xCard unmarshalling.  It will be unmarshalled as a " + XmlType.NAME + " property instead.");
448                            type = new XmlType();
449                            type.setGroup(group);
450                            type.unmarshalXml(subTypes, element, version4, warningsBuf, compatibilityMode);
451                    } catch (EmbeddedVCardException e) {
452                            warningsBuf.add("Property will not be unmarshalled because xCard does not supported embedded vCards.");
453                            return;
454                    } finally {
455                            for (String warning : warningsBuf) {
456                                    addWarning(warning, type.getTypeName(), warnings);
457                            }
458                    }
459    
460                    vcard.addType(type);
461            }

463            /**
464             * Parses the property parameters (aka "sub types").
465             * @param element the property's XML element
466             * @return the parsed parameters
467             */
468            private VCardSubTypes parseSubTypes(Element element) {
469                    VCardSubTypes subTypes = new VCardSubTypes();
470    
471                    List<Element> parametersElements = XmlUtils.toElementList(element.getElementsByTagNameNS(version4.getXmlNamespace(), "parameters"));
472                    for (Element parametersElement : parametersElements) { // foreach "<parameters>" element (there should only be 1 though)
473                            List<Element> paramElements = XmlUtils.toElementList(parametersElement.getChildNodes());
474                            for (Element paramElement : paramElements) {
475                                    String name = paramElement.getLocalName().toUpperCase();
476                                    List<Element> valueElements = XmlUtils.toElementList(paramElement.getChildNodes());
477                                    if (valueElements.isEmpty()) {
478                                            String value = paramElement.getTextContent();
479                                            subTypes.put(name, value);
480                                    } else {
481                                            for (Element valueElement : valueElements) {
482                                                    String value = valueElement.getTextContent();
483                                                    subTypes.put(name, value);
484                                            }
485                                    }
486                            }
487                    }
488    
489                    return subTypes;
490            }

492            /**
493             * Creates the appropriate VCardType instance given the vCard property name.
494             * This method does not unmarshal the type, it just creates the type object.
495             * @param name the property name (e.g. "fn")
496             * @param ns the namespace of the element
497             * @return the type that was created
498             */
499            private VCardType createTypeObject(String name, String ns) {
500                    //parse as a registered extended type class (extended type classes should override standard ones)
501                    Class<? extends VCardType> extendedTypeClass = extendedTypeClasses.get(new QName(ns, name.toLowerCase()));
502                    if (extendedTypeClass != null) {
503                            try {
504                                    return extendedTypeClass.newInstance();
505                            } catch (Exception e) {
506                                    //should never be thrown
507                                    //the type class is checked to see if it has a public, no-arg constructor in the "registerExtendedType" method
508                                    throw new RuntimeException("Extended type class \"" + extendedTypeClass.getName() + "\" MUST have a public, no-arg constructor.");
509                            }
510                    }
511    
512                    //parse as a standard property
513                    Class<? extends VCardType> clazz = TypeList.getTypeClass(name);
514                    if (clazz != null && VCardVersion.V4_0.getXmlNamespace().equals(ns)) {
515                            try {
516                                    return clazz.newInstance();
517                            } catch (Exception e) {
518                                    //should never be thrown
519                                    //all type classes must have public, no-arg constructors
520                                    throw new RuntimeException(e);
521                            }
522                    }
523    
524                    //parse as a RawType
525                    if (name.toUpperCase().startsWith("X-")) {
526                            return new RawType(name);
527                    }
528    
529                    //parse as an XML property
530                    return new XmlType();
531            }

533            /**
534             * Gets the QName from a type class.
535             * @param clazz the type class
536             * @return the QName
537             */
538            private QName getQNameFromTypeClass(Class<? extends VCardType> clazz) {
539                    try {
540                            VCardType type = clazz.newInstance();
541                            QName qname = type.getQName();
542                            if (qname == null) {
543                                    qname = new QName(version4.getXmlNamespace(), type.getTypeName().toLowerCase());
544                            }
545                            return qname;
546                    } catch (Exception e) {
547                            //there is no public, no-arg constructor
548                            throw new RuntimeException(e);
549                    }
550            }

646            /**
647             * Adds a vCard to the XML document.
648             * @param vcard the vCard to add
649             */
650            public void add(VCard vcard) {
651                    ListMultimap<String, VCardType> typesToAdd = new ListMultimap<String, VCardType>(); //group the types by group name (null = no group name)
652    
653                    for (VCardType type : vcard) {
654                            if (addProdId && type instanceof ProdIdType) {
655                                    //do not add the PRODID in the vCard if "addProdId" is true
656                                    continue;
657                            }
658    
659                            if (versionStrict && !type.isSupported(version4)) {
660                                    //do not add the property to the vCard if it is not supported by the target version
661                                    continue;
662                            }
663    
664                            typesToAdd.put(type.getGroup(), type);
665                    }
666    
667                    //add an extended type saying it was generated by this library
668                    if (addProdId) {
669                            EzvcardProdIdType prodId = new EzvcardProdIdType(version4);
670                            typesToAdd.put(prodId.getGroup(), prodId);
671                    }
672    
673                    //marshal each type object
674                    Element vcardElement = createElement("vcard");
675                    for (String groupName : typesToAdd.keySet()) {
676                            Element parent;
677                            if (groupName != null) {
678                                    Element groupElement = createElement("group");
679                                    groupElement.setAttribute("name", groupName);
680                                    vcardElement.appendChild(groupElement);
681                                    parent = groupElement;
682                            } else {
683                                    parent = vcardElement;
684                            }
685    
686                            for (VCardType type : typesToAdd.get(groupName)) {
687                                    try {
688                                            Element typeElement = marshalType(type, vcard);
689                                            parent.appendChild(typeElement);
690                                    } catch (SkipMeException e) {
691                                            //skip property
692                                    } catch (EmbeddedVCardException e) {
693                                            //skip property
694                                    }
695                            }
696                    }
697                    root.appendChild(vcardElement);
698            }
699    
700            /**
701             * Marshals a type object to an XML element.
702             * @param type the type object to marshal
703             * @param vcard the vcard the type belongs to
704             * @return the XML element or null not to add anything to the final XML
705             * document
706             */
707            private Element marshalType(VCardType type, VCard vcard) {
708                    QName qname = type.getQName();
709                    String ns, localPart;
710                    if (qname == null) {
711                            localPart = type.getTypeName().toLowerCase();
712                            ns = version4.getXmlNamespace();
713                    } else {
714                            localPart = qname.getLocalPart();
715                            ns = qname.getNamespaceURI();
716                    }
717                    Element typeElement = createElement(localPart, ns);
718    
719                    //marshal the sub types
720                    VCardSubTypes subTypes = type.marshalSubTypes(version4, compatibilityMode, vcard);
721                    subTypes.setValue(null); //don't include the VALUE parameter (modification of the "VCardSubTypes" object is safe because it's a copy)
722                    if (!subTypes.isEmpty()) {
723                            Element parametersElement = createElement("parameters");
724                            for (Map.Entry<String, List<String>> param : subTypes) {
725                                    String paramName = param.getKey();
726                                    Element parameterElement = createElement(paramName.toLowerCase());
727                                    for (String paramValue : param.getValue()) {
728                                            VCardDataType dataType = parameterDataTypes.get(paramName.toLowerCase());
729                                            String valueElementName = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
730                                            Element parameterValueElement = createElement(valueElementName);
731                                            parameterValueElement.setTextContent(paramValue);
732                                            parameterElement.appendChild(parameterValueElement);
733                                    }
734                                    parametersElement.appendChild(parameterElement);
735                            }
736                            typeElement.appendChild(parametersElement);
737                    }
738    
739                    //marshal the value
740                    type.marshalXml(typeElement, version4, compatibilityMode);
741    
742                    return typeElement;
743            }
744    
745            private List<Element> getVCardElements() {
746                    return getChildElements(root, "vcard");
747            }
748    
749            private List<Element> getChildElements(Element parent, String localName) {
750                    List<Element> elements = new ArrayList<Element>();
751                    for (Element child : XmlUtils.toElementList(parent.getChildNodes())) {
752                            if (localName.equals(child.getLocalName()) && version4.getXmlNamespace().equals(child.getNamespaceURI())) {
753                                    elements.add(child);
754                            }
755                    }
756                    return elements;
757            }

760             * Creates a new XML element.
761             * @param name the name of the XML element
762             * @return the new XML element
763             */
764            private Element createElement(String name) {
765                    return createElement(name, version4.getXmlNamespace());
766            }

768            /**
769             * Creates a new XML element.
770             * @param name the name of the XML element
771             * @param ns the namespace of the XML element
772             * @return the new XML element

776            }
