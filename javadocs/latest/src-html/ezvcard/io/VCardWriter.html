151             * Creates a vCard writer.
152             * @param writer the writer to write the vCard to
153             * @param targetVersion the version that the vCards should conform to
154             * @param foldingScheme the folding scheme to use or null not to fold at all
155             * @param newline the newline sequence to use
156             */
157            public VCardWriter(Writer writer, VCardVersion targetVersion, FoldingScheme foldingScheme, String newline) {
158                    this.writer = new VCardRawWriter(writer, targetVersion, foldingScheme, newline);
159                    this.writer.setProblemsListener(new ProblemsListener() {
160                            public void onParameterValueChanged(String propertyName, String parameterName, String originalValue, String modifiedValue) {
161                                    addWarning("\"" + parameterName + "\" parameter contained one or more characters which are not allowed in vCard " + VCardWriter.this.writer.getVersion() + " parameter values.  These characters were removed.", propertyName);
162                            }
163    
164                            public void onQuotedPrintableEncodingFailed(String propertyName, String propertyValue, EncoderException thrown) {
165                                    addWarning("A unexpected error occurred while encoding the property's value into \"quoted-printable\" encoding.  Value will not be encoded: " + thrown.getMessage(), propertyName);
166                            }
167                    });
168            }

291            /**
292             * Writes a vCard to the stream.
293             * @param vcard the vCard to write
294             * @throws IOException if there's a problem writing to the output stream
295             */
296            public void write(VCard vcard) throws IOException {
297                    warnings.clear();
298                    write(vcard, addProdId);
299            }
300    
301            private void write(VCard vcard, boolean addProdId) throws IOException {
302                    VCardVersion targetVersion = writer.getVersion();
303    
304                    if (targetVersion == VCardVersion.V2_1 || targetVersion == VCardVersion.V3_0) {
305                            if (vcard.getStructuredName() == null) {
306                                    warnings.add("vCard version " + targetVersion + " requires that a structured name property be defined.");
307                            }
308                    }
309    
310                    if (targetVersion == VCardVersion.V3_0 || targetVersion == VCardVersion.V4_0) {
311                            if (vcard.getFormattedName() == null) {
312                                    warnings.add("vCard version " + targetVersion + " requires that a formatted name property be defined.");
313                            }
314                    }
315    
316                    List<VCardType> typesToAdd = new ArrayList<VCardType>();
317                    for (VCardType type : vcard) {
318                            if (addProdId && type instanceof ProdIdType) {
319                                    //do not add the PRODID in the vCard if "addProdId" is true
320                                    continue;
321                            }
322    
323                            //determine if this type is supported by the target version
324                            if (!supportsTargetVersion(type)) {
325                                    addWarning("This property is not supported by vCard version " + targetVersion + " and will not be added to the vCard.  Supported versions are: " + Arrays.toString(type.getSupportedVersions()), type.getTypeName());
326                                    continue;
327                            }
328    
329                            //check for correct KIND value if there are MEMBER types
330                            if (type instanceof MemberType && (vcard.getKind() == null || !vcard.getKind().isGroup())) {
331                                    addWarning("Value must be set to \"group\" if the vCard contains " + MemberType.NAME + " properties.", KindType.NAME);
332                                    continue;
333                            }
334    
335                            typesToAdd.add(type);
336    
337                            //add LABEL types for each ADR type if the target version is 2.1 or 3.0
338                            if (type instanceof AddressType && targetVersion != VCardVersion.V4_0) {
339                                    AddressType adr = (AddressType) type;
340                                    String labelStr = adr.getLabel();
341                                    if (labelStr != null) {
342                                            LabelType label = new LabelType(labelStr);
343                                            for (AddressTypeParameter t : adr.getTypes()) {
344                                                    label.addType(t);
345                                            }
346                                            typesToAdd.add(label);
347                                    }
348                            }
349                    }
350    
351                    //add an extended type saying it was generated by this library
352                    if (addProdId) {
353                            EzvcardProdIdType prodId = new EzvcardProdIdType(targetVersion);
354                            typesToAdd.add(prodId);
355                    }
356    
357                    writer.writeBeginComponent("VCARD");
358                    writer.writeVersion();
359    
360                    List<String> warningsBuf = new ArrayList<String>();
361                    for (VCardType type : typesToAdd) {
362                            //marshal the value
363                            warningsBuf.clear();
364                            String value = null;
365                            VCard nestedVCard = null;
366                            try {
367                                    value = type.marshalText(targetVersion, warningsBuf, compatibilityMode);
368                            } catch (SkipMeException e) {
369                                    warningsBuf.add("Property has requested that it be skipped: " + e.getMessage());
370                                    continue;
371                            } catch (EmbeddedVCardException e) {
372                                    nestedVCard = e.getVCard();
373                            } finally {
374                                    for (String warning : warningsBuf) {
375                                            addWarning(warning, type.getTypeName());
376                                    }
377                            }
378    
379                            //marshal the sub types
380                            warningsBuf.clear();
381                            VCardSubTypes parameters;
382                            try {
383                                    parameters = type.marshalSubTypes(targetVersion, warningsBuf, compatibilityMode, vcard);
384                            } finally {
385                                    for (String warning : warningsBuf) {
386                                            addWarning(warning, type.getTypeName());
387                                    }
388                            }
389    
390                            if (nestedVCard == null) {
391                                    writer.writeProperty(type.getGroup(), type.getTypeName(), parameters, value);
392                            } else {
393                                    if (targetVersion == VCardVersion.V2_1) {
394                                            //write a nested vCard (2.1 style)
395                                            writer.writeProperty(type.getGroup(), type.getTypeName(), parameters, value);
396                                            write(nestedVCard, false);
397                                    } else {
398                                            //write an embedded vCard (3.0 style)
399                                            StringWriter sw = new StringWriter();
400                                            VCardWriter agentWriter = new VCardWriter(sw, targetVersion, null, "\n");
401                                            agentWriter.setAddProdId(false);
402                                            agentWriter.setCompatibilityMode(compatibilityMode);
403                                            try {
404                                                    agentWriter.write(nestedVCard);
405                                            } catch (IOException e) {
406                                                    //writing to a string
407                                            } finally {
408                                                    for (String w : agentWriter.getWarnings()) {
409                                                            addWarning("Problem marshalling embedded vCard value: " + w, type.getTypeName());
410                                                    }
411                                            }
412    
413                                            String vCardStr = sw.toString();
414                                            vCardStr = VCardStringUtils.escape(vCardStr);
415                                            writer.writeProperty(type.getGroup(), type.getTypeName(), parameters, vCardStr);
416                                    }
417                            }
418                    }
419    
420                    writer.writeEndComponent("VCARD");
421            }
422    
423            private boolean supportsTargetVersion(VCardType type) {
424                    for (VCardVersion version : type.getSupportedVersions()) {
425                            if (version == writer.getVersion()) {
426                                    return true;
427                            }
428                    }
429                    return false;
430            }
