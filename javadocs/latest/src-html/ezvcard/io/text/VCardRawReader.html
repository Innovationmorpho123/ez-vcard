043/**
044 * Parses a vCard data stream.
045 * @author Michael Angstadt
046 */
047public class VCardRawReader implements Closeable {
048        private final FoldedLineReader reader;
049        private boolean caretDecodingEnabled = true;
050        private VCardVersion version = VCardVersion.V2_1; //initialize to 2.1, since the VERSION property can exist anywhere in the file in this version
051
052        /**
053         * Creates a new reader.
054         * @param reader the reader to the data stream
055         */
056        public VCardRawReader(Reader reader) {
057                this.reader = new FoldedLineReader(reader);
058        }
059
060        /**
061         * Gets the line number of the last line that was read.
062         * @return the line number
063         */
064        public int getLineNum() {
065                return reader.getLineNum();
066        }
067
068        /**
069         * Gets the vCard version that the reader is currently parsing with.
070         * @return the vCard version
071         */
072        public VCardVersion getVersion() {
073                return version;
074        }
075
076        /**
077         * Parses the next line of the vCard file.
078         * @return the next line or null if there are no more lines
079         * @throws InvalidVersionException if a VERSION property with an invalid
080         * value is encountered
081         * @throws VCardParseException if a line cannot be parsed
082         * @throws IOException if there's a problem reading from the input stream
083         */
084        public VCardRawLine readLine() throws IOException {
085                String line = reader.readLine();
086                if (line == null) {
087                        return null;
088                }
089
090                String group = null;
091                String propertyName = null;
092                VCardParameters parameters = new VCardParameters();
093                String value = null;
094
095                char escapeChar = 0; //is the next char escaped?
096                boolean inQuotes = false; //are we inside of double quotes?
097                StringBuilder buffer = new StringBuilder();
098                String curParamName = null;
099                for (int i = 0; i < line.length(); i++) {
100                        char ch = line.charAt(i);
101
102                        if (escapeChar != 0) {
103                                //this character was escaped
104                                if (escapeChar == '\\') {
105                                        if (ch == '\\') {
106                                                buffer.append(ch);
107                                        } else if (ch == 'n' || ch == 'N') {
108                                                //newlines appear as "\n" or "\N" (see RFC 2426 p.7)
109                                                buffer.append(NEWLINE);
110                                        } else if (ch == '"' && version != VCardVersion.V2_1) {
111                                                //double quotes don't need to be escaped in 2.1 parameter values because they have no special meaning
112                                                buffer.append(ch);
113                                        } else if (ch == ';' && version == VCardVersion.V2_1) {
114                                                //semi-colons can only be escaped in 2.1 parameter values (see section 2 of specs)
115                                                //if a 3.0/4.0 param value has semi-colons, the value should be surrounded in double quotes
116                                                buffer.append(ch);
117                                        } else {
118                                                //treat the escape character as a normal character because it's not a valid escape sequence
119                                                buffer.append(escapeChar).append(ch);
120                                        }
121                                } else if (escapeChar == '^') {
122                                        if (ch == '^') {
123                                                buffer.append(ch);
124                                        } else if (ch == 'n') {
125                                                buffer.append(NEWLINE);
126                                        } else if (ch == '\'') {
127                                                buffer.append('"');
128                                        } else {
129                                                //treat the escape character as a normal character because it's not a valid escape sequence
130                                                buffer.append(escapeChar).append(ch);
131                                        }
132                                }
133                                escapeChar = 0;
134                                continue;
135                        }
136
137                        if (ch == '\\' || (ch == '^' && version != VCardVersion.V2_1 && caretDecodingEnabled)) {
138                                //an escape character was read
139                                escapeChar = ch;
140                                continue;
141                        }
142
143                        if (ch == '.' && group == null && propertyName == null) {
144                                //set the group
145                                group = buffer.toString();
146                                buffer.setLength(0);
147                                continue;
148                        }
149
150                        if ((ch == ';' || ch == ':') && !inQuotes) {
151                                if (propertyName == null) {
152                                        //property name
153                                        propertyName = buffer.toString();
154                                } else {
155                                        //parameter value
156                                        String paramValue = buffer.toString();
157                                        if (version == VCardVersion.V2_1) {
158                                                //2.1 allows whitespace to surround the "=", so remove it
159                                                paramValue = StringUtils.ltrim(paramValue);
160                                        }
161                                        parameters.put(curParamName, paramValue);
162                                        curParamName = null;
163                                }
164                                buffer.setLength(0);
165
166                                if (ch == ':') {
167                                        //the rest of the line is the property value
168                                        if (i < line.length() - 1) {
169                                                value = line.substring(i + 1);
170                                        } else {
171                                                value = "";
172                                        }
173                                        break;
174                                }
175                                continue;
176                        }
177
178                        if (ch == ',' && !inQuotes && version != VCardVersion.V2_1) {
179                                //multi-valued parameter
180                                parameters.put(curParamName, buffer.toString());
181                                buffer.setLength(0);
182                                continue;
183                        }
184
185                        if (ch == '=' && curParamName == null) {
186                                //parameter name
187                                String paramName = buffer.toString();
188                                if (version == VCardVersion.V2_1) {
189                                        //2.1 allows whitespace to surround the "=", so remove it
190                                        paramName = StringUtils.rtrim(paramName);
191                                }
192                                curParamName = paramName;
193                                buffer.setLength(0);
194                                continue;
195                        }
196
197                        if (ch == '"' && version != VCardVersion.V2_1) {
198                                //2.1 doesn't use the quoting mechanism
199                                inQuotes = !inQuotes;
200                                continue;
201                        }
202
203                        buffer.append(ch);
204                }
205
206                if (propertyName == null || value == null) {
207                        throw new VCardParseException(line);
208                }
209
210                if ("VERSION".equalsIgnoreCase(propertyName)) {
211                        VCardVersion version = VCardVersion.valueOfByStr(value);
212                        if (version == null) {
213                                throw new InvalidVersionException(value, line);
214                        }
215                        this.version = version;
216                }
217
218                value = value.trim();
219                return new VCardRawLine(group, propertyName, parameters, value);
220        }
221
222        /**
223         * <p>
224         * Gets whether the reader will decode parameter values that use circumflex
225         * accent encoding (enabled by default). This escaping mechanism allows
226         * newlines and double quotes to be included in parameter values.
227         * </p>
228         * 
229         * <table class="simpleTable">
230         * <tr>
231         * <th>Raw Character</th>
232         * <th>Encoded Character</th>
233         * </tr>
234         * <tr>
235         * <td>{@code "}</td>
236         * <td>{@code ^'}</td>
237         * </tr>
238         * <tr>
239         * <td><i>newline</i></td>
240         * <td>{@code ^n}</td>
241         * </tr>
242         * <tr>
243         * <td>{@code ^}</td>
244         * <td>{@code ^^}</td>
245         * </tr>
246         * </table>
247         * 
248         * <p>
249         * Example:
250         * </p>
251         * 
252         * <pre>
253         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
254         *  sburgh, PA 15212":40.446816;80.00566
255         * </pre>
256         * 
257         * @return true if circumflex accent decoding is enabled, false if not
258         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
259         */
260        public boolean isCaretDecodingEnabled() {
261                return caretDecodingEnabled;
262        }
263
264        /**
265         * <p>
266         * Sets whether the reader will decode parameter values that use circumflex
267         * accent encoding (enabled by default). This escaping mechanism allows
268         * newlines and double quotes to be included in parameter values.
269         * </p>
270         * 
271         * <table class="simpleTable">
272         * <tr>
273         * <th>Raw Character</th>
274         * <th>Encoded Character</th>
275         * </tr>
276         * <tr>
277         * <td>{@code "}</td>
278         * <td>{@code ^'}</td>
279         * </tr>
280         * <tr>
281         * <td><i>newline</i></td>
282         * <td>{@code ^n}</td>
283         * </tr>
284         * <tr>
285         * <td>{@code ^}</td>
286         * <td>{@code ^^}</td>
287         * </tr>
288         * </table>
289         * 
290         * <p>
291         * Example:
292         * </p>
293         * 
294         * <pre>
295         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
296         *  sburgh, PA 15212":geo:40.446816,-80.00566
297         * </pre>
298         * 
299         * @param enable true to use circumflex accent decoding, false not to
300         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
301         */
302        public void setCaretDecodingEnabled(boolean enable) {
303                caretDecodingEnabled = enable;
304        }
305
306        /**
307         * Gets the character encoding of the reader.
308         * @return the character encoding or null if none is defined
309         */
310        public Charset getEncoding() {
311                return reader.getEncoding();
312        }
313
314        /**
315         * Closes the underlying {@link Reader} object.
316         */
317        public void close() throws IOException {
318                reader.close();
319        }
320}
