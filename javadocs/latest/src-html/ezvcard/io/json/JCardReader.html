056/**
057 * <p>
058 * Parses {@link VCard} objects from a JSON data stream (jCard format).
059 * </p>
060 * <p>
061 * <b>Example:</b>
062 * 
063 * <pre class="brush:java">
064 * File file = new File("vcards.json");
065 * JCardReader jcardReader = new JCardReader(file);
066 * VCard vcard;
067 * while ((vcard = jcardReader.readNext()) != null){
068 *   ...
069 * }
070 * jcardReader.close();
071 * </pre>
072 * 
073 * </p>
074 * @author Michael Angstadt
075 * @see <a href="http://tools.ietf.org/html/rfc7095">RFC 7095</a>
076 */
077public class JCardReader extends StreamReader {
078        private final JCardRawReader reader;
079
080        /**
081         * Creates a jCard reader.
082         * @param json the JSON string
083         */
084        public JCardReader(String json) {
085                this(new StringReader(json));
086        }
087
088        /**
089         * Creates a jCard reader.
090         * @param in the input stream to read the vCards from
091         */
092        public JCardReader(InputStream in) {
093                this(utf8Reader(in));
094        }
095
096        /**
097         * Creates a jCard reader.
098         * @param file the file to read the vCards from
099         * @throws FileNotFoundException if the file doesn't exist
100         */
101        public JCardReader(File file) throws FileNotFoundException {
102                this(utf8Reader(file));
103        }
104
105        /**
106         * Creates a jCard reader.
107         * @param reader the reader to read the vCards from
108         */
109        public JCardReader(Reader reader) {
110                this.reader = new JCardRawReader(reader);
111        }
112
113        @Override
114        protected VCard _readNext() throws IOException {
115                if (reader.eof()) {
116                        return null;
117                }
118
119                warnings.clear();
120
121                JCardDataStreamListenerImpl listener = new JCardDataStreamListenerImpl();
122                reader.readNext(listener);
123                VCard vcard = listener.vcard;
124                if (vcard != null && !listener.versionFound) {
125                        warnings.add(reader.getLineNum(), null, 29);
126                }
127                return vcard;
128        }
129
130        public void close() throws IOException {
131                reader.close();
132        }
133
134        private class JCardDataStreamListenerImpl implements JCardDataStreamListener {
135                private VCard vcard = null;
136                private boolean versionFound = false;
137
138                public void beginVCard() {
139                        vcard = new VCard();
140                        vcard.setVersion(VCardVersion.V4_0);
141                }
142
143                public void readProperty(String group, String propertyName, VCardParameters parameters, VCardDataType dataType, JCardValue value) {
144                        if ("version".equalsIgnoreCase(propertyName)) {
145                                //don't unmarshal "version" because we don't treat it as a property
146                                versionFound = true;
147
148                                VCardVersion version = VCardVersion.valueOfByStr(value.asSingle());
149                                if (version != VCardVersion.V4_0) {
150                                        warnings.add(reader.getLineNum(), propertyName, 30);
151                                }
152                                return;
153                        }
154
155                        VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(propertyName);
156                        if (scribe == null) {
157                                scribe = new RawPropertyScribe(propertyName);
158                        }
159
160                        Result<? extends VCardProperty> result;
161                        try {
162                                result = scribe.parseJson(value, dataType, parameters);
163                                for (String warning : result.getWarnings()) {
164                                        warnings.add(reader.getLineNum(), warning, propertyName);
165                                }
166                        } catch (SkipMeException e) {
167                                warnings.add(reader.getLineNum(), propertyName, 22, e.getMessage());
168                                return;
169                        } catch (CannotParseException e) {
170                                scribe = new RawPropertyScribe(propertyName);
171                                result = scribe.parseJson(value, dataType, parameters);
172
173                                VCardProperty property = result.getProperty();
174                                String valueStr = ((RawProperty) property).getValue();
175                                warnings.add(reader.getLineNum(), propertyName, 25, valueStr, e.getMessage());
176                        } catch (EmbeddedVCardException e) {
177                                warnings.add(reader.getLineNum(), propertyName, 31);
178                                return;
179                        }
180
181                        VCardProperty property = result.getProperty();
182                        property.setGroup(group);
183                        vcard.addProperty(property);
184                }
185        }
186}
