037    /**
038     * Marshals properties that contain a list of values.
039     * @param <T> the property class
040     * @author Michael Angstadt
041     */
042    public abstract class ListPropertyScribe<T extends TextListProperty> extends VCardPropertyScribe<T> {
043            public ListPropertyScribe(Class<T> clazz, String propertyName) {
044                    super(clazz, propertyName);
045            }
046    
047            @Override
048            protected VCardDataType _defaultDataType(VCardVersion version) {
049                    return VCardDataType.TEXT;
050            }
051    
052            @Override
053            protected String _writeText(T property, VCardVersion version) {
054                    return list(property.getValues());
055            }
056    
057            @Override
058            protected T _parseText(String value, VCardDataType dataType, VCardVersion version, VCardParameters parameters, List<String> warnings) {
059                    List<String> values = list(value);
060                    return parse(values);
061            }
062    
063            @Override
064            protected void _writeXml(T property, XCardElement parent) {
065                    parent.append(VCardDataType.TEXT.getName().toLowerCase(), property.getValues());
066            }
067    
068            @Override
069            protected T _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
070                    List<String> values = element.all(VCardDataType.TEXT);
071                    if (!values.isEmpty()) {
072                            return parse(values);
073                    }
074    
075                    throw missingXmlElements(VCardDataType.TEXT);
076            }
077    
078            @Override
079            protected JCardValue _writeJson(T property) {
080                    List<String> values = property.getValues();
081                    if (values.isEmpty()) {
082                            return JCardValue.single("");
083                    }
084    
085                    return JCardValue.multi(values);
086            }
087    
088            @Override
089            protected T _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
090                    List<String> values = value.asMulti();
091                    return parse(values);
092            }
093    
094            private T parse(List<String> values) {
095                    T property = _newInstance();
096                    property.getValues().addAll(values);
097                    return property;
098            }
099    
100            protected abstract T _newInstance();
101    }
