067             * Gets the ENCODING sub type. This is used when the type value is encoded
068             * in a form other than plain text.

257            /**
258             * <p>
259             * Gets the preference value. The lower the number, the more preferred this
260             * property instance is compared with other properties in the same vCard of
261             * the same type. If a property doesn't have a preference value, then it is
262             * considered the least preferred.
263             * </p>
264             * 
265             * <p>
266             * In the vCard below, the address on the second row is the most preferred
267             * because it has the lowest PREF value.
268             * </p>
269             * 
270             * <pre>
271             * ADR;TYPE=work;PREF=2:
272             * ADR;TYPE=work;PREF=1:
273             * ADR;TYPE=home:
274             * </pre>
275             * 
276             * <p>
277             * Preference values must be numeric and must be between 1 and 100.
278             * </p>
279             * 
280             * <p>
281             * <b>Supported versions:</b> <code>4.0</code>
282             * </p>
283             * @return the preference value or null if it doesn't exist or null if it
284             * couldn't be parsed into a number
285             */
286            public Integer getPref() {
287                    String pref = first("PREF");
288                    if (pref == null) {
289                            return null;
290                    }
291    
292                    try {
293                            return Integer.valueOf(pref);
294                    } catch (NumberFormatException e) {
295                            return null;
296                    }
297            }

300             * <p>
301             * Sets the preference value. The lower the number, the more preferred this
302             * property instance is compared with other properties in the same vCard of
303             * the same type. If a property doesn't have a preference value, then it is
304             * considered the least preferred.
305             * </p>
306             * 
307             * <p>
308             * In the vCard below, the address on the second row is the most preferred
309             * because it has the lowest PREF value.
310             * </p>
311             * 
312             * <pre>
313             * ADR;TYPE=work;PREF=2:
314             * ADR;TYPE=work;PREF=1:
315             * ADR;TYPE=home:
316             * </pre>
317             * 
318             * <p>
319             * Preference values must be numeric and must be between 1 and 100.
320             * </p>
321             * 
322             * <p>
323             * <b>Supported versions:</b> <code>4.0</code>
324             * </p>
325             * @param pref the preference value or null to remove
326             * @throws IllegalArgumentException if the value is not between 1 and 100
327             */
328            public void setPref(Integer pref) {
329                    if (pref != null && (pref < 1 || pref > 100)) {
330                            throw new IllegalArgumentException("Preference value must be between 1 and 100 inclusive.");
331                    }
332                    String value = (pref == null) ? null : pref.toString();
333                    replace("PREF", value);
334            }

339             * representations of the same type.
340             * </p>
341             * 
342             * <p>
343             * For example, a vCard may contain multiple NOTE types that each have the
344             * same ALTID. This means that each NOTE contains a different representation
345             * of the same information. In the example below, the first three NOTEs have
346             * the same ALTID. They each contain the same message, but each is written
347             * in a different language. The other NOTEs have different (or absent)
348             * ALTIDs, which means they are not associated with the top three.
349             * </p>
350             * 
351             * <pre>
352             * NOTE;ALTID=1;LANGUAGE=en: Hello world!
353             * NOTE;ALTID=1;LANGUAGE=fr: Bonjour tout le monde!
354             * NOTE;ALTID=1;LANGUAGE=es: �Hola, mundo!
355             * NOTE;ALTID=2;LANGUAGE=de: Meine Lieblingsfarbe ist blau.
356             * NOTE;ALTID=2;LANGUAGE=en: My favorite color is blue.
357             * NOTE: This vCard will self-destruct in 5 seconds.
358             * </pre>
359             * 
360             * <p>
361             * <b>Supported versions:</b> <code>4.0</code>
362             * </p>
363             * @return the ALTID or null if it doesn't exist

403             * <p>
404             * Gets the GEO parameter value. This is used to associate global
405             * positioning information with a vCard type. It can be used with the ADR
406             * type.
407             * </p>
408             * <p>
409             * <b>Supported versions:</b> <code>4.0</code>
410             * </p>
411             * @return the latitude (index 0) and longitude (index 1) or null if not
412             * present or null if the parameter value was in an incorrect format
413             */
414            public double[] getGeo() {
415                    String value = first("GEO");
416                    if (value == null) {
417                            return null;
418                    }
419    
420                    try {
421                            GeoUri geoUri = new GeoUri(value);
422                            return new double[] { geoUri.getCoordA(), geoUri.getCoordB() };
423                    } catch (IllegalArgumentException e) {
424                            return null;
425                    }
426            }

522            /**
523             * <p>
524             * Gets all PID parameter values. PIDs can exist on any property where
525             * multiple instances are allowed (such as EMAIL or ADR, but not N because
526             * only 1 instance of N is allowed).
527             * </p>
528             * <p>
529             * When used in conjunction with the CLIENTPIDMAP property, it allows an
530             * individual property instance to be uniquely identifiable. This feature is
531             * made use of when two different versions of the same vCard have to be
532             * merged together (called "synchronizing").
533             * </p>
534             * <p>
535             * <b>Supported versions:</b> <code>4.0</code>
536             * </p>
537             * @return the PID values or empty set if there are none. Index 0 is the
538             * local ID and index 1 is the ID used to reference the CLIENTPIDMAP
539             * property. Index 0 will never be null, but index 1 may be null.
540             */
541            public List<Integer[]> getPids() {
542                    List<String> values = get("PID");
543                    List<Integer[]> pids = new ArrayList<Integer[]>(values.size());
544                    for (String value : values) {
545                            try {
546                                    String split[] = value.split("\\.");
547                                    Integer pid[] = new Integer[2];
548                                    pid[0] = Integer.valueOf(split[0]);
549                                    if (split.length > 1) {
550                                            pid[1] = Integer.valueOf(split[1]);
551                                    }
552                                    pids.add(pid);
553                            } catch (NumberFormatException e) {
554                                    //skip
555                            }
556                    }
557                    return pids;
558            }

676            /**
677             * <p>
678             * Gets the sorted position of this property when it is grouped together
679             * with other properties of the same type. Properties with low index values
680             * are put at the beginning of the sorted list and properties with high
681             * index values are put at the end of the list.
682             * </p>
683             * 
684             * <p>
685             * <b>Supported versions:</b> <code>4.0</code>
686             * </p>
687             * @return the INDEX value or null if it doesn't exist or null if it
688             * couldn't be parsed into a number
689             * @see <a href="http://tools.ietf.org/html/rfc6715">RFC 6715</a>
690             */
691            public Integer getIndex() {
692                    String index = first("INDEX");
693                    if (index == null) {
694                            return null;
695                    }
696    
697                    try {

705             * <p>
706             * Sets the sorted position of this property when it is grouped together
707             * with other properties of the same type. Properties with low index values
708             * are put at the beginning of the sorted list and properties with high
709             * index values are put at the end of the list.
710             * </p>
711             * 
712             * <p>
713             * <b>Supported versions:</b> <code>4.0</code>
714             * </p>
715             * @param index the INDEX value (must be greater than 0) or null to remove
716             * @see <a href="http://tools.ietf.org/html/rfc6715">RFC 6715</a>
717             * @throws IllegalArgumentException if the value is not greater than 0
718             */
719            public void setIndex(Integer index) {
720                    if (index != null && index <= 0) {
721                            throw new IllegalArgumentException("INDEX value must be greater than 0.");
722                    }
723                    String value = (index == null) ? null : index.toString();
724                    replace("INDEX", value);
725            }
