034    /**
035     * Represents a vCard Type that has a TYPE parameter that contains only one
036     * value (for example, {@link PhotoType}).
037     * @param <T> the parameter class
038     * @author Michael Angstadt
039     * @see MultiValuedTypeParameterType
040     */
041    public abstract class SingleValuedTypeParameterType<T extends VCardParameter> extends TypeParameterType<T> {
042            /**
043             * Creates a single-valued type parameter property.
044             * @param name the type name (e.g. "PHOTO")
045             */
046            public SingleValuedTypeParameterType(String name) {
047                    super(name);
048            }
049    
050            /**
051             * Gets the value of the TYPE parameter.
052             * @return the TYPE value or null if it has no TYPE value
053             */
054            public T getType() {
055                    String type = subTypes.getType();
056                    return (type == null) ? null : buildTypeObj(type);
057            }
058    
059            /**
060             * Sets the value of the TYPE parameter.
061             * @param type the TYPE value to set
062             */
063            public void setType(T type) {
064                    subTypes.setType(type.getValue());
065            }
066    }
