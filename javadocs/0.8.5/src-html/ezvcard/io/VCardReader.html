065    /**
066     * <p>
067     * Parses {@link VCard} objects from a plain-text vCard data stream.
068     * </p>
069     * <p>
070     * <b>Example:</b>
071     * 
072     * <pre class="brush:java">
073     * File file = new File("vcards.vcf");
074     * VCardReader vcardReader = new VCardReader(file);
075     * VCard vcard;
076     * while ((vcard = vcardReader.readNext()) != null){
077     *   ...
078     * }
079     * vcardReader.close();
080     * </pre>
081     * 
082     * </p>
083     * @author Michael Angstadt
084     */
085    public class VCardReader implements Closeable {
086            private CompatibilityMode compatibilityMode = CompatibilityMode.RFC;
087            private final List<String> warnings = new ArrayList<String>();
088            private final Map<String, Class<? extends VCardType>> extendedTypeClasses = new HashMap<String, Class<? extends VCardType>>();
089            private final VCardRawReader reader;
090    
091            /**
092             * Creates a reader that parses vCards from a string.
093             * @param str the string to read the vCards from
094             */
095            public VCardReader(String str) {
096                    this(new StringReader(str));
097            }
098    
099            /**
100             * Creates a reader that parses vCards from an input stream.
101             * @param in the input stream to read the vCards from
102             */
103            public VCardReader(InputStream in) {
104                    this(new InputStreamReader(in));
105            }

107            /**
108             * Creates a reader that parses vCards from a file.
109             * @param file the file to read the vCards from
110             * @throws FileNotFoundException if the file doesn't exist
111             */
112            public VCardReader(File file) throws FileNotFoundException {
113                    this(new FileReader(file));
114            }

116            /**
117             * Creates a reader that parses vCards from a reader.
118             * @param reader the reader to read the vCards from
119             */
120            public VCardReader(Reader reader) {
121                    this.reader = new VCardRawReader(reader);

128             * @return true if circumflex accent decoding is enabled, false if not
129             * @see VCardRawReader#isCaretDecodingEnabled()

166            /**
167             * Registers an extended type class.
168             * @param clazz the extended type class to register (MUST have a public,
169             * no-arg constructor)
170             * @throws RuntimeException if the class doesn't have a public, no-arg
171             * constructor
172             */
173            public void registerExtendedType(Class<? extends VCardType> clazz) {
174                    extendedTypeClasses.put(getTypeNameFromTypeClass(clazz), clazz);
175            }
176    
177            /**
178             * Removes an extended type class that was previously registered.
179             * @param clazz the extended type class to remove
180             */
181            public void unregisterExtendedType(Class<? extends VCardType> clazz) {
182                    extendedTypeClasses.remove(getTypeNameFromTypeClass(clazz));
183            }
184    
185            /**
186             * Gets the warnings from the last vCard that was unmarshalled. This list is
187             * reset every time a new vCard is read.
188             * @return the warnings or empty list if there were no warnings
189             */
190            public List<String> getWarnings() {
191                    return new ArrayList<String>(warnings);
192            }
193    
194            /**
195             * Reads the next vCard from the data stream.
196             * @return the next vCard or null if there are no more
197             * @throws IOException if there's a problem reading from the stream
198             */
199            public VCard readNext() throws IOException {
200                    if (reader.eof()) {
201                            return null;
202                    }
203    
204                    warnings.clear();
205    
206                    VCardDataStreamListenerImpl listener = new VCardDataStreamListenerImpl();
207                    reader.start(listener);
208    
209                    return listener.root;
210            }

212            /**
213             * Creates the appropriate {@link VCardType} instance, given the type name.
214             * This method does not unmarshal the type, it just creates the type object.
215             * @param name the type name (e.g. "FN")
216             * @return the Type that was created
217             */
218            private VCardType createTypeObject(String name) {
219                    //parse as a registered extended type class (extended type classes should override standard ones)
220                    Class<? extends VCardType> extendedTypeClass = extendedTypeClasses.get(name.toUpperCase());
221                    if (extendedTypeClass != null) {
222                            try {
223                                    return extendedTypeClass.newInstance();
224                            } catch (Exception e) {
225                                    //should never be thrown
226                                    //the type class is checked to see if it has a public, no-arg constructor in the "registerExtendedType" method
227                                    throw new RuntimeException("Extended type class \"" + extendedTypeClass.getName() + "\" must have a public, no-arg constructor.");
228                            }
229                    }
230    
231                    //parse as a standard property
232                    Class<? extends VCardType> clazz = TypeList.getTypeClass(name);
233                    if (clazz != null) {
234                            try {
235                                    return clazz.newInstance();
236                            } catch (Exception e) {
237                                    //should never be thrown
238                                    //all type classes must have public, no-arg constructors
239                                    throw new RuntimeException(e);
240                            }
241                    }
242    
243                    //parse as a RawType
244                    if (!name.toUpperCase().startsWith("X-")) {
245                            addWarning("Non-standard property found.  Treating it as an extended property.", name);
246                    }
247                    return new RawType(name); //use RawType instead of TextType because we don't want to unescape any characters that might be meaningful to this type
248            }
249    
250            /**
251             * Gets the type name from a type class.
252             * @param clazz the type class
253             * @return the type name
254             */
255            private String getTypeNameFromTypeClass(Class<? extends VCardType> clazz) {
256                    try {
257                            VCardType t = clazz.newInstance();
258                            return t.getTypeName().toUpperCase();
259                    } catch (Exception e) {
260                            //there is no public, no-arg constructor
261                            throw new RuntimeException(e);
262                    }
263            }
264    
265            /**
266             * Assigns names to all nameless parameters. v3.0 and v4.0 requires all
267             * parameters to have names, but v2.1 does not.
268             * @param parameters the parameters
269             */
270            private void handleNamelessParameters(VCardSubTypes parameters) {
271                    List<String> namelessParamValues = parameters.get(null);
272                    for (String paramValue : namelessParamValues) {
273                            String paramName;
274                            if (VCardDataType.find(paramValue) != null) {
275                                    paramName = VCardSubTypes.VALUE;
276                            } else if (EncodingParameter.find(paramValue) != null) {
277                                    paramName = VCardSubTypes.ENCODING;
278                            } else {
279                                    //otherwise, assume it's a TYPE
280                                    paramName = VCardSubTypes.TYPE;
281                            }
282                            parameters.put(paramName, paramValue);
283                    }
284                    parameters.removeAll(null);
285            }
286    
287            /**
288             * <p>
289             * Accounts for multi-valued TYPE parameters being enclosed entirely in
290             * double quotes (for example: ADR;TYPE="home,work").
291             * </p>
292             * <p>
293             * Many examples throughout the 4.0 specs show TYPE parameters being encoded
294             * in this way. This conflicts with the ABNF and is noted in the errata.
295             * This method will split the value by comma incase the vendor implemented
296             * it this way.
297             * </p>
298             * @param parameters the parameters
299             */
300            private void handleQuotedMultivaluedTypeParams(VCardSubTypes parameters) {
301                    //account for multi-valued TYPE parameters being enclosed entirely in double quotes
302                    //e.g. ADR;TYPE="home,work"
303                    for (String typeParam : parameters.getTypes()) {
304                            if (!typeParam.contains(",")) {
305                                    continue;
306                            }
307    
308                            parameters.removeTypes();
309                            for (String splitValue : typeParam.split(",")) {
310                                    parameters.addType(splitValue);
311                            }
312                    }
313            }
314    
315            /**
316             * Decodes the property value if it's encoded in quoted-printable encoding.
317             * Quoted-printable encoding is only supported in v2.1.
318             * @param name the property name
319             * @param parameters the parameters
320             * @param value the property value
321             * @return the decoded property value
322             */
323            private String decodeQuotedPrintable(String name, VCardSubTypes parameters, String value) {
324                    if (parameters.getEncoding() != EncodingParameter.QUOTED_PRINTABLE) {
325                            return value;
326                    }
327    
328                    parameters.setEncoding(null); //remove encoding sub type
329    
330                    QuotedPrintableCodec codec = new QuotedPrintableCodec();
331                    String charset = parameters.getCharset();
332                    try {
333                            if (charset == null) {
334                                    return codec.decode(value);
335                            } else {
336                                    try {
337                                            return codec.decode(value, charset);
338                                    } catch (UnsupportedEncodingException e) {
339                                            addWarning("The specified charset is not supported.  Using default charset instead: " + charset, name);
340                                            return codec.decode(value);
341                                    }
342                            }
343                    } catch (DecoderException e) {
344                            addWarning("Property value was marked as \"quoted-printable\", but it could not be decoded.  Treating the value as plain text.", name);
345                    }
346    
347                    return value;
348            }

350            /**
351             * Closes the underlying {@link Reader} object.
352             */
353            public void close() throws IOException {
354                    reader.close();
355            }
356    
357            private void addWarning(String message) {
358                    addWarning(message, null);
359            }
360    
361            private void addWarning(String message, String propertyName) {
362                    StringBuilder sb = new StringBuilder();
363                    sb.append("Line ").append(reader.getLineNum());
364                    if (propertyName != null) {
365                            sb.append(" (").append(propertyName).append(" property)");
366                    }
367                    sb.append(": ").append(message);
368    
369                    warnings.add(sb.toString());
370            }

372            private class VCardDataStreamListenerImpl implements VCardRawReader.VCardDataStreamListener {
373                    private VCard root;
374                    private final List<LabelType> labels = new ArrayList<LabelType>();
375                    private final List<String> warningsBuf = new ArrayList<String>();
376                    private final LinkedList<VCard> vcardStack = new LinkedList<VCard>();
377                    private EmbeddedVCardException embeddedVCardException;
378    
379                    public void beginComponent(String name) {
380                            if (!"VCARD".equalsIgnoreCase(name)) {
381                                    return;
382                            }
383    
384                            VCard vcard = new VCard();
385    
386                            //initialize version to 2.1, since the VERSION property can exist anywhere in a 2.1 vCard
387                            vcard.setVersion(VCardVersion.V2_1);
388    
389                            vcardStack.add(vcard);
390    
391                            if (root == null) {
392                                    root = vcard;
393                            }
394    
395                            if (embeddedVCardException != null) {
396                                    embeddedVCardException.injectVCard(vcard);
397                                    embeddedVCardException = null;
398                            }
399                    }
400    
401                    public void readVersion(VCardVersion version) {
402                            if (vcardStack.isEmpty()) {
403                                    //not in a "VCARD" component
404                                    return;
405                            }
406    
407                            vcardStack.getLast().setVersion(version);
408                    }
409    
410                    public void readProperty(String group, String name, VCardSubTypes parameters, String value) {
411                            if (vcardStack.isEmpty()) {
412                                    //not in a "VCARD" component
413                                    return;
414                            }
415    
416                            if (embeddedVCardException != null) {
417                                    //the next property was supposed to be the start of a nested vCard, but it wasn't
418                                    embeddedVCardException.injectVCard(null);
419                                    embeddedVCardException = null;
420                            }
421    
422                            handleNamelessParameters(parameters);
423    
424                            handleQuotedMultivaluedTypeParams(parameters);
425    
426                            value = decodeQuotedPrintable(name, parameters, value);
427    
428                            //create the type object
429                            VCardType type = createTypeObject(name);
430                            type.setGroup(group);
431    
432                            //unmarshal the text string into the object
433                            VCard curVCard = vcardStack.getLast();
434                            VCardVersion version = curVCard.getVersion();
435                            warningsBuf.clear();
436                            try {
437                                    type.unmarshalText(parameters, value, version, warningsBuf, compatibilityMode);
438    
439                                    //add to vcard
440                                    if (type instanceof LabelType) {
441                                            //LABELs must be treated specially so they can be matched up with their ADRs
442                                            labels.add((LabelType) type);
443                                            return;
444                                    }
445                            } catch (SkipMeException e) {
446                                    warningsBuf.add("Property has requested that it be skipped: " + e.getMessage());
447                                    return;
448                            } catch (CannotParseException e) {
449                                    warningsBuf.add("Property value could not be parsed.  Property will be saved as an extended type instead." + NEWLINE + "  Value: " + value + NEWLINE + "  Reason: " + e.getMessage());
450                                    type = new RawType(name, value);
451                                    type.setGroup(group);
452                            } catch (EmbeddedVCardException e) {
453                                    //parse an embedded vCard (i.e. the AGENT type)
454    
455                                    if (value.length() == 0 || version == VCardVersion.V2_1) {
456                                            //a nested vCard is expected to be next (2.1 style)
457                                            embeddedVCardException = e;
458                                    } else {
459                                            //the property value should be an embedded vCard (3.0 style)
460                                            value = VCardStringUtils.unescape(value);
461    
462                                            VCardReader agentReader = new VCardReader(value);
463                                            agentReader.setCompatibilityMode(compatibilityMode);
464                                            try {
465                                                    VCard nestedVCard = agentReader.readNext();
466                                                    if (nestedVCard != null) {
467                                                            e.injectVCard(nestedVCard);
468                                                    }
469                                            } catch (IOException e2) {
470                                                    //reading from a string
471                                            } finally {
472                                                    for (String w : agentReader.getWarnings()) {
473                                                            addWarning("Problem unmarshalling nested vCard value: " + w, type.getTypeName());
474                                                    }
475                                                    IOUtils.closeQuietly(agentReader);
476                                            }
477                                    }
478                            } finally {
479                                    for (String warning : warningsBuf) {
480                                            addWarning(warning, name);
481                                    }
482                            }
483    
484                            curVCard.addType(type);
485                    }
486    
487                    public void endComponent(String name) {
488                            if (vcardStack.isEmpty()) {
489                                    //not in a "VCARD" component
490                                    return;
491                            }
492    
493                            if (!"VCARD".equalsIgnoreCase(name)) {
494                                    //not a "VCARD" component
495                                    return;
496                            }
497    
498                            VCard curVCard = vcardStack.removeLast();
499    
500                            //assign labels to their addresses
501                            for (LabelType label : labels) {
502                                    boolean orphaned = true;
503                                    for (AddressType adr : curVCard.getAddresses()) {
504                                            if (adr.getLabel() == null && adr.getTypes().equals(label.getTypes())) {
505                                                    adr.setLabel(label.getValue());
506                                                    orphaned = false;
507                                                    break;
508                                            }
509                                    }
510                                    if (orphaned) {
511                                            curVCard.addOrphanedLabel(label);
512                                    }
513                            }
514    
515                            if (vcardStack.isEmpty()) {
516                                    throw new VCardRawReader.StopReadingException();
517                            }
518                    }
519    
520                    public void invalidLine(String line) {
521                            if (vcardStack.isEmpty()) {
522                                    //not in a "VCARD" component
523                                    return;
524                            }
525    
526                            addWarning("Skipping malformed line: \"" + line + "\"");
527                    }
528    
529                    public void invalidVersion(String version) {
530                            if (vcardStack.isEmpty()) {
531                                    //not in a "VCARD" component
532                                    return;
533                            }
534    
535                            addWarning("Ignoring invalid version value: " + version, "VERSION");
536                    }
537            }
