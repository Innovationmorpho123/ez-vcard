038    /**
039     * Automatically folds lines as they are written.
040     * @author Michael Angstadt
041     */
042    public class FoldedLineWriter extends Writer {
043            private final Writer writer;
044            private int curLineLength = 0;
045            private Integer lineLength;
046            private String indent;
047            private String newline;
048    
049            /**
050             * Creates a folded line writer.
051             * @param writer the writer object to wrap
052             * @param lineLength the maximum length a line can be before it is folded
053             * (excluding the newline), or null disable folding
054             * @param indent the string to prepend to each folded line (e.g. a single
055             * space character)
056             * @param newline the newline sequence to use (e.g. "\r\n")
057             * @throws IllegalArgumentException if the line length is less than or equal
058             * to zero
059             * @throws IllegalArgumentException if the length of the indent string is
060             * greater than the max line length
061             */
062            public FoldedLineWriter(Writer writer, Integer lineLength, String indent, String newline) {
063                    this.writer = writer;
064                    setLineLength(lineLength);
065                    setIndent(indent);
066                    this.newline = newline;
067            }
068    
069            /**
070             * Writes a string, followed by a newline.
071             * @param str the text to write
072             * @throws IOException if there's a problem writing to the output stream
073             */
074            public void writeln(String str) throws IOException {
075                    write(str);
076                    write(newline);
077            }
078    
079            /**
080             * Writes a string.
081             * @param str the string to write
082             * @param quotedPrintable true if the string has been encoded in
083             * quoted-printable encoding, false if not
084             * @return this
085             * @throws IOException if there's a problem writing to the output stream
086             */
087            public FoldedLineWriter append(CharSequence str, boolean quotedPrintable) throws IOException {
088                    write(str, quotedPrintable);
089                    return this;
090            }
091    
092            /**
093             * Writes a string.
094             * @param str the string to write
095             * @param quotedPrintable true if the string has been encoded in
096             * quoted-printable encoding, false if not
097             * @throws IOException if there's a problem writing to the output stream
098             */
099            public void write(CharSequence str, boolean quotedPrintable) throws IOException {
100                    write(str.toString().toCharArray(), 0, str.length(), quotedPrintable);
101            }
102    
103            @Override
104            public void write(char[] cbuf, int off, int len) throws IOException {
105                    write(cbuf, off, len, false);
106            }
107    
108            /**
109             * Writes a portion of an array of characters.
110             * @param buf the array of characters
111             * @param off the offset from which to start writing characters
112             * @param len the number of characters to write
113             * @param quotedPrintable true to convert the string to "quoted-printable"
114             * encoding, false not to
115             * @throws IOException if there's a problem writing to the output stream
116             */
117            public void write(char[] cbuf, int off, int len, boolean quotedPrintable) throws IOException {
118                    //encode to quoted-printable
119                    if (quotedPrintable) {
120                            QuotedPrintableCodec codec = new QuotedPrintableCodec();
121                            try {
122                                    cbuf = codec.encode(new String(cbuf, off, len)).toCharArray();
123                                    off = 0;
124                                    len = cbuf.length;
125                            } catch (EncoderException e) {
126                                    //only thrown if an unsupported charset is passed into the codec
127                                    throw new RuntimeException(e);
128                            }
129                    }
130    
131                    if (lineLength == null) {
132                            //if line folding is disabled, then write directly to the Writer
133                            writer.write(cbuf, off, len);
134                            return;
135                    }
136    
137                    int effectiveLineLength = lineLength;
138                    if (quotedPrintable) {
139                            //"=" must be appended onto each line
140                            effectiveLineLength -= 1;
141                    }
142    
143                    int encodedCharPos = -1;
144                    int start = off;
145                    int end = off + len;
146                    for (int i = start; i < end; i++) {
147                            char c = cbuf[i];
148    
149                            //keep track of the quoted-printable characters to prevent them from being cut in two at a folding boundary
150                            if (encodedCharPos >= 0) {
151                                    encodedCharPos++;
152                                    if (encodedCharPos == 3) {
153                                            encodedCharPos = -1;
154                                    }
155                            }
156    
157                            if (c == '\n') {
158                                    writer.write(cbuf, start, i - start + 1);
159                                    curLineLength = 0;
160                                    start = i + 1;
161                                    continue;
162                            }
163    
164                            if (c == '\r') {
165                                    if (i == end - 1 || cbuf[i + 1] != '\n') {
166                                            writer.write(cbuf, start, i - start + 1);
167                                            curLineLength = 0;
168                                            start = i + 1;
169                                    } else {
170                                            curLineLength++;
171                                    }
172                                    continue;
173                            }
174    
175                            if (c == '=' && quotedPrintable) {
176                                    encodedCharPos = 0;
177                            }
178    
179                            if (curLineLength >= effectiveLineLength) {
180                                    //if the last characters on the line are whitespace, then exceed the max line length in order to include the whitespace on the same line
181                                    //otherwise it will be lost because it will merge with the padding on the next line
182                                    if (Character.isWhitespace(c)) {
183                                            while (Character.isWhitespace(c) && i < end - 1) {
184                                                    i++;
185                                                    c = cbuf[i];
186                                            }
187                                            if (i >= end - 1) {
188                                                    //the rest of the char array is whitespace, so leave the loop
189                                                    break;
190                                            }
191                                    }
192    
193                                    //if we are in the middle of a quoted-printable encoded char, then exceed the max line length in order to print out the rest of the char
194                                    if (encodedCharPos > 0) {
195                                            i += 3 - encodedCharPos;
196                                            if (i >= end - 1) {
197                                                    //the rest of the char array was an encoded char, so leave the loop
198                                                    break;
199                                            }
200                                    }
201    
202                                    writer.write(cbuf, start, i - start);
203                                    if (quotedPrintable) {
204                                            writer.write('=');
205                                    }
206                                    writer.write(newline);
207                                    writer.write(indent);
208                                    curLineLength = indent.length() + 1;
209                                    start = i;
210    
211                                    continue;
212                            }
213    
214                            curLineLength++;
215                    }
216    
217                    writer.write(cbuf, start, end - start);
218            }
219    
220            /**
221             * Closes the writer.
222             */
223            @Override
224            public void close() throws IOException {
225                    writer.close();
226            }
227    
228            /**
229             * Flushes the writer.
230             */
231            @Override
232            public void flush() throws IOException {
233                    writer.flush();
234            }
235    
236            /**
237             * Gets the maximum length a line can be before it is folded (excluding the
238             * newline).
239             * @return the line length or null if folding is disabled
240             */
241            public Integer getLineLength() {
242                    return lineLength;
243            }
244    
245            /**
246             * Sets the maximum length a line can be before it is folded (excluding the
247             * newline).
248             * @param lineLength the line length or null to disable folding
249             * @throws IllegalArgumentException if the line length is less than or equal
250             * to zero
251             */
252            public void setLineLength(Integer lineLength) {
253                    if (lineLength != null && lineLength <= 0) {
254                            throw new IllegalArgumentException("Line length must be greater than 0.");
255                    }
256                    this.lineLength = lineLength;
257            }
258    
259            /**
260             * Gets the string that is prepended to each folded line.
261             * @return the indent string
262             */
263            public String getIndent() {
264                    return indent;
265            }
266    
267            /**
268             * Sets the string that is prepended to each folded line.
269             * @param indent the indent string (e.g. a single space character)
270             * @throws IllegalArgumentException if the length of the indent string is
271             * greater than the max line length
272             */
273            public void setIndent(String indent) {
274                    if (lineLength != null && indent.length() >= lineLength) {
275                            throw new IllegalArgumentException("The length of the indent string must be less than the max line length.");
276                    }
277                    this.indent = indent;
278            }
279    
280            /**
281             * Gets the newline sequence that is used to separate lines.
282             * @return the newline sequence
283             */
284            public String getNewline() {
285                    return newline;
286            }
287    
288            /**
289             * Sets the newline sequence that is used to separate lines
290             * @param newline the newline sequence
291             */
292            public void setNewline(String newline) {
293                    this.newline = newline;
294            }
295    
296            /**
297             * Gets the wrapped {@link Writer} object.
298             * @return the wrapped writer
299             */
300            public Writer getWriter() {
301                    return writer;
302            }
303    }
