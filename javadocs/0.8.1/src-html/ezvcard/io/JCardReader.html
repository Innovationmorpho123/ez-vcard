061    /**
062     * Unmarshals JSON-encoded vCards into {@link VCard} objects.
063     * @author Michael Angstadt
064     * @see <a href="http://tools.ietf.org/html/draft-kewisch-vcard-in-json-01">jCard draft</a>
065     */
066    //@formatter:on
067    public class JCardReader implements IParser {
068            private final Reader reader;
069            private final List<String> warnings = new ArrayList<String>();
070            private JsonParser jp;
071            private boolean vcardstream;
072            private Map<String, Class<? extends VCardType>> extendedTypeClasses = new HashMap<String, Class<? extends VCardType>>();
073            private VCardVersion version = VCardVersion.V4_0;
074    
075            private int propertiesRead;
076            private boolean versionFound;
077

111            //@Override
112            public VCard readNext() throws IOException {
113                    if (jp != null && jp.isClosed()) {
114                            return null;
115                    }
116    
117                    warnings.clear();
118                    versionFound = false;
119                    propertiesRead = 0;
120    
121                    if (jp == null) {
122                            //consume: ["vcardstream", ["vcard",
123                            //or
124                            //consume: ["vcard",
125                            initStream();
126                    } else {
127                            //consume: ["vcard",
128                            if (jp.nextToken() != JsonToken.START_ARRAY) {
129                                    if (jp.getCurrentToken() == null || (vcardstream && jp.getCurrentToken() == JsonToken.END_ARRAY)) {
130                                            //it's the end of the JSON data
131                                            return null;
132                                    } else {
133                                            throw new JCardParseException(JsonToken.START_ARRAY, jp.getCurrentToken());
134                                    }
135                            }
136    
137                            if (jp.nextToken() != JsonToken.VALUE_STRING) {
138                                    throw new JCardParseException(JsonToken.VALUE_STRING, jp.getCurrentToken());
139                            }
140                            String value = jp.getValueAsString();
141                            if (!"vcard".equals(value)) {
142                                    warnings.add("The jCard array must begin with \"vcard\", but it begins with \"" + value + "\".  Ignoring.");
143                            }
144                    }
145    
146                    //start properties array
147                    if (jp.nextToken() != JsonToken.START_ARRAY) {
148                            throw new JCardParseException(JsonToken.START_ARRAY, jp.getCurrentToken());
149                    }
150    
151                    VCard vcard = new VCard();
152                    vcard.setVersion(VCardVersion.V4_0);
153    
154                    List<String> warningsBuf = new ArrayList<String>();
155                    while (jp.nextToken() != JsonToken.END_ARRAY) {
156                            //get property name
157                            if (jp.nextToken() != JsonToken.VALUE_STRING) {
158                                    throw new JCardParseException(JsonToken.VALUE_STRING, jp.getCurrentToken());
159                            }
160                            String propertyName = jp.getValueAsString().toLowerCase();
161    
162                            //get parameters and group
163                            String group = null;
164                            VCardSubTypes subTypes = new VCardSubTypes();
165                            if (jp.nextToken() != JsonToken.START_OBJECT) {
166                                    throw new JCardParseException(JsonToken.START_OBJECT, jp.getCurrentToken());
167                            }
168                            while (jp.nextToken() != JsonToken.END_OBJECT) {
169                                    String parameterName = jp.getText();
170    
171                                    List<String> parameterValues = new ArrayList<String>();
172                                    if (jp.nextToken() == JsonToken.START_ARRAY) {
173                                            //multi-valued parameter
174                                            while (jp.nextToken() != JsonToken.END_ARRAY) {
175                                                    parameterValues.add(jp.getText());
176                                            }
177                                    } else {
178                                            parameterValues.add(jp.getValueAsString());
179                                    }
180    
181                                    if ("group".equalsIgnoreCase(parameterName)) {
182                                            group = parameterValues.get(0);
183                                    } else {
184                                            subTypes.putAll(parameterName, parameterValues);
185                                    }
186                            }
187    
188                            JCardValue jcardValue = new JCardValue();
189    
190                            //get data type
191                            if (jp.nextToken() != JsonToken.VALUE_STRING) {
192                                    throw new JCardParseException(JsonToken.VALUE_STRING, jp.getCurrentToken());
193                            }
194                            jcardValue.setDataType(JCardDataType.get(jp.getText()));
195    
196                            //get property value(s)
197                            boolean structured = false;
198                            if (jp.nextToken() == JsonToken.START_ARRAY) {
199                                    //structured property value (e.g. ["n", {}, "text", ["Doe", "John", "", "", ["Mr", "Dr"]]])
200                                    structured = true;
201                                    jp.nextToken();
202                            }
203                            jcardValue.setStructured(structured);
204                            while (jp.getCurrentToken() != JsonToken.END_ARRAY) {
205                                    List<Object> curValue = new ArrayList<Object>();
206                                    if (jp.getCurrentToken() == JsonToken.START_ARRAY) {
207                                            //multi-valued component
208                                            while (jp.nextToken() != JsonToken.END_ARRAY) {
209                                                    curValue.add(parseValueElement());
210                                            }
211                                    } else {
212                                            curValue.add(parseValueElement());
213                                    }
214                                    if (curValue.isEmpty()) {
215                                            jcardValue.addValues("");
216                                    } else {
217                                            jcardValue.addValues(curValue);
218                                    }
219                                    jp.nextToken();
220                            }
221                            if (structured) {
222                                    //consume the extra END_ARRAY character from the structured value array
223                                    if (jp.nextToken() != JsonToken.END_ARRAY) {
224                                            throw new JCardParseException(JsonToken.END_ARRAY, jp.getCurrentToken());
225                                    }
226                                    if (jcardValue.getValues().isEmpty()) {
227                                            jcardValue.addValues("");
228                                    }
229                            }
230    
231                            if (propertiesRead == 0 && !"version".equals(propertyName)) {
232                                    warnings.add("jCard does not start with the \"version\" property.  Version will be set to " + version);
233                            }
234    
235                            propertiesRead++;
236    
237                            if ("version".equals(propertyName)) {
238                                    Object firstValue = jcardValue.getFirstValue();
239                                    String firstValueStr = (firstValue == null) ? "" : firstValue.toString();
240                                    if (versionFound) {
241                                            warnings.add("Additional \"version\" property encountered: \"" + firstValueStr + "\".  It will be ignored.");
242                                    } else {
243                                            versionFound = true;
244                                            if (!version.getVersion().equals(firstValueStr)) {
245                                                    warnings.add("Invalid value of \"version\" property: " + firstValueStr);
246                                            }
247                                    }
248                                    continue;
249                            }
250    
251                            VCardType type = createTypeObject(propertyName);
252                            type.setGroup(group);
253    
254                            //unmarshal the text string into the object
255                            warningsBuf.clear();
256                            try {
257                                    type.unmarshalJson(subTypes, jcardValue, version, warningsBuf);
258                                    addToVCard(type, vcard);
259                            } catch (SkipMeException e) {
260                                    warningsBuf.add(type.getTypeName() + " property will not be unmarshalled: " + e.getMessage());
261                            } catch (EmbeddedVCardException e) {
262                                    warningsBuf.add(type.getTypeName() + " property will not be unmarshalled: jCard does not supported embedded vCards.");
263                            } finally {
264                                    warnings.addAll(warningsBuf);
265                            }
266                    }
267    
268                    if (!versionFound) {
269                            warnings.add("\"version\" property was missing from the jCard.  Used version " + version.getVersion() + ".");
270                    }
271    
272                    //consume ending of properties array
273                    if (jp.nextToken() != JsonToken.END_ARRAY) {
274                            throw new JCardParseException(JsonToken.END_ARRAY, jp.getCurrentToken());
275                    }
276    
277                    return vcard;
278            }
279    
280            private Object parseValueElement() throws JsonParseException, IOException {
281                    JsonToken curToken = jp.getCurrentToken();
282                    if (curToken == JsonToken.VALUE_FALSE || curToken == JsonToken.VALUE_TRUE) {
283                            return jp.getBooleanValue();
284                    }
285                    if (curToken == JsonToken.VALUE_NUMBER_FLOAT) {
286                            return jp.getDoubleValue();
287                    }
288                    if (curToken == JsonToken.VALUE_NUMBER_INT) {
289                            return jp.getLongValue();
290                    }
291                    if (curToken == JsonToken.VALUE_NULL) {
292                            return "";
293                    }
294                    return jp.getText();
295            }

297            private void initStream() throws JsonParseException, IOException {
298                    JsonFactory factory = new JsonFactory();
299                    jp = factory.createJsonParser(reader);
300    
301                    if (jp.nextToken() != JsonToken.START_ARRAY) {
302                            throw new JCardParseException(JsonToken.START_ARRAY, jp.getCurrentToken());
303                    }
304    
305                    if (jp.nextToken() != JsonToken.VALUE_STRING) {
306                            throw new JCardParseException(JsonToken.VALUE_STRING, jp.getCurrentToken());
307                    }
308                    String value = jp.getValueAsString();
309    
310                    if ("vcardstream".equals(value)) {
311                            vcardstream = true;
312                            if (jp.nextToken() != JsonToken.START_ARRAY) {
313                                    throw new JCardParseException(JsonToken.START_ARRAY, jp.getCurrentToken());
314                            }
315                            if (jp.nextToken() != JsonToken.VALUE_STRING) {
316                                    throw new JCardParseException(JsonToken.VALUE_STRING, jp.getCurrentToken());
317                            }
318    
319                            value = jp.getValueAsString();
320                            if (!"vcard".equals(value)) {
321                                    warnings.add("The jCard array must begin with \"vcard\", but it begins with \"" + value + "\".  Ignoring.");
322                            }
323                    } else if ("vcard".equals(value)) {
324                            vcardstream = false;
325                    } else {
326                            vcardstream = false;
327                            warnings.add("The jCard array must begin with \"vcard\", but it begins with \"" + value + "\".  Ignoring.");
328                    }
329            }

331            /**
332             * Creates the appropriate {@link VCardType} instance, given the type name.
333             * This method does not unmarshal the type, it just creates the type object.
334             * @param name the type name (e.g. "FN")
335             * @return the Type that was created
336             */
337            private VCardType createTypeObject(String name) {
338                    Class<? extends VCardType> clazz = TypeList.getTypeClass(name);
339                    VCardType t;
340                    if (clazz != null) {
341                            try {
342                                    //create a new instance of the class
343                                    t = clazz.newInstance();
344                            } catch (Exception e) {
345                                    //it is the responsibility of the ez-vcard developer to ensure that this exception is never thrown
346                                    //all type classes defined in the ez-vcard library MUST have public, no-arg constructors
347                                    throw new RuntimeException(e);
348                            }
349                    } else {
350                            Class<? extends VCardType> extendedTypeClass = extendedTypeClasses.get(name);
351                            if (extendedTypeClass != null) {
352                                    try {
353                                            t = extendedTypeClass.newInstance();
354                                    } catch (Exception e) {
355                                            //this should never happen because the type class is checked to see if it has a public, no-arg constructor in the "registerExtendedType" method
356                                            throw new RuntimeException("Extended type class \"" + extendedTypeClass.getName() + "\" must have a public, no-arg constructor.");
357                                    }
358                            } else {
359                                    t = new RawType(name); //use RawType instead of TextType because we don't want to unescape any characters that might be meaningful to this type
360                                    if (!name.startsWith("X-")) {
361                                            warnings.add("Non-standard type \"" + name + "\" found.  Treating it as an extended type.");
362                                    }
363                            }
364                    }
365                    return t;
366            }

368            /**
369             * Adds a type object to the vCard.
370             * @param t the type object
371             * @param vcard the vCard
372             */
373            private void addToVCard(VCardType t, VCard vcard) {
374                    Method method = TypeList.getAddMethod(t.getClass());
375                    if (method != null) {
376                            try {
377                                    method.invoke(vcard, t);
378                            } catch (Exception e) {
379                                    //this should NEVER be thrown because the method MUST be public
380                                    throw new RuntimeException(e);
381                            }
382                    } else {
383                            vcard.addExtendedType(t);
384                    }
385            }

387            /**
388             * Gets the type name from a type class.
389             * @param clazz the type class
390             * @return the type name
391             */
392            private String getTypeNameFromTypeClass(Class<? extends VCardType> clazz) {
393                    try {
394                            VCardType t = clazz.newInstance();
395                            return t.getTypeName().toLowerCase();
396                    } catch (Exception e) {
397                            //there is no public, no-arg constructor
398                            throw new RuntimeException(e);
399                    }
400            }
401    
402            //@Override
403            public void registerExtendedType(Class<? extends VCardType> clazz) {
404                    extendedTypeClasses.put(getTypeNameFromTypeClass(clazz), clazz);
405            }
406    
407            //@Override
408            public void unregisterExtendedType(Class<? extends VCardType> clazz) {
409                    extendedTypeClasses.remove(getTypeNameFromTypeClass(clazz));
410            }
411    
412            //@Override
413            public List<String> getWarnings() {
414                    return new ArrayList<String>(warnings);
415            }
