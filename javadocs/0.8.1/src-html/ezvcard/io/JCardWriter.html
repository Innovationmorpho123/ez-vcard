059    /**
060     * Converts vCards to their JSON representations.
061     * @author Michael Angstadt
062     */
063    public class JCardWriter implements Closeable {
064            private static final String newline = System.getProperty("line.separator");
065            private final Writer writer;
066            private JsonGenerator jg;
067            private VCardVersion targetVersion = VCardVersion.V4_0;
068            private final List<String> warnings = new ArrayList<String>();
069            private boolean addProdId = true;
070            private boolean indent = false;
071    
072            /**
073             * Creates a jCard writer.
074             * @param out the output stream to write the vCard to
075             */
076            public JCardWriter(OutputStream out) {
077                    this(new OutputStreamWriter(out));
078            }
079    
080            /**
081             * Creates a hCard writer.
082             * @param file the file to write the vCard to
083             * @throws IOException if there's a problem opening the file
084             */
085            public JCardWriter(File file) throws IOException {
086                    this(new FileWriter(file));
087            }
088    
089            /**
090             * Creates a jCard writer.
091             * @param writer the writer to write the vCard to
092             */
093            public JCardWriter(Writer writer) {
094                    this.writer = writer;
095            }
096    
097            /**
098             * Writes a vCard to the stream.
099             * @param vcard the vCard to write
100             * @throws IOException if there's a problem writing to the output stream
101             */
102            public void write(final VCard vcard) throws IOException {
103                    warnings.clear();
104    
105                    if (jg == null) {
106                            JsonFactory factory = new JsonFactory();
107                            factory.configure(Feature.AUTO_CLOSE_TARGET, false);
108                            jg = factory.createJsonGenerator(writer);
109    
110                            jg.writeStartArray();
111                            jg.writeString("vcardstream");
112                    }
113    
114                    jg.writeStartArray();
115                    indent(2);
116                    jg.writeString("vcard");
117                    jg.writeStartArray();
118    
119                    if (vcard.getFormattedName() == null) {
120                            warnings.add("vCard version " + targetVersion + " requires that a formatted name be defined.");
121                    }
122    
123                    List<VCardType> typesToAdd = new ArrayList<VCardType>();
124                    typesToAdd.add(new TextType("version", targetVersion.getVersion()));
125    
126                    for (VCardType type : vcard) {
127                            if (addProdId && type instanceof ProdIdType) {
128                                    //do not add the PRODID in the vCard if "addProdId" is true
129                                    continue;
130                            }
131    
132                            //determine if this type is supported by the target version
133                            if (!supportsTargetVersion(type)) {
134                                    warnings.add(type.getTypeName() + " is not supported by vCard version " + targetVersion + " and will not be added to the vCard.  Supported versions are: " + Arrays.toString(type.getSupportedVersions()));
135                                    continue;
136                            }
137    
138                            //check for correct KIND value if there are MEMBER types
139                            if (type instanceof MemberType && (vcard.getKind() == null || !vcard.getKind().isGroup())) {
140                                    warnings.add("KIND must be set to \"group\" in order to add MEMBER properties to the vCard.");
141                                    continue;
142                            }
143    
144                            typesToAdd.add(type);
145                    }
146    
147                    //add an extended type saying it was generated by this library
148                    if (addProdId) {
149                            EzvcardProdIdType prodId = new EzvcardProdIdType(targetVersion);
150                            typesToAdd.add(prodId);
151                    }
152    
153                    List<String> warningsBuf = new ArrayList<String>();
154                    for (VCardType type : typesToAdd) {
155                            //marshal the value
156                            warningsBuf.clear();
157                            JCardValue value = null;
158                            try {
159                                    value = type.marshalJson(targetVersion, warningsBuf);
160                            } catch (SkipMeException e) {
161                                    warningsBuf.add(type.getTypeName() + " property will not be marshalled: " + e.getMessage());
162                                    continue;
163                            } catch (EmbeddedVCardException e) {
164                                    warningsBuf.add(type.getTypeName() + " property will not be marshalled: jCard does not supported embedded vCards.");
165                            } finally {
166                                    warnings.addAll(warningsBuf);
167                            }
168    
169                            //marshal the sub types
170                            warningsBuf.clear();
171                            VCardSubTypes subTypes;
172                            try {
173                                    subTypes = type.marshalSubTypes(targetVersion, warningsBuf, CompatibilityMode.RFC, vcard);
174                            } finally {
175                                    warnings.addAll(warningsBuf);
176                            }
177                            subTypes.removeAll(ValueParameter.NAME); //remove all VALUE parameters
178    
179                            //add the group as a parameter
180                            if (type.getGroup() != null) {
181                                    subTypes.put("group", type.getGroup());
182                            }
183    
184                            jg.writeStartArray(); //start property
185                            indent(4);
186    
187                            //write property name
188                            jg.writeString(type.getTypeName().toLowerCase());
189    
190                            //write parameters
191                            jg.writeStartObject();
192                            for (Map.Entry<String, List<String>> entry : subTypes) {
193                                    String name = entry.getKey().toLowerCase();
194                                    List<String> values = entry.getValue();
195                                    if (values.isEmpty()) {
196                                            continue;
197                                    }
198    
199                                    if (values.size() == 1) {
200                                            jg.writeStringField(name, values.get(0));
201                                    } else {
202                                            jg.writeArrayFieldStart(name);
203                                            for (String paramValue : values) {
204                                                    jg.writeString(paramValue);
205                                            }
206                                            jg.writeEndArray();
207                                    }
208                            }
209                            jg.writeEndObject();
210    
211                            //write data type
212                            JCardDataType dataType = value.getDataType();
213                            if (dataType == null) {
214                                    warnings.add(type.getTypeName() + " property does not have a jCard data type associated with it.  Defaulting to \"text\".");
215                                    dataType = JCardDataType.TEXT;
216                            }
217                            jg.writeString(dataType.toString());
218    
219                            //write value
220                            if (value.isStructured()) {
221                                    jg.writeStartArray();
222                            }
223                            if (value.getValues().isEmpty()) {
224                                    jg.writeString("");
225                            } else {
226                                    for (List<Object> theValue : value.getValues()) {
227                                            if (theValue.isEmpty()) {
228                                                    jg.writeString("");
229                                                    continue;
230                                            }
231    
232                                            if (theValue.size() > 1) {
233                                                    jg.writeStartArray();
234                                            }
235                                            for (Object v : theValue) {
236                                                    if (v == null) {
237                                                            jg.writeString("");
238                                                    } else if (v instanceof Byte) {
239                                                            jg.writeNumber((Byte) v);
240                                                    } else if (v instanceof Short) {
241                                                            jg.writeNumber((Short) v);
242                                                    } else if (v instanceof Integer) {
243                                                            jg.writeNumber((Integer) v);
244                                                    } else if (v instanceof Long) {
245                                                            jg.writeNumber((Long) v);
246                                                    } else if (v instanceof Float) {
247                                                            jg.writeNumber((Float) v);
248                                                    } else if (v instanceof Double) {
249                                                            jg.writeNumber((Double) v);
250                                                    } else if (v instanceof Boolean) {
251                                                            jg.writeBoolean((Boolean) v);
252                                                    } else {
253                                                            jg.writeString(v.toString());
254                                                    }
255                                            }
256                                            if (theValue.size() > 1) {
257                                                    jg.writeEndArray();
258                                            }
259                                    }
260                            }
261                            if (value.isStructured()) {
262                                    jg.writeEndArray();
263                            }
264    
265                            jg.writeEndArray(); //end property
266                    }
267    
268                    indent(2);
269                    jg.writeEndArray();
270                    jg.writeEndArray();
271            }
272    
273            /**
274             * Determines if a property is supported by the target version.
275             * @param type the property
276             * @return true if it is supported, false if not
277             */
278            private boolean supportsTargetVersion(VCardType type) {
279                    for (VCardVersion version : type.getSupportedVersions()) {
280                            if (version == targetVersion) {
281                                    return true;
282                            }
283                    }
284                    return false;
285            }
286    
287            /**
288             * Checks to see if pretty-printing is enabled, and adds indentation
289             * whitespace if it is.
290             * @param spaces the number of spaces to indent with
291             * @throws IOException
292             */
293            private void indent(int spaces) throws IOException {
294                    if (indent) {
295                            jg.writeRaw(newline);
296                            for (int i = 0; i < spaces; i++) {
297                                    jg.writeRaw(' ');
298                            }
299                    }
300            }
301    
302            /**
303             * Gets whether or not a "PRODID" type will be added to each vCard, saying
304             * that the vCard was generated by this library.
305             * @return true if it will be added, false if not (defaults to true)
306             */
307            public boolean isAddProdId() {
308                    return addProdId;
309            }
310    
311            /**
312             * Sets whether or not to add a "PRODID" type to each vCard, saying that the
313             * vCard was generated by this library.
314             * @param addProdId true to add this type, false not to (defaults to true)
315             */
316            public void setAddProdId(boolean addProdId) {
317                    this.addProdId = addProdId;
318            }
319    
320            /**
321             * Gets whether or not the JSON will be pretty-printed.
322             * @return true if it will be pretty-printed, false if not (defaults to
323             * false)
324             */
325            public boolean isIndent() {
326                    return indent;
327            }
328    
329            /**
330             * Sets whether or not to pretty-print the JSON.
331             * @param indent true to pretty-print it, false not to (defaults to false)
332             */
333            public void setIndent(boolean indent) {
334                    this.indent = indent;
335            }
336    
337            /**
338             * Gets the warnings from the last vCard that was marshalled. This list is
339             * reset every time a new vCard is written.
340             * @return the warnings or empty list if there were no warnings
341             */
342            public List<String> getWarnings() {
343                    return new ArrayList<String>(warnings);
344            }
345    
346            /**
347             * Ends the jCard data stream, but does <b>not</b> close the underlying
348             * writer.
349             * @throws IOException if there's a problem closing the stream
350             */
351            public void endJsonStream() throws IOException {
352                    if (jg != null) {
353                            indent(0);
354                            jg.writeEndArray();
355                            jg.close();
356                    }
357            }
358    
359            /**
360             * Ends the jCard data stream and closes the underlying writer.
361             * @throws IOException if there's a problem closing the stream
362             */
363            public void close() throws IOException {
364                    endJsonStream();
365                    writer.close();
366            }
