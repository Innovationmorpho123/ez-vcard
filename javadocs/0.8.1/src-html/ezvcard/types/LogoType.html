039    /**
040     * A company logo.
041     * 
042     * <p>
043     * <b>Adding a logo</b>
044     * </p>
045     * 
046     * <pre>
047     * VCard vcard = new VCard();
048     * 
049     * //URL
050     * LogoType logo = new LogoType("http://www.company.com/logo.png", ImageTypeParameter.PNG);
051     * vcard.addLogo(logo);
052     * 
053     * //binary data
054     * byte data[] = ...
055     * logo = new LogoType(data, ImageTypeParameter.PNG);
056     * vcard.addLogo(logo);
057     * 
058     * //if "ImageTypeParameter" does not have the pre-defined constant that you need, then create a new instance
059     * //arg 1: the value of the 2.1/3.0 TYPE parameter
060     * //arg 2: the value to use for the 4.0 MEDIATYPE parameter and for 4.0 data URIs
061     * //arg 3: the file extension of the data type (optional)
062     * ImageKeyTypeParameter param = new ImageTypeParameter("bmp", "image/x-ms-bmp", "bmp");
063     * logo = new LogoType("http://www.company.com/logo.bmp", param);
064     * vcard.addLogo(logo);
065     * </pre>
066     * 
067     * <p>
068     * <b>Getting the logos</b>
069     * </p>
070     * 
071     * <pre>
072     * VCard vcard = ...
073     * 
074     * int fileCount = 0;
075     * for (LogoType logo : vcard.getLogos()){
076     *   //the logo will have either a URL or a binary data
077     *   if (logo.getData() == null){
078     *     System.out.println("Logo URL: " + logo.getUrl());
079     *   } else {
080     *     ImageTypeParameter type = logo.getContentType();
081     *     
082     *     if (type == null) {
083     *       //the vCard may not have any content type data associated with the logo
084     *       System.out.println("Saving a logo file...");
085     *     } else {
086     *       System.out.println("Saving a \"" + type.getMediaType() + "\" file...");
087     *     }
088     *     
089     *     String folder;
090     *     if (type == ImageTypeParameter.PNG){ //it is safe to use "==" instead of "equals()"
091     *       folder = "png-files";
092     *     } else {
093     *       folder = "image-files";
094     *     }
095     *     
096     *     byte data[] = logo.getData();
097     *     String filename = "logo" + fileCount;
098     *     if (type != null && type.getExtension() != null){
099     *      filename += "." + type.getExtension();
100     *     }
101     *     OutputStream out = new FileOutputStream(new File(folder, filename));
102     *     out.write(data);
103     *     out.close();
104     *     fileCount++;
105     *   }
106     * }
107     * </pre>
108     * 
109     * <p>
110     * <b>Property name:</b> <code>LOGO</code>
111     * </p>
112     * <p>
113     * <b>Supported versions:</b> <code>2.1, 3.0, 4.0</code>
114     * </p>
115     * @author Michael Angstadt
116     */
117    public class LogoType extends ImageType {
118            public static final String NAME = "LOGO";
119    
120            public LogoType() {
121                    super(NAME);
122            }
123    
124            /**
125             * @param url the URL to the logo
126             * @param type the content type (e.g. PNG)
127             */
128            public LogoType(String url, ImageTypeParameter type) {
129                    super(NAME, url, type);
130            }
131    
132            /**
133             * @param data the binary data of the logo
134             * @param type the content type (e.g. PNG)
135             */
136            public LogoType(byte[] data, ImageTypeParameter type) {
137                    super(NAME, data, type);
138            }
139    
140            /**
141             * @param in an input stream to the binary data (will be closed)
142             * @param type the content type (e.g. PNG)
143             * @throws IOException if there's a problem reading from the input stream
144             */
145            public LogoType(InputStream in, ImageTypeParameter type) throws IOException {
146                    super(NAME, in, type);
147            }
148    
149            /**
150             * @param file the image file
151             * @param type the content type (e.g. PNG)
152             * @throws IOException if there's a problem reading from the file
153             */
154            public LogoType(File file, ImageTypeParameter type) throws IOException {
155                    super(NAME, file, type);
156            }
157    
158            /**
159             * Gets the language that the address is written in.
160             * @return the language or null if not set
161             * @see VCardSubTypes#getLanguage
162             */
163            public String getLanguage() {
164                    return subTypes.getLanguage();
165            }
166    
167            /**
168             * Sets the language that the address is written in.
169             * @param language the language or null to remove
170             * @see VCardSubTypes#setLanguage
171             */
172            public void setLanguage(String language) {
173                    subTypes.setLanguage(language);
174            }
175    }
