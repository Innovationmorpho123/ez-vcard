042    /**
043     * Represents a vCard type that stores image data (for example, the PHOTO and
044     * LOGO types).
045     * @author Michael Angstadt
046     */
047    public class ImageType extends BinaryType<ImageTypeParameter> {
048            /**
049             * @param typeName the type name (e.g. "PHOTO")
050             */
051            public ImageType(String typeName) {
052                    super(typeName);
053            }
054    
055            /**
056             * @param typeName the type name (e.g. "PHOTO")
057             * @param url the URL to the image
058             * @param type the content type (e.g. JPEG)
059             */
060            public ImageType(String typeName, String url, ImageTypeParameter type) {
061                    super(typeName, url, type);
062            }
063    
064            /**
065             * @param typeName the type name (e.g. "PHOTO")
066             * @param data the binary data of the image
067             * @param type the content type (e.g. JPEG)
068             */
069            public ImageType(String typeName, byte[] data, ImageTypeParameter type) {
070                    super(typeName, data, type);
071            }
072    
073            /**
074             * @param typeName the type name (e.g. "PHOTO")
075             * @param in an input stream to the binary data (will be closed)
076             * @param type the content type (e.g. JPEG)
077             * @throws IOException if there's a problem reading from the input stream
078             */
079            public ImageType(String typeName, InputStream in, ImageTypeParameter type) throws IOException {
080                    super(typeName, in, type);
081            }
082    
083            /**
084             * @param typeName the type name (e.g. "PHOTO")
085             * @param file the image file
086             * @param type the content type (e.g. JPEG)
087             * @throws IOException if there's a problem reading from the file
088             */
089            public ImageType(String typeName, File file, ImageTypeParameter type) throws IOException {
090                    super(typeName, file, type);
091            }
092    
093            @Override
094            protected ImageTypeParameter buildTypeObj(String type) {
095                    ImageTypeParameter param = ImageTypeParameter.valueOf(type);
096                    if (param == null) {
097                            param = new ImageTypeParameter(type, "image/" + type, null);
098                    }
099                    return param;
100            }
101    
102            @Override
103            protected ImageTypeParameter buildMediaTypeObj(String mediaType) {
104                    ImageTypeParameter p = ImageTypeParameter.findByMediaType(mediaType);
105                    if (p == null) {
106                            int slashPos = mediaType.indexOf('/');
107                            String type;
108                            if (slashPos == -1 || slashPos == mediaType.length() - 1) {
109                                    type = "";
110                            } else {
111                                    type = mediaType.substring(slashPos + 1);
112                            }
113                            p = new ImageTypeParameter(type, mediaType, null);
114                    }
115                    return p;
116            }
117    
118            @Override
119            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
120                    String elementName = element.tagName();
121                    if ("img".equals(elementName)) {
122                            String src = element.absUrl("src");
123                            if (src.length() > 0) {
124                                    try {
125                                            DataUri uri = new DataUri(src);
126                                            ImageTypeParameter mediaType = buildMediaTypeObj(uri.getContentType());
127                                            setData(uri.getData(), mediaType);
128                                    } catch (IllegalArgumentException e) {
129                                            //TODO create buildTypeObjFromExtension() method
130                                            setUrl(src, null);
131                                    }
132                            } else {
133                                    throw new SkipMeException("<img> tag does not have a \"src\" attribute.");
134                            }
135                    } else {
136                            super.doUnmarshalHtml(element, warnings);
137                    }
138            }
139    }
