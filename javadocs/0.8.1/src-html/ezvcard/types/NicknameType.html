038    /**
039     * A list of nicknames the person goes by.
040     * 
041     * <p>
042     * <b>Code sample</b>
043     * </p>
044     * 
045     * <pre>
046     * VCard vcard = new VCard();
047     * NicknameType nickname = new NicknameType();
048     * nickname.addValue(&quot;Rick&quot;);
049     * nickname.addValue(&quot;Ricky&quot;);
050     * nickname.addValue(&quot;Bobby&quot;);
051     * vcard.setNickname(nickname);
052     * </pre>
053     * 
054     * <p>
055     * <b>Property name:</b> <code>NICKNAME</code>
056     * </p>
057     * <p>
058     * <b>Supported versions:</b> <code>3.0, 4.0</code>
059     * </p>
060     * @author Michael Angstadt
061     */
062    public class NicknameType extends TextListType implements HasAltId {
063            public static final String NAME = "NICKNAME";
064    
065            public NicknameType() {
066                    super(NAME, ',');
067            }
068    
069            /**
070             * Gets the TYPE parameter.
071             * <p>
072             * <b>Supported versions:</b> <code>4.0</code>
073             * </p>
074             * @return the TYPE value (typically, this will be either "work" or "home")
075             * or null if it doesn't exist
076             */
077            public String getType() {
078                    return subTypes.getType();
079            }
080    
081            /**
082             * Sets the TYPE parameter.
083             * <p>
084             * <b>Supported versions:</b> <code>4.0</code>
085             * </p>
086             * @param type the TYPE value (this should be either "work" or "home") or
087             * null to remove
088             */
089            public void setType(String type) {
090                    subTypes.setType(type);
091            }
092    
093            /**
094             * Gets the language the name is written in.
095             * @return the language or null if not set
096             * @see VCardSubTypes#getLanguage
097             */
098            public String getLanguage() {
099                    return subTypes.getLanguage();
100            }
101    
102            /**
103             * Sets the language the name is written in.
104             * @param language the language or null to remove
105             * @see VCardSubTypes#setLanguage
106             */
107            public void setLanguage(String language) {
108                    subTypes.setLanguage(language);
109            }
110    
111            /**
112             * Gets all PID parameter values.
113             * <p>
114             * <b>Supported versions:</b> <code>4.0</code>
115             * </p>
116             * @return the PID values or empty set if there are none
117             * @see VCardSubTypes#getPids
118             */
119            public List<Integer[]> getPids() {
120                    return subTypes.getPids();
121            }
122    
123            /**
124             * Adds a PID value.
125             * <p>
126             * <b>Supported versions:</b> <code>4.0</code>
127             * </p>
128             * @param localId the local ID
129             * @param clientPidMapRef the ID used to reference the property's globally
130             * unique identifier in the CLIENTPIDMAP property.
131             * @see VCardSubTypes#addPid(int, int)
132             */
133            public void addPid(int localId, int clientPidMapRef) {
134                    subTypes.addPid(localId, clientPidMapRef);
135            }
136    
137            /**
138             * Removes all PID values.
139             * <p>
140             * <b>Supported versions:</b> <code>4.0</code>
141             * </p>
142             * @see VCardSubTypes#removePids
143             */
144            public void removePids() {
145                    subTypes.removePids();
146            }
147    
148            /**
149             * Gets the preference value.
150             * <p>
151             * <b>Supported versions:</b> <code>4.0</code>
152             * </p>
153             * @return the preference value or null if it doesn't exist
154             * @see VCardSubTypes#getPref
155             */
156            public Integer getPref() {
157                    return subTypes.getPref();
158            }
159    
160            /**
161             * Sets the preference value.
162             * <p>
163             * <b>Supported versions:</b> <code>4.0</code>
164             * </p>
165             * @param pref the preference value or null to remove
166             * @see VCardSubTypes#setPref
167             */
168            public void setPref(Integer pref) {
169                    subTypes.setPref(pref);
170            }
171    
172            //@Override
173            public String getAltId() {
174                    return subTypes.getAltId();
175            }
176    
177            //@Override
178            public void setAltId(String altId) {
179                    subTypes.setAltId(altId);
180            }
181    
182            @Override
183            public VCardVersion[] getSupportedVersions() {
184                    return new VCardVersion[] { VCardVersion.V3_0, VCardVersion.V4_0 };
185            }
186    
187            @Override
188            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
189                    addValue(element.value());
190            }
191    }
