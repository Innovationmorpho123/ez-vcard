038    /**
039     * <p>
040     * Represents a date in which some of the components are missing. This is used
041     * to represent reduced accuracy and truncated dates, as defined in ISO8601.
042     * </p>
043     * <p>
044     * A <b>truncated date<b> is a date where the "lesser" components are missing.
045     * For example, "12:30" is truncated because the "seconds" component is missing.
046     * 
047     * <pre>
048     * PartialDate date = PartialDate.time(12, 30, null);
049     * </pre>
050     * 
051     * </p>
052     * <p>
053     * A <b>reduced accuracy date<b> is a date where the "greater" components are
054     * missing. For example, "April 20" is reduced accuracy because the "year"
055     * component is missing.
056     * 
057     * <pre>
058     * PartialDate date = PartialDate.date(null, 4, 20);
059     * </pre>
060     * 
061     * </p>
062     * @author Michael Angstadt
063     */
064    public class PartialDate {
065            private static final int SKIP = -1;
066            private static final int YEAR = 0;
067            private static final int MONTH = 1;
068            private static final int DATE = 2;
069            private static final int HOUR = 3;
070            private static final int MINUTE = 4;
071            private static final int SECOND = 5;
072            private static final int TIMEZONE_HOUR = 6;
073            private static final int TIMEZONE_MINUTE = 7;
074    
075            //@formatter:off
076            private static final Format dateFormats[] = new Format[] {
077                    new Format("(\\d{4})", YEAR),
078                    new Format("(\\d{4})-(\\d{2})", YEAR, MONTH),
079                    new Format("(\\d{4})-?(\\d{2})-?(\\d{2})", YEAR, MONTH, DATE),
080                    new Format("--(\\d{2})-?(\\d{2})", MONTH, DATE),
081                    new Format("--(\\d{2})", MONTH),
082                    new Format("---(\\d{2})", DATE)
083            };
084            //@formatter:on
085    
086            private static final String timezoneRegex = "(([-+]\\d{1,2}):?(\\d{2})?)?";
087    
088            //@formatter:off
089            private static final Format timeFormats[] = new Format[] {
090                    new Format("(\\d{2})" + timezoneRegex, HOUR, SKIP, TIMEZONE_HOUR, TIMEZONE_MINUTE),
091                    new Format("(\\d{2}):?(\\d{2})" + timezoneRegex, HOUR, MINUTE, SKIP, TIMEZONE_HOUR, TIMEZONE_MINUTE),
092                    new Format("(\\d{2}):?(\\d{2}):?(\\d{2})" + timezoneRegex, HOUR, MINUTE, SECOND, SKIP, TIMEZONE_HOUR, TIMEZONE_MINUTE),
093                    new Format("-(\\d{2}):?(\\d{2})" + timezoneRegex, MINUTE, SECOND, SKIP, TIMEZONE_HOUR, TIMEZONE_MINUTE),
094                    new Format("-(\\d{2})" + timezoneRegex, MINUTE, SKIP, TIMEZONE_HOUR, TIMEZONE_MINUTE),
095                    new Format("--(\\d{2})" + timezoneRegex, SECOND, SKIP, TIMEZONE_HOUR, TIMEZONE_MINUTE)
096            };
097            //@formatter:on

230            /**
231             * <p>
232             * Creates a new partial date.
233             * </p>
234             * <p>
235             * The following combinations are not allowed and will result in an
236             * {@link IllegalArgumentException} being thrown:
237             * </p>
238             * <ul>
239             * <li>year, date (without month)</li>
240             * <li>hour, second (without minute)</li>
241             * <li>timezoneMinute (without timezoneHour)</li>
242             * </ul>
243             * @param year the year or null to exclude
244             * @param month the month or null to exclude
245             * @param date the day of the month or null to exclude
246             * @param hour the hour or null to exclude
247             * @param minute the minute or null to exclude
248             * @param second the second or null to exclude
249             * @param timezoneHour the timezone hour or null to exclude
250             * @param timezoneMinute the timezone minute or null to exclude
251             * @throws IllegalArgumentException if an invalid combination is entered or
252             * a component value is invalid (e.g. a negative minute)
253             */
254            public PartialDate(Integer year, Integer month, Integer date, Integer hour, Integer minute, Integer second, Integer timezoneHour, Integer timezoneMinute) {
255                    //check for illegal values
256                    if (month != null && (month < 1 || month > 12)) {
257                            throw new IllegalArgumentException("Month must be between 1 and 12 inclusive.");
258                    }
259                    if (date != null && (date < 1 || date > 31)) {
260                            throw new IllegalArgumentException("Date must be between 1 and 31 inclusive.");
261                    }
262                    if (hour != null && (hour < 0 || hour > 23)) {
263                            throw new IllegalArgumentException("Hour must be between 0 and 23 inclusive.");
264                    }
265                    if (minute != null && (minute < 0 || minute > 59)) {
266                            throw new IllegalArgumentException("Minute must be between 0 and 59 inclusive.");
267                    }
268                    if (second != null && (second < 0 || second > 59)) {
269                            throw new IllegalArgumentException("Second must be between 0 and 59 inclusive.");
270                    }
271                    if (timezoneMinute != null && (timezoneMinute < 0 || timezoneMinute > 59)) {
272                            throw new IllegalArgumentException("Timezone minute must be between 0 and 59 inclusive.");
273                    }
274    
275                    //check for illegal combinations
276                    if (year != null && month == null && date != null) {
277                            throw new IllegalArgumentException("Invalid date component combination: year, date");
278                    }
279                    if (hour != null && minute == null && second != null) {
280                            throw new IllegalArgumentException("Invalid time component combination: hour, second");
281                    }
282                    if (timezoneHour == null && timezoneMinute != null) {
283                            throw new IllegalArgumentException("Timezone minute was specified without an hour component.");
284                    }
285    
286                    //assign values
287                    components[YEAR] = year;
288                    components[MONTH] = month;
289                    components[DATE] = date;
290                    components[HOUR] = hour;
291                    components[MINUTE] = minute;
292                    components[SECOND] = second;
293                    components[TIMEZONE_HOUR] = timezoneHour;
294                    components[TIMEZONE_MINUTE] = timezoneMinute;
295            }

297            /**
298             * Parses a partial date from a string.
299             * @param string the string (e.g. "--0420T15")
300             */
301            public PartialDate(String string) {
302                    String split[] = string.split("T");
303                    boolean success;
304                    if (split.length == 1) {
305                            //date or time
306                            success = parseDate(string) || parseTime(string);
307                    } else if (split[0].length() == 0) {
308                            //time
309                            success = parseTime(split[1]);
310                    } else {
311                            //date and time
312                            success = parseDate(split[0]) && parseTime(split[1]);
313                    }
314    
315                    if (!success) {
316                            throw new IllegalArgumentException("Could not parse date: " + string);
317                    }
318            }

320            private boolean parseDate(String value) {
321                    for (Format regex : dateFormats) {
322                            if (regex.parse(this, value)) {
323                                    return true;
324                            }
325                    }
326                    return false;
327            }
328    
329            private boolean parseTime(String value) {
330                    for (Format regex : timeFormats) {
331                            if (regex.parse(this, value)) {
332                                    return true;
333                            }
334                    }
335                    return false;
336            }

442            /**
443             * Converts this partial date to its ISO 8601 representation.
444             * @param extended true to use extended format, false to use basic
445             * @return the ISO 8601 representation
446             */
447            public String toDateAndOrTime(boolean extended) {
448                    StringBuilder sb = new StringBuilder();
449                    NumberFormat nf = new DecimalFormat("00");
450    
451                    String yearStr = hasYear() ? getYear().toString() : null;
452                    String monthStr = hasMonth() ? nf.format(getMonth()) : null;
453                    String dateStr = hasDate() ? nf.format(getDate()) : null;
454    
455                    String dash = extended ? "-" : "";
456                    if (hasYear() && !hasMonth() && !hasDate()) {
457                            sb.append(yearStr);
458                    } else if (!hasYear() && hasMonth() && !hasDate()) {
459                            sb.append("--").append(monthStr);
460                    } else if (!hasYear() && !hasMonth() && hasDate()) {
461                            sb.append("---").append(dateStr);
462                    } else if (hasYear() && hasMonth() && !hasDate()) {
463                            sb.append(yearStr).append("-").append(monthStr);
464                    } else if (!hasYear() && hasMonth() && hasDate()) {
465                            sb.append("--").append(monthStr).append(dash).append(dateStr);
466                    } else if (hasYear() && !hasMonth() && hasDate()) {
467                            throw new IllegalStateException("Invalid date component combination: year, date");
468                    } else if (hasYear() && hasMonth() && hasDate()) {
469                            sb.append(yearStr).append(dash).append(monthStr).append(dash).append(dateStr);
470                    }
471    
472                    if (hasTimeComponent()) {
473                            sb.append('T');
474    
475                            String hourStr = hasHour() ? nf.format(getHour()) : null;
476                            String minuteStr = hasMinute() ? nf.format(getMinute()) : null;
477                            String secondStr = hasSecond() ? nf.format(getSecond()) : null;
478    
479                            dash = extended ? ":" : "";
480                            if (hasHour() && !hasMinute() && !hasSecond()) {
481                                    sb.append(hourStr);
482                            } else if (!hasHour() && hasMinute() && !hasSecond()) {
483                                    sb.append("-").append(minuteStr);
484                            } else if (!hasHour() && !hasMinute() && hasSecond()) {
485                                    sb.append("--").append(secondStr);
486                            } else if (hasHour() && hasMinute() && !hasSecond()) {
487                                    sb.append(hourStr).append(dash).append(minuteStr);
488                            } else if (!hasHour() && hasMinute() && hasSecond()) {
489                                    sb.append("-").append(minuteStr).append(dash).append(secondStr);
490                            } else if (hasHour() && !hasMinute() && hasSecond()) {
491                                    throw new IllegalStateException("Invalid time component combination: hour, second");
492                            } else if (hasHour() && hasMinute() && hasSecond()) {
493                                    sb.append(hourStr).append(dash).append(minuteStr).append(dash).append(secondStr);
494                            }
495    
496                            if (hasTimezone()) {
497                                    Integer[] timezone = getTimezone();
498                                    if (timezone[1] == null) {
499                                            timezone[1] = 0;
500                                    }
501                                    sb.append(VCardDateFormatter.formatTimeZone(timezone[0], timezone[1], extended));
502                            }
503                    }
504    
505                    return sb.toString();
506            }

510                    final int prime = 31;
511                    int result = 1;
512                    result = prime * result + Arrays.hashCode(components);

518                    if (this == obj)
519                            return true;
520                    if (obj == null)
521                            return false;
522                    if (getClass() != obj.getClass())
523                            return false;

528            }

535            /**
536             * Represents a string format that a partial date can be in.
537             */
538            private static class Format {
539                    private Pattern regex;
540                    private int[] componentIndexes;
541    
542                    /**
543                     * @param regex the regular expression that describes the format
544                     * @param componentIndexes the indexes of the
545                     * {@link PartialDate#components} array to assign the value of each
546                     * regex group to, or -1 to ignore the group
547                     */
548                    public Format(String regex, int... componentIndexes) {
549                            this.regex = Pattern.compile("^" + regex + "$");
550                            this.componentIndexes = componentIndexes;
551                    }
552    
553                    /**
554                     * Tries to parse a given string.
555                     * @param partialDate the {@link PartialDate} object
556                     * @param value the string
557                     * @return true if the string was successfully parsed, false if not
558                     */
559                    public boolean parse(PartialDate partialDate, String value) {
560                            Matcher m = regex.matcher(value);
561                            if (m.find()) {
562                                    for (int i = 0; i < componentIndexes.length; i++) {
563                                            int index = componentIndexes[i];
564                                            if (index == SKIP) {
565                                                    continue;
566                                            }
567    
568                                            int group = i + 1;
569                                            String groupStr = m.group(group);
570                                            if (groupStr != null) {
571                                                    if (groupStr.startsWith("+")) {
572                                                            groupStr = groupStr.substring(1);
573                                                    }
574                                                    partialDate.components[index] = Integer.valueOf(groupStr);
575                                            }
576                                    }
577                                    return true;
578                            }
579                            return false;
580                    }
581            }
